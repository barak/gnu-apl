119Token
120NonscalarFunction_default_identity::eval_identity_fun(Value_P B,
121 sAxis axis) const
122{
123 // axis is already normalized to IO←0
124 // return Z←,/B0 where (B0 , B) is B.
125
126const Shape & shape_B = B->get_shape();
127const sRank rank_B = shape_B.get_rank();
128 if (rank_B < 1) RANK_ERROR; // identity restriction, lrm p. 212
129 if (axis >= rank_B) RANK_ERROR;
130
131 // eval_identity_fun() is supposedly only called when the length of axis is 0
132const ShapeItem axis_len = shape_B.get_shape_item(axis);
133 Assert(axis_len == 0);
134
135const Shape shape_Z = shape_B.without_axis(axis);
136
137 /* the removal of the reduction axis must not create a non-empty result.
138
139 In IBM APL2:
140
141 ┌───── reduction axis
142 │
143 ⍴ ,/ 0 0⍴42 → 0
144 ⍴ ,/ 0 3⍴42 → 0 (not happening here since axis_len == 0)
145 ⍴ ,/ 3 0⍴42 → DOMAIN ERROR (volume would be 3)
146 ⍴ ,/ 0⍴42 → DOMAIN ERROR (volume would be 1)
147 but: ⍴ +/ 0⍴42 → ⍬ and +/ 0⍴42 → 0 (integer scalar 0)
148 ⍴ ×/ 0⍴42 → ⍬ and +/ 0⍴42 → 1 (integer scalar 1)
149 */
150 if (shape_Z.get_rank() && shape_Z.get_volume() > 0) DOMAIN_ERROR;
151
152Value_P Z(shape_Z, LOC);
153 Z->set_default(*B, LOC);
154 Z->check_value(LOC);
155 return Token(TOK_APL_VALUE1, Z);
156}

174//----------------------------------------------------------------------------
175Token
176Bif_F12_RHO::eval_AB(Value_P A, Value_P B) const
177{
178#ifdef PERFORMANCE_COUNTERS_WANTED
179const uint64_t start_1 = cycle_counter();
180#endif
181
182const Shape shape_Z(*A, 0);
183
184 // check that shape_Z is positive
185 //
186 loop(r, shape_Z.get_rank())
187 {
188 if (shape_Z.get_shape_item(r) < 0) DOMAIN_ERROR;
189 }
190
191const ShapeItem len_Z = shape_Z.get_volume();
192
193 if (len_Z <= B->element_count() && // 1. Z is not longer than B
194 B->get_owner_count() == 2 && // 2. B is a temporary value
195 this == Workspace::SI_top()->get_prefix().get_dyadic_fun()) // 3. below
196 {
197 /* Optimization of Z←A⍴B. At this point:
198
199 1. Z is not longer than B, and
200 2. B has only 2 owners:
201 2a. the prefix (who will discard it after we return), and
202 2b. our Value_P B.
203 3. A⍴B was called from a reduction rule (Prefix::reduce_A_F_B())
204
205 We will give up our ownership 2a. on return ret; below, and
206 Prefix::reduce_A_F_B prefix will Prefix::pop_args_push_result()
207 and hence give up its ownership 2b, causeing B to be erased.
208
209 That means that B will no longer be used and that, instead of
210 of copying B into a new Z and then erasing B, we can reshape B
211 in place and return the reshaped B.
212
213 return Token(TOK_APL_VALUE1, B); below will take ownership of B
214 so that Prefix::reduce_A_F_B() won't erase B.
215 */
216 Log(LOG_optimization) CERR << "optimizing A⍴B" << endl;
217
218 // release the no longer used cells of B after shape_Z.
219 //
220 const ShapeItem len_B = B->element_count(); // all Cells
221 ShapeItem rest = len_Z; // Cells remaining
222 if (rest == 0) // Z is empty
223 {
224 rest = 1;
225 if (B->get_cproto().is_pointer_cell())
226 {
227 B->get_cproto().get_pointer_value()->to_type(false);
228 }
229 else
230 {
231 B->get_wproto().init_type(B->get_cproto(), *B, LOC);
232 }
233 }
234
235 // release the Cells after Z
236 while (rest < len_B) B->release(rest++, LOC);
237
238 B->set_shape(shape_Z);
239
240#ifdef PERFORMANCE_COUNTERS_WANTED
241const uint64_t end_1 = cycle_counter();
242 Performance::fs_F12_RHO_AB.add_sample(end_1 - start_1,
243 B->nz_element_count());
244#endif
245
246 return Token(TOK_APL_VALUE1, B);
247 }
248
249#ifdef PERFORMANCE_COUNTERS_WANTED
250Token ret = do_reshape(shape_Z, *B);
251const uint64_t end_1 = cycle_counter();
252 Performance::fs_F12_RHO_AB.add_sample(end_1 - start_1,
253 shape_Z.get_volume());
254 return ret;
255#else
256 return do_reshape(shape_Z, *B);
257#endif
258}
259//----------------------------------------------------------------------------
260Token
261Bif_F12_RHO::do_reshape(const Shape & shape_Z, const Value & B)
262{
263const ShapeItem len_B = B.element_count();
264
265Value_P Z(shape_Z, LOC);
266const ShapeItem len_Z = Z->element_count();
267
268 if (len_B == 0) // empty B: use prototype of B for all cells of Z
269 {
270 loop(z, len_Z) Z->next_ravel_Proto(B.get_cproto());
271 }
272 else
273 {
274 loop(z, len_Z)
275 {
276 Z->next_ravel_Cell(B.get_cravel(z % len_B));
277 }
278 }
279
280 Z->set_default(B, LOC);
281 Z->check_value(LOC);
282 return Token(TOK_APL_VALUE1, Z);
283}

285Token
286Bif_ROTATE::reverse(Value_P B, sAxis axis)
287{
288 if (B->is_scalar())
289 {
290 Token result(TOK_APL_VALUE1, CLONE_P(B, LOC));
291 return result;
292 }
293
294const Shape3 shape_B3(B->get_shape(), axis);
295
296Value_P Z(B->get_shape(), LOC);
297
298 loop(h, shape_B3.h())
299 {
300 // plane h is the first element in B[h;;]
301 //
302 const ShapeItem plane_h = h * shape_B3.m() * shape_B3.l();
303 loop(m, shape_B3.m())
304 {
305 // col_m is the source column in B for Z[h;m;0]
306 const ShapeItem col_m = plane_h
307 + (shape_B3.l() * (shape_B3.m() - m - 1));
308 loop(l, shape_B3.l())
309 {
310 Z->next_ravel_Cell(B->get_cravel(col_m + l));
311 }
312 }
313 }
314
315 Z->set_default(*B.get(), LOC);
316 Z->check_value(LOC);
317 return Token(TOK_APL_VALUE1, Z);
318}
319//----------------------------------------------------------------------------
320Token
321Bif_ROTATE::rotate(Value_P A, Value_P B, sAxis axis)
322{
323int32_t gsh = 0; // global shift (scalar A); 0 means local shift (A) used.
324
325const Shape3 shape_B3(B->get_shape(), axis);
326const Shape shape_A2(shape_B3.h(), shape_B3.l());
327
328 if (A->is_scalar_or_len1_vector())
329 {
330 gsh = A->get_cfirst().get_near_int();
331 if (gsh == 0) // nothing to do.
332 {
333 Token result(TOK_APL_VALUE1, CLONE_P(B, LOC));
334 return result;
335 }
336 }
337 else // otherwise shape A must be shape B with 'axis' removed.
338 {
339 A->get_shape().check_same(B->get_shape().without_axis(axis),
340 E_RANK_ERROR, E_LENGTH_ERROR, LOC);
341 }
342
343
344Value_P Z(B->get_shape(), LOC);
345
346 loop(h, shape_B3.h())
347 loop(m, shape_B3.m())
348 loop(l, shape_B3.l())
349 {
350 ShapeItem src = gsh;
351 if (!src) src = A->get_cravel(l + h*shape_B3.l()).get_near_int();
352 src += shape_B3.m() + m;
353 while (src < 0) src += shape_B3.m();
354 while (src >= shape_B3.m()) src -= shape_B3.m();
355 Z->next_ravel_Cell(B->get_cravel(shape_B3.hml(h, src, l)));
356 }
357
358 Z->set_default(*B.get(), LOC);
359
360 Z->check_value(LOC);
361 return Token(TOK_APL_VALUE1, Z);
362}

407Token
408Bif_F12_TRANSPOSE::eval_AB(Value_P A, Value_P B) const
409{
410 // A should be a scalar or vector.
411 //
412 if (A->get_rank() > 1) RANK_ERROR;
413
414 if (B->is_scalar()) // B is a scalar (so A should be empty)
415 {
416 if (A->element_count() != 0) LENGTH_ERROR;
417 Value_P Z = CLONE_P(B, LOC);
418 Z->check_value(LOC);
419 return Token(TOK_APL_VALUE1, Z);
420 }
421
422const Shape shape_A(*A, Workspace::get_IO());
423 if (shape_A.get_rank() != B->get_rank()) LENGTH_ERROR;
424
425 // the elements in A shall be valid axes of B->
426 loop(r, A->get_rank())
427 {
428 if (shape_A.get_shape_item(r) < 0) DOMAIN_ERROR;
429 if (shape_A.get_shape_item(r) >= B->get_rank()) DOMAIN_ERROR;
430 }
431
432Value_P Z = (shape_A.get_rank() == B->get_rank() && shape_A.is_permutation())
433 ? transpose(shape_A, B.get())
434 : transpose_diag(shape_A, B.get());
435
436 Z->set_default(*B.get(), LOC);
437
438 Z->check_value(LOC);
439 return Token(TOK_APL_VALUE1, Z);
440}
441//----------------------------------------------------------------------------
442Value_P
443Bif_F12_TRANSPOSE::transpose(const Shape & A, const Value * B)
444{
445 // some simple to optimize cases beforehand...
446 //
447 if (A.get_rank() <= 2)
448 {
449 if (A.get_rank() <= 1) // scalar or vector B:
450 {
451 return CLONE(B, LOC);
452 }
453
454 // 2-dimensional matrix
455 //
456 if (A.get_shape_item(0) == 0 &&
457 A.get_shape_item(1) == 1) return CLONE(B, LOC); // identity
458
459 const ShapeItem rows_B = B->get_shape_item(0);
460 const ShapeItem cols_B = B->get_shape_item(1);
461 const Shape shape_Z(cols_B, rows_B);
462 Value_P Z(shape_Z, LOC);
463 loop(rZ, cols_B) // the rows of B are columns of Z
464 loop(cZ, rows_B) // the columns of B are rows of Z
465 Z->next_ravel_Cell(B->get_cravel(rZ + cZ*cols_B));
466 Z->check_value(LOC);
467 return Z;
468 }
469
470 /*
471 A specifies an axis permutation in the "forward" direction, i.e.
472
473 B[i] → Z[A[i]]
474
475 For our next_ravel_XXX() mechanism we need the "backward" direction:
476
477 Z[i] ← B[A⁻¹[i]]
478
479 We therefore use the inverse permutation for A, which exchanges
480 source and destination
481
482 A is already normalized to ⎕IO←0 so its axes are 0, 1, ... in some order
483 */
484
485const Shape shape_inv_A = inverse_permutation(A);
486const Shape & shape_B = B->get_shape();
487const Shape shape_Z = permute(shape_B, shape_inv_A);
488
489Value_P Z(shape_Z, LOC);
490
491 if (shape_Z.is_empty())
492 {
493 Z->set_default(*B, LOC);
494 return Z;
495 }
496
497 for (ArrayIterator b(shape_Z, A); b.more(); ++b)
498 {
499 Z->next_ravel_Cell(B->get_cravel(b.get_ravel_offset()));
500 }
501
502 Z->check_value(LOC);
503 return Z;
504}
505//----------------------------------------------------------------------------
506Value_P
507Bif_F12_TRANSPOSE::transpose_diag(const Shape & A, const Value * B)
508{
509 // check that: ∧/(⍳⌈/0,A)ϵA
510 //
511 // I.e. 0, 1, ... max_A are in A
512 // we search sequentially, since A is small.
513 // we construct shape_Z as we go.
514 //
515Shape shape_Z;
516 {
517 // check that 0 ≤ A[a] ≤ ⍴⍴B, and compute max_A = ⌈/A
518 // i.e. ∧/(ι/0,L)∈L lrm p. 253
519 ShapeItem max_A = 0; // the largest index in A (- qio).
520 loop(ra, A.get_rank())
521 {
522 if (A.get_shape_item(ra) < 0) DOMAIN_ERROR;
523 if (A.get_shape_item(ra) >= B->get_rank()) DOMAIN_ERROR;
524 if (max_A < A.get_shape_item(ra)) max_A = A.get_shape_item(ra);
525 }
526
527 // check that every m < max_A is in A.
528 // the smallest axis in B found is added to shape_Z.
529 //
530 loop(m, max_A + 1)
531 {
532 ShapeItem min_Bm = -1;
533 loop(ra, A.get_rank())
534 {
535 if (m != A.get_shape_item(ra)) continue;
536 const ShapeItem B_ra = B->get_shape_item(ra);
537 if (min_Bm == -1) min_Bm = B_ra;
538 else if (min_Bm > B_ra) min_Bm = B_ra;
539 }
540
541 if (min_Bm == -1) DOMAIN_ERROR; // m not in A
542
543 shape_Z.add_shape_item(min_Bm);
544 }
545 }
546
547Value_P Z(shape_Z, LOC);
548 if (Z->is_empty())
549 {
550 Z->set_default(*B, LOC);
551 return Z;
552 }
553
554const Cell * cB = &B->get_cfirst();
555
556 for (ArrayIterator z(shape_Z); z.more(); ++z)
557 {
558 const Shape idx_B = permute(z.get_shape_offsets(), A);
559 const ShapeItem b = B->get_shape().ravel_pos(idx_B);
560 Z->next_ravel_Cell(cB[b]);
561 }
562
563 Z->check_value(LOC);
564 return Z;
565}
566//----------------------------------------------------------------------------
567Shape
568Bif_F12_TRANSPOSE::inverse_permutation(const Shape & perm)
569{
570 // perm is a permutation of ⎕IO + 0, 1, ...
571 // return the inverse permutaion of perm.
572
573ShapeItem rho[MAX_RANK];
574
575 // 1. set all items to -1.
576 //
577 loop(r, perm.get_rank()) rho[r] = -1;
578
579 // 2. set all items to the shape items of perm
580 //
581 loop(a, perm.get_rank())
582 {
583 const ShapeItem ax = perm.get_shape_item(a);
584 if (ax < 0)
585 {
586 UCS_string & more = MORE_ERROR();
587 more << "Axis " << ax << " is < ⎕IO (="
588 << Workspace::get_IO() << ") in permutation";
589 loop(a, perm.get_rank())
590 more << " " << (Workspace::get_IO() + perm.get_shape_item(a));
591 MORE_ERROR() = more;
592 AXIS_ERROR;
593 }
594
595 if (ax >= perm.get_rank())
596 {
597 UCS_string & more = MORE_ERROR();
598 more << "Axis " << ax << " exceeds rank "
599 << perm.get_rank() << " of shape in permutation";
600 loop(a, perm.get_rank())
601 more << " " << (Workspace::get_IO() + perm.get_shape_item(a));
602 MORE_ERROR() = more;
603 AXIS_ERROR;
604 }
605 if (rho[ax] != -1)
606 {
607 UCS_string & more = MORE_ERROR();
608 more << "Duplictate Axis " << ax << " in permutation";
609 loop(a, perm.get_rank())
610 more << " " << (Workspace::get_IO() + perm.get_shape_item(a));
611 MORE_ERROR() = more;
612 AXIS_ERROR;
613 }
614
615 // everything OK
616 //
617 rho[ax] = a;
618 }
619
620 return Shape(perm.get_rank(), rho);
621}
622//----------------------------------------------------------------------------
623Shape
624Bif_F12_TRANSPOSE::permute(const Shape & sh, const Shape & perm)
625{
626 /* permute sh according to perm.
627
628 perm is a shape specifying a permutation like this:
629
630 0 → perm[0]
631 1 → perm[1]
632 ...
633
634 the result ret is the permuted shape sh:
635
636 ret[0] = sh[perm[0]]
637 ret[1] = sh[perm[1]]
638 ...
639
640 */
641Shape ret;
642
643 loop(r, perm.get_rank())
644 {
645 ret.add_shape_item(sh.get_shape_item(perm.get_shape_item(r)));
646 }
647
648 return ret;
649}

651Token
652Bif_F12_DECODE::eval_AB(Value_P A, Value_P B) const
653{
654 // ρZ is: (¯1↓ρA),1↓ρB
655 // ρρZ is: (0⌈¯1+ρρA) + (0⌈¯1+ρρB)
656 //
657const Shape shape_A1 = A->get_shape().without_last_axis();
658const Shape shape_B1 = B->get_shape().without_first_axis();
659
660const ShapeItem l_len_A = A->get_rank() ? A->get_last_shape_item() : 1;
661const ShapeItem h_len_B = B->get_rank() ? B->get_shape_item(0) : 1;
662
663const ShapeItem h_len_A = shape_A1.get_volume();
664const ShapeItem l_len_B = shape_B1.get_volume();
665
666 if (l_len_A == 0 || h_len_B == 0) // empty result
667 {
668 const Shape shape_Z(shape_A1 + shape_B1);
669 Value_P Z(shape_Z, LOC);
670 Z->check_value(LOC);
671 return Token(TOK_APL_VALUE1, Z);
672 }
673
674 if ((l_len_A != 1) && // cannot scalar-extend A, and
675 (h_len_B != 1) && // cannot scalar-extend B, and
676 (l_len_A != h_len_B)) // the lengths of A and B differ
677 LENGTH_ERROR;
678
679const Shape shape_Z = shape_A1 + shape_B1;
680
681Value_P Z(shape_Z, LOC);
682
683const Cell * cA = &A->get_cfirst();
684
685 loop(h, h_len_A)
686 {
687 // cA ... cA + len_A are used. See if they are complex.
688 //
689 bool complex_A = false;
690 bool integer_A = true;
691 loop(aa, l_len_A)
692 {
693 if (!cA[aa].is_near_real())
694 {
695 complex_A = true;
696 integer_A = false;
697 break;
698 }
699
700 if (!cA[aa].is_near_int()) integer_A = false;
701 }
702
703 loop(l, l_len_B)
704 {
705 // cB, cB + l_len_B, ... are used. See if they are complex
706 //
707 bool complex_B = false;
708 bool integer_B = true;
709 loop(bb, h_len_B)
710 {
711 if (!B->get_cravel(l + bb*l_len_B).is_near_real())
712 {
713 complex_B = true;
714 integer_B = false;
715 break;
716 }
717
718 if (!B->get_cravel(l + bb*l_len_B).is_near_int())
719 integer_B = false;
720 }
721
722 const Cell * cB = &B->get_cravel(l);
723 if (integer_A && integer_B)
724 {
725 const bool overflow = decode_int(*Z, l_len_A, cA,
726 h_len_B, cB, l_len_B);
727 if (!overflow) continue;
728
729 // otherwise: compute as float
730 }
731
732 if (complex_A || complex_B)
733 decode_complex(*Z, l_len_A, cA,
734 h_len_B, cB, l_len_B);
735 else
736 decode_real(*Z, l_len_A, cA,
737 h_len_B, cB, l_len_B);
738 }
739 cA += l_len_A;
740 }
741
742 Z->set_default(*B.get(), LOC);
743 Z->check_value(LOC);
744 return Token(TOK_APL_VALUE1, Z);
745}
746//----------------------------------------------------------------------------
747bool
748Bif_F12_DECODE::decode_int(Value & Z, ShapeItem len_A, const Cell * cA,
749 ShapeItem len_B, const Cell * cB, ShapeItem dB)
750{
751 // decode_int() can easily produce an integer overflow. We keep track
752 // of that by also computing the final result as double and return
753 // true if an integer overflow occurs. The caller shall then call
754 // decode_real() instead.
755 //
756const ShapeItem dec_A = len_A == 1 ? 0 : 1;
757const ShapeItem dec_B = len_B == 1 ? 0 : dB;
758const ShapeItem len = dec_A ? len_A : len_B;
759
760 cA += dec_A*len_A; // let cA point past the lowest weight item in A
761 cB += dec_B*len_B; // let cB point past the lowest weight item in B
762
763APL_Integer value = 0;
764APL_Float value_f = 0.0;
765
766APL_Integer weight = 1;
767APL_Float weight_f = 1.0;
768
769 loop(l, len)
770 {
771 cA -= dec_A;
772 cB -= dec_B;
773
774 if (weight_f > LARGE_INT) return true;
775 if (weight_f < SMALL_INT) return true;
776
777 const APL_Integer vB = cB[0].get_near_int();
778 value = value + weight * vB;
779 value_f = value_f + weight_f * vB;
780 if (value_f > LARGE_INT) return true;
781 if (value_f < SMALL_INT) return true;
782
783 weight = weight * cA[0].get_near_int();
784 weight_f = weight_f * cA[0].get_near_int();
785 }
786
787 Z.next_ravel_Int(value);
788
789 return false; // no overflow
790}

794 ShapeItem len_B, const Cell * cB, ShapeItem dB)

796const ShapeItem dec_A = len_A == 1 ? 0 : 1;
797const ShapeItem dec_B = len_B == 1 ? 0 : dB;

799
800 cA += dec_A*len_A; // let cA point past the lowest weight item in A
801 cB += dec_B*len_B; // let cB point past the lowest weight item in B

814}

839//----------------------------------------------------------------------------
840Token
841Bif_F12_ENCODE::eval_AB(Value_P A, Value_P B) const
842{
843 // A⊤B: every number in B is represented in a number system with
844 // radix A. ⍴Z ←→ (⍴A, ⍴B) and Z[...;...b...] = A⊤B[...b...]
845
846 if (A->is_scalar()) return Bif_F12_STILE::fun->eval_AB(A, B);
847
848const ShapeItem ec_A = A->element_count();
849const ShapeItem ec_B = B->element_count();
850const Shape shape_Z = A->get_shape() + B->get_shape();
851Value_P Z(shape_Z, LOC);
852
853 if (ec_A == 0 || ec_B == 0) // empty A or B
854 {
855 Z->check_value(LOC);
856 return Token(TOK_APL_VALUE1, Z);
857 }
858
859const ShapeItem aL = A->get_shape_item(0); // first (LSB) dimension of A
860const ShapeItem aH = ec_A/aL; // remaining (MSB) dimensions
861
862const double qct = Workspace::get_CT();
863
864ConstRavel_P iA(A, true);
865 loop(a, aH)
866 {
867 const Cell * cB = &B->get_cfirst();
868
869 // find largest Celltype in A, starting with CT_INT and maybe
870 // "increasing" it to CT_FLOAT or CT_COMPLEX as needed.
871 //
872 CellType ct_a = CT_INT;
873 loop(h, aL)
874 {
875 const CellType ct = A->get_cravel(a + h*aH).get_cell_type();
876 if (ct == CT_INT) ;
877 else if (ct == CT_FLOAT) { if (ct_a == CT_INT) ct_a = ct; }
878 else if (ct == CT_COMPLEX) ct_a = CT_COMPLEX;
879 else DOMAIN_ERROR;
880 }
881
882 for (ConstRavel_P iB(B, true); +iB; ++iB)
883 {
884 CellType ct = ct_a;
885 const CellType ct_b = cB->get_cell_type();
886 if (ct_b == CT_INT) ;
887 else if (ct_b == CT_FLOAT) { if (ct == CT_INT) ct = ct_b; }
888 else if (ct_b == CT_COMPLEX) ct = CT_COMPLEX;
889 else DOMAIN_ERROR;
890
891 if (ct == CT_INT) // A and B are both integer
892 encode_Int(*Z, aL, aH, iA, iB);
893 else if (ct == CT_FLOAT) // A and B are not APL_Complex
894 encode_Flt(*Z, aL, aH, iA, iB, qct);
895 else // A or B contain APL_Complex
896 encode_Cpx(*Z, aL, aH, iA, iB, qct);
897 }
898 ++iA;
899 }
900
901 Z->set_default(*B.get(), LOC);
902
903 Z->check_value(LOC);
904 return Token(TOK_APL_VALUE1, Z);
905}
906//----------------------------------------------------------------------------
907void
908Bif_F12_ENCODE::encode_Int(Value & Z, ShapeItem aL, ShapeItem aH,
909 const ConstRavel_P & iA, const ConstRavel_P & iB)
910{
911const ShapeItem dZ = aH*iB.get_length();
912Cell * cZ = &Z.get_wravel(iB()
913 + iA()*iB.get_length()) // = the current Z (near bottom)
914 + aL*dZ; // + the size of Z (above top)
915
916const Cell * cA = &*iA + iA.get_length(); // the end of A (+aH)
917
918 /* unfortunately the less significant weights of the number system base A
919 are located at higher indices of A. We therefore work downwards from
920 the higher indices of A resp. Z...
921
922 ← ← ... ← .. ← cA // pre-decremented by aH
923 ↓
924 ┌────┬────┬─────┬────┐
925 │ a0 │ a1 │ ... │ aN │ // number base (MSB...LSB)
926 └────┴────┴─────┴────┘
927 ↓ ↓ ↓ ↓
928 bi ← bi ← ... ← bi // divided by ai
929 ↓ ↓ ↓
930 ┌────┬────┬─────┬────┐
931 │ z0 │ z1 │ ... │ zN │ // remainders
932 └────┴────┴─────┴────┘
933 ↑ ↑ ↑ ↑
934 ← ← ... ← .. ← cZ // pre-decremented by dZ
935 */
936
937APL_Integer bi = iB->get_int_value(); // the value being decoded
938 loop(a, aL)
939 {
940 cA -= aH;
941 cZ -= dZ;
942
943 const IntCell cC(bi);
944 cC.bif_residue(cZ, cA);
945
946 if (cA->get_int_value() == 0)
947 {
948 bi = 0;
949 }
950 else
951 {
952 bi -= cZ->get_int_value();
953 bi /= cA->get_int_value();
954 }
955 }
956}
957//----------------------------------------------------------------------------
958void
959Bif_F12_ENCODE::encode_Flt(Value & Z, ShapeItem aL, ShapeItem aH,
960 const ConstRavel_P & iA,
961 const ConstRavel_P & iB, double qct)
962{
963const ShapeItem dZ = aH*iB.get_length();
964Cell * cZ = &Z.get_wravel(iB()
965 + iA()*iB.get_length()) // = the current Z
966 + aL*dZ; // + the size of Z
967
968const Cell * cA = &*iA + iA.get_length(); // the end of A (+1)
969
970 // work downwards from the higher indices (see encode_Int())...
971 //
972APL_Float bf = iB->get_real_value(); // the value being decoded
973 loop(a, aL)
974 {
975 cA -= aH;
976 cZ -= dZ;
977
978 const FloatCell cC(bf);
979 cC.bif_residue(cZ, cA);
980
981 if (cA->is_near_zero())
982 {
983 bf = 0.0;
984 }
985 else
986 {
987 bf -= cZ->get_real_value();
988 bf /= cA->get_real_value();
989 }
990 }

1320//----------------------------------------------------------------------------
1321Token
1322Bif_F1_EXECUTE::execute_statement(UCS_string & statement)
1323{

1325
1326 // check for commands
1327 //
1328 if (statement.size() &&
1329 (statement[0] == UNI_R_PARENT ||
1330 statement[0] == UNI_R_BRACK)) return execute_command(statement);
1331
1332ExecuteList * fun = ExecuteList::fix(statement.no_pad(), LOC);
1333 if (fun == 0) SYNTAX_ERROR;
1334
1335 Log(LOG_UserFunction__execute) fun->print(CERR);
1336
1337 // important special case: ⍎ of an APL literal value
1338 //
1339 if (fun->get_body().size() == 2 &&
1340 fun->get_body()[0].get_Class() == TC_VALUE)
1341 {

1423ShapeItem
1424Bif_F12_UNION::append_zone(const Cell ** cells_Z, const Cell ** cells_B,
1425 Zone_list & B_from_to, double qct)
1426{
1427const Cell ** Z0 = cells_Z;
1428
1429 while (B_from_to.size())
1430 {
1431 const Zone zone = B_from_to.back();
1432 const ShapeItem zone_count = zone.count();
1433 Assert(zone_count > 0);
1434 B_from_to.pop_back();
1435
1436 const ShapeItem B_from = zone.from;
1437 const ShapeItem B_to = zone.to;
1438
1439 // Find the smallest pointer (i.e. smallest Cell address, not the
1440 // smallest Cell value) in the zone. This pointer will go into Z and
1441 // it will kill all its neighbours that are equal within qct.
1442 //
1443
1444#if 0
1445 // this fails when used with char cells but is quite handy for
1446 // testing the algorithm!
1447 Q1(LOC)
1448 fprintf(stderr, "%ld-element zone:\n", B_to - B_from);
1449 for (ShapeItem j = B_from; j < B_to; ++j)
1450 fprintf(stderr, "[%ld] value: %.12f\n", j,
1451 cells_B[j]->get_real_value());
1452#endif
1453
1454
1455 // the by far most likely cases are zones with 1 or,
1456 // already less likely, 2 cells. These cases can be handled
1457 // without searching the smallest element and are, for
1458 // performance reasons, handled beforehand.
1459 //
1460 if (zone_count == 1)
1461 {
1462 *cells_Z++ = cells_B[B_from];
1463 continue; // zone done
1464 }
1465
1466 if (zone_count == 2)
1467 {
1468 if (cells_B[B_from] < cells_B[B_from + 1])
1469 *cells_Z++ = cells_B[B_from];
1470 else
1471 *cells_Z++ = cells_B[B_from + 1];
1472 continue; // zone done
1473 }
1474
1475 ShapeItem smallest = B_from;
1476 for (ShapeItem bb = B_from + 1; bb < B_to; ++bb)
1477 {
1478 if (cells_B[smallest] > cells_B[bb]) smallest = bb;
1479 }
1480
1481 // See if the zone is transitive or not.
1482 //
1483 if (cells_B[smallest]->equal(*cells_B[B_to - 1], qct))
1484 {
1485 // the zone is transitive (i.e. all cells are equal within ⎕CT).
1486 // The unique of the zone is then the smallest pointer (i.e.first,
1487 // not the smallest Cell) in the zone.
1488 //
1489 *cells_Z++ = cells_B[smallest];
1490 continue; // zone done
1491 }
1492
1493 // the zone is not transitive, i.e. cells_B[B_from] is NOT equal to
1494 // cells_B[B_to-1]. Divide the zone into 3 (possibly empty) zones:
1495 //
1496 // a. cells < smallest within ⎕CT,
1497 // b. cells = smallest within ⎕CT, and
1498 // c. cells > smallest within ⎕CT
1499
1500 // Start with the middle zone b. This zone initially contains only
1501 // smallest and its then blown up with elements that are equal (within)
1502 // ⎕CT) to smallest.
1503 //
1504 const Cell * first_B = cells_B[smallest];
1505 *cells_Z++ = first_B;
1506 ShapeItem from1 = smallest; // initial start of zone b.
1507 ShapeItem to1 = smallest; // initial end of zone b.
1508
1509 // decrement from1 as long as its cell is within ⎕CT of smallest.
1510 // That makes all
1511 while (from1 > B_from && first_B->equal(*cells_B[from1], qct)) --from1;
1512 if (from1 > B_from) // non-empty zone before smallest
1513 {
1514 const Zone smaller(B_from, from1);
1515 B_from_to.push_back(smaller);
1516 continue; // zone done
1517 }
1518
1519 // increment to1 as long as its cell is within ⎕CT of smallest.
1520 //
1521 while (to1 < B_to && first_B->equal(*cells_B[to1], qct)) ++to1;
1522 if (to1 < B_to) // non-empty zone after smallest
1523 {
1524 Zone larger(to1, B_to);
1525 B_from_to.push_back(larger);
1526 continue; // zone done
1527 }
1528 }
1529
1530 return cells_Z - Z0;
1531}

1533Token
1534Bif_F12_UNION::eval_B(Value_P B) const
1535{
1536 if (B->get_rank() > 1) RANK_ERROR;
1537
1538const ShapeItem len_B = B->element_count();
1539 if (len_B <= 1) return Token(TOK_APL_VALUE1, CLONE_P(B, LOC));
1540
1541const double qct = Workspace::get_CT();
1542
1543 // 1. construct a vector of Cell pointers and sort it so that the
1544 // cells being pointed to are sorted ascendingly.
1545 //
1546 // For efficiency we allocate both the Cell pointers for argument B and
1547 // for the result Z. Sorting is first done with ⎕CT←0; ⎕CT will be
1548 // considered later on.
1549 //
1550const Cell ** cells_B = new const Cell *[2*len_B];
1551 if (cells_B == 0) WS_FULL;
1552
1553const Cell ** cells_Z = cells_B + len_B;
1554ShapeItem len_Z = 0;
1555
1556 for (ConstRavel_P b(B, true); +b; ++b) cells_B[b()] = &*b;
1557 Heapsort<const Cell*>::sort(cells_B, len_B, 0, Cell::A_greater_B);
1558
1559 // 2. divide the cells into zones. A zone is a sequence of cells
1560 // B[from] (including) ... B[to] (excluding) so that the difference
1561 // between B[i] and B[i+1] is < ⎕CT.
1562 //
1563 // Then append the unique element(s) of each zone to cells_Z.
1564 //
1565ShapeItem from = 0;
1566Zone_list from_to;
1567 for (ShapeItem b = 1; b < len_B; ++b)
1568 {
1569 if (cells_B[b]->equal(*cells_B[b-1], qct))
1570 {
1571 // cells_B[b] belongs to the zone
1572 //
1573 continue;
1574 }
1575
1576 const Zone ft0(from, b);
1577 from_to.push_back(ft0);
1578 len_Z += append_zone(cells_Z + len_Z, cells_B, from_to, qct);
1579 Assert(from_to.size() == 0);
1580 from = b;
1581 continue; // next b (in the next zone)
1582 }
1583
1584 if (from < len_B) // the rest
1585 {
1586 const Zone ft0(from, len_B);
1587 from_to.push_back(ft0);
1588 len_Z += append_zone(cells_Z + len_Z, cells_B, from_to, qct);
1589 Assert(from_to.size() == 0);
1590 }
1591
1592 // 3. cells_Z now contains only the unique cells in cells_B, but sorted by
1593 // cell contnt. Sort cells_Z by address (= position in B) so that the
1594 // original order in B is reconstructed
1595 //
1596 Heapsort<const Cell *>::sort(cells_Z, len_Z, 0, Cell::compare_ptr);
1597
1598 // 4. construct the result.
1599 //
1600Value_P Z(len_Z, LOC);
1601 loop(z, len_Z) Z->next_ravel_Cell(*cells_Z[z]);
1602 delete[] cells_B; // also deletes cells_Z
1603 Z->check_value(LOC);
1604 return Token(TOK_APL_VALUE1, Z);
1605}

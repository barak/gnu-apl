39//----------------------------------------------------------------------------
40Token
41Quad_TF::eval_AB(Value_P A, Value_P B) const
42{
43 // A should be an integer scalar or 1-element_vector with value 1 or 2
44 //
45 if (A->get_rank() > 0) RANK_ERROR;
46 if (A->element_count() != 1) LENGTH_ERROR;
47
48const APL_Integer mode = A->get_cfirst().get_int_value();
49const UCS_string symbol_name(*B.get());
50
51Value_P Z;
52
53 if (mode == 1)
54 {
55 const bool inverse = is_inverse(symbol_name);
56 if (inverse) Z = tf1_inv(symbol_name);
57 else Z = tf1(symbol_name);
58 }
59 else if (mode == 2)
60 {
61 const bool inverse = is_inverse(symbol_name);
62 if (!inverse) return tf2(symbol_name);
63 Z = Value_P(tf2_inverse(symbol_name), LOC);
64 }
65 else if (mode == 3)
66 {
67 Z = tf3(symbol_name);
68 }
69 else
70 {
71 DOMAIN_ERROR;
72 }
73
74 if (!Z) DOMAIN_ERROR;
75
76 Z->check_value(LOC);
77 return Token(TOK_APL_VALUE1, Z);
78}
79//----------------------------------------------------------------------------
80bool
81Quad_TF::is_inverse(const UCS_string & maybe_name)
82{
83 // a forward ⎕TF contains only a function or variable name.
84 // an inverse ⎕TF contains at least one non-symbol character
85 //
86 loop(n, maybe_name.size())
87 {
88 if (n == 0 && Avec::is_quad(maybe_name[n])) continue;
89 if (Avec::is_symbol_char(maybe_name[n])) continue;
90
91 return true; // inverse ⎕TF
92 }
93
94 // all chars in val were symbol characters. Therefore it is a forward ⎕TF
95 return false;
96}
97//----------------------------------------------------------------------------
98Value_P
99Quad_TF::tf1(const UCS_string & name)
100{
101const NamedObject * obj = Workspace::lookup_existing_name(name);
102 if (obj == 0) return Str0(LOC);
103
104const Function * function = obj->get_function();
105
106 if (function)
107 {
108 if (obj->is_user_defined()) return tf1(name, *function);
109
110 // quad function or primitive: return ''
111 return Str0(LOC);
112 }
113
114
115 if (const Symbol * symbol = obj->get_symbol())
116 {
117 Value_P value = symbol->get_apl_value();
118 if (+value) return tf1(name, value);
119
120 /* not a variable: fall through */
121 }
122
123 return Str0(LOC);
124}
125//----------------------------------------------------------------------------
126Value_P
127Quad_TF::tf1(const UCS_string & var_name, Value_P val)
128{
129const bool is_char_array = val->get_cfirst().is_character_cell();
130UCS_string ucs(is_char_array ? UNI_C : UNI_N);
131
132 ucs.append(var_name);
133 ucs.append(UNI_SPACE);
134 ucs.append_number(val->get_rank()); // rank
135
136 loop(r, val->get_rank())
137 {
138 ucs.append(UNI_SPACE);
139 ucs.append_number(val->get_shape_item(r)); // shape
140 }
141
142const ShapeItem ec = val->element_count();
143
144 if (is_char_array)
145 {
146 ucs.append(UNI_SPACE);
147
148 loop(e, ec)
149 {
150 const Cell & cell = val->get_cravel(e);
151 if (!cell.is_character_cell())
152 {
153 return Str0(LOC);
154 }
155
156 ucs.append(cell.get_char_value());
157 }
158 }
159 else // number
160 {
161 loop(e, ec)
162 {
163 ucs.append(UNI_SPACE);
164
165 const Cell & cell = val->get_cravel(e);
166 if (cell.is_integer_cell())
167 {
168 const int sign = ucs.size();
169 ucs.append_number(cell.get_int_value());
170 if (ucs[sign] == '-') ucs[sign] = UNI_OVERBAR;
171 }
172 else if (cell.is_near_real())
173 {
174 PrintContext pctx(PR_APL_MIN, Quad_PP_TF, MAX_Quad_PW);
175 bool scaled = true;
176 UCS_string ucs1(cell.get_real_value(), scaled, pctx);
177 ucs.append(ucs1);
178 }
179 else if (cell.is_numeric())
180 {
181 PrintContext pctx(PR_APL_MIN, Quad_PP_TF, MAX_Quad_PW);
182 bool scaled = true;
183 UCS_string ucs1(cell.get_real_value(), scaled, pctx);
184 ucs.append(ucs1);
185
186 ucs.append(UNI_J);
187
188 ucs1 = UCS_string(cell.get_imag_value(), scaled, pctx);
189 ucs.append(ucs1);
190 }
191 else
192 {
193 MORE_ERROR() << "Non-number in 1 ⎕TF N record";
194 return Value_P();
195 }
196 }
197 }
198
199 return Value_P(ucs, LOC);
200}
201//----------------------------------------------------------------------------
202Value_P
203Quad_TF::tf1(const UCS_string & fun_name, const Function & fun)
204{
205const UCS_string text = fun.canonical(false);
206UCS_string_vector lines;
207const size_t max_len = text.to_vector(lines);
208
209UCS_string ucs;
210 ucs.append(UNI_F);
211 ucs.append(fun_name);
212 ucs.append(UNI_SPACE);
213 ucs.append_number(2); // rank
214 ucs.append(UNI_SPACE);
215 ucs.append_number(lines.size()); // rows
216 ucs.append(UNI_SPACE);
217 ucs.append_number(max_len); // cols
218 ucs.append(UNI_SPACE);
219
220 loop(l, lines.size())
221 {
222 ucs.append(lines[l]);
223 loop(c, max_len - lines[l].size()) ucs.append(UNI_SPACE);
224 }
225
226 return Value_P(ucs, LOC);
227}
228//----------------------------------------------------------------------------
229Value_P
230Quad_TF::tf1_inv(const UCS_string & ravel)
231{
232const int len = ravel.size();
233
234 if (len < 2)
235 {
236 MORE_ERROR() << "1 ⎕TF record too short";
237 return Value_P();
238 }
239
240 // mode should be 'F', 'N', or 'C'.
241const Unicode mode = ravel[0];
242 if (mode != 'C' && mode != 'F' && mode != 'N')
243 {
244 MORE_ERROR() << "1 ⎕TF record type not F, N, or C";
245 return Value_P();
246 }
247
248UCS_string name(ravel[1]);
249 if (!Avec::is_quad(name[0]) && !Avec::is_first_symbol_char(name[0]))
250 {
251 MORE_ERROR() << "Bad variable name in 1 ⎕TF record";
252 return Value_P();
253 }
254
255ShapeItem idx = 2;
256 while (idx < len && Avec::is_symbol_char(ravel[idx]))
257 name.append(ravel[idx++]);
258
259 if (Avec::is_quad(name[0]) && name.size() == 1)
260 {
261 MORE_ERROR() << "Bad variable name in 1 ⎕TF record";
262 return Value_P();
263 }
264
265 if (ravel[idx++] != UNI_SPACE)
266 {
267 MORE_ERROR() << "missing space (before rank) in 1 ⎕TF record";
268 return Value_P();
269 }
270
271ShapeItem idx0 = idx;
272sRank rank = 0;
273 while (idx < len && Avec::is_digit(ravel[idx]))
274 rank = 10 * rank + ravel[idx++] - UNI_0;
275
276 if (rank == 0 && idx0 == idx)
277 {
278 MORE_ERROR() << "missing rank in 1 ⎕TF record";
279 return Value_P();
280 }
281
282 if (ravel[idx++] != UNI_SPACE)
283 {
284 MORE_ERROR() << "missing space (after rank) in 1 ⎕TF record";
285 return Value_P();
286 }
287
288 if (rank > MAX_RANK)
289 {
290 MORE_ERROR() << "max. rank exceeded in 1 ⎕TF record";
291 return Value_P();
292 }
293
294Shape shape;
295 loop(r, rank)
296 {
297 idx0 = idx;
298 ShapeItem sh = 0;
299 while (idx < len && Avec::is_digit(ravel[idx]))
300 sh = 10 * sh + ravel[idx++] - UNI_0;
301 if (sh == 0 && idx0 == idx) // no shape
302 {
303 MORE_ERROR() << "too few shape items in 1 ⎕TF record";
304 return Value_P();
305 }
306
307 if (ravel[idx++] != UNI_SPACE)
308 {
309 MORE_ERROR() << "missing space (in shape) in 1 ⎕TF record";
310 return Value_P();
311 }
312 shape.add_shape_item(sh);
313 }
314
315const Symbol * symbol = 0;
316NameClass nc = NC_UNUSED_USER_NAME;
317const NamedObject * sym_or_fun = Workspace::lookup_existing_name(name);
318 if (sym_or_fun) // existing name
319 {
320 symbol = sym_or_fun->get_symbol();
321 nc = sym_or_fun->get_NC();
322 if (symbol && symbol->is_readonly())
323 {
324 MORE_ERROR() << "symbol cannot be modified in 1 ⎕TF record";
325 return Value_P();
326 }
327 }
328
329const int data_chars = len - idx;
330
331 if (mode == UNI_F) // function record
332 {
333 if (rank != 2)
334 {
335 MORE_ERROR() << "function text is not a matrix in 1 ⎕TF record";
336 return Value_P();
337 }
338
339 if (nc != NC_UNUSED_USER_NAME &&
340 nc != NC_FUNCTION &&
341 nc != NC_OPERATOR)
342 {
343 MORE_ERROR() << "symbol is an existing variable in 1 ⎕TF F record";
344 return Value_P();
345 }
346
347 if (data_chars != shape.get_volume())
348 {
349 MORE_ERROR() << "item count mismatch in 1 ⎕TF N record";
350 return Value_P();
351 }
352
353 Value_P new_val(shape, LOC);
354 loop(d, data_chars) new_val->next_ravel_Char(ravel[idx + d]);
355 new_val->check_value(LOC);
356
357 Token t = Quad_FX::do_eval_B(new_val.get());
358 }
359 else if (mode == UNI_C) // char array
360 {
361 if (data_chars != shape.get_volume())
362 {
363 MORE_ERROR() << "item count mismatch in 1 ⎕TF C record";
364 return Value_P();
365 }
366 if (nc != NC_UNUSED_USER_NAME && nc != NC_VARIABLE)
367 {
368 MORE_ERROR() <<
369 "symbol is existing and not a variable in 1 ⎕TF C record";
370 return Value_P();
371 }
372
373 Value_P new_val(shape, LOC);
374 loop(d, data_chars) new_val->next_ravel_Char(ravel[idx + d]);
375
376 new_val->check_value(LOC);
377
378 if (symbol == 0) symbol = Workspace::lookup_symbol(name);
379 const_cast<Symbol *>(symbol)->assign(new_val, false, LOC);
380 }
381 else if (mode == UNI_N) // numeric array
382 {
383 if (nc != NC_UNUSED_USER_NAME && nc != NC_VARIABLE)
384 {
385 MORE_ERROR() << "symbol cannot be assigned in 1 ⎕TF N record";
386 return Value_P();
387 }
388
389 UCS_string data(ravel, idx, len - idx);
390 Tokenizer tokenizer(PM_EXECUTE, LOC, false);
391 Token_string tos;
392 if (tokenizer.tokenize(data, tos) != E_NO_ERROR)
393 {
394 MORE_ERROR() << "tokenization failed in 1 ⎕TF N record";
395 return Value_P();
396 }
397 if (size_t(tos.size()) != size_t(shape.get_volume()))
398 {
399 MORE_ERROR() << "item count mismatch in 1 ⎕TF N record";
400 return Value_P();
401 }
402
403 // check that all token are numeric...
404 //
405 loop(t, tos.size())
406 {
407 const TokenTag tag = tos[t].get_tag();
408 if (tag == TOK_INTEGER) continue;
409 if (tag == TOK_REAL) continue;
410 if (tag == TOK_COMPLEX) continue;
411 MORE_ERROR() << "Non-number in 1 ⎕TF N record";
412 return Value_P();
413 }
414
415 // at this point, we have a valid inverse 1 ⎕TF.
416 //
417 Value_P ZZ(shape, LOC);
418
419 loop(t, tos.size())
420 {
421 const Token & tok = tos[t];
422 const TokenTag tag = tok.get_tag();
423 if (tag == TOK_INTEGER)
424 ZZ->next_ravel_Int(tok.get_int_val());
425 else if (tag == TOK_REAL)
426 ZZ->next_ravel_Float(tok.get_flt_val());
427 else if (tag == TOK_COMPLEX)
428 ZZ->next_ravel_Complex(tok.get_cpx_real(),
429 tok.get_cpx_imag());
430 else Assert(0); // since checked above
431 }
432
433 ZZ->check_value(LOC);
434
435 if (!symbol) symbol = Workspace::lookup_symbol(name);
436
437 const_cast<Symbol *>(symbol)->assign(ZZ, false, LOC);
438 }
439 else Assert(0); // since checked above
440
441 return Value_P(name, LOC);
442}
443//----------------------------------------------------------------------------
444Token
445Quad_TF::tf2(const UCS_string & name)
446{
447const NamedObject * obj = Workspace::lookup_existing_name(name);
448 if (obj == 0)
449 {
450 Log(LOG_Quad_TF) CERR << "bad name in tf2(" << name << ")" << endl;
451 return Token(TOK_APL_VALUE1, Str0(LOC));
452 }
453
454const Function * function = obj->get_function();
455
456 if (function)
457 {
458 if (obj->is_user_defined())
459 {
460 UCS_string ucs = tf2_fun(name, *function);
461 Value_P Z(ucs, LOC);
462 Z->check_value(LOC);
463 return Token(TOK_APL_VALUE1, Z);
464 }
465
466 // quad function or primitive: return ''
467 return Token(TOK_APL_VALUE1, Str0(LOC));
468 }
469
470const Symbol * symbol = obj->get_symbol();
471 if (symbol)
472 {
473 Value_P value = symbol->get_apl_value();
474 if (+value) return tf2_var(name, *value);
475
476 /* not a variable: fall through */
477 }
478
479 Log(LOG_Quad_TF) CERR << "error in tf2(" << name << ")" << endl;
480 return Token(TOK_APL_VALUE1, Str0(LOC));
481}
482//----------------------------------------------------------------------------
483Token
484Quad_TF::tf2_var(const UCS_string & var_name, const Value & B)
485{
486 Log(LOG_Quad_TF) CERR << "tf2_var(" << var_name << ")" << endl;
487
488UCS_string ucs_value;
489 if (B.is_scalar() && !B.is_simple_scalar())
490 {
491 const Cell & cell = B.get_cfirst();
492 Assert(cell.is_pointer_cell());
493 tf2_value(0, ucs_value, *cell.get_pointer_value(), 1);
494 }
495 else
496 {
497 tf2_value(0, ucs_value, B, 0);
498 }
499
500 Assert(ucs_value[0] == UNI_L_PARENT);
501 Assert(ucs_value[ucs_value.size() - 1] == UNI_R_PARENT);
502
503UCS_string ucs(var_name);
504 ucs.append(UNI_LEFT_ARROW);
505
506 // copy ucs_value except the outer parrentheses
507 for (ShapeItem v = 1; v < (ucs_value.size() - 1); ++v)
508 ucs.append(ucs_value[v]);
509
510 Log(LOG_Quad_TF) CERR << "success in tf2_var(): " << ucs << endl;
511Value_P Z(ucs, LOC);
512 Z->check_value(LOC);
513 return Token(TOK_APL_VALUE1, Z);
514}
515//----------------------------------------------------------------------------
516UCS_string
517Quad_TF::tf2_inverse(const UCS_string & ravel)
518{
519Token_string tos;
520 tos.push_back(Token(TOK_L_PARENT, int64_t(0)));
521
522 Log(LOG_Quad_TF) CERR << "inverse ⎕TF2: " << ravel << endl;
523
524 try
525 {
526 UCS_string ucs1 = no_UCS(ravel);
527 const Parser parser(PM_EXECUTE, LOC, false);
528 parser.parse(ucs1, tos);
529 }
530 catch(...)
531 {
532 Log(LOG_Quad_TF) CERR << "parse error in tf2_inverse()" << endl;
533 return UCS_string();
534 }
535
536 // tos is now ( VAR ← ... ) or: ( ⎕FX val ...
537 // make it VAR ← ( ... ) or: leave as is
538 //
539 {
540 if (tos.size() >= 3 && tos[1].get_tag() != TOK_Quad_FX)
541 {
542 tos[0] = tos[1]; // move VAR
543 tos[1] = tos[2]; // move ←
544 tos[2] = Token(TOK_L_PARENT, int64_t(0));
545 }
546 }
547 tos.push_back(Token(TOK_R_PARENT, int64_t(0)));
548
549 // simplify tos as much as possible. We replace A⍴B by reshaped B
550 // and glue values together.
551 //
552 tf2_reduce(tos);
553
554 Log(LOG_Quad_TF)
555 {
556 CERR << "inverse ⎕TF2: tos[" << tos.size() << "] after reduce() is: ";
557 tos.print(CERR, false);
558 CERR << endl;
559 }
560
561 if (tos.size() < 2)
562 {
563 Log(LOG_Quad_TF)
564 CERR << "short tos in tf2_inverse()" << tos.size() << endl;
565 return UCS_string(); // too short for an inverse 2⎕TF
566 }
567
568 // it could happen that some system variable ⎕XX which is not known by
569 // GNU APL is read. This case is parsed as ⎕ XX ← ...
570 // Issue a warning in that case
571 //
572 if (tos[0].get_tag() == TOK_Quad_Quad &&
573 tos[1].get_Class() == TC_SYMBOL)
574 {
575 UCS_string new_var_or_fun = tos[1].get_sym_ptr()->get_name();
576 CERR << "*** Unknown system variable ⎕" << new_var_or_fun
577 << " in 2⎕TF / )IN (assignment ignored)" << endl;
578 return new_var_or_fun;
579 }
580
581 /* we expect either VAR ← VALUE
582 or: ( ⎕FX fun-text )
583
584 Try VAR ← VALUE first.
585 */
586 if (tos[0].get_Class() == TC_SYMBOL && tos[1].get_Class() == TC_ASSIGN)
587 {
588 // at this point, we expect SYM ← VALUE.
589 //
590 if (tos.size() != 3) return UCS_string();
591 if (tos[2].get_Class() != TC_VALUE) return UCS_string();
592
593 tos[0].get_sym_ptr()->assign(tos[2].get_apl_val(), true, LOC);
594 Log(LOG_Quad_TF) CERR << "valid inverse 2 ⎕TF" << endl;
595 return tos[0].get_sym_ptr()->get_name(); // valid 2⎕TF
596 }
597
598 // try dyadic ⎕FX (native function)
599 //
600 // Then tos[5] is: ( VALUE ⎕FX VALUE )
601 //
602 if (tos.size() == 5 &&
603 tos[1].get_Class() == TC_VALUE &&
604 tos[2].get_tag() == TOK_Quad_FX &&
605 tos[3].get_Class() == TC_VALUE)
606 {
607 const Value * fname = tos[1].get_apl_val().get();
608 const Value * so_path = tos[3].get_apl_val().get();
609
610 const Token tok = Quad_FX::do_eval_AB(so_path, fname);
611 if (tok.get_Class() == TC_VALUE) // ⎕FX successful
612 {
613 Value_P val = tok.get_apl_val();
614 Log(LOG_Quad_TF)
615 CERR << "valid inverse 2 ⎕TF (native function):" << endl;
616 return UCS_string(*val.get());
617 }
618
619 Log(LOG_Quad_TF)
620 CERR << "invalid inverse 2 ⎕TF (native function):" << endl;
621 return UCS_string();
622 }
623
624 /* monadic ⎕FX. at this point we should have:
625
626 tos[4] = ( ⎕FX Value )
627 */
628
629 Log(LOG_Quad_TF)
630 CERR << "inverse 2 ⎕TF (monadic ⎕FX):" << endl;
631 if (tos.size() != 4) return UCS_string();
632 if (tos[1].get_tag() != TOK_Quad_FX) return UCS_string();
633 if (tos[2].get_Class() != TC_VALUE) return UCS_string();
634
635static const int eprops[] = { 0, 0, 0, 0 };
636const Token tok = Quad_FX::do_quad_FX(eprops, tos[2].get_apl_val().get(),
637 UTF8_string("2 ⎕TF"), true);
638
639 if (tok.get_Class() != TC_VALUE)
640 {
641 Log(LOG_Quad_TF)
642 CERR << "Quad_FX::do_quad_FX() failed in tf2_inverse()" << endl;
643 return UCS_string(); // error in ⎕FX
644 }
645
646 // ⎕FX succeeded (and the token returned is the name of the function)
647 //
648 return UCS_string(*tok.get_apl_val().get());
649}
650//----------------------------------------------------------------------------
651void
652Quad_TF::tf2_shape(UCS_string & ucs, const Shape & shape, ShapeItem nesting)
653{
654 ucs.append(UNI_L_PARENT);
655 loop(n, nesting) ucs.append(UNI_SUBSET); // ⊂...
656
657 // scalars are ''⍴SCALAR but ''⍴ has no effect and can be omitted
658
659 if (shape.get_rank()) // non-scalar
660 {
661 loop(r, shape.get_rank())
662 {
663 if (r) ucs.append(UNI_SPACE);
664 ucs.append_number(shape.get_shape_item(r));
665 }
666
667 ucs.append(UNI_RHO);
668 }
669}
670//----------------------------------------------------------------------------
671void
672Quad_TF::tf2_value(int level, UCS_string & ucs, const Value & value,
673 ShapeItem nesting)
674{
675 Log(LOG_Quad_TF)
676 {
677 char cc[100];
678 snprintf(cc, sizeof(cc), "tf2_value() initial level %u\n", level);
679 CERR << "tf2_value(): ucs before at level " << level << ": "
680 << ucs << endl << cc;
681 value.print_boxed(CERR, 0);
682 }
683
684
685 // some (but not all) empty vectors
686 if (value.is_empty())
687 {
688 const Cell & cell = value.get_cfirst();
689 if (cell.is_character_cell())
690 {
691 ucs.append(UNI_L_PARENT);
692 ucs.append(UNI_SINGLE_QUOTE);
693 ucs.append(UNI_SINGLE_QUOTE);
694 ucs.append(UNI_R_PARENT);
695 return;
696 }
697 }
698
699const ShapeItem ec = value.nz_element_count();
700
701 // emit e.g. ( shape ⍴
702 //
703 tf2_shape(ucs, value.get_shape(), nesting);
704 if (value.NOTCHAR()) tf2_ravel(level, ucs, ec, &value.get_cfirst());
705 else tf2_all_char_ravel(level, ucs, value);
706 ucs.append(UNI_R_PARENT); // close corresponding '(' from tf2_shape()
707
708 Log(LOG_Quad_TF)
709 {
710 CERR << "tf2_value(): ucs after at level " << level
711 << ": " << ucs << endl;
712 }
713 return;
714}
715//----------------------------------------------------------------------------
716void
717Quad_TF::tf2_ravel(int level, UCS_string & ucs, const ShapeItem len,
718 const Cell * cells)
719{
720 Assert(len > 0);
721
722 loop(e, len)
723 {
724 if (e) ucs.append(UNI_SPACE);
725 const Cell & cell = *cells++;
726
727 if (cell.is_pointer_cell())
728 {
729 Value_P sub_val = cell.get_pointer_value();
730 ShapeItem nesting = 0;
731 while (sub_val->is_scalar())
732 {
733 Assert(sub_val->get_cfirst().is_pointer_cell());
734 ++nesting;
735 sub_val = sub_val->get_cfirst().get_pointer_value();
736 }
737
738 tf2_value(level + 1, ucs, *sub_val, nesting);
739 }

742 DOMAIN_ERROR;
743 }

835void

837{
838ShapeItem skipped = 0;
839
840 loop(s, tos.size())
841 {
842 if (s < ShapeItem(tos.size() - 1) && tos[s].get_tag() == TOK_Quad_UCS)
843 {
844 s += 1; skipped += 1; // skip ⎕UCS
845 for (; s < ShapeItem(tos.size()) &&
846 tos[s].get_Class() == TC_VALUE; ++s)
847 {
848 tf2_toggle_UCS(*tos[s].get_apl_val().get());
849 tos[s - skipped].move_1(tos[s], LOC);
850 }
851 }
852
853 if (skipped) tos[s - skipped].move_1(tos[s], LOC);
854
855 }
856
857 if (skipped)
858 {
859 Log(LOG_Quad_TF)
860 CERR << "tf2_reduce_UCS() has skipped "
861 << skipped << " token" << endl;
862

864 }

867//----------------------------------------------------------------------------
868ShapeItem
869Quad_TF::tf2_toggle_UCS(Value & val)
870{
871ShapeItem error_count = 0;
872
873 loop(e, val.nz_element_count())
874 {
875 const Cell & cell = val.get_cravel(e);
876 if (cell.is_character_cell()) // char → integer
877 {
878 val.set_ravel_Int(e, cell.get_char_value());
879 }
880 else if (cell.is_integer_cell()) // integer → char
881 {
882 val.set_ravel_Char(e, Unicode(cell.get_int_value()));
883 }
884 else if (cell.is_pointer_cell()) // nested
885 {
886 error_count += tf2_toggle_UCS(*cell.get_pointer_value());
887 }
888 else
889 {
890 ++error_count;
891 }
892 }
893
894 if (val.is_empty()) val.to_type(false); // unlikely
895
896 return error_count;

898//----------------------------------------------------------------------------
899bool
900Quad_TF::tf2_reduce_RHO(Token_string & tos)
901{
902ShapeItem skipped = 0;
903
904 loop(s, tos.size())
905 {
906 // we replace (A⍴B) by B reshaped to A.
907 //
908 if ((s + 3) >= ShapeItem(tos.size()) || // too short
909 tos[s ].get_Class() != TC_VALUE || // not A
910 tos[s + 1].get_tag() != TOK_F12_RHO || // not ⍴
911 tos[s + 2].get_Class() != TC_VALUE || // not B
912 tos[s + 3].get_tag() != TOK_R_PARENT) // not )
913 {
914 // no match: copy the token (but never to itself)
915 //
916 if (skipped) tos[s - skipped].move_1(tos[s], LOC);
917 continue;
918 }
919
920 Shape sh;
921 {
922 const Value & aval = *tos[s].get_apl_val();
923 sh = Shape(aval, /* ⎕IO */ 0);
924 tos[s].extract_apl_val(LOC);
925 }
926 s += 2; skipped += 2; // skip ( A
927
928 Value_P bval = tos[s].get_apl_val();
929 if (sh.get_volume() == bval->element_count()) // same volume
930 {
931 tos[s - skipped].move_1(tos[s], LOC);
932 tos[s - skipped].ChangeTag(TOK_APL_VALUE1);
933 bval->set_shape(sh);
934 }
935 else
936 {
937 Token t = Bif_F12_RHO::do_reshape(sh, *bval);
938 tos[s - skipped].move_1(t, LOC);
939 }
940 }
941
942 if (skipped)
943 {
944 Log(LOG_Quad_TF)
945 CERR << "tf2_reduce_RHO() has skipped "
946 << skipped << " token" << endl;
947
948 tos.resize(tos.size() - skipped);
949 }
950
951 return skipped > 0;
952}
953//----------------------------------------------------------------------------
954bool
955Quad_TF::tf2_reduce_sequence(Token_string & tos)
956{
957ShapeItem skipped = 0;
958
959// tos.print(CERR);
960
961 // replace N - ⎕IO - ⍳ K by N N+1 ... N+K-1
962 // 0 1 2 3 4 5
963 //
964 loop(s, tos.size())
965 {
966 if ((s + 5) >= ShapeItem(tos.size()) || // too short
967 tos[s ].get_Class() != TC_VALUE || // not N
968 tos[s + 1].get_tag() != TOK_F12_MINUS || // not -
969 tos[s + 2].get_tag() != TOK_Quad_IO || // not ⎕IO
970 tos[s + 3].get_tag() != TOK_F12_MINUS || // not -
971 tos[s + 4].get_tag() != TOK_F12_INDEX_OF || // not ⍳
972 tos[s + 5].get_Class() != TC_VALUE || // not K
973 !tos[s ].get_apl_val()->is_int_scalar() || // N not integer
974 !tos[s + 5].get_apl_val()->is_int_scalar()) // K not integer
975 {
976 if (skipped) // dont copy to itself
977 tos[s - skipped].move_1(tos[s], LOC);
978 continue;
979 }
980
981 const APL_Integer N = tos[s].get_apl_val()->get_cfirst()
982 .get_int_value();
983 const APL_Integer K = tos[s + 5].get_apl_val()->get_cfirst()
984 .get_int_value();
985
986 loop(j, 6) tos[s + j].clear(LOC);
987
988 Value_P sequence(K, LOC);
989 loop(k, K) sequence->next_ravel_Int(N + k);
990 sequence->check_value(LOC);
991 Token tok(TOK_APL_VALUE1, sequence);
992 tos[s - skipped].move_2(tok, LOC);
993 s += 5; skipped += 5;
994 }
995
996 if (skipped)
997 {
998 Log(LOG_Quad_TF)
999 CERR << "tf2_reduce_sequence() has skipped "
1000 << skipped << " token" << endl;
1001
1002 tos.resize(tos.size() - skipped);
1003 }
1004
1005 return skipped > 0;
1006}
1007//----------------------------------------------------------------------------
1008bool
1009Quad_TF::tf2_reduce_sequence1(Token_string & tos)
1010{
1011ShapeItem skipped = 0;
1012
1013// tos.print(CERR);
1014
1015 // replace N - M × ⎕IO - ⍳ K by M(N N+1 ... N+K-1)
1016 // 0 1 2 3 4 5 6 7
1017 //
1018 loop(s, tos.size())
1019 {
1020 if ((s + 7) >= ShapeItem(tos.size()) || // too short
1021 tos[s ].get_Class() != TC_VALUE || // not N
1022 tos[s + 1].get_tag() != TOK_F12_MINUS || // not -
1023 tos[s + 2].get_Class() != TC_VALUE || // not K
1024 tos[s + 3].get_tag() != TOK_F12_TIMES || // not ×
1025 tos[s + 4].get_tag() != TOK_Quad_IO || // not ⎕IO
1026 tos[s + 5].get_tag() != TOK_F12_MINUS || // not -
1027 tos[s + 6].get_tag() != TOK_F12_INDEX_OF || // not ⍳
1028 tos[s + 7].get_Class() != TC_VALUE || // not K
1029 !tos[s ].get_apl_val()->is_int_scalar() || // N not integer
1030 !tos[s + 2].get_apl_val()->is_int_scalar() || // M not integer
1031 !tos[s + 7].get_apl_val()->is_int_scalar()) // K not integer
1032 {
1033 if (skipped) // dont copy to itself
1034 tos[s - skipped].move_1(tos[s], LOC);
1035 continue;
1036 }
1037
1038 const APL_Integer N = tos[s].get_apl_val()->get_cfirst()
1039 .get_int_value();
1040 const APL_Integer M = tos[s + 2].get_apl_val()->get_cfirst()
1041 .get_int_value();
1042 const APL_Integer K = tos[s + 7].get_apl_val()->get_cfirst()
1043 .get_int_value();
1044
1045 loop(j, 6) tos[s + j].clear(LOC);
1046
1047 Value_P sequence(K, LOC);
1048 loop(k, K) sequence->next_ravel_Int(M * (N + k));
1049 sequence->check_value(LOC);
1050 Token tok(TOK_APL_VALUE1, sequence);
1051 tos[s - skipped].move_2(tok, LOC);
1052 s += 7; skipped += 7;
1053 }
1054
1055 if (skipped)
1056 {
1057 Log(LOG_Quad_TF)
1058 CERR << "tf2_reduce_sequence1() has skipped "
1059 << skipped << " token" << endl;
1060
1061 tos.resize(tos.size() - skipped);
1062 }
1063 return skipped > 0;
1064}

1070
1071 // replace ⊂ B ) by an enclosed B )
1072 //

1113//----------------------------------------------------------------------------
1114bool
1115Quad_TF::tf2_reduce_COMMA(Token_string & tos)
1116{
1117ShapeItem skipped = 0;
1118
1119 loop(s, tos.size())
1120 {
1121 // we replace , B by B reshaped or A , B by AB
1122 //
1123 if (s < ShapeItem(tos.size() - 1) &&
1124 tos[s].get_tag() == TOK_F12_COMMA && // ,
1125 tos[s + 1].get_Class() == TC_VALUE) // B
1126 {
1127 const ShapeItem d_1 = s - skipped - 1;
1128 if (s > 0 && tos[d_1].get_tag() == TOK_R_PARENT) // value to be
1129 {
1130 // TOK_R_PARENT will become a value: just copy it for now
1131 //
1132 if (skipped) tos[s - skipped].move_1(tos[s], LOC);
1133 continue;
1134 }
1135
1136 if (s > 0 && tos[d_1].get_Class() == TC_VALUE) // value
1137 {
1138 Value_P A = tos[d_1].get_apl_val();
1139 Value_P B = tos[s + 1].get_apl_val();
1140 Assert(A->get_rank() <= 1);
1141 Assert(B->get_rank() <= 1);
1142 Value_P Z(A->element_count() + B->element_count(), LOC);
1143 loop(a, A->element_count())
1144 Z->next_ravel_Cell(A->get_cravel(a));
1145 loop(b, B->element_count())
1146 Z->next_ravel_Cell(B->get_cravel(b));
1147
1148 tos[s + 1].clear(LOC);
1149 Token tok_AB(TOK_APL_VALUE1, Z);
1150 tos[d_1].move_2(tok_AB, LOC);
1151 s += 1; skipped += 2; // skip , (of A , B) but not B
1152 continue; // don't move_1() below
1153 }
1154
1155 // monadic , B
1156 s += 1; skipped += 1; // skip ,
1157
1158 Value_P bval = tos[s].get_apl_val();
1159 const Shape sh(bval->element_count());
1160 bval->set_shape(sh);
1161
1162 tos[s].ChangeTag(TOK_APL_VALUE1);
1163 tos[s - skipped].move_1(tos[s], LOC);
1164 continue;
1165 }
1166
1167 if (skipped) tos[s - skipped].move_1(tos[s], LOC);
1168 }
1169
1170 if (skipped)
1171 {
1172 Log(LOG_Quad_TF)
1173 CERR << "tf2_reduce_COMMA() has skipped "
1174 << skipped << " token" << endl;
1175
1176 tos.resize(tos.size() - skipped);
1177 }
1178
1179 return skipped > 0;
1180}

1264//----------------------------------------------------------------------------
1265bool
1266Quad_TF::tf2_reduce_parentheses(Token_string & tos)
1267{
1268ShapeItem skipped = 0;
1269
1270 loop(s, tos.size())
1271 {
1272 // we replace ( B ) by B
1273 //
1274 if ((s + 2) < ShapeItem(tos.size()) &&
1275 tos[s].get_tag() == TOK_L_PARENT &&

1278 {

1281 if (dest.get_tag() == TOK_APL_VALUE3)
1282 {
1283 // final result of strand expression: enclose it
1284 //
1285 Value_P B = dest.get_apl_val();
1286 if (!B->is_simple_scalar()) // unless simple scalar
1287 {
1288 dest.extract_apl_val(LOC); // we will override it
1289 Value_P Z(LOC);
1290 Z->next_ravel_Pointer(B.get());
1291 Z->check_value(LOC);
1292 new (&dest) Token(TOK_APL_VALUE3, Z);
1293 }
1294 }
1295 s += 2; skipped += 2;
1296 continue;
1297 }
1298
1299 if (skipped) // dont copy to itself
1300 tos[s - skipped].move_1(tos[s], LOC);
1301 }
1302
1303 if (skipped)
1304 {
1305 Log(LOG_Quad_TF)
1306 CERR << "tf2_reduce_parentheses() has skipped "

1309 }
1310

1313//----------------------------------------------------------------------------
1314bool
1315Quad_TF::tf2_glue(Token_string & tos)
1316{
1317ShapeItem skipped = 0;
1318
1319 loop(s, tos.size()) // ⍴ must not be first or last
1320 {
1321 // the first value (if any) survives
1322 //
1323 if (skipped) // dont copy to itself
1324 tos[s - skipped].move_1(tos[s], LOC);
1325
1326 if (tos[s - skipped].get_Class() != TC_VALUE) continue;
1327
1328 // subsequent values (if any) are glued to the first value which
1329 // is now tos[s - skipped]
1330 //
1331 while ((s + 1) < ShapeItem(tos.size()) &&
1332 tos[s + 1].get_Class() == TC_VALUE)
1333 {
1334 Token t;
1335 t.move_1(tos[s - skipped], LOC);
1336 Value::glue(tos[s - skipped], t, tos[s + 1], LOC);
1337 s += 1; skipped += 1;
1338 }
1339 }
1340
1341 if (skipped)
1342 {
1343 Log(LOG_Quad_TF)
1344 CERR << "tf2_glue() has skipped " << skipped << " token" << endl;
1345
1346 tos.resize(tos.size() - skipped);
1347 }
1348
1349 return skipped > 0;
1350}

1381

1384
1385 ucs.append_UTF8("⎕FX");

1393//----------------------------------------------------------------------------
1394void
1395Quad_TF::tf2_char_vec(UCS_string & ucs, const UCS_string & vec)
1396{
1397 if (vec.size() == 0) return;
1398
1399bool in_UCS = false;
1400 ucs.append_UTF8("'");
1401
1402 loop(v, vec.size())
1403 {
1404 const Unicode uni = vec[v];
1405 const bool need_UCS = Avec::need_UCS(uni);
1406 if (in_UCS != need_UCS) // mode changed
1407 {
1408 if (in_UCS) ucs.append_UTF8("),'"); // UCS() → 'xxx'
1409 else ucs.append_UTF8("',(⎕UCS"); // 'xxx' → UCS()

40//----------------------------------------------------------------------------
41ErrorCode
42Parser::parse(const UCS_string & input, Token_string & tos) const
43{
44 // convert input characters into token
45 //
46Token_string tos1;
47
48 {
49 Tokenizer tokenizer(pmode, LOC, macro);
50 if (const ErrorCode ec = tokenizer.tokenize(input, tos1)) return ec;
51 }
52
53 // special case: single token (to speed up ⍎)
54 //
55 if (tos1.size() == 1)
56 {
57 const ErrorCode err = parse_statement(tos1);
58 if (err == E_NO_ERROR) tos.push_back(tos1[0]);
59 return err;
60 }
61
62 return parse(tos1, tos);
63}
64//----------------------------------------------------------------------------
65ErrorCode
66Parser::parse(const Token_string & input, Token_string & tos) const
67{
68 Log(LOG_parse)
69 {
70 CERR << "parse 1 [" << input.size() << "]: ";
71 input.print(CERR, true);
72 CERR << endl;
73 }
74
75 // split input line into statements (separated by ◊)
76 //
77std::vector<Token_string *> statements;
78 {
79 Token_string * stat = new Token_string();
80 loop(idx, input.size())
81 {
82 const Token & tok = input[idx];
83 if (tok.get_tag() == TOK_DIAMOND)
84 {
85 statements.push_back(stat);
86 stat = new Token_string();
87 }
88 else
89 {
90 stat->push_back(tok);
91 }
92 }
93 statements.push_back(stat);
94 }
95
96 loop(s, statements.size())
97 {
98 Token_string * stat = statements[s];
99 if (const ErrorCode err = parse_statement(*stat))
100 {
101 while (size_t(s) < statements.size())
102 {
103 stat = statements[s++];
104 delete stat;
105 }
106 return err;
107 }
108
109 if (s) tos.push_back(Token(TOK_DIAMOND));
110
111 loop(t, stat->size())
112 {
113 tos.push_back(Token());
114 tos[tos.size() - 1].move_1((*stat)[t], LOC);
115 }
116 delete stat;
117 }
118
119 return E_NO_ERROR;
120}
121//----------------------------------------------------------------------------
122ErrorCode
123Parser::parse_statement(Token_string & tos)
124{
125 // 1. convert (X) into X and ((X...)) into (X...)
126 //
127 remove_nongrouping_parantheses(tos);
128 remove_void_token(tos);
129
130 Log(LOG_parse)
131 {
132 CERR << "parse 2 [" << tos.size() << "]: ";
133 tos.print(CERR, true);
134 }
135
136 // 2. convert groups like '(' val val...')' into single APL values
137 // A group is defined as 2 or more values enclosed in '(' and ')'.
138 // For example: (1 2 3) or (4 5 (6 7) 8 9)
139 //
140 for (int progress = true; progress;)
141 {
142 progress = collect_groups(tos);
143 if (progress) remove_void_token(tos);
144 }
145
146 Log(LOG_parse)
147 {
148 CERR << "parse 3 [" << tos.size() << "]: ";
149 tos.print(CERR, true);
150 }
151
152 // 3. convert vectors like 1 2 3 or '1' 2 '3' into single APL values
153 //
154 collect_constants(tos);
155 remove_void_token(tos);
156
157 // special case: single APL value (to speed up ⍎)
158 //
159 if (tos.size() == 1)
160 {
161 Log(LOG_parse) CERR << "parse 3a: single value " << tos[0] << endl;;
162 return E_NO_ERROR;
163 }
164
165 Log(LOG_parse)
166 {
167 CERR << "parse 4 [" << tos.size() << "]: ";
168 tos.print(CERR, true);
169 }
170
171 // 4. mark symbol left of ← as LSYMB
172 //
173 mark_lsymb(tos);
174 Log(LOG_parse)
175 {
176 CERR << "parse 5 [" << tos.size() << "]: ";
177 tos.print(CERR, true);
178 }
179
180 // 5. replace bitwise functons ⊤∧, ⊤∨, ⊤⍲, and ⊤⍱ by their bitwise variant
181 //
182 replace_bitwise_functions(tos);
183 remove_void_token(tos);
184 Log(LOG_parse)
185 {
186 CERR << "parse 6 [" << tos.size() << "]: ";
187 tos.print(CERR, true);
188 }
189
190 // 6. update distances between (), [], and {}
191 //
192 {
193 const ErrorCode ec = match_par_bra(tos, false);
194 if (ec != E_NO_ERROR)
195 {
196 loop(t, tos.size()) tos[t].clear(LOC);
197 return ec;
198 }
199 }
200
201 return E_NO_ERROR;
202}
203//----------------------------------------------------------------------------
204void
205Parser::collect_constants(Token_string & tos)
206{
207 Log(LOG_collect_constants)
208 {
209 CERR << "collect_constants [" << tos.size() << " token] in: ";
210 tos.print(CERR, true);
211 }
212
213 // convert several items in vector notation into a single APL value
214 //
215 loop (t, tos.size())
216 {
217 size_t to;
218 switch(tos[t].get_tag())
219 {
220 case TOK_APL_VALUE1:
221 case TOK_APL_VALUE3:
222 case TOK_CHARACTER:
223 case TOK_COMPLEX:
224 case TOK_INTEGER:
225 case TOK_REAL:
226 break; // continue below
227
228 default: continue; // nex token
229 }
230
231 // at this point, t is the first item. Collect subsequenct items.
232 //
233 for (to = t + 1; to < tos.size(); ++to)
234 {
235#if 1
236 // Note: ISO 13751 gives an example with 1 2 3[2] ↔ 2
237 //
238 // In contrast, the binding rules stated in IBM's apl2lrm.pdf
239 // would give 3[2] ↔ RANK ERROR.
240 //
241 // We follow apl2lrm; to enable ISO behavior write #if 0 above.
242 //
243
244 // if tos[to + 1] is [ then [ binds stronger than
245 // vector notation and we stop collecting.
246 //
247 if (size_t(to + 1) < tos.size() &&
248 tos[to + 1].get_tag() == TOK_L_BRACK) break;
249#endif
250 const TokenTag tag = tos[to].get_tag();
251
252 if (tag == TOK_CHARACTER) continue;
253 if (tag == TOK_INTEGER) continue;
254 if (tag == TOK_REAL) continue;
255 if (tag == TOK_COMPLEX) continue;
256 if (tag == TOK_APL_VALUE1) continue;
257 if (tag == TOK_APL_VALUE3) continue;
258
259 // no more values: stop collecting
260 //
261 break;
262 }
263
264 create_value(tos, t, to - t);
265 }
266
267 Log(LOG_collect_constants)
268 {
269 CERR << "collect_constants [" << tos.size() << " token] out: ";
270 tos.print(CERR, true);
271 }
272}
273//----------------------------------------------------------------------------
274bool
275Parser::collect_groups(Token_string & tos)
276{
277 Log(LOG_collect_constants)
278 {
279 CERR << "collect_groups [" << tos.size() << " token] in: ";
280 tos.print(CERR, true);
281 }
282
283int opening = -1;
284 loop(t, tos.size())
285 {
286 switch(tos[t].get_tag())
287 {
288 case TOK_CHARACTER:
289 case TOK_INTEGER:
290 case TOK_REAL:
291 case TOK_COMPLEX:
292 case TOK_APL_VALUE1:
293 case TOK_APL_VALUE3:
294 continue;
295
296 case TOK_L_PARENT:
297 // we remember the last opening '(' so that we know where
298 // the group started when we see a ')'. This causes the
299 // deepest ( ... ) to be grouped first.
300 opening = t;
301 continue;
302
303 case TOK_R_PARENT:
304 if (opening == -1) continue; // ')' without '('
305 if (opening == t - 1) SYNTAX_ERROR; // '(' ')'
306
307 tos[opening].clear(LOC); // invalidate '('
308 tos[t].clear(LOC); // invalidate ')'
309 create_value(tos, opening + 1, t - opening - 1);
310
311 // we removed parantheses, so we close the value.
312 //
313 if (tos[opening + 1].get_tag() == TOK_APL_VALUE3)
314 tos[opening + 1].ChangeTag(TOK_APL_VALUE1);
315
316 Log(LOG_collect_constants)
317 {
318 CERR << "collect_groups [" << tos.size()
319 << " token] out: ";
320 tos.print(CERR, true);
321 }
322 return true;
323
324 default: // nothing group'able
325 opening = -1;
326 continue;
327 }
328 }
329
330 return false;
331}
332//----------------------------------------------------------------------------
333int
334Parser::find_closing_bracket(const Token_string & tos, int pos)
335{
336 Assert(tos[pos].get_tag() == TOK_L_BRACK);
337
338int others = 0;
339
340 for (size_t p = pos + 1; p < tos.size(); ++p)
341 {
342 Log(LOG_find_closing)
343 CERR << "find_closing_bracket() sees " << tos[p] << endl;
344
345 if (tos[p].get_tag() == TOK_R_BRACK)
346 {
347 if (others == 0) return p;
348 --others;
349 }
350 else if (tos[p].get_tag() == TOK_R_BRACK) ++others;
351 }
352
353 SYNTAX_ERROR;
354}

424//----------------------------------------------------------------------------
425void
426Parser::remove_nongrouping_parantheses(Token_string & tos)
427{
428 //
429 // 1. replace ((X...)) by: (X...)
430 // 2. replace (X) by: X for a single token X,
431 //
432 for (bool progress = true; progress; progress = false)
433 {
434 loop(t, int(tos.size()) - 2)
435 {
436 if (tos[t].get_Class() != TC_L_PARENT) continue;
437
438 const int closing = find_closing_parent(tos, t);
439
440 // check for case 1.
441 //
442 if (tos[t + 1].get_Class() == TC_L_PARENT)
443 {
444 // tos[t] is (( ...
445 //
446 const int closing_1 = find_closing_parent(tos, t + 1);
447 if (closing == (closing_1 + 1))
448 {
449 // tos[closing_1] is )) ...
450 // We have case 1. but not for example ((...)(...))
451 // remove redundant tos[t] and tos[closing] because
452 // ((...)) are not "not separating"
453 //
454 progress = true;
455 tos[t].clear(LOC);
456 tos[closing].clear(LOC);
457 continue;
458 }
459 }
460
461 // check for case 2.
462 //
463 if (closing != (t + 2)) continue;
464
465
466 // case 2. We have tos[t] = ( X ) ...
467 //
468 // (X) : "not grouping" if X is a scalar.
469 // If X is non-scalar, enclose it
470 //
471 progress = true;
472 tos[t + 2].move_1(tos[t + 1], LOC);
473
474 // we "remember" the nongrouping parantheses to disambiguate
475 // e,g, SYM/xxx from (SYM)/xxx
476 //
477 if (tos[t + 2].get_tag() == TOK_SYMBOL)
478 tos[t + 2].ChangeTag(TOK_P_SYMB);
479 tos[t + 1].clear(LOC);
480 tos[t].clear(LOC);
481 ++t; // skip tos[t + 1]
482 }
483 }
484}
485//----------------------------------------------------------------------------
486void
487Parser::replace_bitwise_functions(Token_string & tos)
488{
489 // replace bitwise functions (⊤∧ ⊤∨ ⊤⍲ ⊤⍱ ⊤= or ⊤≠) by their
490 // own token
491 //
492 loop(t, int(tos.size()) - 1)
493 {
494 if (tos[t].get_tag() != TOK_F12_ENCODE) continue;
495
496 switch(tos[t+1].get_tag())
497 {
498 case TOK_F2_AND:
499 tos[t] = Token(TOK_F2_AND_B, &Bif_F2_AND_B::_fun);
500 tos[++t].clear(LOC);
501 continue;
502
503 case TOK_F2_OR:
504 tos[t] = Token(TOK_F2_OR_B, &Bif_F2_OR_B::_fun);
505 tos[++t].clear(LOC);
506 continue;
507
508 case TOK_F2_NAND:
509 tos[t] = Token(TOK_F2_NAND_B,&Bif_F2_NAND_B::_fun);
510 tos[++t].clear(LOC);
511 continue;
512
513 case TOK_F2_NOR:
514 tos[t] = Token(TOK_F2_NOR_B, &Bif_F2_NOR_B::_fun);
515 tos[++t].clear(LOC);
516 continue;
517
518 case TOK_F2_EQUAL:
519 tos[t] = Token(TOK_F2_EQUAL_B, &Bif_F2_EQUAL_B::_fun);
520 tos[++t].clear(LOC);
521 continue;
522
523 case TOK_F2_UNEQU:
524 tos[t] = Token(TOK_F2_UNEQ_B, &Bif_F2_UNEQ_B::_fun);
525 tos[++t].clear(LOC);
526 continue;
527
528 default: break;
529 }
530 }

532//----------------------------------------------------------------------------
533bool
534Parser::check_if_value(const Token_string & tos, int pos)
535{
536 // figure if tos[pos] (the token left of /. ⌿. \. or ⍀) is the end of a
537 // function (and then return false) or the end of a value (and then
538 // return true).
539 //
540 switch(tos[pos].get_Class())
541 {
542 case TC_ASSIGN: // e.g. ←/ (actually syntax error)
543 case TC_R_ARROW: // e.g. →/ (actually syntax error)
544 case TC_L_BRACK: // e.g. [/ (actually syntax error)
545 case TC_END: // e.g. / (actually syntax error)
546 case TC_L_PARENT: // e.g. (/ (actually syntax error)
547 case TC_VALUE: // e.g. 5/
548 case TC_RETURN: // e.g. / (actually syntax error)
549 case TC_OPER2: // e.g. ./
550 return true; // tos[pos] is at the end of a value
551
552 case TC_R_BRACK: // e.g. +[1]/2 or 2/[1]2
553 {
554 const int pos1 = find_opening_bracket(tos, pos);
555 if (pos1 == 0) return true; // this is a syntax error
556 return check_if_value(tos, pos1 - 1);
557 }
558
559 case TC_R_PARENT: // e.g. (2+3)/2 or 1 2 3 (2+3)/2
560 {
561 if (pos == 0) return true; // (actually syntax error)
562 return check_if_value(tos, pos - 1);
563 }
564
565 case TC_SYMBOL: // e.g. A/2 or FOO/2
566 if (tos[pos].get_tag() == TOK_ALPHA) return true;
567 if (tos[pos].get_tag() == TOK_CHI) return true;
568 if (tos[pos].get_tag() == TOK_OMEGA) return true;
569 if (tos[pos].get_tag() == TOK_ALPHA_U) return false;
570 if (tos[pos].get_tag() == TOK_OMEGA_U) return false;
571 return (tos[pos].get_tag() == TOK_P_SYMB); // if value
572
573 default: break;
574 }
575
576 return false; // tos[pos] is at the end of a function
577}
578//----------------------------------------------------------------------------
579void
580Parser::remove_void_token(Token_string & tos)
581{
582size_t dst = 0;
583
584 loop(src, tos.size())
585 {
586 if (tos[src].get_tag() == TOK_VOID) continue;
587 if (src != ShapeItem(dst)) tos[dst].move_1(tos[src], LOC);
588 ++dst;
589 }
590
591 tos.resize(dst);
592}
593//----------------------------------------------------------------------------
594ErrorCode
595Parser::match_par_bra(Token_string & tos, bool backwards)
596{
597std::vector<ShapeItem> stack;
598 loop(s, tos.size())
599 {
600 const ShapeItem t = backwards ? (tos.size() - 1) - s : s;
601 ErrorCode ec;
602 TokenClass tc_peer;
603 switch(tos[t].get_Class())
604 {
605 // for [ ( or { push the position onto stack
606 case TC_L_BRACK: if (tos[t].get_tag() != TOK_L_BRACK) continue;
607 case TC_L_PARENT:
608 case TC_L_CURLY: stack.push_back(t);
609 continue;
610
611 case TC_R_BRACK: ec = E_UNBALANCED_BRACKET;
612 tc_peer = TC_L_BRACK;
613 break;
614
615 case TC_R_PARENT: ec = E_UNBALANCED_PARENT;
616 tc_peer = TC_L_PARENT;
617 break;
618
619 case TC_R_CURLY: ec = E_UNBALANCED_CURLY;
620 tc_peer = TC_L_CURLY;
621 break;
622
623 default: continue;
624 }
625
626 // at this point, a closing ), ], or } was detected
627 //
628 if (stack.size() == 0) return ec;
629
630 const ShapeItem t1 = stack.back();
631 stack.pop_back();
632
633 if (tos[t1].get_Class() != tc_peer) return ec;
634
635 const ShapeItem diff = (t > t1) ? t - t1 : t1 - t;
636 tos[t].set_int_val2(diff);
637 tos[t1].set_int_val2(diff);
638 }
639
640 // if there are unmatched items: return syntax error of the outer token
641 //
642 if (stack.size())
643 {
644 const TokenClass outer = tos[stack[0]].get_Class();
645 if (outer == TC_L_BRACK) return E_UNBALANCED_BRACKET;
646 if (outer == TC_L_PARENT) return E_UNBALANCED_PARENT;
647 return E_UNBALANCED_CURLY;
648 }
649
650 return E_NO_ERROR;
651}
652//----------------------------------------------------------------------------
653void
654Parser::create_value(Token_string & tos, int pos, int count)
655{
656 Log(LOG_create_value)
657 {
658 CERR << "create_value(" << __LINE__ << ") tos[" << tos.size()
659 << "] pos " << pos << " count " << count << " in:";
660 tos.print(CERR, true);
661 }
662
663 if (count == 1) create_scalar_value(tos[pos]);
664 else create_vector_value(tos, pos, count);
665
666 Log(LOG_create_value)
667 {
668 CERR << "create_value(" << __LINE__ << ") tos[" << tos.size()
669 << "] pos " << pos << " count " << count << " out:";
670 tos.print(CERR, true);
671 }
672}
673//----------------------------------------------------------------------------
674void
675Parser::create_scalar_value(Token & output)
676{
677 switch(output.get_tag())
678 {
679 case TOK_CHARACTER:
680 {
681 Value_P scalar(LOC);
682
683 scalar->next_ravel_Char(output.get_char_val());
684 scalar->check_value(LOC);
685 Token tok(TOK_APL_VALUE3, scalar);
686 output.move_1(tok, LOC);
687 }
688 return;
689
690 case TOK_INTEGER:
691 {
692 Token tok(TOK_APL_VALUE3, IntScalar(output.get_int_val(), LOC));
693 output.move_1(tok, LOC);
694 }
695 return;
696
697 case TOK_REAL:
698 {
699 Token tok(TOK_APL_VALUE3,
700 FloatScalar(output.get_flt_val(), LOC));
701 output.move_1(tok, LOC);
702 }
703 return;
704
705 case TOK_COMPLEX:
706 {
707 Token tok(TOK_APL_VALUE3,
708 ComplexScalar(output.get_cpx_real(),
709 output.get_cpx_imag(), LOC));
710 output.move_1(tok, LOC);
711 }
712 return;
713
714 case TOK_APL_VALUE1:
715 case TOK_APL_VALUE3:
716 return;
717
718 default: break;
719 }
720
721 CERR << "Unexpected token " << output.get_tag() << ": " << output << endl;
722 Assert(0 && "Unexpected token");
723}
724//----------------------------------------------------------------------------
725void
726Parser::create_vector_value(Token_string & tos, int pos, int count)
727{
728Value_P Z(count, LOC);
729
730 loop(l, count)
731 {
732 Token & tok = tos[pos + l];
733
734 switch(tok.get_tag())
735 {
736 case TOK_CHARACTER:
737 Z->next_ravel_Char(tok.get_char_val());
738 tok.clear(LOC); // invalidate token
739 break;
740
741 case TOK_INTEGER:
742 Z->next_ravel_Int(tok.get_int_val());
743 tok.clear(LOC); // invalidate token
744 break;
745
746 case TOK_REAL:
747 Z->next_ravel_Float(tok.get_flt_val());
748 tok.clear(LOC); // invalidate token
749 break;
750
751 case TOK_COMPLEX:
752 Z->next_ravel_Complex(tok.get_cpx_real(),
753 tok.get_cpx_imag());
754 tok.clear(LOC); // invalidate token
755 break;
756
757 case TOK_APL_VALUE1:
758 case TOK_APL_VALUE3:
759 Z->next_ravel_Value(tok.get_apl_val().get());
760 tok.clear(LOC); // invalidate token
761 break;
762
763 default: FIXME;
764 }
765 }
766
767 Z->check_value(LOC);
768Token tok(TOK_APL_VALUE3, Z);
769
770 tos[pos].move_1(tok, LOC);
771
772 Log(LOG_create_value)
773 {
774 CERR << "create_value [" << tos.size() << " token] out: ";
775 tos.print(CERR, true);
776 }
777}
778//----------------------------------------------------------------------------
780void
781Parser::mark_lsymb(Token_string & tos)
782{
783 loop(ass, tos.size())
784 {
785 if (tos[ass].get_Class() != TC_ASSIGN) continue;
786
787 // found ← in VAR VAR)← move backwards. Before that we handle the special
788 // case of vector specification, i.e. (SYM SYM ... SYM) ← value
789 //
790 if (ass >= 3 && tos[ass - 1].get_Class() == TC_R_PARENT &&
791 tos[ass - 2].get_Class() == TC_SYMBOL &&
792 tos[ass - 3].get_Class() == TC_SYMBOL)
793 {
794 // first make sure that this is really a vector specification.
795 //
796 const int syms_to = ass - 2;
797 int syms_from = ass - 3;
798 bool is_vector_spec = true;
799 bool is_selective_spec = false;
800 for (int a1 = ass - 4; a1 >= 0; --a1)
801 {
802 const TokenClass tc = tos[a1].get_Class();
803 if (tc == TC_L_PARENT) // end of (...)←
804 {
805 syms_from = a1 + 1;
806 break;
807 }
808 if (tc == TC_SYMBOL) continue;
809 if (tc == TC_FUN12 || tc == TC_OPER1 || tc == TC_OPER2)
810 {
811 is_selective_spec = true;
812 is_vector_spec = false;
813 break;
814 }
815
816 // something else
817 //
818 is_vector_spec = false;
819 break;
820 }
821
822 if (is_selective_spec == is_vector_spec) // none or both
823 {
824 MORE_ERROR() <<
825 "Left of )← seems to be neither a vector "
826 "specification nor a selective specvification";
827 LEFT_SYNTAX_ERROR;
828 }
829
830 // if this is a vector specification, then mark all symbols
831 // inside ( ... ) as TOK_LSYMB2.
832 //
833 if (is_vector_spec)
834 {
835 for (int a1 = syms_from; a1 <= syms_to; ++a1)
836 tos[a1].ChangeTag(TOK_LSYMB2);
837 }
838
839 continue;
840 }
841
842 int bracks = 0;
843 for (int prev = ass - 1; prev >= 0; --prev)
844 {
845 switch(tos[prev].get_Class())
846 {
847 case TC_R_BRACK: ++bracks; break;
848 case TC_L_BRACK: --bracks; break;
849 case TC_SYMBOL: if (bracks) break; // inside [ ... ]
850 if (tos[prev].get_tag() == TOK_SYMBOL)
851 {
852 tos[prev].ChangeTag(TOK_LSYMB);
853 }
854 // it could be that a system variable
855 // is assigned, e.g. ⎕SVE←0. In that case
856 // (actually always) we simply stop here.
857 //
858
859 /* fall-through */
860
861 case TC_END: prev = 0; // stop prev loop
862 break;
863
864 default: break;
865 }
866 }
867 }

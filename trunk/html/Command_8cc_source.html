64//----------------------------------------------------------------------------
65void
66Command::process_line()
67{
68UCS_string accu; // for new-style multiline strings
69UCS_string prompt = Workspace::get_prompt();
70 for (;;)
71 {
72 UCS_string line;
73 bool eof = false;
74 InputMux::get_line(LIM_ImmediateExecution, prompt,
75 line, eof, LineInput::get_history());
76
77 if (eof) CERR << "EOF at " << LOC << endl;
78
79 if (line.ends_with("\"\"\""))
80 {
81 if (accu.size() == 0) // start of multi-line
82 {
83 accu = line;
84 prompt.prepend(UNI_RIGHT_ARROW);
85 accu.resize(line.size() - 3); // discard """
86 accu.append(UNI_SPACE);
87 }
88 else // end of multi-line
89 {
90 accu.pop_back(); // trailing " "
91 process_line(accu);
92 return;
93 }
94 }
95 else if (accu.size()) // inside multi-line
96 {
97 accu.append_ASCII("\"");
98 accu.append(line.do_escape(true));
99 accu.append_ASCII("\" ");
100 }
101 else // normal input line
102 {
103 process_line(line);
104 return;
105 }
106 }
107}
108//----------------------------------------------------------------------------
109void
110Command::process_line(UCS_string & line)
111{
112 line.remove_leading_whitespaces();
113 if (line.size() == 0) return; // empty input line
114
115 switch(line[0])
116 {
117 case UNI_R_PARENT: // regular command, e.g. )SI
118 do_APL_command(COUT, line);
119 if (line.size()) break;
120 return;
121
122 case UNI_R_BRACK: // debug command, e.g. ]LOG
123 do_APL_command(CERR, line);
124 if (line.size()) break;
125 return;
126
127 case UNI_NABLA: // e.g. ∇FUN
128 Nabla::edit_function(line);
129 return;
130
131 case UNI_NUMBER_SIGN: // e.g. # comment
132 case UNI_COMMENT: // e.g. ⍝ comment
133 return;
134
135 default: break;
136 }
137
138 ++APL_expression_count;
139 do_APL_expression(line);
140}
141//----------------------------------------------------------------------------
142bool
143Command::do_APL_command(ostream & out, UCS_string & line)
144{
145const UCS_string line1(line); // the original line
146
147 // split line into command and arguments
148 //
149UCS_string cmd; // the command without arguments
150int len = 0;
151 line.copy_black(cmd, len);
152
153UCS_string arg(line, len, line.size() - len);
154UCS_string_vector args = split_arg(arg);
155 line.clear();
156 if (!cmd.starts_iwith(")MORE"))
157 {
158 // clear )MORE info unless command is )MORE
159 //
160 Workspace::more_error().clear();
161 }
162
163#define cmd_def(cmd_str, code, garg, _hint) \
164 if (cmd.starts_iwith(cmd_str)) \
165 { if (check_params(out, cmd_str, args.size(), garg)) return true; \
166 code; return false; }
167#include "Command.def"
168
169 // check for user defined commands...
170 //
171 loop(u, Workspace::get_user_commands().size())
172 {
173 if (cmd.starts_iwith(Workspace::get_user_commands()[u].prefix))
174 {
175 do_USERCMD(out, line, line1, cmd, args, u);
176 return true;
177 }
178 }
179
180 out << "BAD COMMAND" << endl;
181 return false;
182}
183//----------------------------------------------------------------------------
184bool
185Command::check_params(ostream & out, const char * command, int argc,
186 const char * args)
187{
188 // allow everything for ]USERCMD
189 //
190 if (!strcmp(command, "]USERCMD")) return false;
191
192 // analyze args to figure the number of parametes expected.
193 //
194int mandatory_args = 0;
195int opt_args = 0;
196int brackets = 0;
197bool in_param = false;
198bool many = false;
199
200UCS_string args_ucs(args);
201 loop (a, args_ucs.size()) switch(args_ucs[a])
202 {
203 case '[': ++brackets; in_param = false; continue;
204 case ']': --brackets; in_param = false; continue;
205 case '|': in_param = false;
206 if (brackets) --opt_args;
207 else --mandatory_args;
208 continue;
209 case '.':
210 if (a < (args_ucs.size() - 2) &&
211 args_ucs[a + 1] == '.' &&
212 args_ucs[a + 2] == '.') many = true;
213 continue;
214 case '0' ... '9':
215 case '_':
216 case 'A' ... 'Z':
217 case 'a' ... 'z':
218 case UNI_OVERBAR:
219 case '-':
220 if (!in_param) // start of a name or range
221 {
222 if (brackets) ++opt_args;
223 else ++mandatory_args;
224 in_param = true;
225 }
226 continue;
227
228 case ' ': in_param = false;
229 continue;
230
231 default: Q1(args_ucs[a]) Q1(int(args_ucs[a]))
232 }
233
234 if (argc < mandatory_args) // too few parameters
235 {
236 out << "BAD COMMAND+" << endl;
237 MORE_ERROR() << "missing parameter(s) in command " << command
238 << ". Usage:\n"
239 << " " << command << " " << args;
240 return true;
241 }
242
243 if (many) return false;
244
245 if (argc > (mandatory_args + opt_args)) // too many parameters
246 {
247 out << "BAD COMMAND+" << endl;
248 MORE_ERROR() << "too many (" << argc<< ") parameter(s) in command "
249 << command << ". Usage:\n"
250 << " " << command << " " << args;
251 return true;
252 }
253
254 return false; // OK
255}
256//----------------------------------------------------------------------------
257void
258Command::do_APL_expression(UCS_string & line)
259{
260 Workspace::more_error().clear();
261
262Executable * statements = 0;
263 try
264 {
265 statements = StatementList::fix(line, LOC);
266 }
267 catch (Error err)
268 {
269 UERR << Error::error_name(err.get_error_code());
270 if (Workspace::more_error().size()) UERR << UNI_PLUS;
271 UERR << endl;
272 if (*err.get_error_line_2())
273 {
274 COUT << " " << err.get_error_line_2() << endl
275 << " " << err.get_error_line_3() << endl;
276 }
277
278 err.print(UERR, LOC);
279 delete statements;
280 return;
281 }
282 catch (...)
283 {
284 CERR << "*** Command::process_line() caught other exception ***"
285 << endl;
286 delete statements;
287 cmd_OFF(0);
288 }
289
290 if (statements == 0)
291 {
292 COUT << "main: Parse error." << endl;
293 return;
294 }
295
296 // At this point, the user command was parsed correctly.
297 // check for Escape (→)
298 //
299 {
300 const Token_string & body = statements->get_body();
301 if (body.size() == 3 &&
302 body[0].get_tag() == TOK_ESCAPE &&
303 body[1].get_Class() == TC_END &&
304 body[2].get_tag() == TOK_RETURN_STATS)
305 {
306 // remove all SI entries up to (including) the next immediate
307 // execution context
308 //
309 for (bool goon = true; goon;)
310 {
311 StateIndicator * si = Workspace::SI_top();
312 if (si == 0) break; // SI empty
313
314 goon = si->get_parse_mode() != PM_STATEMENT_LIST;
315 si->escape(); // pop local vars of user defined functions
316 Workspace::pop_SI(LOC);
317 }
318
319 delete statements;
320 return;
321 }
322 }
323
324// statements->print(CERR);
325
326 // push a new context for the statements.
327 //
328 Workspace::push_SI(statements, LOC);
329 finish_context();
330}
331//----------------------------------------------------------------------------
332void
333Command::finish_context()
334{
335 for (;;)
336 {
337 //
338 // NOTE that the entire SI may change while executing this loop.
339 // We should therefore avoid references to SI entries.
340 //
341 Token token = Workspace::SI_top()->get_executable()->execute_body();
342
343// Q1(token)
344
345 // start over if execution has pushed a new SI entry
346 //
347 if (token.get_tag() == TOK_SI_PUSHED) continue;
348
349check_EOC:
350 if (Quad_FIO::benchmark_cycles_from) // a benchmark is ongoing
351 {
352 if (StateIndicator * si = Workspace::SI_top()->get_parent())
353 {
354 if (si->is_safe_execution_start())
355 {
356 // at this point token is the result of a defined
357 // function which is being benchmarked. Replace the
358 // result by the number of CPU cycles executed since
359 // the start of the benchmark.
360 //
361 const uint64_t to = cycle_counter();
362 const uint64_t from = Quad_FIO::benchmark_cycles_from;
363 const uint64_t diff = to - from;
364 Value_P val = IntScalar(diff, LOC);
365 token.~Token(); // free the value (if any)
366 new (&token) Token(TOK_APL_VALUE1, val);
367 si->clear_safe_execution();
368 }
369 }
370 }
371 else if (Workspace::SI_top()->is_safe_execution_start())
372 {
373 if (Quad_FIO::benchmark_cycles_from)
374 {
375 // ⎕FIO[-1] has started a cycle measurement. Read the CPU
376 // cycle counter, compute the cycle difference, stop the
377 // measurement, and return the the cycle difference as int
378 // scalar...
379 //
380 const uint64_t to = cycle_counter();
381 const uint64_t diff = to - Quad_FIO::benchmark_cycles_from;
382 token = Token(TOK_APL_VALUE1, IntScalar(diff, LOC));
383 Quad_FIO::benchmark_cycles_from = 0;
384 }
385 else
386 {
387 Quad_EC::eoc(token);
388 }
389 }
390
391 // the far most frequent cases are TC_VALUE and TOK_VOID
392 // so we handle them first.
393 //
394 if (token.get_Class() == TC_VALUE || token.get_tag() == TOK_VOID )
395 {
396 if (Workspace::SI_top()->get_parse_mode() == PM_STATEMENT_LIST)
397 {
398 if (attention_is_raised())
399 {
400 clear_attention_raised(LOC);
401 clear_interrupt_raised(LOC);
402 ATTENTION;
403 }
404
405 break; // will return to calling context
406 }
407
408 Workspace::pop_SI(LOC);
409
410 // we are back in the calling SI. There should be a TOK_SI_PUSHED
411 // token at the top of stack. Replace it with the result from
412 // the called (just poped) SI.
413 //
414 {
415 Prefix & prefix =
416 Workspace::SI_top()->get_prefix();
417 Assert(prefix.at0().get_tag() == TOK_SI_PUSHED);
418
419 new (&prefix.tos().tok) Token(token);
420 }
421 if (attention_is_raised())
422 {
423 clear_attention_raised(LOC);
424 clear_interrupt_raised(LOC);
425 ATTENTION;
426 }
427
428 continue;
429 }
430
431 if (token.get_tag() == TOK_BRANCH)
432 {
433 const Function_Line line = Function_Line(token.get_int_val());
434 if (line == Function_Retry &&
435 Workspace::SI_top()->get_parse_mode() == PM_STATEMENT_LIST &&
436 Workspace::SI_top()->get_parent())
437 {
438 Workspace::pop_SI(LOC);
439 Workspace::SI_top()->retry(LOC);
440 continue;
441 }
442
443 StateIndicator * si = Workspace::SI_top_fun();
444
445 if (si == 0)
446 {
447 MORE_ERROR() <<
448 "branch back into function (→N) without suspended function";
449 SYNTAX_ERROR; // →N without function,
450 }
451
452 // pop contexts above defined function
453 //
454 while (si != Workspace::SI_top()) Workspace::pop_SI(LOC);
455
456 si->goon(line, LOC);
457 continue;
458 }
459
460 if (token.get_tag() == TOK_ESCAPE)
461 {
462 // remove all SI entries up to (including) the next immediate
463 // execution context
464 //
465 for (bool goon = true; goon;)
466 {
467 StateIndicator * si = Workspace::SI_top();
468 if (si == 0) break; // SI empty
469
470 goon = si->get_parse_mode() != PM_STATEMENT_LIST;
471 si->escape(); // pop local vars of user defined functions
472 Workspace::pop_SI(LOC);
473 }
474 return;
475 }
476
477 if (token.get_tag() == TOK_ERROR)
478 {
479 if (token.get_int_val() == E_COMMAND_PUSHED)
480 {
481 Workspace::pop_SI(LOC);
482 UCS_string pushed_command = Workspace::get_pushed_Command();
483 process_line(pushed_command);
484 pushed_command.clear();
485 Workspace::push_Command(pushed_command); // clear in
486 return;
487 }
488
489 // clear attention and interrupt flags
490 //
491 clear_attention_raised(LOC);
492 clear_interrupt_raised(LOC);
493
494 // check for safe execution mode. Unroll all SI entries that
495 // have the same safe_execution_count, except the last
496 // unroll the SI stack.
497 //
498 if (Workspace::SI_top()->get_safe_execution())
499 {
500 // SI_top() is in save execution mode. Pop it and all
501 // callers with the same get_safe_execution() level.
502 //
503 // after pop'ing the SI entries only the original SI
504 // (which has set the save execution mode) shall remain
505 // the SI stack.
506 //
507 StateIndicator * si = Workspace::SI_top();
508 const int sex_level = si->get_safe_execution();
509 while (si->get_parent() && sex_level ==
510 si->get_parent()->get_safe_execution())
511 {
512 si = si->get_parent();
513 Workspace::pop_SI(LOC);
514 }
515
516 goto check_EOC;
517 }
518
519 // if suspend is not allowed then pop all SI entries that
520 // don't allow suspend
521 //
522 if (Workspace::SI_top()->get_executable()->cannot_suspend())
523 {
524 Error err = StateIndicator::get_error(Workspace::SI_top());
525 while (Workspace::SI_top()->get_executable()
526 ->cannot_suspend())
527 {
528 Workspace::pop_SI(LOC);
529 }
530
531 if (Workspace::SI_top())
532 {
533 StateIndicator::get_error(Workspace::SI_top()) = err;
534 }
535 }
536
537 if (Workspace::get_error()->get_print_loc() == 0) // not printed
538 {
539 Workspace::get_error()->print(CERR, LOC);
540 }
541 else
542 {
543 // CERR << "ERROR printed twice" << endl;
544 }
545
546 if (Workspace::SI_top()->get_level() == 0)
547 {
548 Value::erase_stale(LOC);
549 IndexExpr::erase_stale(LOC);
550 }
551 return;
552 }
553
554 // we should not come here.
555 //
556 Q1(token) Q1(token.get_Class()) Q1(token.get_tag()) FIXME;
557 }
558
559 // pop the context for the statements
560 //
561 Workspace::pop_SI(LOC);
562}

654//----------------------------------------------------------------------------
655void
656Command::cmd_CHECK(ostream & out)
657{
658 // 1. erase stale functions from failed ⎕EX
659 //
660 {
661 bool erased = false;
662 if (const int stale = Workspace::cleanup_expunged(CERR, erased))
663 {
664 out << "WARNING - " << stale << " stale functions ("
665 << (erased ? "" : "not ") << "erased)" << endl;
666 }
667 else
668 {
669 out << "OK - no stale functions" << endl;
670 }
671 }
672
673 // 2. print stale values (if any)
674 //
675 {
676 if (const int stale = Value::print_stale(CERR))
677 {
678 out << "ERROR - " << stale << " stale values" << endl;
679 IO_Files::apl_error(LOC);
680 }
681 else
682 {
683 out << "OK - no stale values" << endl;
684 }
685 }
686
687 // 3. print stale index expressions (if any)
688 {
689 if (const int stale = IndexExpr::print_stale(CERR))
690 {
691 out << "ERROR - " << stale << " stale indices" << endl;
692 IO_Files::apl_error(LOC);
693 }
694 else
695 {
696 out << "OK - no stale indices" << endl;
697 }
698 }
699
700 // 4. discover duplicate parents. In the old clone() scheme every nested value // has (at most) one parent. In the new clone() scheme, however, a nested
701 // value can be reused and have multiple parents.
702 {
703#ifndef NEW_CLONE // old clone
704 // 4a. create a { parent = 0, value } vector<val_val> of all values
705 //
706 std::vector<val_val> values;
707 ShapeItem duplicate_parents = 0;
708 for (const DynamicObject * obj =
709 DynamicObject::get_all_values()->get_next();
710 obj != DynamicObject::get_all_values(); obj = obj->get_next())
711 {
712 const Value * val = static_cast<const Value *>(obj);
713
714 val_val vv = { 0, val }; // no parent
715 values.push_back(vv);
716 }
717
718 // 4b. sort vector<val_val> values by address so we can bsearch it.
719 //
720 Heapsort<val_val>::sort(&values[0], values.size(), 0,
721 &val_val::compare_val_val);
722 loop(v, (values.size() - 1))
723 Assert(&values[v].child < &values[v + 1].child);
724
725 // 4c. set parents of pointer cells
726 //
727 loop(v, values.size()) // for every .child (acting as parent here)
728 {
729 const Value * val = values[v].child;
730 const ShapeItem ec = val->nz_element_count();
731 loop(e, ec) // for every ravel cell of the (parent-) value
732 {
733 const Cell & cP = val->get_cravel(e);
734 if (!cP.is_pointer_cell()) continue; // not a parent
735
736 const Value * sub = cP.get_pointer_value().get();
737 Assert1(sub);
738
739 val_val * vvp = reinterpret_cast<val_val *>
740 (bsearch(sub, &values[0], values.size(),
741 sizeof(val_val), val_val::compare_val_val1));
742 Assert(vvp);
743 if (vvp->parent == 0) // child has no parent (OK)
744 {
745 vvp->parent = val;
746 continue;
747 }
748
749 // the child already has a parent (which is bad).
750 // print its history to help figuring why.
751 //
752 ++duplicate_parents;
753 out << "Value * vvp=" << voidP(vvp) << " already has parent "
754 << voidP(vvp->parent) << " when checking Value * val="
755 << voidP(vvp) << endl;
756
757 out << "History of the child:" << endl;
758 VH_entry::print_history(out, *vvp->child, LOC);
759 out << "History of the first parent:" << endl;
760 VH_entry::print_history(out, *vvp->parent, LOC);
761 out << "History of the second parent:" << endl;
762 VH_entry::print_history(out, *val, LOC);
763 out << endl;
764 }
765 }
766
767 if (duplicate_parents)
768 {
769 out << "ERROR - " << duplicate_parents
770 << " duplicate parents" << endl;
771 IO_Files::apl_error(LOC);
772 }
773 else
774#endif
775 {
776 out << "OK - no duplicate parents" << endl;
777 }
778 }
779
780 // 5. discover strange Cells and counters.
781 //
782 Value::check_all_Cells(out);
783}

845 UCS_string ws(UNI_SPACE);
846 ws += src[0]; // Nwsname

849 ws += UNI_SPACE;
850 usv.push_back(ws);
851 }

872 //

877 // silently return if wsname is already contained in the copy_once_table
878 //

881 // add it to the table;
882 //
883 CERR << "NEW )COPY_ONCE workspace: "
884 << Unicode(libref + UNI_0) << " " << wsname << endl;

886 copy_once_table.push_back(lib_wsname);

1400//----------------------------------------------------------------------------
1401void
1402Command::cmd_IN(ostream & out, UCS_string_vector & args, bool protection)
1403{
1404 // Command is:
1405 //
1406 // IN filename [objects...]
1407
1408UCS_string fname = args[0];
1409 args[0] = args.back();
1410 args.pop_back();
1411
1412UTF8_string filename = LibPaths::get_lib_filename(LIB_NONE, fname, true,
1413 ".atf", 0);
1414
1415FILE * in = fopen(filename.c_str(), "r");
1416 if (in == 0) // open failed: try filename.atf unless already .atf
1417 {
1418 UTF8_string fname_utf8(fname);
1419 CERR << ")IN " << fname_utf8.c_str()
1420 << " failed: " << strerror(errno) << endl;
1421
1422 char cc[200];
1423 snprintf(cc, sizeof(cc),
1424 "command )IN: could not open file %s for reading: %s",
1425 fname_utf8.c_str(), strerror(errno));
1426 Workspace::more_error() << cc;
1427 return;
1428 }
1429
1430UTF8 buffer[80];
1431int idx = 0;
1432
1433transfer_context tctx(protection);
1434
1435 for (;;)
1436 {
1437 const int cc = fgetc(in);
1438 if (cc == EOF) break;
1439 if (idx == 0 && cc == 0x0A) // optional LF
1440 {
1441 // CERR << "CRLF" << endl;
1442 continue;
1443 }
1444
1445 if (idx < 80)
1446 {
1447 if (idx < 72) buffer[idx++] = cc;
1448 else buffer[idx++] = 0;
1449 continue;
1450 }
1451
1452 if (cc == 0x0D || cc == 0x15) // ASCII or EBCDIC
1453 {
1454 tctx.is_ebcdic = (cc == 0x15);
1455 tctx.process_record(buffer, args);
1456
1457 idx = 0;
1458 ++tctx.recnum;
1459 continue;
1460 }
1461
1462 CERR << "BAD record charset (neither ASCII nor EBCDIC)" << endl;
1463 break;
1464 }
1465
1466 fclose(in);
1467}

1569DIR *
1570Command::open_LIB_dir(UTF8_string & path, ostream & out,
1571 const UCS_string_vector & args)
1572{
1573 // args can be one of:
1574 // example:
1575 // 1. )LIB
1576 // 2. N )LIB 1
1577 // 3. )LIB directory-name )LIB /usr/lib/...
1578 //
1579
1580UCS_string arg("0");
1581 if (args.size()) arg = args[0];
1582
1583 if (args.size() == 0) // case 1.
1584 {
1585 path = LibPaths::get_lib_dir(LIB0);
1586 }
1587 else if (arg.size() == 1 &&
1588 Avec::is_digit(Unicode(arg[0]))) // case 2.
1589 {
1590 path = LibPaths::get_lib_dir(LibRef(arg[0] - '0'));
1591 }
1592 else // case 3.
1593 {
1594 path = UTF8_string(arg);
1595 }
1596
1597 // follow symbolic links, but not too often (because symbolic links may
1598 // form an endless loop)...
1599 //
1600 loop(depth, 20)
1601 {
1602 char buffer[FILENAME_MAX + 1];
1603 const ssize_t len = readlink(path.c_str(), buffer, FILENAME_MAX);
1604 if (len <= 0) break; // not a symlink
1605
1606 buffer[len] = 0;
1607 if (buffer[0] == '/') // absolute path
1608 {
1609 path = UTF8_string(buffer);
1610 }
1611 else // relative path
1612 {
1613 path += '/';
1614 path.append_UTF8(UTF8_string(buffer));
1615 }
1616 }
1617
1618DIR * dir = opendir(path.c_str());
1619
1620 if (dir == 0)
1621 {
1622 const char * why = strerror(errno);
1623 out << "IMPROPER LIBRARY REFERENCE '" << arg << "': " << why << endl;
1624
1625 MORE_ERROR() <<
1626 "path '" << path << "' could not be opened as directory: " << why;
1627 return 0; // error
1628 }
1629
1630 return dir;
1631}

1649//----------------------------------------------------------------------------
1650void
1651Command::lib_common(ostream & out, const UCS_string_vector & args_range,
1652 int variant)
1653{
1654 // 1. check for (and then extract) an optional range parameter...
1655 //
1656UCS_string_vector args;
1657const UCS_string * range = 0;
1658 loop(a, args_range.size())
1659 {
1660 const UCS_string & arg = args_range[a];
1661 bool is_range = false;
1662 bool is_path = false;
1663 loop(aa, arg.size())
1664 {
1665 const Unicode uni = arg[aa];
1666 if (uni == UNI_MINUS)
1667 {
1668 is_range = true;
1669 break;
1670 }
1671
1672 if (!Avec::is_symbol_char(uni))
1673 {
1674 is_path = true;
1675 break;
1676 }
1677 }
1678
1679 if (is_path) // normal (non-range) arg
1680 {
1681 args.push_back(arg);
1682 }
1683 else if (!is_range) // normal (non-range) arg
1684 {
1685 args.push_back(arg);
1686 }
1687 else if (range) // second non-range arg
1688 {
1689 MORE_ERROR() <<
1690 "multiple range parameters in )LIB or ]LIB command";
1691 return;
1692 }
1693 else
1694 {
1695 range = &arg;
1696 }
1697 }
1698
1699UCS_string from;
1700UCS_string to;
1701 if (range)
1702 {
1703 const bool bad_from_to = parse_from_to(from, to, *range);
1704 if (bad_from_to)
1705 {
1706 CERR << "bad range argument" << endl;
1707 MORE_ERROR() << "bad range argument " << *range
1708 << ", expecting from-to";
1709 return;
1710 }
1711 }
1712
1713 // 2. open directory
1714 //
1715UTF8_string path;
1716DIR * dir = open_LIB_dir(path, out, args);
1717 if (dir == 0) return;
1718
1719 // 3. collect files and directories
1720 //
1721UCS_string_vector files;
1722UCS_string_vector directories;
1723
1724 for (;;)
1725 {
1726 dirent * entry = readdir(dir);
1727 if (entry == 0) break; // directory done
1728
1729 UTF8_string filename_utf8(entry->d_name);
1730 UCS_string filename(filename_utf8);
1731
1732 // check range (if any)...
1733 //
1734 if (from.size() && filename.lexical_before(from)) continue;
1735 if (to.size() && to.lexical_before(filename)) continue;
1736
1737 if (is_directory(entry, path))
1738 {
1739 if (filename_utf8[0] == '.') continue;
1740 filename.append(UNI_SLASH);
1741 directories.push_back(filename);
1742 continue;
1743 }
1744
1745 const int dlen = strlen(entry->d_name);
1746 if (variant == 1)
1747 {
1748 if (filename_utf8.ends_with(".apl"))
1749 {
1750 filename.resize(filename.size() - 4); // skip extension
1751 files.push_back(filename);
1752 }
1753 else if (filename_utf8.ends_with(".xml"))
1754 {
1755 filename.resize(filename.size() - 4); // skip extension
1756 files.push_back(filename);
1757 }
1758 }
1759 else
1760 {
1761 if (filename[0] == '.') continue; // skip dot files
1762 if (filename[dlen - 1] == '~') continue; // and editor backups
1763 files.push_back(filename);
1764 }
1765 }
1766 closedir(dir);
1767
1768 // 4. sort directories and filenames alphabetically and append files
1769 // to directories
1770 //
1771 directories.sort();
1772 files.sort();
1773 loop(f, files.size())
1774 {
1775 if (directories.size() && directories.back() == files[f])
1776 {
1777 // there were some file.apl and file.xml. Skip the second
1778 //
1779 continue;
1780 }
1781 directories.push_back(files[f]);
1782 }
1783
1784 // 5. list directories first, then files
1785 //
1786
1787 // figure column widths
1788 //
1789 enum { tabsize = 4 };
1790
1791std::vector<int> col_widths;
1792 directories.compute_column_width(tabsize, col_widths);
1793
1794 loop(c, directories.size())
1795 {
1796 const size_t col = c % col_widths.size();
1797 out << directories[c];
1798 if (col == size_t(col_widths.size() - 1) ||
1799 c == ShapeItem(directories.size() - 1))
1800 {
1801 // last column or last item: print newline
1802 //
1803 out << endl;
1804 }
1805 else
1806 {
1807 // intermediate column: print spaces
1808 //
1809 const int len = tabsize*col_widths[col] - directories[c].size();
1810 Assert(len > 0);
1811 loop(l, len) out << " ";
1812 }
1813 }
1814}

28PointerCell::PointerCell(Value * sub_val, Value & cell_owner)
29{
30 Assert(!sub_val->is_simple_scalar());
31
32 new (&value.pval.valp) Value_P(sub_val, LOC);
33 value.pval.owner = &cell_owner;
34
35 Assert(value.pval.owner != sub_val); // typical cut-and-paste error
36 cell_owner.increment_pointer_cell_count();
37 cell_owner.add_subcount(sub_val->nz_element_count());
38}

56void
57PointerCell::init_other(void * other, Value & other_owner,
58 const char * loc) const
59{
60 Assert(other); // the new PointerCell to be created
61
62Value_P sub; // instantiate beforehand so that sub is 0 if clone() fails
63
64 sub = CLONE_P(get_pointer_value(), loc);
65 Assert(+sub);
66 new (other) PointerCell(sub.get(), other_owner);
67}
68//----------------------------------------------------------------------------
69void
70PointerCell::release(const char * loc)
71{
72 // 1. update the PointerCell related counters in the owner of this
73 // PointerCell (since this Cell is no longer a PointerCell).
74 //
75 value.pval.owner->decrement_pointer_cell_count();
76 value.pval.owner->add_subcount(-get_pointer_value()->nz_element_count());
77
78 // 2. decrement the owner_count of our sub-value via reset() (since this
79 // PointerCell releases the ownership of the sub-value.
80 //
81 value.pval.valp.reset();
82
83 // 3. not needed but to be on the safe side: make this Cell an IntCell
84 // (in case someone still points to it).
85 //
86 IntCell::z0(this);
87}
88//----------------------------------------------------------------------------
89bool
90PointerCell::equal(const Cell & other, double qct) const
91{
92 if (!other.is_pointer_cell()) return false;
93
94Value_P A = get_pointer_value();
95Value_P B = other.get_pointer_value();
96
97 if (!A->same_shape(*B)) return false;
98
99const ShapeItem count = A->nz_element_count();
100 loop(c, count)
101 if (!A->get_cravel(c).equal(B->get_cravel(c), qct)) return false;
102
103 return true;
104}

109 if (compare(other) == COMP_GT) return true;
110 if (compare(other) == COMP_LT) return false;
111 return this > &other;

113//----------------------------------------------------------------------------
114Comp_result
115PointerCell::compare(const Cell & other) const
116{
117 if (other.get_cell_type() & CT_SIMPLE) // nested > numeric > char
118 return COMP_GT;
119
120 if (other.get_cell_type() != CT_POINTER) DOMAIN_ERROR;
121
122 // at this point both cells are pointer cells.
123 //
124Value_P v1 = get_pointer_value();
125Value_P v2 = other.get_pointer_value();
126
127 // compare ranks
128 //
129 if (v1->get_rank() != v2->get_rank()) // ranks differ
130 return v1->get_rank() > v2->get_rank() ? COMP_GT : COMP_LT;
131
132 // same rank, compare shapes
133 //
134 loop(r, v1->get_rank())
135 {
136 const ShapeItem axis1 = v1->get_shape_item(r);
137 const ShapeItem axis2 = v2->get_shape_item(r);
138 if (axis1 != axis2) // axis r differs
139 return axis1 > axis2 ? COMP_GT : COMP_LT;
140 }
141
142 // same rank and shape, compare ravel
143 //
144const Cell * C1 = &v1->get_cfirst();
145const Cell * C2 = &v2->get_cfirst();
146 loop(e, v1->nz_element_count())
147 {
148 if (const Comp_result comp = C1++->compare(*C2++)) return comp;
149 }
150
151 // everthing equal
152 //
153 return COMP_EQ;
154}

183PrintBuffer
184PointerCell::character_representation(const PrintContext & pctx) const
185{
186Value_P val = get_pointer_value();
187 Assert(+val);
188
189 if (pctx.get_style() & PST_QUOTE_CHARS)
190 {
191 if (val->is_char_vector()) return PrintBuffer(*val, pctx, 0);
192 }
193
194 if (pctx.get_style() == PR_APL_FUN) // APL function display
195 {
196 const ShapeItem ec = val->element_count();
197 UCS_string ucs;
198
199 if (val->is_char_vector())
200 {
201 ucs.append(UNI_SINGLE_QUOTE);
202 loop(e, ec)
203 {
204 const Unicode uni = val->get_cravel(e).get_char_value();
205 ucs.append(uni);
206 if (uni == UNI_SINGLE_QUOTE) ucs.append(uni); // ' -> ''
207 }
208 ucs.append(UNI_SINGLE_QUOTE);
209 }
210 else
211 {
212 ucs.append(UNI_L_PARENT);
213 loop(e, ec)
214 {
215 PrintBuffer pb = val->get_cravel(e).
216 character_representation(pctx);
217 ucs.append(UCS_string(pb, 0, Workspace::get_PW()));
218
219 if (e < ec - 1) ucs.append(UNI_SPACE);
220 }
221 ucs.append(UNI_R_PARENT);
222 }
223
224 ColInfo ci;
225 PrintBuffer ret(ucs, ci);
226 ret.get_info().int_len = ret.get_column_count();
227 ret.get_info().real_len = ret.get_column_count();
228 return ret;
229 }
230
231PrintBuffer ret(*val, pctx, 0);
232 ret.get_info().flags &= ~CT_MASK;
233 ret.get_info().flags |= CT_POINTER;
234
235 if (pctx.get_style() & PST_CS_INNER)
236 {
237 const ShapeItem ec = val->element_count();
238 if (ec == 0) // empty value
239 {
240 int style = pctx.get_style();
241 loop(r, val->get_rank())
242 {
243 if (val->get_shape_item(r) == 0) // an empty axis
244 {
245 if (r == val->get_rank() - 1) style |= PST_EMPTY_LAST;
246 else style |= PST_EMPTY_NLAST;
247 }
248 }
249
250 // do A←(1⌈⍴A)⍴⊂↑A
251 Value_P proto = val->prototype(LOC);
252
253 // compute a shape like ⍴ val but 0 replaced with 1.
254 //
255 Shape sh(val->get_shape());
256 loop(r, sh.get_rank())
257 {
258 if (sh.get_shape_item(r) == 0) sh.set_shape_item(r, 1);
259 }
260
261 Value_P proto_reshaped(sh, LOC);
262 const ShapeItem len = proto_reshaped->nz_element_count();
263
264 // store proto in the first ravel item, and copies of proto in
265 // the subsequent ravel items
266 //
267 loop(rv, len) proto_reshaped->next_ravel_Value(proto.get());
268
269 proto_reshaped->check_value(LOC);
270 ret = PrintBuffer(*proto_reshaped, pctx, 0);
271 ret.add_frame(PrintStyle(style), proto_reshaped->get_shape(),
272 proto_reshaped->compute_depth());
273 }
274 else
275 {
276 if (!(pctx.get_style() & PST_QUOTE_CHARS && val->is_char_array()))
277 {
278 ret.add_frame(pctx.get_style(), val->get_shape(),
279 val->compute_depth());
280 }
281 }
282 }
283
284 ret.get_info().int_len = ret.get_column_count();
285 ret.get_info().real_len = ret.get_column_count();
286 return ret;
287}

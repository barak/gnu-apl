38void
39InputFile::open_current_file()
40{
41 if (files_todo.size() && files_todo[0].file == 0)
42 {
43 if (!strcmp(current_filename(), "-"))
44 files_todo[0].file = stdin;
45 else if (!strcmp(current_filename(), "stdin"))
46 files_todo[0].file = stdin;
47 else
48 files_todo[0].file = fopen(current_filename(), "r");
49
50 files_todo[0].line_no = 0;
51 }
52}
53//----------------------------------------------------------------------------
54void
55InputFile::close_current_file()
56{
57 if (files_todo.size()) // there are input files
58 {
59 if (files_todo[0].from_COPY)
60 {
61 files_todo[0].from_COPY = false;
62 --Bif_F1_EXECUTE::copy_pending;
63 }
64
65 if (files_todo[0].file) // and the first file is open
66 {
67 if (files_todo[0].file != stdin)
68 {
69 fclose(files_todo[0].file);
70 files_todo[0].file = 0;
71 files_todo[0].line_no = -1;
72 }
73 }
74 }
75}
76//----------------------------------------------------------------------------
77void
78InputFile::randomize_files()
79{
80 {
81 timeval now;
82 gettimeofday(&now, 0);
83 srandom(now.tv_sec + now.tv_usec);
84 }
85
86 // check that all file are test files
87 //
88 loop(f, files_todo.size())
89 {
90 if (files_todo[f].test == false) // not a test file
91 {
92 CERR << "Cannot randomise testfiles when a mix of -T"
93 " and -f is used (--TR ignored)" << endl;
94 return;
95 }
96 }
97
98 for (size_t done = 0; done < 4*files_todo.size();)
99 {
100 const int n1 = random() % files_todo.size();
101 const int n2 = random() % files_todo.size();
102 if (n1 == n2) continue;
103
104 InputFile f1 = files_todo[n1];
105 InputFile f2 = files_todo[n2];
106
107 const char * ff1 = strrchr(f1.filename.c_str(), '/');
108 if (!ff1++) ff1 = f1.filename.c_str();
109 const char * ff2 = strrchr(f2.filename.c_str(), '/');
110 if (!ff2++) ff2 = f2.filename.c_str();
111
112 // the order of files named AAA... or ZZZ... shall not be changed
113 //
114 if (strncmp(ff1, "AAA", 3) == 0) continue;
115 if (strncmp(ff1, "ZZZ", 3) == 0) continue;
116 if (strncmp(ff2, "AAA", 3) == 0) continue;
117 if (strncmp(ff2, "ZZZ", 3) == 0) continue;
118
119 // at this point f1 and f2 shall be swapped.
120 //
121 files_todo[n1] = f2;
122 files_todo[n2] = f1;
123 ++done;
124 }
125}

134bool
135InputFile::check_filter(const UTF8_string & line)
136{
137bool ret = in_matched;
138UCS_string ucs_line(line);
139 ucs_line.remove_leading_and_trailing_whitespaces();
140
141 if (in_function)
142 {
143 if (ucs_line.size() && ucs_line[0] == UNI_NABLA) // end of a function
144 {
145 in_function = false;
146 in_variable = false;
147 in_matched = false;
148 }
149 return ret;
150 }
151
152 if (in_variable)
153 {
154 if (ucs_line.size() == 0) // end of a vriable
155 {
156 in_function = false;
157 in_variable = false;
158 in_matched = false;
159 }
160 return ret;
161 }
162
163 if (ucs_line.size() && (ucs_line[0] == UNI_NABLA)) // new function
164 {
165 in_function = true;
166 in_variable = false;
167 in_matched = false;
168 ucs_line = ucs_line.drop(1);
169 UserFunction_header uh(ucs_line, false);
170 const UCS_string fun_name = uh.get_name();
171 loop(n, object_filter.size())
172 {
173 if (fun_name == object_filter[n])
174 {
175 in_matched = true;
176 return true;
177 }
178 }
179 return false;
180 }
181
182 if (ucs_line.size() && (Avec::is_quad(ucs_line[0]) ||
183 Avec::is_first_symbol_char(ucs_line[0]))) // new variable
184 {
185 in_function = false;
186 in_variable = true;
187 in_matched = false;
188 loop(u, ucs_line.size())
189 {
190 if (ucs_line[u] != UNI_LEFT_ARROW) continue; // not ←
191
192 ucs_line.resize(u);
193 ucs_line.remove_leading_and_trailing_whitespaces();
194 loop(n, object_filter.size())
195 {
196 if (ucs_line == object_filter[n])
197 {
198 in_matched = true;
199 return true;
200 }
201 }
202 }
203 }
204
205 return ret;
206}

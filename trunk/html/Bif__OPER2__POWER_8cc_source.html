47//----------------------------------------------------------------------------
48// the eval_form_2() function is for LO ⍣ N B and A LO ⍣ N B variants
49// (with condition function RO and worker function LO)
50Token
51Bif_OPER2_POWER::eval_form_2(Value_P A, Token & _LO, Token & _RO, Value_P B)
52{
53Function_P LO = _LO.get_function(); Assert(LO);
54Function_P RO = _RO.get_function(); Assert(RO);
55
56 if (!LO->has_result()) DOMAIN_ERROR;
57 if (!RO->has_result()) DOMAIN_ERROR;
58
59 if (LO->may_push_SI() || RO->may_push_SI()) // user-defined or macro
60 {
61 if (!A) return Macro::get_macro(Macro::MAC_Z__LO_POWER_RO_B)
62 ->eval_LRB (_LO, _RO, B);
63 else return Macro::get_macro(Macro::MAC_Z__A_LO_POWER_RO_B)
64 ->eval_ALRB(A, _LO, _RO, B);
65 }
66
67 // primitive LO and RO
68 //
69 for (;;)
70 {
71 const Token result_LO = !A ? LO->eval_B(B) : LO->eval_AB(A, B);
72 if (result_LO.get_tag() == TOK_ERROR) return result_LO;
73
74 Assert(result_LO.get_Class() == TC_VALUE);
75 Value_P LO_Z = result_LO.get_apl_val();
76
77 const Token result_RO = RO->eval_AB(LO_Z, B);
78 if (result_RO.get_tag() == TOK_ERROR) return result_RO;
79
80 Assert(result_RO.get_Class() == TC_VALUE);
81 Value_P condition = result_RO.get_apl_val();
82 if (condition->is_scalar_extensible() &&
83 condition->get_cfirst().is_near_bool() &&
84 condition->get_cfirst().get_near_int() == 1)
85 return Token(TOK_APL_VALUE1, LO_Z);
86
87 if (interrupt_is_raised())
88 {
89 clear_attention_raised(LOC);
90 clear_interrupt_raised(LOC);
91 INTERRUPT
92 }
93
94 B = LO_Z;
95 LO_Z.clear(LOC);
96 }
97}
98//----------------------------------------------------------------------------
99// the eval_form_1() function is for LO ⍣ N B and A LO ⍣ N B variants
100// (with numeric RO and worker function LO)
101Token
102Bif_OPER2_POWER::eval_form_1(Value_P A, Token & _LO, Value_P N, Value_P B)
103{
104Function_P LO = _LO.get_function();
105 Assert(LO);
106
107 Assert(+N);
108 if (N->element_count() != 1)
109 {
110 if (N->get_rank() > 1) RANK_ERROR;
111 else LENGTH_ERROR;
112 }
113
114 if (!N->get_cfirst().is_near_int()) DOMAIN_ERROR;
115ShapeItem repeat_cnt = N->get_cfirst().get_checked_near_int();
116
117 // special cases: 0, negative, and 1
118 //
119 if (repeat_cnt == 0)
120 {
121 Value_P Z(B);
122 return Token(TOK_APL_VALUE1, Z);
123 }
124
125 if (repeat_cnt < 0) // inverse
126 {
127 Function_P inverse = LO->get_dyadic_inverse();
128 if (inverse == 0) DOMAIN_ERROR; // no inverse for LO
129
130 LO = inverse;
131 repeat_cnt = - repeat_cnt;
132 }
133
134 if (repeat_cnt == 1)
135 {
136 if (!A) return LO->eval_B(B);
137 else return LO->eval_AB(A, B);
138 }
139
140 // at this point, repeat_cnt > 1
141 //
142 if (LO->may_push_SI()) // user-defined or macro
143 {
144 Token N(TOK_APL_VALUE1, IntScalar(repeat_cnt, LOC));
145 if (LO->has_result())
146 if (!A) return Macro::get_macro(Macro::MAC_Z__LO_POWER_N_B)
147 ->eval_LRB ( _LO, N, B);
148 else return Macro::get_macro(Macro::MAC_Z__A_LO_POWER_N_B)
149 ->eval_ALRB(A, _LO, N, B);
150 else
151 if (!A) return Macro::get_macro(Macro::MAC_LO_POWER_N_B)
152 ->eval_LRB ( _LO, N, B);
153 else return Macro::get_macro(Macro::MAC_A_LO_POWER_N_B)
154 ->eval_ALRB(A, _LO, N, B);
155 }
156
157 for (;;)
158 {
159 Token result = !A ? LO->eval_B(B)
160 : LO->eval_AB(A, B);
161
162 if (result.get_tag() == TOK_ERROR) return result;
163 Assert(result.get_Class() == TC_VALUE);
164 if (--repeat_cnt == 0) return result;
165 B = result.get_apl_val();
166 }
167}

90class Statistics_record
91{
92public:
94 Statistics_record() { reset(); }
95
97 Statistics_record(uint64_t cnt, uint64_t da, double da2)
98 : count(cnt),
99 data(da),
100 data2(da2)
101 {}
102
104 void reset()
105 {
106 count = 0;
107 data = 0;
108 data2 = 0;
109 }
110
112 void add_sample(uint64_t val)
113 {
114 ++count;
115 data += val;
116 data2 += val*val;
117 }
118
120 void print(ostream & out);
121
123 void save_record(ostream & outf);
124
126 uint64_t get_count() const
127 { return count; }
128
130 uint64_t get_sum() const
131 { return data; }
132
134 uint64_t get_average() const
135 { return average(data, count); }
136
138 double get_sum2() const
139 { return data2; }
140
142 static uint64_t average(uint64_t sum, uint64_t count)
143 { return count ? sum/count : 0; }
144
146 static void print5(ostream & out, uint64_t num);
147
148protected:
150 uint64_t count;
151
153 uint64_t data;
154
156 double data2; // can grow quickly!
157};

246class CellFunctionStatistics : public Statistics
247{
248public:
250 CellFunctionStatistics(Pfstat_ID _id)
251 : Statistics(_id)
252 { reset(); }
253
255 virtual void reset()
256 {
257 first.reset();
258 subsequent.reset();
259 }
260
262 void add_sample(uint64_t val, bool subseq)
263 {
264 if (subseq) subsequent.add_sample(val);
265 else first.add_sample(val);
266 }
267
269 virtual void print(ostream & out);
270
272 virtual void save_data(ostream & outf, const char * perf_name);
273
275 virtual const Statistics_record * get_first_record() const
276 { return &first; }
277
279 virtual const Statistics_record * get_record() const
280 { return &subsequent; }
281
283 uint64_t get_sum() const
284 { return first.get_sum() + subsequent.get_sum(); }
285
287 double get_sum2() const
288 { return first.get_sum2() + subsequent.get_sum2(); }
289
291 uint64_t get_count1() const
292 { return first.get_count(); }
293
295 uint64_t get_countN() const
296 { return subsequent.get_count(); }
297
299 uint64_t get_count() const
300 { return first.get_count() + subsequent.get_count(); }
301
302protected:
304 Statistics_record first;
305
307 Statistics_record subsequent;

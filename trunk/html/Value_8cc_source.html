66//----------------------------------------------------------------------------
67void
68Value::init_ravel()
69{
70 owner_count = 0;
71 fetcher = &cell_fetcher;
72 pointer_cell_count = 0;
73 nz_subcell_count = 0;
74 check_ptr = 0;
75 IntCell::z0(short_value);
76 ravel = short_value;
77
78 ++value_count;
79 if (Quad_SYL::value_count_limit &&
80 Quad_SYL::value_count_limit < APL_Integer(value_count))
81 {
82 MORE_ERROR() <<
83"the system limit on the APL value count (as set in ⎕SYL["
84<< (Quad_SYL::SYL_VALUE_COUNT_LIMIT + Workspace::get_IO())
85<< ";2]) was reached\n"
86"(and to avoid lock-up, this system limit in ⎕SYL was automatically cleared).";
87
88 // reset the limit so that we don't get stuck here.
89 //
90 Quad_SYL::value_count_limit = 0;
91 set_attention_raised(LOC);
92 set_interrupt_raised(LOC);
93 }
94
95const ShapeItem length = shape.get_volume();
96
97 // small values always succeed...
98 //
99 if (length <= SHORT_VALUE_LENGTH_WANTED)
100 {
101 check_ptr = charP(this) + 7;
102 return;
103 }
104
105 // large Value. If the compiler uses 4-byte pointers, then do not allow APL
106 // values that are too large. The reason is that new[] may return
107 // a non-0 pointer (thus pretending everything is OK), but a subsequent
108 // attempt to initialize the value might then throw a segfault.
109 //
110 enum { MAX_LEN = 2000000000 / sizeof(Cell) };
111 if (length > MAX_LEN && sizeof(void *) < 6)
112 throw_apl_error(E_WS_FULL, alloc_loc);
113
114 if (Quad_SYL::ravel_count_limit &&
115 Quad_SYL::ravel_count_limit < APL_Integer(total_ravel_count + length))
116 {
117CERR << "*** Quad_SYL::ravel_count_limit hit ***" << endl;
118 // make sure that the value is properly initialized
119 //
120 new (&shape) Shape();
121 ravel = short_value;
122 IntCell::zI(ravel, 42);
123
124 MORE_ERROR() <<
125"the system limit on the total ravel size (as set in ⎕SYL["
126<< (Quad_SYL::SYL_RAVEL_BYTES_LIMIT + Workspace::get_IO())
127<< ";2]) was reached\n"
128"(and to avoid lock-up, this system limit in ⎕SYL was automatically cleared).";
129 // reset the limit so that we don't get stuck here.
130 //
131 Quad_SYL::ravel_count_limit = 0;
132 set_attention_raised(LOC);
133 set_interrupt_raised(LOC);
134 }
135
136 alloc_size = length * sizeof(Cell);
137 ravel = reinterpret_cast<Cell *>(new char[alloc_size]);
138
139/*
140 ravel = 0; // assume new() fails
141// try
142// {
143 ravel = reinterpret_cast<Cell *>(new char[length * sizeof(Cell)]);
144 if (ravel == 0) // new failed (without trowing an exception)
145 {
146 Log(LOG_Value_alloc)
147 CERR << "new char[" << (length * sizeof(Cell))
148 << "] (aka. long ravel allocation) returned 0 at " LOC
149 << endl;
150
151 MORE_ERROR() << "The instatiation of a Value object succeeded, "
152 "but allocation of its (large) ravel failed.";
153 new (&shape) Shape();
154 ravel = short_value;
155 throw_apl_error(E_WS_FULL, alloc_loc);
156 }
157 }
158 catch (...)
159 {
160 // for some unknown reason, this object gets cleared after
161 // throwing the E_WS_FULL below (which destroys the prev and
162 // next pointers. We therefore unlink() the object here (where
163 // prev and next pointers are still intact).
164 //
165 unlink();
166
167 Log(LOG_Value_alloc)
168 {
169 CERR << "new char[" << (length * sizeof(Cell))
170 << "] (aka. long ravel allocation) threw an exception at " LOC
171 << endl;
172 }
173
174 MORE_ERROR() << "The instatiation of a Value object succeeded, "
175 "but allocation of its (large) ravel failed.";
176 new (&shape) Shape();
177 ravel = short_value;
178 throw_apl_error(E_WS_FULL, alloc_loc);
179 }
180*/
181
182 // init the first ravel element to (prototype) 0 so that we can avoid
183 // many empty checks all over the place
184 //
185 IntCell::z0(&get_wproto());
186 check_ptr = charP(this) + 7;
187 total_ravel_count += length;
188}
189//----------------------------------------------------------------------------
190bool
191Value::check_WS_FULL(const char * args, ShapeItem requested_cell_count,
192 const char * loc)
193{
194const int64_t used_memory
195 = (total_ravel_count + requested_cell_count) * sizeof(Cell)
196 + (value_count + 1) * sizeof(Value)
197 + Workspace::SI_entry_count() * sizeof(StateIndicator);
198
199 if (int64_t((Quad_WA::total_memory*Quad_WA::WA_scale/100)) >
200 (used_memory + Quad_WA::WA_margin)) return false; // OK
201
202 Log(LOG_Value_alloc) CERR
203 << " value_count: " << value_count << endl
204 << " total_ravel_count: " << total_ravel_count << " cells" << endl
205 << " new cell_count: " << requested_cell_count << " cells" << endl
206 << " total_memory: " << Quad_WA::total_memory << " bytes" << endl
207 << " used_memory: " << used_memory << " bytes" << endl
208 << " ⎕WA margin: " << Quad_WA::WA_margin << " bytes" << endl
209 << " ⎕WA scale: " << Quad_WA::WA_scale << "%" << endl
210
211 << " at " << LOC << endl;
212 return true;
213}

400//----------------------------------------------------------------------------
401Value::~Value()
402{
403 ADD_EVENT(this, VHE_Destruct, 0, LOC);
404 unlink();
405
406 if (flags & VF_packed)
407 {
408 uint8_t * bits = reinterpret_cast<uint8_t *>(ravel);
409 delete[] bits;
410 ravel = 0;
411 Assert(check_ptr == charP(this) + 7);
412 check_ptr = 0;
413 return;
414 }
415
416const ShapeItem length = nz_element_count();
417
418#if APL_Float_is_class
419
420 // APL_Float is a class, therefore we need to release all Cells
421
422#else
423
424 // APL_Float is NOT a class, Therefore release only PointerCells
425 if (get_pointer_cell_count() > 0)
426
427#endif
428 {
429 Cell * cZ = &get_wfirst();
430 loop(c, length)
431 {
432 if (get_pointer_cell_count() == 0) break;
433 cZ++->release(LOC);
434 }
435 }
436
437 Assert1(get_pointer_cell_count() == 0);
438
439 --value_count;
440
441 if (ravel == 0) return; // new() failed
442
443 if (ravel != short_value) // long value
444 {
445 total_ravel_count -= length;
446 delete [] ravel;
447 }
448
449 Assert(check_ptr == charP(this) + 7);
450 check_ptr = 0;
451}
452//----------------------------------------------------------------------------
453Value_P
454Value::get_cellrefs(const char * loc)
455{
456 /* Create a non-nested (!) (left-) value Z from this value.
457 Z has the same shape and consists entirely of LvalCells that point
458 to the corresponding cells of this value.
459
460 NOTE that if a cell of this value is PointerCell, then get_cellrefs()
461 will simply point to that cell rather than recursing into its sub-value
462 (which may happen at a later point in time, though).
463
464 The reson for delaying the recursion into sub-values is that the
465 subsequent processing of Z may erase those sub-values and therefore
466 doing it now may render useless later.
467 */
468
469Value_P Z(get_shape(), loc);
470
471 if (const ShapeItem ec = element_count())
472 {
473 loop(e, ec) Z->next_ravel_Cell(LvalCell(&get_wravel(e), this));
474 }
475 else // prototype
476 {
477 Z->set_ravel_Cell(0, LvalCell(&get_wproto(), this));
478 }
479
480 Z->check_value(LOC);
481 return Z;
482}
483//----------------------------------------------------------------------------
484void
485Value::assign_cellrefs(Value_P new_value)
486{
487const ShapeItem dest_count = element_count();
488 if (dest_count == 0) return; // nothing to assign
489
490const int dest_incr = (dest_count == 1) ? 0 : 1;
491
492const ShapeItem value_count = new_value->nz_element_count();
493const int src_incr = (new_value->nz_element_count() == 1) ? 0 : 1;
494
495 if (src_incr && dest_incr && (dest_count != value_count)) LENGTH_ERROR;
496
497 /* this: a value containing LvalCells and possibly PointerCells.
498
499 test with:
500
501 C←"001011010010010110010100100001000010001101" ◊ ((C='0')/C)←' ' ◊ C
502 K←'RED' 'WHITE' 'BLUE' ◊ (↑K)←'YELLOW' ◊ K
503 V←'v' 'vw' 'vwx' 'vwxyz' ◊ (2↑¨V)←5 ◊ 4 ⎕CR V
504
505 */
506
507 // consistency check: all items are LvalCells
508 //
509 check_lval_consistency();
510
511 if (is_scalar() && !new_value->is_scalar())
512 {
513 Cell * const C0 = &get_wscalar();
514 if (!C0->is_lval_cell()) LEFT_SYNTAX_ERROR;
515 const LvalCell * LVC0 = reinterpret_cast<const LvalCell *>(C0);
516 Cell * target = LVC0->get_lval_value(); // can be 0!
517 Value * owner = LVC0->get_cell_owner();
518 if (target) target->release(LOC); // free sub-values etc (if any)
519
520 new (target) PointerCell(new_value.get(), *owner);
521 return;
522 }
523
524const Cell * src = &new_value->get_cfirst();
525 loop(d, dest_count)
526 {
527 Cell & C = get_wravel(d);
528 if (C.is_pointer_cell())
529 {
530 /*
531 NOTE: At this point C is one of the left cell in a selective
532 assignment, say:
533
534 (f1 f2 ... fn VAR) ← new_value.
535
536 The initial cellrefs (i.e. of VAR) contain no PointerCells
537 (though possibly Lval cells pointing to PointerCells).
538
539 Therefore C must have been created by one of the fi and then
540 all sub-values C must also be Lval Cells.
541 */
542 Value * sub = C.get_pointer_value().get(); // right-value
543 loop(s, sub->nz_element_count())
544 {
545 Cell * Csub = &sub->get_wravel(s);
546 if (!Csub->is_lval_cell()) LEFT_SYNTAX_ERROR;
547
548 const LvalCell * LVC =
549 reinterpret_cast<const LvalCell *>(Csub);
550 if (Cell * target = LVC->get_lval_value()) // can be 0!
551 {
552 target->release(LOC); // free sub-values etc (if any)
553
554 // erase the pointee when overriding a pointer-cell.
555 //
556 Value * owner = LVC->get_cell_owner();
557 target->init(*src, *owner, LOC);
558 }
559 }
560 }
561 else if (C.is_lval_cell())
562 {
563 const LvalCell & LVC = reinterpret_cast<const LvalCell &>(C);
564 if (Cell * target = C.get_lval_value()) // target can be 0!
565 {
566 target->release(LOC); // free sub-values etc (if any)
567
568 // erase the pointee when overriding a pointer-cell.
569 //
570 Value * owner = LVC.get_cell_owner();
571 target->init(*src, *owner, LOC);
572 }
573 }
574 else LEFT_SYNTAX_ERROR;
575 src += src_incr;
576 }
577}
578//----------------------------------------------------------------------------
579void
580Value::check_lval_consistency() const
581{
582const ShapeItem ec = nz_element_count();
583
584 // left value rules:
585 //
586 // 1. every item is either an LvalCell, or a PointerCell
587 // 2. for every LvalCells:
588 // 2a. either cell is 0 and owner is 0,
589 // 2c, or cell is ≠0 and owner is ≠0 and cell lies in the ravel of owner
590 //
591 loop(e, ec)
592 {
593 const Cell * cell = &get_cravel(e); // rarely
594 if (cell->is_pointer_cell())
595 {
596 cell->get_pointer_value()->check_lval_consistency();
597 }
598 else if (cell->is_lval_cell())
599 {
600 const LvalCell * LVC = reinterpret_cast<const LvalCell *>(cell);
601 LVC->check_consistency();
602 }
603 else // error, e.g. 3{⍺+2←⍵}4
604 {
605 MORE_ERROR() << "mal-formed selective specification";
606 SYNTAX_ERROR;
607 }
608 }

610//----------------------------------------------------------------------------
611Cell *
612Value::get_member(const vector<const UCS_string *> & members, Value * & owner,
613 bool create_if_needed, bool throw_error)
614{
615 owner = this;
616
617 // members[members.size() - 1] == this
618 //
619 for (int m = members.size() - 2; m >= 0; --m)
620 {
621 if (!owner->is_member())
622 {
623 UCS_string & more = MORE_ERROR()
624 << "member access: non-structured variable "
625 << *members.back() << " has no member ";
626 more.append_members(members, 0);
627 if (throw_error) DOMAIN_ERROR;
628 else return 0;
629 }
630
631 const UCS_string & member_ucs = *members[m];
632 if (owner->get_rank() != 2)
633 {
634 if (!throw_error) return 0;
635 UCS_string & more = MORE_ERROR() << "member access: the rank of ";
636 more.append_members(members, m);
637 more << " is not 2.\n"
638 "Expecting an N×2 matrix of member,value pairs.";
639 RANK_ERROR;
640 }
641
642 if (owner->get_cols() != 2)
643 {
644 if (!throw_error) return 0;
645 UCS_string & more = MORE_ERROR()
646 << "member access: the number of columns of ";
647 more.append_members(members, m);
648 more << " is not 2.\n"
649 "Expecting an N×2 matrix of member,value pairs.";
650 LENGTH_ERROR;
651 }
652
653 Cell * member_cell = owner->get_member_data(member_ucs);
654 if (member_cell) // existing member
655 {
656 if (m == 0) return member_cell; // final member
657
658 // more members coming. Then member_cell should point to a
659 // structured sub-member
660 //
661 if (!member_cell->is_pointer_cell() ||
662 !member_cell->get_pointer_value()->is_member())
663 {
664 if (!throw_error) return 0;
665 UCS_string & more = MORE_ERROR()
666 << "member access: member " << member_ucs
667 << " exists in ";
668 more.append_members(members, m);
669 more << "but its (internal) value is not nested";
670 DOMAIN_ERROR;
671 }
672
673 // next member
674 //
675 owner = member_cell->get_pointer_value().get();
676 }
677 else // member does not exist
678 {
679 if (!create_if_needed) // give up
680 {
681 if (!throw_error) return 0; // silent
682
683 UCS_string & more = MORE_ERROR()
684 << "member access: structure ";
685 more.append_members(members, m + 1);
686 more << " has no member '" << member_ucs << "'.";
687 VALUE_ERROR;
688 }
689
690 // create new member row...
691
692 // find an unused slot in owner. get_new_member() is guaranteed
693 // to find one (by possibly increasing the ravel of owner.
694 //
695 Cell * member_data = owner->get_new_member(member_ucs);
696 if (m == 0) return member_data;
697
698 // more members coming
699 //
700 Value_P member_sub = EmptyStruct(LOC);
701 new (member_data) PointerCell(member_sub.get(), *owner);
702 owner = member_sub.get();
703 } // if (member_cell == 0)
704 }
705
706 FIXME; // not reached
707}
708//----------------------------------------------------------------------------
709ShapeItem
710Value::get_all_members_count() const
711{
712 Assert(is_structured());
713
714ShapeItem count = 0;
715const ShapeItem rows = get_rows();
716 loop(r, rows)
717 {
718 const Cell & member_name_cell = get_cravel(2*r);
719 if (!member_name_cell.is_pointer_cell()) continue; // unused row
720 {
721 ++count;
722 const Cell & member_data_cell = get_cravel(2*r + 1);
723 if (member_data_cell.is_pointer_cell()) // nested value
724 {
725 const Value & subval = *member_data_cell.get_pointer_value();
726 if (!subval.is_structured()) continue;
727 count += subval.get_all_members_count();
728 }
729 }
730 }
731
732 return count;
733}
734//----------------------------------------------------------------------------
735void
736Value::used_members(std::vector<ShapeItem> & result, bool sorted) const
737{
738 Assert(is_structured());
739
740 // 1. colllect the used columns...
741 //
742const ShapeItem rows = get_rows();
743 result.reserve(rows);
744 loop(r, rows)
745 {
746 const Cell & member_name_cell = get_cravel(2*r);
747 if (!member_name_cell.is_pointer_cell()) continue; // unused
748
749 result.push_back(r);
750 }
751
752 if (!sorted) return;
753
754 // make result[m] ≤ result[m+k] for all k > 0 and all m ≥ 0
755 //
756 loop(m, result.size()) // for all m
757 {
758 ShapeItem small = m; // assume that m is already the smallest
759 for (size_t j = m + 1; j < result.size(); ++j) // for all j = m + k
760 {
761 if (get_cravel(2*result[small]).greater(get_cravel(2*result[j])))
762 small = j;
763 }
764
765 if (small != m) // there was an even smaller item above m
766 {
767 const ShapeItem tmp = result[m];
768 result[m] = result[small];
769 result[small] = tmp;
770 }
771 }
772
773}
774//----------------------------------------------------------------------------
775void
776Value::sorted_members(std::vector<ShapeItem> & result,
777 const Unicode * filters) const
778{
779 // we take advantage of the fact that the positions in the member
780 // prefixes are limited, so we can sort them in linear time.
781 //
782 // CAUTION: This function works only for members with a _NNN prefix
783 // indication the sorting order (as produced by ⎕XML) @@@
784 //
785 Assert(is_structured());
786
787const Unicode filters_all[] = { UNI_DELTA_UNDERBAR, UNI_DELTA,
788 UNI_UNDERSCORE, Unicode_0 };
789
790 if (filters == 0) filters = filters_all;
791
792const ShapeItem rows = get_rows();
793const ShapeItem max_member_count = get_all_members_count() + 1; // ⎕IO←0 or 1
794const ShapeItem alloc = 2*max_member_count; // one for ⍙, one for ∆ and _
795ShapeItem * sorted = new ShapeItem[alloc];
796 if (sorted == 0) WS_FULL;
797
798ShapeItem * sorted_attributes = sorted; // ⍙
799ShapeItem * sorted_nodes = sorted + max_member_count; // ∆ and _
800
801 loop(a, alloc) sorted[a] = -1;
802
803 loop(r, rows)
804 {
805 const Cell & member_name_cell = get_cravel(2*r);
806 if (!member_name_cell.is_pointer_cell()) continue; // unused
807
808 const Value & val = *member_name_cell.get_pointer_value();
809 ShapeItem member_pos; // the position in the XML file
810 Unicode category;
811 UCS_string name;
812 Quad_XML::split_name(&category, &member_pos, &name, val);
813
814 Assert(member_pos <= max_member_count);
815 loop(f, 4) // at most filters ⍙, ∆, and _
816 {
817 const Unicode filter = filters[f];
818 if (filter == Unicode_0) break;
819 if (filter == category)
820 {
821 if (category == UNI_DELTA_UNDERBAR)
822 sorted_attributes[member_pos] = r;
823 else
824 sorted_nodes[member_pos] = r;
825 break;
826 }
827 }
828 }
829
830 loop(a, alloc)
831 {
832 if (sorted[a] != -1) result.push_back(sorted[a]);
833 }
834
835 delete [] sorted;
836}

944//----------------------------------------------------------------------------
945Cell *
946Value::get_new_member(const UCS_string & new_member_name)
947{
948 // find an unused slot in owner. Return its data cell (= its name cell + 1)
949 //
950 for (;;) // until an unused row was found
951 {
952 const ShapeItem rows = get_rows(); // changed by double_ravel() !
953 ShapeItem row = new_member_name.FNV_hash() % rows;
954 loop(r, 14)
955 {
956 if (++row >= rows) row = 0;
957 Cell * cell = &get_wravel(2*row);
958 if (cell->is_integer_cell() &&
959 cell->get_int_value() == 0) // unused row
960 {
961 Value_P name_val(new_member_name, LOC);
962 new (cell) PointerCell(name_val.get(), *this);
963 return cell + 1;
964 }
965 }
966
967 /* at this point a cluster of 14 consecutive used rows was hit.
968 Common wisdom (Knuth) has it, that a hash table should maintain
969 fill level (i.e. used rows ÷ all rows) below 50%.
970
971 For a hash table with N rows and an unknown fill level (like this
972 one) we could either:
973
974 A1. first determine the fill level (an O(N) operation), and
975 A2. double the table size (another O(n) operation) if the fill
976 level is > 50 %,
977
978 or:
979
980 B. double the table size regardless of the fill level (and
981 an 1:16384 chance that the doubling was premature).
982
983
984 We choose B. because it is simpler and because prematurely
985 doubling the table could very well amortize itself later.
986 */
987
988 double_ravel(LOC);
989 }
990}
991//----------------------------------------------------------------------------
992void
993Value::double_ravel(const char * loc)
994{
995Cell * const old_ravel = ravel;
996
997 Assert(is_member());
998const char * del = 0;
999 if (ravel != short_value) del = reinterpret_cast<char *>(ravel);
1000
1001 Assert(get_rank() == 2);
1002 Assert(get_cols() == 2);
1003
1004const ShapeItem old_rows = get_rows();
1005const ShapeItem new_rows = 2*old_rows;
1006const ShapeItem new_cells = 2*new_rows;
1007
1008Cell * doubled = new Cell[new_cells];
1009 loop(n, new_cells) IntCell::z0(doubled + n);
1010 valid_ravel_items = new_cells;
1011 shape.set_shape_item(0, new_rows);
1012 ravel = doubled;
1013
1014 loop(r, old_rows)
1015 {
1016 // transfer the member name. The old member_name_cell must be released
1017 // since get_new_member() below creates a new one in the new ravel.
1018 //
1019 Cell & member_name_cell = old_ravel[2*r];
1020 if (!member_name_cell.is_pointer_cell()) continue; // unused
1021 Assert(member_name_cell.is_pointer_cell());
1022 UCS_string member_name(*member_name_cell.get_pointer_value());
1023 member_name_cell.release(LOC);
1024
1025 // transfer the member value. memcpy() should work because ownership
1026 // remains the same
1027 //
1028 void * dest = get_new_member(member_name);
1029 void * old_member_data_cell = old_ravel + 2*r + 1;
1030 memcpy(dest, old_member_data_cell, sizeof(Cell));
1031 }
1032
1033 delete [] del; // del is char *, so no cell destructor is called
1034}

1037Value::get_lval_cellowner() const
1038{
1039 /* this Value may or may not be a left value. If it is, then its
1040 top-level (!) ravel is a mix of:
1041
1042 1. PointerCells pointing to (not yet get_cellrefs()ed) right Values,
1043 2a. valid LvalCells (from get_cellrefs()), or
1044 2b. invalid LvalCells(0, 0) (e.g. from ↑ overtake).

1048 loop(e, nz_element_count())

1643//----------------------------------------------------------------------------
1644Value_P
1645Value::index(const IndexExpr & IX) const
1646{
1647 // if IX.value_count() were 1 (as in A[X] as opposed to A[X1;...]) then
1648 // it should have been parsed as an axis and Value::index(Value_P X) should
1649 // have been called instead of this one. This is an internal error.
1650
1651 if (is_member())
1652 {
1653 // for a structured variable VAR, we only allow VAR[1;] to obtain
1654 // the valid member names...
1655 //
1656 if (IX.get_rank() != 2) RANK_ERROR;
1657 if (+IX.values[1]) // not elided
1658 {
1659 MORE_ERROR() << "member access: second index is not elided";
1660 LENGTH_ERROR; // not elided
1661 }
1662
1663 if (IX.values[0]->element_count() != 1)
1664 {
1665 MORE_ERROR() << "member access: first index too long";
1666 LENGTH_ERROR; // not 1 columns
1667 }
1668
1669 const APL_Integer col = IX.values[0]->get_cfirst().get_int_value();
1670 if (col != Workspace::get_IO()) INDEX_ERROR;
1671
1672 // count the number of member names.
1673 //
1674 ShapeItem member_count = 0;
1675 const ShapeItem rows = get_rows();
1676 loop(row, rows)
1677 {
1678 if (get_cravel(2*row).is_pointer_cell()) ++member_count;
1679 }
1680
1681 Value_P Z(member_count, LOC);
1682 loop(row, rows)
1683 {
1684 if (get_cravel(2*row).is_pointer_cell())
1685 Z->next_ravel_Cell(get_cravel(2*row));
1686 }
1687
1688 if (member_count == 0) // no valid members (last member )ERASEd)
1689 new (&Z->get_wfirst()) PointerCell(Idx0(LOC).get(), *Z);
1690
1691 Z->check_value(LOC);
1692 return Z;
1693 }
1694
1695 Assert(!IX.is_axis()); // should have called index(Value_P X)
1696
1697 if (get_rank() != IX.get_rank()) RANK_ERROR; // ISO p. 158
1698
1699 // Notes:
1700 //
1701 // 1. IX is parsed from right to left: B[I2;I1;I0] --> I0 I1 I2
1702 // the shapes of this and IX are then related as follows:
1703 //
1704 // this IX
1705 // ---------------
1706 // 0 rank-1 (rank = IX->value_count())
1707 // 1 rank-2
1708 // ... ...
1709 // rank-2 1
1710 // rank-1 0
1711 // ---------------
1712 //
1713 // 2. shape Z is the concatenation of all shapes in IX
1714 // 3. rank Z is the sum of all ranks in IX
1715
1716 // construct result rank_Z and shape_Z.
1717 // We go from higher indices of IX to lower indices (Note 1.)
1718 //
1719Shape shape_Z;
1720 loop(this_r, get_rank())
1721 {
1722 const ShapeItem idx_r = get_rank() - this_r - 1;
1723
1724 Value_P I = IX.values[idx_r];
1725 if (!I)
1726 {
1727 shape_Z.add_shape_item(this->get_shape_item(this_r));
1728 }
1729 else
1730 {
1731 loop(s, I->get_rank())
1732 shape_Z.add_shape_item(I->get_shape_item(s));
1733 }
1734 }
1735
1736 // check that all indices are valid
1737 //
1738 IX.check_index_range(get_shape());
1739
1740MultiIndexIterator mult(get_shape(), IX);
1741
1742Value_P Z(shape_Z, LOC);
1743const ShapeItem ec_z = Z->element_count();
1744
1745 if (ec_z == 0) // empty result
1746 {
1747 Z->set_default(*this, LOC);
1748 Z->check_value(LOC);
1749 return Z;
1750 }
1751
1752 // construct iterators.
1753 // We go from lower indices to higher indices in IX, which
1754 // means from higher indices to lower indices in this and Z
1755 //
1756 loop(z, ec_z)
1757 Z->next_ravel_Cell(get_cravel(mult++));
1758
1759 Assert(!mult.more());
1760 Z->check_value(LOC);
1761 return Z;
1762}

1766{

1778 {

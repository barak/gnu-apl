143#if HAVE_RDTSC
144inline uint64_t cycle_counter()
145{
146uint32_t lo, hi;
147 __asm__ __volatile__ ("rdtsc" : "=a" (lo), "=d" (hi));
148 return uint64_t(hi) << 32 | lo;
149}
150
151#elif HAVE_GETTIMEOFDAY
152
153inline uint64_t cycle_counter()
154{
155timeval tv;
156 gettimeofday(&tv, 0);
157 return tv.tv_sec * 1000000ULL + tv.tv_usec;
158}
159#else // neither HAVE_RDTSC nor HAVE_GETTIMEOFDAY
160#define cycle_counter() 0
161#endif // HAVE_RDTSC

191class Probe
192{
193public:
195 enum { PROBE_COUNT = 100,
196 PROBE_LEN = 20
197 };
198
200 Probe()
201 { init(); }
202
204 void init()
205 {
206 idx = 0;
207 start_p = &dummy;
208 stop_p = &dummy;
209 }
210
212 static int init(int p)
213 { if (p >= PROBE_COUNT) return -3;
214 probes[p].init();
215 return 0;
216 }
217
219 static void init_all()
220 { loop(p, PROBE_COUNT) init(p); }
221
223 void start()
224 {
225 if (idx < PROBE_LEN)
226 {
227 measurement & m = measurements[idx];
228 start_p = &m.cycles_from;
229 stop_p = &m.cycles_to;
230 }
231 else
232 {
233 start_p = &dummy;
234 stop_p = &dummy;
235 }
236
237 // write to *start_p and *stop_p so that they are loaded into the cache
238 //
239 *stop_p = cycle_counter();
240 *start_p = cycle_counter();
241
242 // now the real start of the measurment
243 //
244 *start_p = cycle_counter();
245 }
246
248 void stop()
249 {
250 // the real end of the measurment
251 //
252 *stop_p = cycle_counter();
253 ++idx;
254 }
255
257 int64_t get_time(int m) const
258 { if (m >= idx) return -1;
259 const int64_t diff = measurements[m].cycles_to
260 - measurements[m].cycles_from;
261 if (diff < 0) return -2;
262 return diff;
263 }
264
266 static int get_time(int p, int m)
267 { if (p >= PROBE_COUNT) return -3;
268 return probes[p].get_time(m);
269 }
270
272 int64_t get_start(int m) const
273 { if (m >= idx) return -1;
274 return measurements[m].cycles_from;
275 }
276
278 static int get_start(int p, int m)
279 { if (p >= PROBE_COUNT) return -3;
280 return probes[p].get_start(m);
281 }
282
284 int64_t get_stop(int m) const
285 { if (m >= idx) return -1;
286 return measurements[m].cycles_to;
287 }
288
290 static int get_stop(int p, int m)
291 { if (p >= PROBE_COUNT) return -3;
292 return probes[p].get_stop(m);
293 }
294
296 static int get_length(int p)
297 { if (p >= PROBE_COUNT) return -3;
298 return probes[p].idx;
299 }
300

302 static Probe & P_1;
303 static Probe & P_2;

305 static Probe & P_4;
306 static Probe & P_5;

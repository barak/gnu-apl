42//----------------------------------------------------------------------------
43void
44LibPaths::init(const char * argv0, bool logit)
45{
46 logit && CERR << "\ninitializing paths from argv[0] = " << argv0 << endl;
47
48 compute_bin_path(argv0, logit);
49 search_APL_lib_root();
50
51 loop(d, LIB_MAX)
52 {
53 if (root_from_env) lib_dirs[d].cfg_src = LibDir::CSRC_ENV;
54 else if (root_from_pwd) lib_dirs[d].cfg_src = LibDir::CSRC_PWD;
55 else lib_dirs[d].cfg_src = LibDir::CSRC_NONE;
56 }
57}
58//----------------------------------------------------------------------------
59void
60LibPaths::compute_bin_path(const char * argv0, bool logit)
61{
62 // compute APL_bin_path from argv0
63 //
64 if (strchr(argv0, '/') == 0)
65 {
66 // if argv0 contains no / then realpath() seems to prepend the current
67 // directory to argv0 (which is wrong since argv0 may be in $PATH).
68 //
69 // we fix this by searching argv0 in $PATH
70 //
71 const char * path = getenv("PATH"); // must NOT be modified
72
73 if (path)
74 {
75 logit && CERR << "initializing paths from $PATH = "
76 << path << endl;
77
78 // we must not modify path, so we copy it to path1 and
79 // replace the semicolons in path1 by 0. That converts
80 // p1;p2; ... into a sequence of 0-terminated strings
81 // p1 p2 ... The variable next points to the start of each
82 // string.
83 //
84 const size_t plen = strlen(path);
85 std::string path1;
86 path1.reserve(plen + 1);
87 loop(p, (plen + 1)) path1 += path[p];
88 char * next = &path1[0];
89 for (;;)
90 {
91 char * semi = strchr(next, ':');
92 if (semi) *semi = 0;
93 UTF8_string filename;
94 for (const char * n = next; *n; ++n) filename += *n;
95 filename += '/';
96 for (const char * a = argv0; *a; ++a) filename += *a;
97
98 if (access(filename.c_str(), X_OK) == 0)
99 {
100 strncpy(APL_bin_path, filename.c_str(),
101 sizeof(APL_bin_path) - 1);
102 APL_bin_path[sizeof(APL_bin_path) - 1] = 0;
103
104 char * slash = strrchr(APL_bin_path, '/');
105 Assert(slash); // due to %s/%s above
106 *slash = 0;
107 APL_bin_name = slash + 1;
108 goto done;
109 }
110
111 if (semi == 0) break;
112 next = semi + 1;
113 }
114 }
115 else
116 {
117 logit && CERR << "initializing paths from $PATH failed because "
118 "it was not set" << endl;
119 }
120 }
121
122 unused = realpath(argv0, APL_bin_path);
123 APL_bin_path[APL_PATH_MAX] = 0;
124 {
125 char * slash = strrchr(APL_bin_path, '/');
126 if (slash) { *slash = 0; APL_bin_name = slash + 1; }
127 else { APL_bin_name = APL_bin_path; }
128 }
129
130 // if we have a PWD and it is a prefix of APL_bin_path then replace PWD
131 // by './'
132 //
133 if (const char * PWD = getenv("PWD")) // we have a pwd
134 {
135 logit && CERR << "initializing paths from $PWD = " << PWD << endl;
136 const int PWD_len = strlen(PWD);
137 if (!strncmp(PWD, APL_bin_path, PWD_len) && PWD_len > 1)
138 {
139 strcpy(APL_bin_path + 1, APL_bin_path + PWD_len);
140 APL_bin_path[0] = '.';
141 }
142 }
143 else
144 {
145 logit && CERR << "initializing paths from $PWD failed because "
146 "it was not set" << endl;
147 }
148
149done:
150 logit && CERR << "APL_bin_path is: " << APL_bin_path << endl
151 << "APL_bin_name is: " << APL_bin_name << endl;
152}
153//----------------------------------------------------------------------------
154bool
155LibPaths::is_lib_root(const char * dir)
156{
157char filename[APL_PATH_MAX + 1];
158
159 snprintf(filename, sizeof(filename), "%s/workspaces", dir);
160 if (access(filename, F_OK)) return false;
161
162 snprintf(filename, sizeof(filename), "%s/wslib1", dir);
163 if (access(filename, F_OK)) return false;
164
165 return true;
166}
167//----------------------------------------------------------------------------
168void
169LibPaths::search_APL_lib_root()
170{
171 APL_lib_root[0] = 0;
172
173const char * path = getenv("APL_LIB_ROOT");
174 if (path)
175 {
176 unused = realpath(path, APL_lib_root);
177 root_from_env = true;
178 return;
179 }
180
181 root_from_pwd = true;
182
183 // search from "." to "/" for a valid lib-root
184 //
185 unused = realpath(".", APL_lib_root);
186 while (strlen(APL_lib_root))
187 {
188 if (is_lib_root(APL_lib_root)) return; // lib-root found
189 if (char * s = strrchr(APL_lib_root, '/')) *s = 0;
190 else if (char * s = strrchr(APL_lib_root, '\\')) *s = 0;
191 else
192 {
193 CERR << "*** Cannot locate APL_lib_root: no / or \\ in "
194 << APL_lib_root << endl;
195 break;
196 }
197 }
198
199 unused = realpath(".", APL_lib_root);
200}

214//----------------------------------------------------------------------------
215UTF8_string
216LibPaths::get_lib_dir(LibRef libref)
217{
218 switch(lib_dirs[libref].cfg_src)
219 {
220 case LibDir::CSRC_NONE: return UTF8_string();
221
222 case LibDir::CSRC_ENV:
223 case LibDir::CSRC_PWD: break; // continue below
224
225 case LibDir::CSRC_PREF_SYS:
226 case LibDir::CSRC_PREF_HOME:
227 case LibDir::CSRC_CMD: return lib_dirs[libref].dir_path;
228 }
229
230UTF8_string ret(APL_lib_root);
231 if (libref == LIB0) // workspaces
232 {
233 const UTF8_string subdir("/workspaces");
234 ret.append_UTF8(subdir);
235 }
236 else // wslibN
237 {
238 const UTF8_string subdir("/wslib");
239 ret.append_UTF8(subdir);
240 ret += libref + '0';
241 }
242
243 return ret;
244}

246void
247LibPaths::maybe_warn_ambiguous(int name_has_extension, const UTF8_string name,
248 const char * ext1, const char * ext2)
249{
250 if (name_has_extension) return; // extension was provided
251 if (ext2 == 0) return; // no second extension
252
253UTF8_string filename_ext2 = name;
254 filename_ext2.append_ASCII(ext2);
255 if (access(filename_ext2.c_str(), F_OK)) return; // not existing
256
257 CERR << endl
258 << "WARNING: filename " << name << endl
259 << " is ambiguous because another file" << endl << " "
260 << filename_ext2 << endl
261 << " exists as well. Using the first (.xml) file." << endl << endl;
262}
263//----------------------------------------------------------------------------
264UTF8_string
265LibPaths::get_lib_filename(LibRef lib, const UTF8_string & name,
266 bool existing, const char * ext1, const char * ext2)
267{
268 // check if name has one of the extensions ext1 or ext2 already.
269 //
270int name_has_extension = 0; // assume name has neither extension ext1 nor ext2
271 if (name.ends_with(ext1)) name_has_extension = 1;
272 else if (name.ends_with(ext2)) name_has_extension = 2;
273
274 if (name.starts_with("/") ||
275 name.starts_with("./") ||
276 name.starts_with("../"))
277 {
278 // paths from / or ./ are fallbacks for the case where the library
279 // path setup is wrong. So that the user can survive by using an
280 // explicit path
281 //
282 if (name_has_extension) return name;
283
284 UTF8_string filename(name);
285 if (access(filename.c_str(), F_OK) == 0) return filename;
286
287 if (ext1)
288 {
289 UTF8_string filename_ext1 = name;
290 filename_ext1.append_ASCII(ext1);
291 if (!access(filename_ext1.c_str(), F_OK))
292 {
293 // filename_ext1 exists, but filename_ext2 may exist as well.
294 // warn user unless an explicit extension was given.
295 //
296 maybe_warn_ambiguous(name_has_extension, name, ext1, ext2);
297 return filename_ext1;
298 }
299 }
300
301 if (ext2)
302 {
303 UTF8_string filename_ext2 = name;
304 filename_ext2.append_ASCII(ext2);
305 if (!access(filename_ext2.c_str(), F_OK)) return filename_ext2;
306 }
307
308 // neither ext1 nor ext2 worked: return original name
309 //
310 filename = name;
311 return filename;
312 }
313
314UTF8_string filename = get_lib_dir(lib);
315 filename += '/';
316 filename.append_UTF8(name);
317
318 if (name_has_extension) return filename;
319
320 if (existing)
321 {
322 // file ret is supposed to exist (and will be openend read-only).
323 // If it does return filename otherwise filename.extension.
324 //
325 if (access(filename.c_str(), F_OK) == 0) return filename;
326
327 if (ext1)
328 {
329 UTF8_string filename_ext1 = filename;
330 filename_ext1.append_ASCII(ext1);
331 if (!access(filename_ext1.c_str(), F_OK))
332 {
333 maybe_warn_ambiguous(name_has_extension,
334 filename, ext1, ext2);
335 return filename_ext1;
336 }
337 }
338
339 if (ext2)
340 {
341 UTF8_string filename_ext2 = filename;
342 filename_ext2.append_ASCII(ext2);
343 if (!access(filename_ext2.c_str(), F_OK)) return filename_ext2;
344 }
345
346 return filename; // without ext
347 }
348 else
349 {
350 // file may or may not exist (and will be created if not).
351 // therefore checking the existence does not work.
352 // check that the file ends with ext1 or ext2 if provided
353 //
354 if (name_has_extension) return filename;
355
356 if (ext1) filename.append_ASCII(ext1);
357 else if (ext2) filename.append_ASCII(ext2);
358 return filename;
359 }
360}

362bool
363LibPaths::is_present(LibRef lib)
364{
365 // try to open dir and return false if that fails
366 //
367UTF8_string path = LibPaths::get_lib_dir(lib);
368DIR * dir = opendir(path.c_str());
369 if (!dir) return false;
370
371 closedir(dir);
372 return true;
373}

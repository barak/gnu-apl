43//----------------------------------------------------------------------------
44Token
45Quad_FX::do_eval_AB(const Value * A, const Value * B)
46{
47 CHECK_SECURITY(disable_native_functions);
48
49 if (A->get_rank() > 1) RANK_ERROR;
50
51 // dyadic ⎕FX supports the following formats:
52 //
53 // 1. "libname.so" ⎕FX "APL-name" (native function)
54 //
55 // exec properties creator
56 // ------------------------
57 // 2a. N ⎕FX "APL-text" N N N N "⎕FX"
58 // 2b. N "creator" ⎕FX "APL-text" N N N N "creator"
59 // 2c. N1 N2 N3 N4 ⎕FX "APL-text" N1 N2 N3 N4 "⎕FX"
60 // 2d. N1 N2 N3 N4 "creator" ⎕FX "APL-text" N1 N2 N3 N4 "creator"
61 //
62
63 if (A->is_char_string())
64 return Token(TOK_APL_VALUE1, do_native_FX(A, -1, B));
65
66int eprops[4];
67UTF8_string creator("⎕FX");
68
69 switch(A->element_count())
70 {
71 case 2: // format 2b.
72 {
73 const Value & C = *A->get_cravel(1).get_pointer_value().get();
74 UCS_string creator_ucs(C);
75 creator = UTF8_string(creator_ucs);
76 }
77 /* no break */
78 case 1: // format 2a.
79 eprops[0] = A->get_cfirst().get_int_value();
80 if (eprops[0] < 0) DOMAIN_ERROR;
81 if (eprops[0] > 1) DOMAIN_ERROR;
82 eprops[3] = eprops[2] = eprops[1] = eprops[0];
83 break;
84
85 case 5: // format 2d.
86 {
87 const Value & C = *A->get_cravel(4).get_pointer_value().get();
88 UCS_string creator_ucs(C);
89 creator = UTF8_string(creator_ucs);
90 }
91 /* no break */
92 case 4: // format 2c.
93 loop(e, 4)
94 {
95 eprops[e] = A->get_cravel(e).get_int_value();
96 if (eprops[e] < 0) DOMAIN_ERROR;
97 if (eprops[e] > 1) DOMAIN_ERROR;
98 }
99 break;
100
101 default: LENGTH_ERROR;
102 }
103
104 return do_quad_FX(eprops, B, creator, false);
105}

119//----------------------------------------------------------------------------
120Token
121Quad_FX::do_quad_FX(const int * exec_props, const Value * B,
122 const UTF8_string & creator, bool tolerant)
123{
124 if (B->get_rank() > 2) RANK_ERROR;
125 if (B->get_rank() < 1) RANK_ERROR;
126
127UCS_string text;
128
129 // ⎕FX accepts two kinds of B argments:
130 //
131 // 1. A vector whose elements are the (nested) lines of the function, or
132 // 2. A character matrix whose rows are the lines of the function.
133 //
134 // we convert each format into text, which is a UCS string with
135 // lines separated by ASCII_LF.
136 //
137 if (B->compute_depth() >= 2) // case 1: vector of simple character vectors
138 {
139 const ShapeItem rows = B->element_count();
140 loop(row, rows)
141 {
142 const Cell & cell = B->get_cravel(row);
143 if (cell.is_character_cell())
144 {
145 const Unicode uni = cell.get_char_value();
146 if (uni > UNI_SPACE) text.append(uni);
147 text.append(UNI_LF);
148 continue;
149 }
150
151 // row has more than 1 chatacter, so it must be nested
152 if (!cell.is_pointer_cell())
153 {
154 MORE_ERROR() << "⎕FX: Function line " << row
155 << " is not a string";
156 DOMAIN_ERROR;
157 }
158
159 Value_P line = cell.get_pointer_value();
160 Assert(+line);
161
162 Log(LOG_quad_FX)
163 {
164 CERR << "[" << setw(2) << row << "] " << *line << endl;
165 }
166
167 if (line->is_char_vector())
168 {
169 const ShapeItem line_len = line->element_count();
170 bool skipping = false;
171 loop(l, line_len)
172 {
173 const Cell & c = line->get_cravel(l);
174 if (!c.is_character_cell())
175 {
176 MORE_ERROR() << "non-char in line at " LOC;
177 DOMAIN_ERROR;
178 }
179
180 const Unicode uni = c.get_char_value();
181 if (l == 0 || skipping)
182 skipping = (uni <= UNI_SPACE);
183 if (!skipping) text.append(uni);
184 }
185 }
186 else if (line->is_scalar())
187 {
188 const Cell & c1 = line->get_cfirst();
189 if (!c1.is_character_cell())
190 {
191 MORE_ERROR() << "non-char in line at " LOC;
192 DOMAIN_ERROR;
193 }
194 const Unicode uni = c1.get_char_value();
195 if (uni > UNI_SPACE) text.append(uni);
196 }
197 else
198 {
199 line->print_boxed(CERR, 0);
200 MORE_ERROR() << "bad line at " LOC;
201 DOMAIN_ERROR;
202 }
203
204 text.append(UNI_LF);
205 }
206 }
207 else // case 2: simple character matrix
208 {
209 const ShapeItem rows = B->get_rows();
210 const ShapeItem cols = B->get_cols();
211 const Cell * cB = &B->get_cfirst();
212
213 loop(row, rows)
214 {
215 bool skipping = false;
216 loop(col, cols)
217 {
218 const Unicode uni = cB++->get_char_value();
219 if (col == 0 || skipping)
220 skipping = (uni <= UNI_SPACE);
221 if (!skipping) text.append(uni);
222 }
223 text.append(UNI_LF);
224 }
225 }
226
227 return do_quad_FX(exec_props, text, creator, tolerant);
228}
229//----------------------------------------------------------------------------
230Token
231Quad_FX::do_quad_FX(const int * exec_props, const UCS_string & text,
232 const UTF8_string & creator, bool tolerant)
233{
234int error_line = 0;
235UserFunction * fun = UserFunction::fix(text, error_line, false, LOC, creator,
236 tolerant);
237 if (fun == 0)
238 {
239 Value_P Z(LOC);
240 Z->next_ravel_Int(error_line + Workspace::get_IO());
241 Z->check_value(LOC);
242 return Token(TOK_APL_VALUE1, Z);
243 }
244
245 fun->set_exec_properties(exec_props);
246
247const UCS_string fun_name = fun->get_name();
248Value_P Z(fun_name, LOC);
249
250 Z->check_value(LOC);
251 return Token(TOK_APL_VALUE1, Z);
252}
253//----------------------------------------------------------------------------
254Value_P
255Quad_FX::do_native_FX(const Value * A, sAxis axis, const Value * B)
256{
257 if (uprefs.safe_mode) DOMAIN_ERROR;
258
259const UCS_string so_name = A->get_UCS_ravel();
260const UCS_string function_name = B->get_UCS_ravel();
261
262 if (so_name.size() == 0) LENGTH_ERROR;
263 if (function_name.size() == 0) LENGTH_ERROR;
264
265NativeFunction * fun = NativeFunction::fix(so_name, function_name);
266 if (fun == 0) return IntScalar(0, LOC);
267
268 return CLONE(B, LOC);
269}

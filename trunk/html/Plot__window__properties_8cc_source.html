28 int verbosity)
29 : line_count(data->get_row_count()),
30 plot_data(*data),
31# define gdef(_ty, na, val, _descr) na(val),
32# include "Quad_PLOT.def"
33 window_width(pa_border_L + origin_X + pa_width + pa_border_R),
34 window_height(pa_border_T + origin_Y + pa_height + pa_border_B),
35 min_X(data->get_min_X()),
36 max_X(data->get_max_X()),
37 min_Y(data->get_min_Y()),
38 max_Y(data->get_max_Y()),
39 min_Z(data->get_min_Z()),
40 max_Z(data->get_max_Z()),
41 scale_X(0),
42 scale_Y(0),
43 scale_Z(0),
44 tile_X(0),
45 tile_Y(0),
46 tile_Z(0),
47 verbosity(0),
48 rangeX_type(NO_RANGE),
49 rangeY_type(NO_RANGE),
50 rangeZ_type(NO_RANGE)
51{
52 if (verbosity & SHOW_DRAW)
53 CERR << setw(20) << "min_X: " << min_X << endl
54 << setw(20) << "max_X: " << max_X << endl
55 << setw(20) << "min_Y: " << min_Y << endl
56 << setw(20) << "max_Y: " << max_Y << endl
57 << setw(20) << "min_Z: " << min_Z << endl
58 << setw(20) << "max_Z: " << max_Z << endl;
59
60 line_properties = new Plot_line_properties *[line_count + 1];
61 loop (l, line_count) line_properties[l] = new Plot_line_properties(l);
62 line_properties[line_count] = 0;
63
64 // set the origin_X and origin_Y properties back to 0 so that they
65 // affect only surface plots.
66 if (!data->surface)
67 {
68 set_origin_X(0);
69 set_origin_Y(0);
70 }
71
72 update(verbosity);
73}

85Plot_window_properties::set_window_size(Pixel_X width, Pixel_Y height)
86{
87 // pa_width and pa_height are the size of the plot area (without borders).
88 //

94
95bool
96Plot_window_properties::update(int verbosity)
97{
98 // this function is called for 2 reasons:
99 //
100 // 1. when constructor Plot_window_properties::Plot_window_properties()
101 // is called in plot_main(), i.e. when the the plot window is created·
102 // for the first time, and
103 //
104 // 2. when the user resizes the plot window.
105 //
106 // In both cases the caller has most likely changed pa_width or pa_height,
107 // and this function recomputes all variables that depend on them.
108 //
109 window_width = pa_border_L + origin_X + pa_width + pa_border_R;
110 window_height = pa_border_T + origin_Y + pa_height + pa_border_B;
111
112 min_X = plot_data.get_min_X();
113 max_X = plot_data.get_max_X();
114 min_Y = plot_data.get_min_Y();
115 max_Y = plot_data.get_max_Y();
116 min_Z = plot_data.get_min_Z();
117 max_Z = plot_data.get_max_Z();
118
119 // at this point the plot variables were computed from ionly the plot data
120 // (aka. auto-scaled). Now override them with user preferences from the plot
121 // attributes A (of A ⎕PLOT B). The XXX_valid() functions tell us whether
122 // or not an attribute was provided in A.
123 //
124 if (rangeX_min_valid()) min_X = rangeX_min;
125 if (rangeX_max_valid()) max_X = rangeX_max;
126 if (rangeY_min_valid()) min_Y = rangeY_min;
127 if (rangeY_max_valid()) max_Y = rangeY_max;
128 if (rangeZ_min_valid()) min_Z = rangeZ_min;
129 if (rangeZ_max_valid()) max_Z = rangeZ_max;
130
131 if (min_X >= max_X)
132 {
133 MORE_ERROR() << "empty X range in A ⎕PLOT B";
134 return true;
135 }
136
137 if (min_Y >= max_Y)
138 {
139 MORE_ERROR() << "empty Y range " << min_Y << ":" << max_Y
140 << " in A ⎕PLOT B";
141 return true;
142 }
143
144 if (min_Z >= max_Z)
145 {
146 MORE_ERROR() << "empty Z range in A ⎕PLOT B";
147 return true;
148 }
149
150double delta_X = max_X - min_X;
151double delta_Y = max_Y - min_Y;
152double delta_Z = max_Z - min_Z;
153const double orig_len = sqrt(origin_X*origin_X + origin_Y*origin_Y);
154
155 // first approximation for value → pixel factor
156 //
157 scale_X = pa_width / delta_X; // pixels per X-value
158 scale_Y = pa_height / delta_Y; // pixels per Y-value
159 scale_Z = orig_len ? orig_len / delta_Z : 1.0; // pixels per Z-value
160
161 if (verbosity & SHOW_DRAW)
162 CERR << setw(20) << "delta_X (1): " << delta_X << " (Xmax-Xmin)" << endl
163 << setw(20) << "delta_Y (1): " << delta_Y << " (Ymax-Ymin)" << endl
164 << setw(20) << "delta_Z (1): " << delta_Z << " (Zmax-Zmin)" << endl
165 << setw(20) << "scale_X (1): " << scale_X << " pixels/X" << endl
166 << setw(20) << "scale_Y (1): " << scale_Y << " pixels/Y" << endl
167 << setw(20) << "scale_Z (1): " << scale_Z << " pixels/Z" << endl;
168
169const double tile_X_raw = gridX_pixels / scale_X;
170const double tile_Y_raw = gridY_pixels / scale_Y;
171const double tile_Z_raw = gridZ_pixels / scale_Z;
172
173 if (verbosity & SHOW_DRAW)
174 CERR << setw(20) << "tile_X_raw: " << tile_X_raw
175 << "(∆X between grid lines)" << endl
176 << setw(20) << "tile_Y_raw: " << tile_Y_raw
177 << "(∆Y between grid lines)" << endl
178 << setw(20) << "tile_Z_raw: " << tile_Z_raw
179 << "(∆Z between grid lines)" << endl;
180
181 tile_X = round_up_125(tile_X_raw);
182 tile_Y = round_up_125(tile_Y_raw);
183 tile_Z = round_up_125(tile_Z_raw);
184
185const int min_Xi = floor(min_X / tile_X);
186const int min_Yi = floor(min_Y / tile_Y);
187const int min_Zi = floor(min_Z / tile_Z);
188const int max_Xi = ceil(max_X / tile_X);
189const int max_Yi = ceil(max_Y / tile_Y);
190const int max_Zi = ceil(max_Z / tile_Z);
191 gridX_last = 0.5 + max_Xi - min_Xi;
192 gridY_last = 0.5 + max_Yi - min_Yi;
193 gridZ_last = 0.5 + max_Zi - min_Zi;
194 min_X = tile_X * floor(min_X / tile_X);
195 min_Y = tile_Y * floor(min_Y / tile_Y);
196 min_Z = tile_Z * floor(min_Z / tile_Z);
197 max_X = tile_X * ceil(max_X / tile_X);
198 max_Y = tile_Y * ceil(max_Y / tile_Y);
199 max_Z = tile_Z * ceil(max_Z / tile_Z);
200
201 delta_X = max_X - min_X;
202 delta_Y = max_Y - min_Y;
203 delta_Z = max_Z - min_Z;
204
205 // final value → pixel factor
206 //
207 scale_X = pa_width / delta_X;
208 scale_Y = pa_height / delta_Y;
209 scale_Z = orig_len ? orig_len / delta_Z : 1.0;
210
211 if (verbosity & SHOW_DRAW)
212 CERR << setw(20) << "min_X (2): " << min_X << endl
213 << setw(20) << "max_X (2): " << max_X << endl
214 << setw(20) << "min_Y (2): " << min_Y << endl
215 << setw(20) << "max_Y (2): " << max_Y << endl
216 << setw(20) << "min_Z (2): " << min_Z << endl
217 << setw(20) << "max_Z (2): " << max_Z << endl
218 << setw(20) << "delta_X (2): " << delta_X << " (Xmax-Xmin)" << endl
219 << setw(20) << "delta_Y (2): " << delta_Y << " (Ymax-Ymin)" << endl
220 << setw(20) << "delta_Z (2): " << delta_Z << " (Zmax-Zmin)" << endl
221 << setw(20) << "scale_X (2): " << scale_X << " pixels/X" << endl
222 << setw(20) << "scale_Y (2): " << scale_Y << " pixels/Y" << endl
223 << setw(20) << "scale_Z (2): " << scale_Z << " pixels/Z" << endl
224 << setw(20) << "tile_X (2): " << tile_X
225 << "(∆X between grid lines)" << endl
226 << setw(20) << "tile_Y (2): " << tile_Y
227 << "(∆Y between grid lines)" << endl
228 << setw(20) << "tile_Z (2): " << tile_Z
229 << "(∆Z between grid lines)" << endl;
230
231 return false; // OK
232}

257//----------------------------------------------------------------------------
258const char *
259Plot_window_properties::set_attribute(const char * att_and_val)
260{
261 /*
262 called with leading whitespaces in att_and_val removed.
263
264 att_and_val is a string specifying one of:
265
266 1. a window attribute, e.g.: origin_X: 100 , or
267 2. a line attribute, e.g.: point_color-1: #000000 , or
268 3. a color gradient, e.g.: color_level-50: #00FF00
269
270 The value (right of ':') can be
271
272 an integer value, like: 100
273 a color, like: #000000 or #000
274 a string, like: "Window Title"
275
276 return 0 on success or else an appropriate error string.
277 */
278
279 // find ':' which separates the attribute name and the attribute value.
280 //
281const char * colon = strchr(att_and_val, ':');
282 if (colon == 0) return "Expecting 'attribute: value' but no : found";
283
284 // skip leading whitespace before the attribute value
285 //
286const char * value = colon + 1;
287 while (*value && (*value <= ' ')) ++value;
288
289 // check for optional line number resp. color level
290 //
291int propnum = 0;
292const char * minus = strchr(att_and_val, '-');
293int line_number = -1;
294
295 if (minus && minus < colon) // line attribute or color gradient
296 {
297 // it is not unlikely that the user wants to use the same attributes
298 // for different plots that may differ in the number of lines. We
299 // therefore silently ignore such over-specified attribute rather
300 // than returning an error string (which then raises a DOMAIN error).
301 //
302 if (!strncmp(att_and_val, "color_level-", 12))
303 {
304 const int level = strtoll(minus + 1, 0, 10);
305 if (level < 0) return "color gradient level < 0%";
306 if (level > 100) return "color gradient level > 100%";
307 const char * error = 0;
308 const Color color = Plot_data::Color_from_str(value, error);
309 if (error) return error;
310
311 level_color grad = { level, color };
312
313 // insert grad into gradient, but keeping gradient sorted
314 // by increasing level
315 //
316 loop(g, gradient.size())
317 {
318 if (gradient[g].level == grad.level)
319 return "duplicate gradient level";
320
321 if (gradient[g].level > grad.level)
322 {
323 gradient.insert(gradient.begin() + g, grad);
324 return 0;
325 }
326 }
327 gradient.push_back(grad);
328 return 0; // OK
329 }
330
331 // figure the line number in the attribute name, for example in
332 // line_color-2. No line number means all lines.
333 //
334 const int line = strtoll(minus + 1, 0, 10) - Workspace::get_IO();
335 if (line < 0) return "line number ≤ ⎕IO";
336 if (line >= line_count) return 0; // silently ignore
337 line_number = line;
338 }
339
340# define ldef(ty, na, val, _descr) \
341 ++propnum; \
342 if (Plot_line_properties::is_line_property(att_and_val, #na)) \
343 { const char * error = 0; \
344 if (line_number == -1) \
345 set_all_ ## na(Plot_data::ty ## _from_str(value, error), \
346 propnum); \
347 else \
348 { line_properties[line_number]->set_ ## na(Plot_data::ty \
349 ## _from_str(value, error)); \
350 properties_set.push_back(line_number << 8 | propnum); } \
351 return error; \
352 }
353
354# define gdef(ty, na, _val, _descr) \
355 if (!strncmp(#na, att_and_val, colon - att_and_val)) \
356 { const char * error = 0; \
357 set_ ## na(Plot_data::ty ## _from_str(value, error)); \
358 return error; \
359 }
360# include "Quad_PLOT.def"
361
362 return "Bad or unknown window attribute.";
363}

366Plot_window_properties::can_be_set(uint16_t line, uint16_t propnum)
367{
368 loop(ps, properties_set.size())
369 {
370 if (properties_set[ps] == (line << 8 | propnum)) return false;

375//----------------------------------------------------------------------------
376const char *
377Plot_window_properties::set_attribute(const UCS_string & att, const Cell & val)
378{
379 // replace trailing _NNN in att with -NNN.
380 //
381UCS_string att1 = att;
382bool trailing_NNN = false;
383 for (int a = att1.size() - 1; a >= 0; --a)
384 {
385 const Unicode cc = att1[a];
386 if (!Avec::is_digit(cc))
387 {
388 trailing_NNN = cc == UNI_UNDERSCORE;
389 if (trailing_NNN) att1[a] = UNI_MINUS;
390 break;
391 }
392 }
393
394 if (att1.size() > 40) // most are < 20
395 {
396 return "attribute too long";
397 }
398
399char att_and_value[100];
400UTF8_string attname_utf(att1);
401const char * attname_cp = attname_utf.c_str();
402 if (val.is_integer_cell()) // integer attribute value
403 {
404 const int ival = val.get_int_value();
405 snprintf(att_and_value, sizeof(att_and_value),
406 "%s: %d", attname_cp, ival);
407 return set_attribute(att_and_value);
408 }
409 else if (val.is_pointer_cell()) // string value
410 {
411 UCS_string val_ucs = val.get_pointer_value()->get_UCS_ravel();
412 UTF8_string attval_utf(val_ucs);
413 const char * attval_cp = attval_utf.c_str();
414 snprintf(att_and_value, sizeof(att_and_value),
415 "%s: %s", attname_cp, attval_cp);
416 return set_attribute(att_and_value);
417 }
418 else
419 {
420 return "Bad attribute value type";
421 }
422
423}
424//----------------------------------------------------------------------------
425double
426Plot_window_properties::round_up_125(double val)
427{
428int expo = 0;
429 while (val >= 10) { val /= 10; ++expo; }
430 while (val < 1) { val *= 10; --expo; }
431
432 // at this point 1 ≤ val < 10.
433 //
434 if (expo > -18 && expo < 18) // use integer arithmetic for 10^expo
435 {
436 const int abs_expo = expo < 0 ? -expo : expo;
437 uint64_t expo_val = 1;
438 loop(a, abs_expo) expo_val *= 10;
439
440 if (val <= 2.0) return expo < 0 ? 2.0/expo_val : 2.0*expo_val;
441 if (val <= 5.0) return expo < 0 ? 5.0/expo_val : 5.0*expo_val;
442 return expo < 0 ? 10.0/expo_val : 10.0*expo_val;
443 }
444 else // use float arithmetic for 10^expo
445 {
446 const double expo_val = pow(10, expo);
447 if (val <= 2.0) return expo < 0 ? 2.0/expo_val : 2.0*expo_val;
448 if (val <= 5.0) return expo < 0 ? 5.0/expo_val : 5.0*expo_val;
449 return expo < 0 ? 10.0/expo_val : 10.0*expo_val;
450 }
451}
452//----------------------------------------------------------------------------
453uint32_t
454Plot_window_properties::get_color(double alpha) const
455{
456 Assert(alpha >= 0.0);
457 Assert(alpha <= 1.0);
458
459 if (gradient.size() == 0) // no gradient specified: use gray-scale
460 return uint32_t(255*alpha) << 16
461 | uint32_t(255*alpha) << 8
462 | uint32_t(255*alpha);
463
464 if (gradient.size() == 1) // only one gradient specified: use it
465 return gradient[0].rgb;
466
467 loop(g, gradient.size())
468 {
469 const double level_g = 0.01*gradient[g].level;
470 if (alpha == level_g) return gradient[g].rgb;
471 if (alpha < level_g) // level_g is the first level above alpha
472 {
473 if (g == 0) return gradient[0].rgb; // use first gradient
474
475 // interpolate
476 //
477 const double l0 = 0.01*gradient[g-1].level; // previous level
478 const double beta1 = (alpha - l0) / (level_g - l0);
479 Assert(beta1 >= 0.0);
480 Assert(beta1 <= 1.0);
481 const double beta0 = 1.0 - beta1;
482 const uint32_t rgb0 = gradient[g-1].rgb;
483 const uint32_t rgb1 = gradient[g].rgb;
484 const uint32_t red = beta1*(rgb1 >> 16 & 0xFF)
485 + beta0*(rgb0 >> 16 & 0xFF);
486 const uint32_t grn = beta1*(rgb1 >> 8 & 0xFF)
487 + beta0*(rgb0 >> 8 & 0xFF);
488 const uint32_t blu = beta1*(rgb1 & 0xFF)
489 + beta0*(rgb0 & 0xFF);
490 return red << 16 | grn << 8 | blu;
491 }
492 }
493
494 // alpha is above last gradient: use last gradient
495 //
496 return gradient.back().rgb;
497}
498//----------------------------------------------------------------------------
499Pixel_XY
500Plot_window_properties::valXYZ2pixelXY(double X, double Y, double Z) const
501{
502const double phi = atan2(get_origin_Y(), get_origin_X());
503const Pixel_Z pz = valZ2pixel(Z - get_min_Z());
504const Pixel_X px = valX2pixel(X - get_min_X()) + get_origin_X() - pz*cos(phi);
505const Pixel_X py = valY2pixel(Y - get_min_Y()) + pz*sin(phi);
506
507 return Pixel_XY(px, py);
508}

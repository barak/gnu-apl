37Value_P
38Bif_F12_TAKE::first(const Value & B)
39{
40const Cell & first_B = B.get_cfirst();
41 if (B.element_count() == 0) // empty value: return prototype
42 {
43 if (first_B.is_lval_cell()) // (↑...)←V
44 {
45 Value_P Z(LOC);
46 Z->next_ravel_Cell(first_B);
47 return Z;
48 }
49
50 // normal (right-) value
51 //
52 Value_P Z = B.prototype(LOC);
53 Z->check_value(LOC);
54 return Z;
55 }
56
57 if (!first_B.is_pointer_cell()) // simple cell
58 {
59 Value_P Z(LOC);
60 Z->get_wscalar().init(first_B, *Z, LOC);
61 Z->check_value(LOC);
62 return Z;
63 }
64
65Value_P v1 = first_B.get_pointer_value();
66Value * v1_owner = v1->get_lval_cellowner();
67 if (v1_owner) // B is a left value
68 {
69 Value_P B1(LOC);
70 B1->next_ravel_Pointer(v1.get());
71 B1->check_value(LOC);
72
73 Value_P Z(LOC);
74 Z->next_ravel_Lval(&B1->get_wscalar(), v1_owner);
75
76 Z->check_value(LOC);
77 return Z;
78 }
79 else
80 {
81 const ShapeItem ec = v1->element_count();
82 Value_P Z(v1->get_shape(), LOC);
83 if (ec == 0) Z->set_default(*v1, LOC);
84
85 loop(e, ec) Z->next_ravel_Cell(v1->get_cravel(e));
86
87 Z->check_value(LOC);
88 return Z;
89 }
90}

92Token
93Bif_F12_TAKE::eval_AXB(Value_P A, Value_P X, Value_P B) const
94{
95 if (A->get_rank() > 1) RANK_ERROR;
96 if (X->get_rank() > 1) RANK_ERROR;
97
98const ShapeItem len_A = A->element_count();
99const ShapeItem len_X = X->element_count();
100 if (len_A != len_X) LENGTH_ERROR;
101
102 if (len_X == 0) // no axes
103 {
104 Token result(TOK_APL_VALUE1, CLONE_P(B, LOC));
105 return result;
106 }
107
108 // construct the left argument of Bif_F12_TAKE::fill(). Start with ⍴B and
109 // then replace corresponding shape items with A[X[x]].
110 //
111const APL_Integer qio = Workspace::get_IO();
112const AxesBitmap axes_X = X->to_bitmap("A ↑[X] B", B->get_rank());
113Shape sh_take = B->get_shape(); // start with ⍴B
114 loop(x, len_X) // for exery axis X[x] in X
115 {
116 const APL_Integer axis = X->get_cravel(x).get_near_int() - qio;
117 const APL_Integer alen = A->get_cravel(x).get_near_int();
118 sh_take.set_shape_item(axis, alen);
119 }
120
121 return Token(TOK_APL_VALUE1, do_take(sh_take, *B, axes_X));
122}

124Token
125Bif_F12_TAKE::eval_XB(Value_P X, Value_P B) const
126{
127 // ↑[X] B ←→ ((⍴X)⍴1)↑[X] B (GNU APL only)
128 if (X->get_rank() > 1) AXIS_ERROR;
129
130const AxesBitmap axes_X = X->to_bitmap("↑[X] B", B->get_rank());
131
132 // construct the left argument of Bif_F12_TAKE::fill(). Start with ⍴B and
133 // then replace corresponding shape items with 1.
134 //
135Shape sh_take = B->get_shape(); // start with ⍴B
136 loop(b, B->get_rank()) // for exery axis X[x] in X
137 {
138 if (axes_X & 1 << b) sh_take.set_shape_item(b, 1);
139 }
140
141 return Token(TOK_APL_VALUE1, do_take(sh_take, *B, axes_X));
142}

144Token
145Bif_F12_TAKE::eval_AB(Value_P A, Value_P B) const
146{
147Shape ravel_A1(*A, /* ⎕IO */ 0); // checks 1 ≤ ⍴⍴A and ⍴A ≤ MAX_RANK
148
149 if (B->is_scalar())
150 {
151 Shape shape_B1;
152 loop(a, ravel_A1.get_rank()) shape_B1.add_shape_item(1);
153 Value_P B1 = CLONE_P(B, LOC); // so that we can set_shape()
154 B1->set_shape(shape_B1);
155 return Token(TOK_APL_VALUE1, do_take(ravel_A1, *B1, false));
156 }
157 else
158 {
159 if (ravel_A1.get_rank() != B->get_rank()) LENGTH_ERROR;
160 return Token(TOK_APL_VALUE1, do_take(ravel_A1, *B, false));
161 }
162}

164Value_P
165Bif_F12_TAKE::do_take(const Shape & ravel_A1, const Value & B,
166 AxesBitmap axes)
167{
168 // ravel_A1 can have negative items (for take from the end).
169 //
170Value_P Z(ravel_A1.abs(), LOC);
171
172 if (ravel_A1.is_empty()) Z->set_default(B, LOC);
173 else fill(ravel_A1, *Z, B, axes);
174 Z->check_value(LOC);
175 return Z;
176}
177//----------------------------------------------------------------------------
178void
179Bif_F12_TAKE::fill(const Shape & shape_Zi, Value & Z,
180 const Value & B, AxesBitmap axes)
181{
182 for (TakeDropIterator i(true, shape_Zi, B.get_shape()); i.more(); ++i)
183 {
184 const ShapeItem offset = i();
185 if (offset != -1) // valid cell
186 {
187 Z.next_ravel_Cell(B.get_cravel(offset));
188 }
189 else if (axes) // overtake from axis
190 {
191 const ShapeItem offset = i.axis_proto(axes);
192 Z.next_ravel_Proto(B.get_cravel(offset));
193 }
194 else // overtake from ↑B
195 {
196 Z.next_ravel_Proto(B.get_cproto());
197 }
198 }
199}

201Token
202Bif_F12_DROP::eval_AB(Value_P A, Value_P B) const
203{
204 if (A->get_rank() > 1) RANK_ERROR;
205
206const Shape ravel_A(*A, /* ⎕IO */ 0);
207
208 if (B->is_scalar())
209 {
210 /*
211 A scalar B is taken as ((⍴⍴B)⍴1)⍴B.
212
213 Z←A↓B with scalar B has an element count() of either 0 or 1:
214
215 If all items of A are are 0 then nothing is dropped) and therefore
216 the result is the scalar B.
217
218 Otherwise at least one axis A[j] is ≠ 0 dropped and the result
219 is empty.
220
221 ⍴⍴Z is ⍴,A and ⍴Z[j] ←→ 0 ⌈ (⍴B)[j] - ∣ A[j]
222 */
223 Shape shape_Z;
224 loop(r, ravel_A.get_rank())
225 {
226 // (⍴Zi)[r] r'th either 1 (if nothing is dropped)
227 // or else 1 (possibly overdrop).
228 if (ravel_A.get_shape_item(r)) shape_Z.add_shape_item(0);
229 else shape_Z.add_shape_item(1);
230 }
231
232 Value_P Z(shape_Z, LOC);
233
234 Z->set_ravel_Cell(0, B->get_cfirst());
235 if (shape_Z.get_volume() == 0) Z->to_type(false);
236 Z->check_value(LOC);
237 return Token(TOK_APL_VALUE1, Z);
238 }
239
240 if (ravel_A.get_rank() != B->get_rank()) LENGTH_ERROR;
241
242Shape shape_Z;
243 loop(r, ravel_A.get_rank())
244 {
245 const ShapeItem sA = ravel_A.get_shape_item(r); // A[r]
246 const ShapeItem sB = B->get_shape_item(r); // (⍴B[r]
247 const ShapeItem pA = sA < 0 ? -sA : sA; // ∣ A[r]
248 if (pA >= sB) shape_Z.add_shape_item(0); // over-drop
249 else shape_Z.add_shape_item(sB - pA); // normal drop
250 }
251
252Value_P Z(shape_Z, LOC);
253 if (shape_Z.is_empty()) // empty Z, e.g. from overdrop
254 {
255 Value_P Z(shape_Z, LOC);
256 Z->set_default(*B.get(), LOC);
257 Z->check_value(LOC);
258 return Token(TOK_APL_VALUE1, Z);
259 }
260
261 for (TakeDropIterator i(false, ravel_A, B->get_shape()); i.more(); ++i)
262 {
263 const ShapeItem offset = i();
264 Z->next_ravel_Cell(B->get_cravel(offset));
265 }
266
267 Z->check_value(LOC);
268 return Token(TOK_APL_VALUE1, Z);
269}

271Token
272Bif_F12_DROP::eval_AXB(Value_P A, Value_P X, Value_P B) const
273{
274 if (X->element_count() == 0) // no axes
275 {
276 Value_P Z = CLONE_P(B, LOC);
277 Token result(TOK_APL_VALUE1, Z);
278 return result;
279 }
280
281 if (X->get_rank() > 1) INDEX_ERROR;
282
283const uint64_t len_X = X->element_count();
284 if (len_X > MAX_RANK) INDEX_ERROR;
285 if (len_X == 0) INDEX_ERROR;
286
287 if (A->get_rank() > 1) RANK_ERROR;
288
289uint64_t len_A = A->element_count();
290 if (len_A != len_X) LENGTH_ERROR;
291
292const APL_Integer qio = Workspace::get_IO();
293
294 // init ravel_A = shape_B and seen.
295 //
296Shape ravel_A(B->get_shape());
297bool seen[MAX_RANK];
298 loop(r, B->get_rank()) seen[r] = false;
299
300 loop(r, len_X)
301 {
302 const APL_Integer a = A->get_cravel(r).get_near_int();
303 const APL_Integer x = X->get_cravel(r).get_near_int() - qio;
304
305 if (x >= B->get_rank()) INDEX_ERROR;
306 if (seen[x]) INDEX_ERROR;
307 seen[x] = true;
308
309 const ShapeItem amax = B->get_shape_item(x);
310 if (a >= amax) ravel_A.set_shape_item(x, 0);
311 else if (a >= 0) ravel_A.set_shape_item(x, a - amax);
312 else if (a > -amax) ravel_A.set_shape_item(x, amax + a);
313 else ravel_A.set_shape_item(x, 0);
314 }
315
316 return Token(TOK_APL_VALUE1,
317 Bif_F12_TAKE::do_take(ravel_A, *B, 0));
318}

320ShapeItem
321TakeDropIterator::axis_proto(AxesBitmap axes) const
322{
323 /* compute the offset for the prototype of current_offset. The APL2
324 language reference somehow suggests that for A↑[X] B the axes of B
325 shall be enclosed (to give ⊂[X] B) and then the prototype of the
326 enclosed sub-array shall be used as the fill item for overtake.
327
328 This would mean that the offsets of the prototype is the sum of the
329 weighted offsets of the axes in X. Interestingly we get the output
330 shown in the language reference only if we use the sum of the weighted
331 offsets of the axes NOT in X.
332 0
333 */
334ShapeItem ret = 0;
335 loop(a, ref_B.get_rank())
336 {
337 if (axes & 1 << a) // axis a in X
338 {
339 const _ftwc & ftwc_a = ftwc[a];
340 ret += ftwc_a.current * ftwc_a.weight;
341 }
342 }
343
344 return ret;
345}

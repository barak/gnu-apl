42UserFunction::UserFunction(const UCS_string txt, const char * loc,
43 const UTF8_string & _creator, bool tolerant,
44 bool macro)
45 : Function(ID_USER_SYMBOL, TOK_FUN2),
46 Executable(txt, true, PM_FUNCTION, loc),
47 header(txt, macro),
48 creator(_creator),
49 error_line(0), // assume header is wrong
50 error_info("Unspecified")
51{
52 if (header.get_error())
53 {
54 error_info = header.get_error_info();
55 return;
56 }
57
58 set_creation_time(now());
59
60 exec_properties[0] = 0;
61 exec_properties[1] = 0;
62 exec_properties[2] = 0;
63 exec_properties[3] = 0;
64
65 line_starts.push_back(Function_PC_0); // will be set later.
66
67 if (header.get_error() != E_NO_ERROR) // bad header
68 {
69 error_info = header.get_error_info();
70 return;
71 }
72
73 // set Function::tag
74 //
75 if (header.RO()) tag = TOK_OPER2;
76 else if (header.LO()) tag = TOK_OPER1;
77 else if (header.A()) tag = TOK_FUN2;
78 else if (header.B()) tag = TOK_FUN1;
79 else tag = TOK_FUN0;
80
81 parse_body(loc, tolerant, macro);
82 if (error_line > 0)
83 {
84 error_info = "Error in function body";
85 return;
86 }
87
88 if (uprefs.discard_indentation) // really ?
89 {
90 loop(l, get_text_size())
91 {
92 UCS_string line = get_text(l);
93 line.remove_leading_and_trailing_whitespaces();
94 set_text(l, line);
95 }
96 }
97
98 error_line = -1; // no error
99 error_info = 0;
100}

102UserFunction::UserFunction(Fun_signature sig, int lambda_num,
103 const UCS_string & text, Token_string & lambda_body)
104 : Function(ID_USER_SYMBOL, TOK_FUN0),
105 Executable(sig, lambda_num, text, LOC),
106 header(sig, lambda_num),
107 creator(UNI_LAMBDA),
108 error_line(0),
109 error_info("Unspecified")
110{
111 set_creation_time(now());
112
113 exec_properties[0] = 0;
114 exec_properties[1] = 0;
115 exec_properties[2] = 0;
116 exec_properties[3] = 0;
117
118 if (header.get_error() != E_NO_ERROR) // bad header
119 {
120 error_info = header.get_error_info();
121 return;
122 }
123
124 if (header.RO()) tag = TOK_OPER2;
125 else if (header.LO()) tag = TOK_OPER1;
126 else if (header.A()) tag = TOK_FUN2;
127 else if (header.B()) tag = TOK_FUN1;
128 else tag = TOK_FUN0;
129
130 while (lambda_body.size() > 2 &&
131 lambda_body.back().get_Class() == TC_SYMBOL &&
132 lambda_body[lambda_body.size() - 2].get_tag() == TOK_SEMICOL)
133 {
134 header.add_local_var(lambda_body.back().get_sym_ptr());
135 lambda_body.pop_back(); // varname
136 lambda_body.pop_back(); // semicolon
137 }
138
139 // order of local vars is reversed. Fix that.
140 //
141 header.reverse_local_vars();
142
143 parse_body_line(Function_Line_0, lambda_body, false, false, LOC);
144 setup_lambdas();
145 line_starts.push_back(Function_PC(lambda_body.size() - 1));
146 line_starts.push_back(Function_PC_0);
147 error_line = -1; // no error
148 error_info = 0;
149}

576{
577 // Sort lines
578 //
579std::vector<bool> ts_lines;
580
581 loop(ts, line_starts.size()) ts_lines.push_back(false);

584 Function_Line line = lines[ll];

602 {
603 if (ts_lines[ts])
604 trace_lines.push_back(Function_Line(ts));
605 }
606 }
607
608 parse_body(LOC, false, false);

610//----------------------------------------------------------------------------
611ErrorCode
612UserFunction::transform_multi_line_strings()
613{
614 /* convert a set of lines like
615
616 [k+1] PREFIX "L1
617 [k+2] L2 ...
618 [k+N] Ln" SUFFIX
619
620 into:
621
622 [k+1] PREFIX "L1" "L2" ... "LN" SUFFIX
623 [...] (empty)
624 [k+N] (empty)
625 */
626
627enum Line_status
628 {
629 Function_header = 0,
630 APL_text = 1,
631 Start_of_string = 2,
632 Inside_string = 3,
633 End_of_string = 4
634 };
635
636char status[get_text_size()]; status[0] = Function_header;
637Line_status current = APL_text;
638
639 // determine line status of each line...
640 //
641 for (int l = 1; l < get_text_size(); ++l)
642 {
643 const int count = get_text(l).double_quote_count(false);
644 if (count & 1) // start or end of string
645 {
646 if (current == APL_text) // start of multi-line string
647 {
648 status[l] = Start_of_string;
649 current = Inside_string;
650 }
651 else // end of multi-line string
652 {
653 status[l] = End_of_string;
654 current = APL_text;
655 }
656 }
657 else // no status change
658 {
659 status[l] = current;
660 }
661 }
662
663 if (current == Start_of_string || current == Inside_string)
664 {
665 // multi-line string started but not ended.
666 //
667 return E_DEFN_ERROR;
668 }
669
670 // modify lines...
671 //
672 for (int l = 1; l < get_text_size();)
673 {
674 if (status[l] == APL_text) { ++l; continue; }
675 const int start = l;
676 Assert1(status[l] == Start_of_string);
677 UCS_string accu = get_text(l++);
678 accu << "\" ";
679 while (status[l] == Inside_string)
680 {
681 accu << " \"" << get_text(l).do_escape(true) << "\"";
682 clear_text(l++);
683 }
684 Assert(status[l] == End_of_string);
685 accu << "\"" << get_text(l);
686 set_text(start, accu);
687 clear_text(l++);
688 }
689
690 // remove trailing empty lines...
691 //
692 while (get_text_size() && get_text(get_text_size() - 1).size() == 0)
693 text.pop_back();
694
695// CERR << endl;
696// loop(l, get_text_size()) CERR << "[" << l << "] " << get_text(l) << endl;
697
698 return E_NO_ERROR;
699}
700//----------------------------------------------------------------------------
701ErrorCode
702UserFunction::transform_multi_line_strings_3()
703{
704 /* convert a set of lines like
705
706 [k+1] PREFIX """
707 [k+2] L2 ...
708 [k+N] """
709
710 into:
711
712 [k+1] PREFIX "L2" ... "LN"
713 [...] (empty)
714 [k+N] (empty)
715
716 In contrast to transform_multi_line_strings(), line k+N may only
717 consist of spaces and the terminating """.
718 */
719
720enum Line_status
721 {
722 Function_header = 0,
723 APL_text = 1,
724 Start_of_string = 2,
725 Inside_string = 3,
726 End_of_string = 4
727 };
728
729char status[get_text_size()]; status[0] = Function_header;
730Line_status current = APL_text;
731
732 // determine line status of each line...
733 //
734 for (ShapeItem l = 1; l < get_text_size(); ++l)
735 {
736 const UCS_string & line = get_text(l);

1131//----------------------------------------------------------------------------
1132UserFunction *
1133UserFunction::fix_lambda(Symbol & var, const UCS_string & text)
1134{

1136
1137 text a normal (non-lambda) function like:
1138
1139 λ←⍺ λ1 ⍵;LOCAL
1140 λ← ⍺+⍵
1141
1142 which has a slightly different lambda body like:
1143
1144 λ← ⍺+⍵;LOCAL
1145
1146 */
1147int signature = SIG_FUN | SIG_Z;
1148int t = 0;
1149
1150ShapeItem semi = -1;
1151 while (t < text.size())
1152 {
1153 switch(text[t++])
1154 {
1155 case UNI_CHI: signature |= SIG_X; continue;
1156 case UNI_OMEGA: signature |= SIG_B; continue;
1157 case UNI_ALPHA_UNDERBAR: signature |= SIG_LO; continue;
1158 case UNI_OMEGA_UNDERBAR: signature |= SIG_RO; continue;
1159 case UNI_ALPHA: signature |= SIG_A; continue;
1160
1161 case UNI_SEMICOLON: if (semi == -1) semi = t - 1;
1162 continue;
1163
1164 case UNI_LF: break; // header done
1165 default: continue;
1166 }
1167
1168 break; // header done
1169 }
1170
1171 // discard leading spaces
1172 //
1173 while (t < text.size() && text[t] == UNI_SPACE) ++t;
1174
1175UCS_string body_text;
1176 for (; t < text.size(); ++t) body_text.append(text[t]);
1177
1178 while (body_text.back() == UNI_LF) body_text.pop_back();
1179
1180Token_string body;
1181 {
1182 Token ret_lambda(TOK_RETURN_SYMBOL, &Workspace::get_v_LAMBDA());
1183 body.push_back(ret_lambda);
1184 const int64_t trace = 0;
1185 Token tok_endl(TOK_ENDL, trace);
1186 body.push_back(tok_endl);
1187 }
1188
1189 if (semi != -1)
1190 {
1191 for (ShapeItem s = semi; text[s] != UNI_LF; ++s)
1192 {
1193 body_text.append(text[s]);
1194 }
1195 }
1196
1197const Parser parser(PM_FUNCTION, LOC, false);
1198const ErrorCode ec = parser.parse(body_text, body);
1199 if (ec)
1200 {
1201 CERR << "Parsing '" << body_text << "' failed" << endl;
1202 return 0;
1203 }
1204
1205UserFunction * ufun = new UserFunction(Fun_signature(signature), 0,
1206 body_text, body);

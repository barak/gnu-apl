79//----------------------------------------------------------------------------
80void
81Workspace::push_SI(const Executable * fun, const char * loc)
82{
83 Assert1(fun);
84
85 if (Quad_SYL::si_depth_limit && SI_top() &&
86 Quad_SYL::si_depth_limit <= SI_top()->get_level())
87 {
88 MORE_ERROR() <<
89"the system limit on SI depth (as set in ⎕SYL["
90<< (Quad_SYL::SYL_SI_DEPTH_LIMIT + Workspace::get_IO())
91<< ";2]) was reached\n"
92"(and to avoid lock-up, this system limit in ⎕SYL was automatically cleared).";
93
94
95 Quad_SYL::si_depth_limit = 0;
96 set_attention_raised(LOC);
97 set_interrupt_raised(LOC);
98 }
99
100 if (Value::check_WS_FULL(__FUNCTION__, 1000, LOC))
101 {
102 if (SI_top() && SI_top()->get_executable() &&
103 the_workspace.SI_entry_count() > 100)
104 MORE_ERROR() <<
105 "Value::check_WS_FULL() complained when calling a defined function.\n"
106 "This was most likely caused by some infinite recursion involving "
107 << SI_top()->get_executable()->get_name();
108 else
109 MORE_ERROR() <<
110 "Value::check_WS_FULL() complained when calling a defined function";
111 WS_FULL;
112 }
113
114 try
115 {
116 StateIndicator * old_top = SI_top();
117 the_workspace.top_SI = new StateIndicator(fun, old_top);
118 if (the_workspace.top_SI == 0)
119 {
120 MORE_ERROR() <<
121 "malloc() → 0 when calling a defined function";
122 the_workspace.top_SI = old_top;
123 WS_FULL;
124 }
125 }
126 catch (const std::bad_alloc & e)
127 {
128 if (SI_top() && SI_top()->get_executable() &&
129 the_workspace.SI_entry_count() > 100)
130 MORE_ERROR() <<
131 "std::bad_alloc exception when calling a defined function\n"
132 "This was most likely caused by some infinite recursion involving "
133 << SI_top()->get_executable()->get_name();
134 else
135 MORE_ERROR() <<
136 "std::bad_alloc exception when calling a defined function";
137 WS_FULL;
138 }
139
140 Log(LOG_StateIndicator__push_pop)
141 {
142 CERR << "Push SI[" << SI_top()->get_level() << "] "
143 << "pmode=" << fun->get_parse_mode()
144 << " exec=" << fun << " "
145 << fun->get_name();
146
147 CERR << " new SI is " << voidP(SI_top())
148 << " at " << loc << endl;
149 }
150}
151//----------------------------------------------------------------------------
152void
153Workspace::pop_SI(const char * loc)
154{
155 Assert(SI_top());
156const Executable * exec = SI_top()->get_executable();
157 Assert1(exec);
158
159 Log(LOG_StateIndicator__push_pop)
160 {
161 CERR << "Pop SI[" << SI_top()->get_level() << "] "
162 << "pmode=" << exec->get_parse_mode()
163 << " exec=" << exec << " ";
164
165 if (exec->get_exec_ufun()) CERR << exec->get_exec_ufun()->get_name();
166 else CERR << SI_top()->get_parse_mode_name();
167 CERR << " " << voidP(SI_top())
168 << " at " << loc << endl;
169 }
170
171 // remove the top SI
172 //
173StateIndicator * del = SI_top();
174 the_workspace.top_SI = del->get_parent();
175 delete del;
176}
177//----------------------------------------------------------------------------
178uint64_t
179Workspace::get_RL(uint64_t mod)
180{
181 // we discard random numbers >= max_rand in order to avoid a bias
182 // towards small numbers
183 //
184const uint64_t max_rand = 0xFFFFFFFFFFFFFFFFULL - (0xFFFFFFFFFFFFFFFFULL % mod);
185uint64_t rand = the_workspace.v_Quad_RL.get_random();
186
187 do { rand ^= rand >> 11;
188 rand ^= rand << 29;
189 rand ^= rand >> 14;
190 } while (rand >= max_rand);
191
192 return rand % mod;
193}

232//----------------------------------------------------------------------------
233Token
234Workspace::immediate_execution(bool exit_on_error)
235{
236 for (;;)
237 {
238 try
239 {
240 Command::process_line();
241 }
242 catch (Error & err)
243 {
244 if (!err.get_print_loc())
245 {
246 if (err.get_error_code() != E_DEFN_ERROR)
247 {
248 err.print_em(UERR, LOC);
249 CERR << __FUNCTION__ << "() caught APL error "
250 << HEX(err.get_error_code()) << " ("
251 << err.error_name(err.get_error_code()) << ")"
252 << endl;
253
254 IO_Files::apl_error(LOC);
255 }
256 }
257 if (exit_on_error) return Token(TOK_OFF);
258 }
259 catch (const std::bad_alloc & e)
260 {
261 CERR << "*** " << __FUNCTION__
262 << "() caught bad_alloc: " << e.what() << " ***" << endl;
263 if (the_workspace.SI_entry_count() > 100)
264 CERR << ")SI depth is " << the_workspace.SI_entry_count()
265 << "; )SIC is strongly recommended !!!"
266 << endl;
267 try { WS_FULL } catch (...) {}
268 if (exit_on_error) return Token(TOK_OFF);
269 }
270 catch (...)
271 {
272 CERR << "*** " << __FUNCTION__
273 << "() caught other exception ***" << endl;
274 IO_Files::apl_error(LOC);
275 if (exit_on_error) return Token(TOK_OFF);
276 }
277 }
278
279 return Token(TOK_ESCAPE);
280}
281//----------------------------------------------------------------------------
282const NamedObject *
283Workspace::lookup_existing_name(const UCS_string & name)
284{
285 if (name.size() == 0) return 0;
286
287 if (Avec::is_quad(name[0])) // distinguished name
288 {
289 int len;
290 Token tok = get_quad(name, len);
291 if (len == 1) return 0;
292 if (name.size() != len) return 0;
293 if (tok.get_Class() == TC_SYMBOL) return tok.get_sym_ptr();
294 if (tok.get_Class() == TC_FUN0) return tok.get_function();
295 if (tok.get_Class() == TC_FUN1) return tok.get_function();
296 if (tok.get_Class() == TC_FUN2) return tok.get_function();
297
298 Assert(0);
299 }
300
301 // user defined variable or function
302 //
303Symbol * sym = the_workspace.symbol_table.lookup_existing_symbol(name);
304 if (sym == 0) return 0;
305
306 switch(sym->get_NC())
307 {
308 case NC_VARIABLE: return sym;
309
310 case NC_FUNCTION:
311 case NC_OPERATOR: return sym->get_function();
312 default: return 0;
313 }
314}
315//----------------------------------------------------------------------------
316Symbol *
317Workspace::lookup_existing_symbol(const UCS_string & symbol_name)
318{
319 if (symbol_name.size() == 0) return 0;
320
321 if (Avec::is_quad(symbol_name[0])) // distinguished name
322 {
323 int len;
324 Token tok = get_quad(symbol_name, len);
325 if (symbol_name.size() != len) return 0;
326 if (tok.get_Class() != TC_SYMBOL) return 0; // system function
327
328 return tok.get_sym_ptr();
329 }
330
331 return the_workspace.symbol_table.lookup_existing_symbol(symbol_name);
332}
333//----------------------------------------------------------------------------
334Token
335Workspace::get_quad(const UCS_string & ucs, int & len)
336{
337 if (ucs.size() == 0 || !Avec::is_quad(ucs[0]))
338 {
339 len = 0;
340 return Token();
341 }
342
343UCS_string name(UNI_Quad_Quad);
344 len = 1;
345
346SystemName * longest = 0;
347
348 for (ShapeItem u = 1; u < ucs.size(); ++u)
349 {
350 Unicode uni = ucs[u];
351 if (!Avec::is_symbol_char(uni)) break;
352
353 if (uni >= 'a' && uni <= 'z') uni = Unicode(uni - 0x20);
354 name.append(uni);
355 SystemName * dn =
356 the_workspace.distinguished_names.lookup_existing_symbol(name);
357
358 if (dn) // ss is a valid distinguished name
359 {
360 longest = dn;
361 len = name.size();
362 }
363 }
364
365 if (longest == 0) return Token(TOK_Quad_Quad, &the_workspace.v_Quad_Quad);
366
367 if (longest->get_variable()) return longest->get_variable()->get_token();
368 else return longest->get_function()->get_token();
369}
370//----------------------------------------------------------------------------
371StateIndicator *
372Workspace::oldest_exec(const Executable * exec)
373{
374StateIndicator * ret = 0;
375
376 for (StateIndicator * si = SI_top(); si; si = si->get_parent())
377 if (exec == si->get_executable()) ret = si; // maybe not yet oldest
378
379 return ret;
380}
381//----------------------------------------------------------------------------
382bool
383Workspace::is_called(const UCS_string & funname)
384{
385 // return true if the current-referent of funname is pendant
386 //
387Symbol * current_referent = lookup_existing_symbol(funname);
388 if (current_referent == 0) return false; // no such symbol
389
390 Assert(current_referent->get_Id() == ID_USER_SYMBOL);
391
392const NameClass nc = current_referent->get_NC();
393 if (nc != NC_FUNCTION && nc != NC_OPERATOR) return false;
394
395const Function * fun = current_referent->get_function();
396 Assert(fun);
397
398const UserFunction * ufun = fun->get_func_ufun();
399 if (!ufun) return false; // not a defined function
400
401 for (const StateIndicator * si = SI_top(); si; si = si->get_parent())
402 {
403 if (si->uses_function(ufun)) return true;
404 }
405
406 return false;
407}
408//----------------------------------------------------------------------------
409void
410Workspace::write_OUT(FILE * out, uint64_t & seq, const UCS_string_vector
411 & objects)
412{
413 // if objects is empty then write all user defined objects and some system
414 // variables
415 //
416 if (objects.size() == 0) // all objects plus some system variables
417 {
418 get_v_Quad_CT().write_OUT(out, seq);
419 get_v_Quad_FC().write_OUT(out, seq);
420 get_v_Quad_IO().write_OUT(out, seq);
421 get_v_Quad_LX().write_OUT(out, seq);
422 get_v_Quad_PP().write_OUT(out, seq);
423 get_v_Quad_PR().write_OUT(out, seq);
424 get_v_Quad_RL().write_OUT(out, seq);
425
426 get_symbol_table().write_all_symbols(out, seq);
427 }
428 else // only specified objects
429 {
430 loop(o, objects.size())
431 {
432 const NamedObject * obj = lookup_existing_name(objects[o]);
433 if (obj == 0) // not found
434 {
435 COUT << ")OUT: " << objects[o] << " NOT SAVED (not found)"
436 << endl;
437 continue;
438 }
439
440 if (obj->get_Id() == ID_USER_SYMBOL) // user defined name
441 {
442 const Symbol * sym = lookup_existing_symbol(objects[o]);
443 Assert(sym);
444 sym->write_OUT(out, seq);
445 }
446 else // distinguished name
447 {
448 const Symbol * sym = obj->get_symbol();
449 if (sym == 0)
450 {
451 COUT << ")OUT: " << objects[o]
452 << " NOT SAVED (not a variable)" << endl;
453 continue;
454 }
455
456 sym->write_OUT(out, seq);
457 }
458 }
459 }

519//----------------------------------------------------------------------------
520int
521Workspace::cleanup_expunged(ostream & out, bool & erased)
522{
523const int ret = the_workspace.expunged_functions.size();
524
525 if (SI_entry_count() > 0)
526 {
527 out << "SI not cleared (size " << SI_entry_count()
528 << "): not deleting ⎕EX'ed functions (try )SIC first)" << endl;
529 erased = false;
530 return ret;
531 }
532
533 while(the_workspace.expunged_functions.size())
534 {
535 const UserFunction * ufun = the_workspace.expunged_functions.back();
536 the_workspace.expunged_functions.pop_back();
537 out << "finally deleting " << ufun->get_name() << "...";
538 delete ufun;
539 out << " OK" << endl;
540 }
541
542 erased = true;
543 return ret;
544}
545//----------------------------------------------------------------------------
546void
547Workspace::clear_WS(ostream & out, bool silent)
548{
549 // remove user-defined commands
550 //
551 get_user_commands().clear();
552
553 // clear the )COPY_ONCE table

555

576#define ro_sv_def(x, _str, _txt)
577#include "SystemVariable.def"
578
579 // at this point all values should have been gone.
580 // complain about those that still exist, and remove them.
581 //
582// Value::erase_all(out);
583
584#define rw_sv_def(x, _str, _txt) \
585 { get_v_ ##x().~x(); new (&get_v_ ##x()) x; }

601}

622//----------------------------------------------------------------------------
623void
624Workspace::save_WS(ostream & out, LibRef libref, const UCS_string & WS_name,
625 bool name_from_WSID)
626{
627UTF8_string filename = LibPaths::get_lib_filename(libref, WS_name, false,
628 ".xml", 0);
629
630 // dont )SAVE if WS_name differs from wsid and the file exists
631 //
632const bool file_exists = access(filename.c_str(), W_OK) == 0;
633 if (file_exists)
634 {
635 if (WS_name.compare(the_workspace.WS_name) != 0) // names differ
636 {
637 out << "NOT SAVED: THIS WS IS "
638 << the_workspace.WS_name << endl;
639
640 MORE_ERROR() << "the workspace was not saved because:\n"
641 << " the workspace name '" << the_workspace.WS_name
642 << "' of )WSID\n does not match the name '" << WS_name
643 << "' used in the )SAVE command\n"
644 << " and the workspace file\n " << filename
645 << "\n already exists. Use )WSID " << WS_name << " first.";
646 return;
647 }
648 }
649
650 if (uprefs.backup_before_save && backup_existing_file(filename.c_str()))
651 {
652 COUT << "NOT SAVED: COULD NOT CREATE BACKUP FILE "
653 << filename << endl;
654 return;
655 }
656
657 // at this point it is OK to rename and save the workspace
658 //
659ofstream outf(filename.c_str(), ofstream::out);
660 if (!outf.is_open()) // open failed
661 {
662 CERR << "Unable to )SAVE workspace '" << WS_name
663 << "'. " << strerror(errno) << endl;
664 return;
665 }
666
667 the_workspace.WS_name = WS_name;
668
669 Log(LOG_archive) CERR << "constructing XML_Saving_Archive." << endl;
670XML_Saving_Archive ar(outf);
671 Log(LOG_archive) CERR << "saving XML_Saving_Archive..." << endl;
672 ar.save();
673 Log(LOG_archive) CERR << "done XML_Saving_Archive." << endl;
674
675 // print time and date to COUT
676 get_v_Quad_TZ().print_timestamp(out, now());
677 if (name_from_WSID) out << " " << the_workspace.WS_name;
678 out << endl;
679}
680//----------------------------------------------------------------------------
681bool
682Workspace::backup_existing_file(const char * filename)
683{
684 // 1. if file 'filename' does not exist then no backup is needed.
685 //
686 if (access(filename, F_OK) != 0) return false; // OK
687
688UTF8_string backup_filename = filename;
689 backup_filename.append_ASCII(".bak");
690
691 // 2. if backup file exists then remove it...
692 //
693 if (access(backup_filename.c_str(), F_OK) == 0) // backup exists
694 {
695 const int err = unlink(backup_filename.c_str());
696 if (err)
697 {
698 CERR << "Could not remove backup file " << backup_filename
699 << ": " << strerror(errno) << endl;
700 return true; // error
701 }
702
703 if (access(backup_filename.c_str(), F_OK) == 0) // still exists
704 {
705 CERR << "Could not remove backup file " << backup_filename << endl;
706 return true; // error
707 }
708 }
709
710 // 3. rename file to file.bak
711 //
712const int err = rename(filename, backup_filename.c_str());
713 if (err) // rename failed
714 {
715 CERR << "Could not rename file " << filename
716 << " to " << backup_filename
717 << ": " << strerror(errno) << endl;
718 return true; // error
719 }
720
721 if (access(filename, F_OK) == 0) // file still exists
722 {
723 CERR << "Could not rename file " << filename
724 << " to " << backup_filename << endl;
725 return true; // error
726 }
727
728 return false; // OK
729}
730//----------------------------------------------------------------------------
731void
732Workspace::load_DUMP(ostream & out, const UTF8_string & filename, int fd,
733 LX_mode with_LX, bool silent,
734 UCS_string_vector * object_filter)
735{
736 Log(LOG_command_IN)
737 CERR << "loading )DUMP file " << filename << "..." << endl;
738
739 {
740 struct stat st;
741 fstat(fd, &st);
742 const APL_time_us when = 1000000ULL * st.st_mtime;

956// )LOAD WS, set ⎕LX of loaded WS on success
957void
958Workspace::load_WS(ostream & out, LibRef libref, const UCS_string & WS_name,
959 UCS_string & quad_lx, bool silent)
960{
961 // )LOAD WS_name WS_name /absolute or relative
962 // )LOAD libnum WS_name WS_name relative
963 //
964 if (UTF8_string(WS_name).ends_with(".bak"))
965 {
966 out << "BAD COMMAND+" << endl;
967 MORE_ERROR() << WS_name <<
968 " is a backup file! You need to rename it before it can be )LOADed";
969 return;
970 }
971
972UTF8_string filename = LibPaths::get_lib_filename(libref, WS_name, true,
973 ".xml", ".apl");
974
975int dump_fd = -1;
976XML_Loading_Archive in(out, filename.c_str(), dump_fd);
977
978 if (dump_fd != -1) // WS_name.apl
979 {
980 the_workspace.clear_WS(out, true);
981
982 Log(LOG_command_IN) out << "LOADING " << WS_name << " from file '"
983 << filename << "' ..." << endl;
984
985 load_DUMP(out, filename, dump_fd, do_LX, silent, 0); // closes dump_fd
986
987 // )DUMP files have no )WSID so create one from the filename
988 //
989 const char * wsid_start = strrchr(filename.c_str(), '/');
990 if (wsid_start == 0) wsid_start = filename.c_str();
991 else ++wsid_start; // skip /
992 const char * wsid_end = filename.c_str() + filename.size();
993 if (wsid_end > (wsid_start - 4) &&
994 wsid_end[-4] == '.' &&
995 wsid_end[-3] == 'a' &&
996 wsid_end[-2] == 'p' &&
997 wsid_end[-1] == 'l') wsid_end -= 4; // skip .apl extension
998 const UTF8_string wsid_utf8(utf8P(wsid_start), wsid_end - wsid_start);
999 const UCS_string wsid_ucs(wsid_utf8);
1000 wsid(out, wsid_ucs, libref, true);
1001
1002 // we cant set ⎕LX because it was not executed yet.
1003 return;
1004 }
1005 else // WS_name.xml
1006 {
1007 if (!in.is_open()) // open failed
1008 {
1009 out << ")LOAD " << WS_name << " (file " << filename
1010 << ") failed: " << strerror(errno) << endl;
1011 return;
1012 }
1013
1014 Log(LOG_command_IN) out << "LOADING " << WS_name << " from file '"
1015 << filename << "' ..." << endl;
1016
1017 // got open file. We assume that from here on everything will be fine.
1018 // clear current WS and load it from file
1019 //
1020#ifdef DYNAMIC_LOG_WANTED
1021 const bool logit = Log_status(LID_archive);
1022 the_workspace.clear_WS(out, true);
1023 Log_control(LID_archive, logit);
1024#else
1025 the_workspace.clear_WS(out, !LOG_archive);
1026#endif
1027
1028 in.read_Workspace(silent);
1029 }
1030
1031 if (Workspace::get_LX().size()) quad_lx = Workspace::get_LX();
1032}

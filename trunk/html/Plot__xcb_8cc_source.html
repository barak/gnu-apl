118//----------------------------------------------------------------------------
119string_width_height::string_width_height(const Plot_context & pctx,
120 const char * string)
121{
122const size_t string_length = strlen(string);
123xcb_char2b_t xcb_str[string_length + 10];
124
125 loop(s, string_length)
126 {
127 xcb_str[s].byte1 = 0;
128 xcb_str[s].byte2 = string[s];
129 }
130
131const xcb_query_text_extents_cookie_t cookie =
132 xcb_query_text_extents(pctx.conn, pctx.font, string_length, xcb_str);
133
134 if ( xcb_query_text_extents_reply_t * reply =
135 xcb_query_text_extents_reply(pctx.conn, cookie, 0))
136 {
137 width = reply->overall_width;
138 height = reply->font_ascent + reply->font_descent;
139 // origin = reply->font_ascent;
140 free(reply);
141 }
142 else
143 {
144 puts("XCB ERROR: xcb_query_text_extents_reply() failed.");
145 width = 24;
146 height = 13;
147 }
148}

254char *
255format_tick(double val)
256{
257static char cc[40];
258char * cp = cc;
259 if (val == 0)
260 {
261 *cp++ = '0';
262 *cp++ = 0;
263 return cc;
264 }
265
266 if (val < 0)
267 {
268 *cp++ = '-';
269 val = -val;
270 }
271
272const char * unit = 0;
273 if (val >= 1e3)
274 {
275 if (val >= 1e15) ;
276 else if (val >= 1e12) { unit = "T"; val *= 1e-12; }
277 else if (val >= 1e9) { unit = "G"; val *= 1e-9; }
278 else if (val >= 1e6) { unit = "M"; val *= 1e-6; }
279 else { unit = "k"; val *= 1e-3; }
280 }
281 else
282 {
283
284 if (val >= 1e0) { unit = ""; }
285 else if (val >= 1e-3) { unit = "m"; val *= 1e3; }
286 else if (val >= 1e-6) { unit = "μ"; val *= 1e6; }
287 else if (val >= 1e-9) { unit = "n"; val *= 1e9; }
288 else if (val >= 1e-12) { unit = "p"; val *= 1e12; }
289 }
290
291 if (unit == 0) // very large or very small
292 {
293 snprintf(cp, sizeof(cc) - 1, "%.2E", val);
294 return cc;
295 }
296
297 // at this point: 1 ≤ val < 1000
298 //
299 if (val < 10) snprintf(cp, sizeof(cc) - 1, "%.2f%s", val, unit);
300 else if (val < 100) snprintf(cp, sizeof(cc) - 1, "%.1f%s", val, unit);
301 else snprintf(cp, sizeof(cc) - 1, "%.0f%s", val, unit);
302 return cc;
303}

321//----------------------------------------------------------------------------
322inline double
323intersection(double Ax, double Ay, double Bx, double By,
324 double Cx, double Cy, double Dx, double Dy)
325{
344 /* first, translate A, B, and D by -C, so that C=(0, 0):
345
346 A
347 \ B
348 \ /
349 M
350 / \
351 / \
352 0 D
353 */
354 Ax -= Cx; Bx -= Cx; Dx -= Cx;
355 Ay -= Cy; By -= Cy; Dy -= Cy;
356
357 /* Thus:
358
359 (Ax-Bx) + alpha*(Dx-Ax) == beta*(-Bx)
360 (Ay-By) + alpha*(Dy-Ay) == beta*(-By)
361
362 (Ax-Bx)*(Dy-Ay) + alpha*(Dx-Ax)*(Dy-Ay) == beta*(-Bx)*(Dy-Ay)
363 (Ay-By)*(Dx-Ax) + alpha*(Dy-Ay)*(Dx-Ax) == beta*(-By)*(Dx-Ax)
364 ======================================= ==================
365 (Ax-Bx)*(Dy-Ay) - (Ay-By)*(Dx-Ax) == beta*((-Bx)*(Dy-Ay) - (-By)*(Dx-Ax))
366 **/
367
368const double numer = (Ax-Bx)*(Dy-Ay) - (Ay-By)*(Dx-Ax); // left side
369const double denom = ((-Bx)*(Dy-Ay) - (-By)*(Dx-Ax)); // right factor
370 if (denom == 0.0)
371 {
372 Ax += Cx; Bx += Cx; Dx += Cx;
373 Ay += Cy; By += Cy; Dy += Cy;
374 MORE_ERROR() << "⎕PLOT: one of the X-Z squares has bad coordinates:"
375 << "P1=(" << Ax << ":" << Ay << "), "
376 "P2=(" << Bx << ":" << By << "), "
377 "P3=(" << Cx << ":" << Cy << ") "
378 "P4=(" << Dx << ":" << Dy << ")";
379 DOMAIN_ERROR;
380 }
381
382 return /* beta == */ numer/denom;
383}

408void
409draw_triangle(const Plot_context & pctx, int verbosity,
410 Pixel_XY P0, double H0, Pixel_XY P1, Pixel_XY P2, double H12)
411{
412 // draw a triangle with P1 and P2 at the same level H12
413 //
414 Assert(H0 >= 0.0); Assert(H0 <= 1.0);
415 Assert(H12 >= 0.0); Assert(H12 <= 1.0);
416
417 if (verbosity & SHOW_DRAW)
418 CERR << " ∆2: P0(" << P0.x << ":" << P0.y << ") @H0=" << H0
419 << " P1(" << P1.x << ":" << P1.y << ") @H12=" << H12
420 << " P2(" << P2.x << ":" << P2.y << ") @H12=" << H12 << endl;
421
422const double d1 = P0.distance2(P1);
423const double d2 = P0.distance2(P2);
424const int steps = sqrt(d1 > d2 ? d1 : d2);
425const double dH = (H12 - H0) / steps;
426 loop(s, steps)
427 {
428 const double alpha = H0 + s*dH;
429 const double beta = 1.0 * s/ steps;
430 const uint32_t color = pctx.w_props.get_color(alpha);
431 set_GC_foreground(pctx, pctx.fill, color);
432
433 const Pixel_XY P0_P1(P0.x + beta * (P1.x - P0.x),
434 P0.y + beta * (P1.y - P0.y));
435 const Pixel_XY P0_P2(P0.x + beta * (P2.x - P0.x),
436 P0.y + beta * (P2.y - P0.y));
437 draw_line(pctx, pctx.fill, P0_P1, P0_P2);
438 }
439}
440//----------------------------------------------------------------------------
442void
443draw_triangle(const Plot_context & pctx, int verbosity, Pixel_XY P0, double H0,
444 Pixel_XY P1, double H1, Pixel_XY P2, double H2)
445{
446 // draw a triangle with P0, P1 and P2 at levels H0, H1, and H2
447
448 Assert(H0 >= 0); Assert(H0 <= 1.0);
449 Assert(H1 >= 0); Assert(H1 <= 1.0);
450 Assert(H2 >= 0); Assert(H2 <= 1.0);
451
452 if (verbosity & SHOW_DRAW)
453 CERR << "\n∆1: P0(" << P0.x << ":" << P0.y << ")@H=" << H0
454 << " P1(" << P1.x << ":" << P1.y << ")@H=" << H1
455 << " P2(" << P2.x << ":" << P2.y << ")@H=" << H2 << endl;
456
457 if (H0 < H1) pv_swap(P0, H0, P1, H1); // then H0 >= H1
458 if (H0 < H2) pv_swap(P0, H0, P2, H2); // then H0 >= H2
459 if (H1 < H2) pv_swap(P1, H1, P2, H2); // then H1 >= H2
460
461 // here H0 >= H1 >= H2
462 //
463 Assert(H0 >= H1);
464 Assert(H1 >= H2);
465
466 if (H0 == H1) // P0 and P1 have the same height
467 {
468 draw_triangle(pctx, verbosity, P2, H2, P0, P1, H0);
469 }
470 else if (H1 == H2) // P1 and P2 have the same height
471 {
472 draw_triangle(pctx, verbosity, P0, H0, P1, P2, H1);
473 }
474 else // P2 lies below P1
475 {
476 const double alpha = (H0 - H1) / (H0 - H2); // alpha → 1 as P1 → P2
477 Assert(alpha >= 0.0);
478 Assert(alpha <= 1.0);
479
480 // compute the point P on P0-P2 that has the heigth H1 (of P1)
481 const Pixel_XY P(P0.x + alpha*(P2.x - P0.x),
482 P0.y + alpha*(P2.y - P0.y));
483 draw_triangle(pctx, verbosity, P0, H0, P1, P, H1);
484 draw_triangle(pctx, verbosity, P2, H2, P1, P, H1);
485 }
486}

489void
490draw_point(const Plot_context & pctx, const Pixel_XY & xy, Color canvas_color,
491 const Plot_line_properties & l_props)
492{
493const Pixel_X point_size = l_props.get_point_size(); // outer point diameter
494const Pixel_X point_size2 = l_props.get_point_size2(); // inner point diameter
495const int16_t half = point_size >> 1; // outer point radius
496const int point_style = l_props.get_point_style();
497const Color point_color = l_props.get_point_color();
498
499 // set outer point color for all point styles (and maybe inner point colors
500 // later
501 //
502 set_GC_foreground(pctx, pctx.point, point_color);
503 if (point_style == 1) // circle
504 {
505 xcb_arc_t arc = { I1616(xy.x - half, xy.y - half), // positions
506 point_size, point_size, // diameters
507 0, 360 << 6 }; // angles
508 xcb_poly_fill_arc(pctx.conn, pctx.window, pctx.point, 1, &arc);
509
510 if (point_size2) // hollow
511 {
512 const int16_t half2 = point_size2 >> 1;
513 xcb_arc_t arc2 = { I1616(xy.x - half2, xy.y - half2), // pos
514 point_size2, point_size2, // dia
515 0, int16_t(360 << 6) }; // ang
516 set_GC_foreground(pctx, pctx.point, canvas_color);
517 xcb_poly_fill_arc(pctx.conn, pctx.window, pctx.point, 1, &arc2);
518 set_GC_foreground(pctx, pctx.point, point_color);
519 }
520 }
521 else if (point_style == 2) // triangle ∆
522 {
523 const int16_t center_y = xy.y - 2*point_size/3;
524 const int16_t hypo_y = half*1.732;
525 const xcb_point_t points[3] = { { int16_t(xy.x), center_y },
526 { half, hypo_y },
527 { int16_t(-2*half), 0 } };
528 xcb_fill_poly(pctx.conn, pctx.window, pctx.point,
529 XCB_POLY_SHAPE_CONVEX,
530 XCB_COORD_MODE_PREVIOUS, ITEMS(points));
531 if (point_size2) // hollow
532 {
533 const int16_t half2 = point_size2 >> 1;
534 const int16_t center2_y = xy.y - 2*point_size2/3;
535 const int16_t hypo2_y = half2*1.732;
536 const xcb_point_t points2[3] = {
537 { int16_t(xy.x), center2_y },
538 { int16_t(half2), hypo2_y },
539 { int16_t(-2*half2), 0 } };
540 set_GC_foreground(pctx, pctx.point, canvas_color);
541 xcb_fill_poly(pctx.conn, pctx.window, pctx.point,
542 XCB_POLY_SHAPE_CONVEX,
543 XCB_COORD_MODE_PREVIOUS, ITEMS(points2));
544 set_GC_foreground(pctx, pctx.point, point_color);
545 }
546 }
547 else if (point_style == 3) // triangle ∇
548 {
549 const int16_t center_y = xy.y + 2*point_size/3;
550 const int16_t hypo_y = - half*1.732;
551 const xcb_point_t points[3] = {
552 { int16_t(xy.x), center_y },
553 { int16_t(half), hypo_y },
554 { int16_t(-2*half), 0 } };
555 xcb_fill_poly(pctx.conn, pctx.window, pctx.point,
556 XCB_POLY_SHAPE_CONVEX,
557 XCB_COORD_MODE_PREVIOUS, ITEMS(points));
558 if (point_size2) // hollow
559 {
560 const int16_t half2 = point_size2 >> 1;
561 const int16_t center2_y = xy.y + 2*point_size2/3;
562 const int16_t hypo2_y = - half2*1.732;
563 const xcb_point_t points2[3] = {
564 { int16_t(xy.x), center2_y },
565 { int16_t(half2), hypo2_y },
566 { int16_t(-2*half2), 0 } };
567 set_GC_foreground(pctx, pctx.point, canvas_color);
568 xcb_fill_poly(pctx.conn, pctx.window, pctx.point,
569 XCB_POLY_SHAPE_CONVEX,
570 XCB_COORD_MODE_PREVIOUS, ITEMS(points2));
571 set_GC_foreground(pctx, pctx.point, point_color);
572 }
573 }
574 else if (point_style == 4) // caro
575 {
576 const xcb_point_t points[4] = {
577 { int16_t(xy.x), int16_t(xy.y - half) },
578 { half, half },
579 { int16_t(-half), half },
580 { int16_t(-half), int16_t(-half) } };
581 xcb_fill_poly(pctx.conn, pctx.window, pctx.point,
582 XCB_POLY_SHAPE_CONVEX,
583 XCB_COORD_MODE_PREVIOUS, ITEMS(points));
584 if (point_size2) // hollow
585 {
586 const int16_t half2 = point_size2 >> 1;
587 const xcb_point_t points2[4] = {
588 { I1616(xy.x, xy.y - half2) },
589 { I1616( half2, half2) },
590 { I1616(-half2, half2) },
591 { I1616(-half2, -half2) } };
592 set_GC_foreground(pctx, pctx.point, canvas_color);
593 xcb_fill_poly(pctx.conn, pctx.window, pctx.point,
594 XCB_POLY_SHAPE_CONVEX,
595 XCB_COORD_MODE_PREVIOUS,
596 ITEMS(points2));
597 set_GC_foreground(pctx, pctx.point, point_color);
598 }
599 }
600 else if (point_style == 5) // square
601 {
602 const xcb_point_t points[4] = {
603 { I1616(xy.x - half, xy.y - half) },
604 { I1616(point_size, 0) },
605 { I1616(0, point_size) },
606 { I1616(-point_size, 0) } };
607 xcb_fill_poly(pctx.conn, pctx.window, pctx.point,
608 XCB_POLY_SHAPE_CONVEX,
609 XCB_COORD_MODE_PREVIOUS, ITEMS(points));
610 if (point_size2) // hollow
611 {
612 const int16_t half2 = point_size2 >> 1;
613 const xcb_point_t points2[4] =
614 { { int16_t(xy.x - half2), int16_t(xy.y - half2) },
615 { int16_t(point_size2), 0 },
616 { 0, int16_t(point_size2) },
617 { int16_t(-point_size2), 0 } };
618 set_GC_foreground(pctx, pctx.point, canvas_color);
619 xcb_fill_poly(pctx.conn, pctx.window, pctx.point,
620 XCB_POLY_SHAPE_CONVEX,
621 XCB_COORD_MODE_PREVIOUS, ITEMS(points2));
622 set_GC_foreground(pctx, pctx.point, point_color);
623 }
624 }
625
626 // otherwise: draw no point
627}

630void
631draw_legend(const Plot_context & pctx, const Plot_window_properties & w_props)
632{
633Plot_line_properties const * const * l_props = w_props.get_line_properties();
634const int line_count = w_props.get_line_count();
635
636 // estimate the box size from the length of the longest legend string
637 //
638const char * longest_legend = "";
639 for (int l = 0; l < line_count; ++l)
640 {
641 const char * lstr = l_props[l]->get_legend_name().c_str();
642 if (strlen(longest_legend) < strlen(lstr)) longest_legend = lstr;
643 }
644
645string_width_height longest_size(pctx, longest_legend);
646
647const Color canvas_color = w_props.get_canvas_color();
648
649 // every legend looks like: ---o--- legend_name
650 // | | | |

864//----------------------------------------------------------------------------
866void
867draw_surface_lines(const Plot_context & pctx,
868 const Plot_window_properties & w_props,
869 const Plot_data & data)
870{
871const Color canvas_color = w_props.get_canvas_color();
872Plot_line_properties const * const * l_props = w_props.get_line_properties();
873const Plot_line_properties & lp0 = *l_props[0];
874
875 // 1. setup graphic contexts
876 //
877 set_GC_foreground(pctx, pctx.line, lp0.get_line_color());
878 set_GC_line_width(pctx, pctx.line, lp0.get_line_width());
879
880 set_GC_foreground(pctx, pctx.point, lp0.get_point_color());
881 set_GC_line_width(pctx, pctx.point, lp0.get_point_size());
882
883 // 2. draw areas between plot lines...
884 //
885const double Hmin = w_props.get_min_Y();
886const double dH = w_props.get_max_Y() - Hmin; // 100%
887const int verbosity = w_props.get_verbosity();
888
889 loop(row, data.get_row_count() - 1)
890 loop(col, data[row].get_N() - 1)
891 {
892 if (verbosity & SHOW_DATA)
893 CERR << "B[" << row << ";" << col << "]"
894 << " X=" << data.get_X(row, col)
895 << " Y=" << data.get_Y(row, col)
896 << " Z=" << data.get_Z(row, col) << endl;
897
898 if (!w_props.get_gradient().size()) continue; // no gradient
899// if (row != 3) continue; // show only given row
900// if (col != 0) continue; // show only given column
901
902 const double X0 = data.get_X(row, col);
903 const double Y0 = data.get_Y(row, col);
904 const double Z0 = data.get_Z(row, col);
905 const double H0 = (Y0 - Hmin)/dH;
906 const Pixel_XY P0 = w_props.valXYZ2pixelXY(X0, Y0, Z0);
907
908 const double X1 = data.get_X(row, col + 1);
909 const double Y1 = data.get_Y(row, col + 1);
910 const double Z1 = data.get_Z(row, col + 1);
911 const double H1 = (Y1 - Hmin)/dH;
912 const Pixel_XY P1 = w_props.valXYZ2pixelXY(X1, Y1, Z1);
913
914 const double X2 = data.get_X(row + 1, col);
915 const double Y2 = data.get_Y(row + 1, col);
916 const double Z2 = data.get_Z(row + 1, col);
917 const double H2 = (Y2 - Hmin)/dH;
918 const Pixel_XY P2 = w_props.valXYZ2pixelXY(X2, Y2, Z2);

921 const double Y3 = data.get_Y(row + 1, col + 1);
922 const double Z3 = data.get_Z(row + 1, col + 1);

925
926
927 /* the surface has 4 points P0, P1, P2, and P3, but they are not
928 necessarily coplanar. We could draw the surface in 2 ways:
929
930 A. find the intersection PM of P0-P3 and P1-P2 on the bottom
931 plane Y=0, interpolate the respective heights, and draw
932 4 triangles from PM to each of the four edges of the square
933
934 B. fold the surface at its shorter edge into 2 triangles that
935 have the edge in common. (triangles are always coplanar).
936
937 Option B seems to look better.
938 */
939
940#if 0
941 /* Option A...
942
943 interpolate middle point PM with height HM and split the
944 square P0-P1-P2-P3 into 4 triangles with common point PM
945 and edges P0-P1, P1-P3, P3-P2, and P2-P0 respectively.

947 const double beta = intersection(X0,Z0, X1,Z1, X2,Z2, X3,Z3);

949 const double YM = Y1 + beta*(Y2 - Y1);

959 /* Option B...
960
961 fold surface-square along its shorter edge
962 */
963

966 draw_triangle(pctx, verbosity, P1, H1, P0, H0, P3, H3);
967 draw_triangle(pctx, verbosity, P2, H2, P0, H0, P3, H3);
968 }
969 else // P1-P2 is shorter
970 {
971 draw_triangle(pctx, verbosity, P0, H0, P1, H1, P2, H2);
972 draw_triangle(pctx, verbosity, P3, H3, P1, H1, P2, H2);
973 }
974#endif
975 }
976
977 // 3. draw lines between plot points
978 //
979 loop(row, data.get_row_count() - 1)
980 loop(col, data[row].get_N() - 1)
981 {
982 if (verbosity & SHOW_DATA)
983 CERR << "data[" << row << "," << col << "]"
984 << " X=" << data.get_X(row, col)
985 << " Y=" << data.get_Y(row, col)
986 << " Z=" << data.get_Z(row, col) << endl;
987
988 const double X0 = data.get_X(row, col);
989 const double Y0 = data.get_Y(row, col);
990 const double Z0 = data.get_Z(row, col);
991 const Pixel_XY P0 = w_props.valXYZ2pixelXY(X0, Y0, Z0);
992
993 const double X1 = data.get_X(row, col + 1);
994 const double Y1 = data.get_Y(row, col + 1);
995 const double Z1 = data.get_Z(row, col + 1);
996 const Pixel_XY P1 = w_props.valXYZ2pixelXY(X1, Y1, Z1);
997
998 const double X2 = data.get_X(row + 1, col);
999 const double Y2 = data.get_Y(row + 1, col);
1000 const double Z2 = data.get_Z(row + 1, col);
1001 const Pixel_XY P2 = w_props.valXYZ2pixelXY(X2, Y2, Z2);
1002
1003 const double X3 = data.get_X(row + 1, col + 1);
1004 const double Y3 = data.get_Y(row + 1, col + 1);
1005 const double Z3 = data.get_Z(row + 1, col + 1);
1006 const Pixel_XY P3 = w_props.valXYZ2pixelXY(X3, Y3, Z3);
1007
1008#if 0
1009 // debug: plot points as vertical lines
1010 //
1011 const Pixel_XY xy0 = w_props.valXYZ2pixelXY(data.get_X(row, col),
1012 w_props.get_min_Y(),
1013 data.get_Z(row, col));
1014 draw_line(pctx, pctx.line, xy0, P0);
1015 continue;
1016#endif
1017 draw_line(pctx, pctx.line, P0, P1);
1018 if (row == (data.get_row_count() - 2)) // last row
1019 draw_line(pctx, pctx.line, P2, P3);
1020

1022 if (col == (data[row].get_N() - 2)) // last column

1080//----------------------------------------------------------------------------
1082void
1083save_file(const char * outfile, Display * dpy, int window,
1084 const Plot_context & pctx)
1085{
1086 // figure position and size...
1087 //
1088int x, y; // position of the plot window in its parent
1089 {
1090 Window child;
1091 XTranslateCoordinates(dpy, window, pctx.screen->root,
1092 0, 0, &x, &y, &child );
1093
1094 /*
1095 XWindowAttributes xwa;
1096 XGetWindowAttributes(dpy, window, &xwa);
1097 CERR << "POS: " << (x - xwa.x) << ":" << (y - xwa.y) << endl;
1098 */
1099 }
1100
1101Window geo_root;
1102int geo_x, geo_y;
1103unsigned int geo_w, geo_h;
1104unsigned int geo_border_w, geo_depth;
1105 if (XGetGeometry(dpy, window, &geo_root, &geo_x, &geo_y, &geo_w, &geo_h,
1106 &geo_border_w, &geo_depth)) // success
1107 {
1108 /*
1109 CERR << endl
1110 << " POS: " << geo_x << ":" << geo_y << endl
1111 << " SIZE: " << geo_w << "x" << geo_h << endl
1112 << " bw: " << geo_border_w << endl
1113 << " bits/pixel=" << geo_depth << endl;
1114 */
1115 }
1116 else // error
1117 {
1118 CERR << " XGetGeometry() failed" << endl;
1119 MORE_ERROR() << "XGetGeometry() failed in save_file()";
1120 DOMAIN_ERROR;
1121 }
1122
1123 // Note: the point geo_x:geo_y seems to be the top-left corner of the
1124 // plotarea.
1125 // Therefore we assume that the window border has a thickness of geo_x
1126 // left, right, and below the plotarea and a thickness of geo_y above
1127 // (caused by the window caption etc.).
1128 //
1129 // To save the window including its borders we have to start at point
1130 // (x - geo_x):(y - geo_y) and add the borders to width:height.
1131 //
1132const unsigned int width = geo_x + geo_w + geo_x;
1133const unsigned int height = geo_y + geo_h + geo_x;
1134const unsigned long plane_mask = AllPlanes;
1135const int format = XYPixmap;
1136const int screen = XDefaultScreen(dpy);
1137XImage * image = XGetImage(dpy, XRootWindow(dpy, screen),
1138 x - geo_x, y - geo_y,
1139 width, height, plane_mask, format);
1140
1141 if (image == 0)
1142 {
1143 MORE_ERROR() << "XGetImage() failed in save_file()";
1144 DOMAIN_ERROR;
1145 }
1146
1147 // write .bmp file...
1148 //
1149UTF8_string outfile_utf8(outfile);
1150 if (!outfile_utf8.ends_with(".bmp")) outfile_utf8.append_ASCII(".bmp");
1151
1152 errno = 0;
1153FILE * bmp = fopen(outfile_utf8.c_str(), "w");
1154 if (bmp == 0)
1155 {
1156 CERR << "open " << outfile << ": " << strerror(errno) << endl;
1157 MORE_ERROR() << "save_file() failed: cannot open output file "
1158 << outfile;
1159 DOMAIN_ERROR;
1160 }
1161
1162 // compute some .bmp related sizes...
1163 //
1164const int pixel_count = width * height;
1165
1166 enum
1167 {
1168 file_header_size = 14,
1169 dib_header_size = 40,
1170 pixel_offset = file_header_size + dib_header_size
1171 };
1172
1173const int part_DWORD = (width*3) & 3; // bytes in final scanline DWORD
1174const int pad_DWORD = 3 & (4-part_DWORD); // pad bytes per scanline
1175const int pad_bytes = height*pad_DWORD;
1176const int file_size = file_header_size
1177 + dib_header_size
1178 + 3*pixel_count
1179 + pad_bytes;
1180
1181int file_bytes = 0;
1182 {
1183 const uint8_t file_header[file_header_size] =
1184 {
1185 uint8_t('B'),
1186 uint8_t('M'),
1187 LE4(file_size),
1188 LE4(0), // reserved
1189 LE4(pixel_offset), // offset
1190 };
1191 file_bytes += fwrite(file_header, 1, sizeof(file_header), bmp);
1192 };
1193
1194 {
1195 const uint8_t dib_header[dib_header_size] =
1196 {
1197 LE4(dib_header_size),
1198 LE4(width),
1199 LE4(height),
1200 LE2(1), // number of planes
1201 LE2(24), // bits per pixel
1202 LE4(0), // no compression
1203 LE4(0), // image size (0 if no compression)
1204 LE4(0), // Xpixels per meter
1205 LE4(0), // Ypixels per meter
1206 LE4(0), // colors used
1207 LE4(0), // importantcolors
1208 };
1209 file_bytes += fwrite(dib_header, 1, sizeof(dib_header), bmp);
1210 }
1211
1212const Colormap cmap = XDefaultColormap(dpy, screen);
1213 for (int y = height - 1; y >= 0; --y)
1214 {
1215 uint8_t scanline[3*width + 10];
1216 uint8_t * s = scanline;
1217 for (unsigned int x = 0; x < width; ++x)

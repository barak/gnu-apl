166//----------------------------------------------------------------------------
167bool
168UserPreferences::is_APL_script(const char * filename)
169{
170 /* according to man execve():
171
172 An interpreter script is a text file that has execute permission
173 enabled and whose first line is of the form:
174
175 #! interpreter [optional-arg]
176
177 */
178
179 if (access(filename, X_OK)) return false;
180int fd = open(filename, O_RDONLY);
181 if (fd == -1) return false;
182
183char buf[2];
184const size_t len = read(fd, buf, sizeof(buf));
185 close(fd);
186
187 if (len != 2) return false;
188 if (buf[0] != '#') return false;
189 if (buf[1] != '!') return false;
190 return true;
191}

204//----------------------------------------------------------------------------
205bool
206UserPreferences::parse_argv_1()
207{
208bool log_startup = false;
209 for (size_t a = 1; a < expanded_argv.size(); )
210 {
211 const char * opt = expanded_argv[a++];
212 const char * val = (a < expanded_argv.size()) ? expanded_argv[a] : 0;
213
214 if (!strcmp(opt, "-l"))
215 {
216 ++a;
217 if (!val)
218 {
219 CERR << "-l without log level" << endl;
220 exit(a);
221 }
222
223 log_startup = (atoi(val) == LID_startup);
224 continue;
225 }
226
227 if (!strcmp(opt, "-p"))
228 {
229 ++a;
230 if (!val)
231 {
232 CERR << "-p without profile number" << endl;
233 exit(a);
234 }
235
236 user_profile = atoi(val);
237 continue;
238 }
239
240 if (!strcmp(opt, "-C"))
241 {
242 ++a;
243 if (!val)
244 {
245 CERR << "-C without directory" << endl;
246 exit(a);
247 }
248
249#ifndef WINDOWS
250 if (chroot(val))
251 {
252 CERR << "chroot(" << val << ") failed: "
253 << strerror(errno) << endl;
254 exit(a);
255 }
256#endif // WINDOWS
257
258 if (chdir("/"))
259 {
260 CERR << "chdir(\"/\") failed: "
261 << strerror(errno) << endl;
262 exit(a);
263 }
264
265 continue;
266 }
267
268#ifndef WINDOWS
269 if (!strcmp(opt, "-u"))
270 {
271 ++a;
272 if (!val)
273 {
274 CERR << "-u without user ID" << endl;
275 exit(a);
276 }
277 if (setuid(strtoll(val, 0, 10)))
278 {
279 CERR << "setuid(" << val << ") failed: "
280 << strerror(errno) << endl;
281 exit(a);
282 }
283 continue;
284 }
285#endif // WINDOWS
286 }
287
288 return log_startup;
289}

883 loop(a, argc)
884 {
885 original_argv.push_back(argv[a]);
886 expanded_argv.push_back(argv[a]);
887 }
888
889 if (argc <= 1) // program name argv[0] only, hence no arguments to expand
890 {
891 return;
892 }
893
894 if (is_APL_script(argv[1])) // case 2: script without argument
895 {
896 script_argc = 1;
897 return;
898 }
899
900 if (!is_APL_script(argv[2])) return; // not run from a script
901
902 /* at this point:
903
904 argv[1] is the optional argument of the interpreter on the shebang line,
905 argv[2] is the name of the script, and
906 argv[3] ... are the command line arguments given by the user
907 when starting the script.
908 expand argv[1], stopping at --
909 */
910const char * apl_args = argv[1]; // the args after e.g. /usr/bin/apl
911 script_argc = 2;
912 if (!strchr(apl_args, ' ')) // single option
913 {
914 return;
915 }
916
917 // remove expanded_argv[1] and insert the expanded expanded_argv[1] instead
918 expanded_argv.erase(expanded_argv.begin() + 1);
919 --script_argc;
920 for (int index = 1;;)
921 {
922 // skip leading whitespace
923 //
924 while (*apl_args && *apl_args <= ' ') ++apl_args;
925
926 if (*apl_args == 0) break;
927
928 if (!strcmp(apl_args, "--")) // "--" at end of apl_args
929 {
930 expanded_argv.insert(expanded_argv.begin() + index++, "--");
931 ++script_argc;
932 break; // done
933 }
934
935 if (!strncmp(apl_args, "-- ", 2) &&
936 apl_args[2] <= ' ') // "--" somewhere in apl_args
937 {
938 expanded_argv.insert(expanded_argv.begin() + index++, "--");
939 ++script_argc;
940 apl_args += 2; // skip "--"
941 continue;
942 }
943
944 const char * arg_end = strchr(apl_args, ' ');
945 if (arg_end) // more arguments
946 {
947 const int arg_len = arg_end - apl_args;
948 const char * arg = strndup(apl_args, arg_len);
949 expanded_argv.insert(expanded_argv.begin() + index++, arg);
950 ++script_argc;
951 apl_args += arg_len;
952 }
953 else // last argument
954 {
955 const char * arg = strdup(apl_args);
956 expanded_argv.insert(expanded_argv.begin() + index++, arg);
957 ++script_argc;
958 break;
959 }
960 }

244 void set_action(const Token & result)
245 {
246 switch(result.get_Class())
247 {
248 case TC_VALUE:
249 case TC_VOID:
250 case TC_END:
251 case TC_FUN2:
252 action = RA_CONTINUE;
253 return;
254
255 case TC_RETURN:
256 // result was TOK_RETURN_EXEC, TOK_RETURN_STATS,
257 // TOK_RETURN_VOID, or TOK_RETURN_SYMBOL. The current
258 // context is xomplete and may or may not have produced
259 // a value.
260 //
261 action = RA_RETURN;
262 return;
263
264 case TC_SI_LEAVE:
265 //
266 // result was TOK_SI_PUSHED or TOK_ERROR
267 if (result.get_tag() == TOK_ERROR) action = RA_RETURN;
268 else action = RA_SI_PUSHED;
269 return;
270
271 default: CERR << "CLASS = " << result.get_Class()
272 << " at " << LOC << endl;
273 TODO;
274 }

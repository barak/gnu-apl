40//----------------------------------------------------------------------------
41Token
42Quad_GTK::eval_AB(Value_P A, Value_P B) const
43{
44 CHECK_SECURITY(disable_Quad_GTK);
45
46 if (A->get_rank() > 1) RANK_ERROR;
47 if (B->get_rank() > 1) RANK_ERROR;
48 if (A->is_char_array() && B->is_char_array())
49 {
50 const UCS_string css_filename = A->get_UCS_ravel();
51 const UCS_string gui_filename = B->get_UCS_ravel();
52 const int fd = open_window(gui_filename, &css_filename);
53 return Token(TOK_APL_VALUE1, IntScalar(fd, LOC));
54 }
55
56 if (!B->is_int_scalar())
57 {
58 MORE_ERROR() << "A ⎕GTK B expects B to be an integer scalar"
59 " or a text vector";
60 DOMAIN_ERROR;
61 }
62
63const int function = B->get_cfirst().get_int_value();
64int fd = -1;
65 switch(function)
66 {
67 case 0: // close window/GUI
68 if (!A->is_int_scalar()) goto bad_fd;
69 fd = A->get_cfirst().get_int_value();
70 return Token(TOK_APL_VALUE1, close_window(fd));
71
72 case 3: // increase verbosity
73 if (!A->is_int_scalar()) goto bad_fd;
74 fd = A->get_cfirst().get_int_value();
75 if (write_TL0(fd, 7))
76 {
77 CERR << "write(Tag 7) failed in Ah ⎕GTK 3" << endl;
78 return Token(TOK_APL_VALUE1, IntScalar(-3, LOC));
79 }
80 return Token(TOK_APL_VALUE1, IntScalar(0, LOC));
81
82 case 4: // decrease verbosity
83 if (!A->is_int_scalar()) goto bad_fd;
84 fd = A->get_cfirst().get_int_value();
85 if (write_TL0(fd, 8))
86 {
87 CERR << "write(Tag 8) failed in Ah ⎕GTK 4" << endl;
88 return Token(TOK_APL_VALUE1, IntScalar(-4, LOC));
89 }
90 return Token(TOK_APL_VALUE1, IntScalar(0, LOC));
91
92 default: MORE_ERROR() << "Invalid function number Bi=" << function
93 << " in A ⎕GTK Bi";
94 DOMAIN_ERROR;
95 }
96
97 MORE_ERROR() << "Unexpected A or B in A ⎕GTK B";
98 DOMAIN_ERROR;
99
100bad_fd:
101 MORE_ERROR() << "Ah ⎕GTK " << function
102 << " expects a handle (i.e. an integer scalar) Ah";
103 DOMAIN_ERROR;
104}
105//----------------------------------------------------------------------------
106Token
107Quad_GTK::eval_B(Value_P B) const
108{
109 CHECK_SECURITY(disable_Quad_GTK);
110
111 if (B->get_rank() > 1) RANK_ERROR;
112 if (B->is_char_array())
113 {
114 const UCS_string gui_filename = B->get_UCS_ravel();
115 const int fd = open_window(gui_filename, /* no CSS */ 0);
116 return Token(TOK_APL_VALUE1, IntScalar(fd, LOC));
117 }
118
119 if (!B->is_int_scalar())
120 {
121 MORE_ERROR() << "⎕GTK B expects B to be an integer scalar"
122 " or a text vector";
123 DOMAIN_ERROR;
124 }
125
126const int function = B->get_cfirst().get_int_value();
127 switch(function)
128 {
129 case 0: // list of open fds
130 return Token(TOK_APL_VALUE1, window_list());
131
132 case 1: // blocking poll for next event
133 case 2: // non-blocking wait for next event
134 poll_all();
135 if (function == 2 && event_queue.size() == 0) // non-blocking
136 return Token(TOK_APL_VALUE1, IntScalar(0, LOC));
137
138 // blocking
139 //
140 while (event_queue.size() == 0 && !interrupt_is_raised())
141 poll_all();
142
143 // at this point either event_queue.size() > 0 or an interrupt
144 // was raised
145 //
146 if (event_queue.size() == 0) // hence interrupt was raised
147 {
148 clear_interrupt_raised(LOC);
149 return Token(TOK_APL_VALUE1, IntScalar(0, LOC));
150 }
151
152 {
153 UCS_string HWF = event_queue[0]; // fd, widget : fun
154 event_queue.erase(0);
155
156 // split (Unicode)Handle,"widget:function"
157 // into a 3-element APL vector Handle (⊂"widget") (⊂"function")
158 //
159 UCS_string_vector args;
160 UCS_string arg;
161 for (ShapeItem j = 1; j < HWF.size(); ++j)
162 {
163 if (HWF[j] == UNI_COLON)
164 {
165 args.push_back(arg);
166 arg.clear();
167 }
168 else
169 {
170 arg.append(HWF[j]);
171 }
172 }
173 args.push_back(arg);
174
175 Value_P Z(1 + args.size(), LOC);
176 Z->next_ravel_Char(HWF[0]);
177 loop(a, args.size())
178 {
179 Value_P Za(args[a], LOC);
180 Z->next_ravel_Pointer(Za.get());
181 }
182 Z->check_value(LOC);
183 return Token(TOK_APL_VALUE1, Z);
184 }
185
186 default: MORE_ERROR() << "Invalid function number Bi=" << function
187 << " in ⎕GTK Bi";
188 DOMAIN_ERROR;
189 }
190
191 MORE_ERROR() << "Unexpected B in ⎕GTK B";
192 DOMAIN_ERROR;
193}
194//----------------------------------------------------------------------------
195Token
196Quad_GTK::eval_AXB(Value_P A, Value_P X, Value_P B) const
197{
198 CHECK_SECURITY(disable_Quad_GTK);
199
200 if (B->get_rank() > 1) RANK_ERROR;
201
202UTF8_string window_id; // e.g. "entry1"
203const int fd = resolve_window(X.get(), window_id);
204 write_TLV(fd, 6, window_id); // select wodget
205
206int fun = FNUM_INVALID;
207 if (B->is_int_scalar()) fun = B->get_cfirst().get_int_value();
208 else if (B->is_char_string()) fun = resolve_fun_name(window_id, B.get());
209 else
210 {
211 MORE_ERROR() << "A ⎕GTK[X] B expects B to be an integer scalar"
212 " or a text vector";
213 DOMAIN_ERROR;
214 }
215
216int command_tag = -1;
217int response_tag = -1;
218Gtype Atype = gtype_V;
219
220 switch(fun)
221 {
222 case FNUM_INVALID: DOMAIN_ERROR;
223 case FNUM_0: return Token(TOK_APL_VALUE1, IntScalar(fd, LOC));
224
225#define gtk_event_def(ev_ename, ...)
226#define gtk_fun_def(glade_ID, gtk_class, gtk_function, _ZAname,_Z, A,_help) \
227 case FNUM_ ## gtk_class ## _ ## gtk_function: \
228 command_tag = Command_ ## gtk_class ## _ ## gtk_function; \
229 response_tag = Response_ ## gtk_class ## _ ## gtk_function; \
230 Atype = gtype_ ## A; \
231 break;
232#include "Gtk/Gtk_map.def"
233
234 default:
235 MORE_ERROR() << "Bad function B in A ⎕GTK B (B='"
236 << *B << ", fun=" << fun;
237 DOMAIN_ERROR;
238 }
239
240 if (Atype == gtype_V) VALENCE_ERROR;
241 if (A->get_rank() > 1) RANK_ERROR;
242
243UCS_string ucs_A;
244 if (fun == FNUM_GtkDrawingArea_draw_commands)
245 {
246 loop(a, A->element_count())
247 {
248 const Cell & cell = A->get_cravel(a);
249 if (!cell.is_pointer_cell())
250 {
251 MORE_ERROR() << "A ⎕GTK " << fun
252 << " expects A to be a vector of "
253 "draw commands (strings)";
254 DOMAIN_ERROR;
255 }
256 Value_P command = cell.get_pointer_value();
257 ucs_A.append(UCS_string(*command));
258 ucs_A.append(UNI_LF);
259 }
260 }
261 else
262 {
263 if (!A->is_char_string())
264 {
265 MORE_ERROR() << "A ⎕GTK[X] B expects A to be a text vector";
266 DOMAIN_ERROR;
267 }
268
269 ucs_A = UCS_string(*A);
270 }
271UTF8_string utf_A(ucs_A);
272 write_TLV(fd, command_tag, utf_A);
273 return Token(TOK_APL_VALUE1, poll_response(fd, response_tag));
274}

278{
279 CHECK_SECURITY(disable_Quad_GTK);
280
281 if (B->get_rank() > 1) RANK_ERROR;
282
283UTF8_string window_id; // e.g. "entry1"
284const int fd = resolve_window(X.get(), window_id);
285 write_TLV(fd, 6, window_id); // select wodget
286
287int fun = FNUM_INVALID;
288 if (B->is_int_scalar()) fun = B->get_cfirst().get_int_value();
289 else if (B->is_char_string()) fun = resolve_fun_name(window_id, B.get());
290 else DOMAIN_ERROR;
291
292int command_tag = -1;
293int response_tag = -1;
294Gtype Atype = gtype_V;
295
296 switch(fun)
297 {
298 case 0:
299 write_TLV(fd, 4, window_id);
300 return Token(TOK_APL_VALUE1, IntScalar(fd, LOC));
301
302#define gtk_event_def(ev_ename, ...)
303#define gtk_fun_def(glade_ID, gtk_class, gtk_function, _ZAname, _Z, A, _help) \
304 case FNUM_ ## gtk_class ## _ ## gtk_function: \
305 command_tag = Command_ ## gtk_class ## _ ## gtk_function; \
306 response_tag = Response_ ## gtk_class ## _ ## gtk_function; \
307 Atype = gtype_ ## A; \
308 break;
309#include "Gtk/Gtk_map.def"
310 }
311
312 if (Atype != gtype_V) VALENCE_ERROR;
313
314 write_TL0(fd, command_tag); // command
315 return Token(TOK_APL_VALUE1, poll_response(fd, response_tag));
316}
317//----------------------------------------------------------------------------
318Value_P
319Quad_GTK::read_fd(int fd, int tag)
320{
321 // tag == -1 indicates a poll for ANY tag (= an event as opposed to a
322 // command response). In that case the poll is non-blocking.
323
324char TLV[1000];
325
326 errno = 0;
327const ssize_t rx_len = read(fd, TLV, sizeof(TLV));
328
329 if (rx_len < 8)
330 {
331 MORE_ERROR() << "read() failed in Quad_GTK::read_fd(): "
332 << strerror(errno);
333 DOMAIN_ERROR;
334 }
335
336const int TLV_tag = (TLV[0] & 0xFF) << 24 | (TLV[1] & 0xFF) << 16
337 | (TLV[2] & 0xFF) << 8 | (TLV[3] & 0xFF);
338
339
340const int V_len = (TLV[4] & 0xFF) << 24 | (TLV[5] & 0xFF) << 16
341 | (TLV[6] & 0xFF) << 8 | (TLV[7] & 0xFF);
342
343 Assert(rx_len == (V_len + 8));
344
345char * V = TLV + 8;
346 V[V_len] = 0;
347
348 // 1. check for events from generic_callback()
349 //
350 if (TLV_tag == Event_widget_fun || // "H:button1:clicked"
351 TLV_tag == Event_widget_fun_id_class || // dito + :id:class
352 TLV_tag == Event_toplevel_window_done)
353 {
354 // V is a string of the form H%s:%s:%s:% where H is a placeholder
355 // for the fd over which we have received V.
356 //
357 // replace H by the actual fd and store the result in event_queue.
358 //
359 UTF8_string data_utf; // H:widget:callback
360 loop(v, V_len) data_utf += V[v];
361 UCS_string data_ucs(data_utf);
362 data_ucs[0] = (Unicode(fd));
363
364 event_queue.push_back(data_ucs);
365
366 return Value_P(); // i.e. NULL
367 }
368
369 if (tag == -1) // not waiting for a specific tag
370 {
371 CERR << "*** ⎕GTK: ignoring event: " << V << endl;
372 return Value_P(); // i.e. NULL
373 }
374
375 if (TLV_tag != tag)
376 {
377 CERR << "Got unexpected tag " << TLV_tag
378 << " when waiting for response " << tag << endl;
379 return Value_P();
380 }
381
382 // TLV_tag is the expected one. Strip off the response offset
383 // Response_0 (= 2000) from the TLV_tag to get the function number
384 // and return APL vector function,"result-value"
385 //
386UTF8_string utf;
387 loop(u, V_len) utf += V[u];
388 UCS_string ucs(utf);
389
390Value_P Z(1 + ucs.size(), LOC);
391 Z->next_ravel_Int(TLV_tag - Response_0);
392 loop(u, ucs.size()) Z->next_ravel_Char(ucs[u]);
393 Z->check_value(LOC);
394 return Z;
395}
396//----------------------------------------------------------------------------
397void
398Quad_GTK::poll_all()
399{
400const int count = open_windows.size();
401pollfd fds[count];
402 loop(w, count)
403 {
404 fds[w].fd = open_windows[w].fd;
405#ifdef POLLRDHUP
406 fds[w].events = POLLIN | POLLRDHUP;
407#else
408 fds[w].events = POLLIN;
409#endif
410 fds[w].revents = 0;
411 }
412
413 errno = 0;
414const int ready = poll(fds, count, 0);
415 if (ready == 0) return;
416 if (ready > 0)
417 {
418 loop(w, count)
419 {
420 if (fds[w].revents) read_fd(fds[w].fd, -1);
421 }
422 }
423 else
424 {
425 if (errno == EINTR) return; // user hits ^C
426 CERR << "*** poll() failed: " << strerror(errno) << endl;
427 }
428}
429//----------------------------------------------------------------------------
430Value_P
431Quad_GTK::poll_response(int fd, int tag)
432{
433again:
434
435pollfd pfd;
436 pfd.fd = fd;
437#ifdef POLLRDHUP
438 pfd.events = POLLIN | POLLRDHUP;
439#else
440 pfd.events = POLLIN;
441#endif
442 pfd.revents = 0;
443
444const int ready = poll(&pfd, 1, 0);
445 if (ready == 0)
446 {
447 usleep(100000);
448 goto again;
449 }
450
451 if (ready > 0 && pfd.revents)
452 {
453 Value_P Z = read_fd(fd, tag);
454 if (!Z) goto again;
455 return Z;
456 }
457
458 CERR << "*** poll() failed" << endl;
459 DOMAIN_ERROR;
460}
461//----------------------------------------------------------------------------
462Value_P
463Quad_GTK::window_list()
464{
465Value_P Z(open_windows.size(), LOC);
466 loop(w, open_windows.size())
467 {
468 const APL_Integer fd = open_windows[w].fd;
469 Z->next_ravel_Int(fd);
470 }
471
472 Z->set_proto_Int();
473 Z->check_value(LOC);
474 return Z;
475}
476//----------------------------------------------------------------------------
477int
478Quad_GTK::resolve_window(const Value * X, UTF8_string & window_id)
479{
480 if (X->get_rank() > 1) RANK_ERROR;
481const int fd = X->get_cfirst().get_int_value();
482
483 // verify that ↑X is an open window...
484 //
485bool window_valid = false;
486 loop(w, open_windows.size())
487 {
488 if (fd == open_windows[w].fd)
489 {
490 window_valid = true;
491 break;
492 }
493 }
494
495 if (!window_valid)
496 {
497 MORE_ERROR() << "Invalid window " << fd
498 << " in Quad_GTK::resolve_window()";
499 DOMAIN_ERROR;
500 }
501
502 loop(i, X->element_count())
503 {
504 if (i) window_id += X->get_cravel(i).get_char_value();
505 }
506
507 return fd;
508}
509//----------------------------------------------------------------------------
510Quad_GTK::Fnum
511Quad_GTK::resolve_fun_name(UTF8_string & window_id, const Value * B)
512{
513 // window_id is a class and an instance number, such as entry1
514 // Determine the length of the class prefix
515 //
516int wid_len = window_id.size();
517 while (wid_len &&
518 window_id[wid_len - 1] >= '0' &&
519 window_id[wid_len - 1] <= '9') --wid_len;
520
521const UCS_string ucs_B(*B);
522UTF8_string utf_B(ucs_B);
523const char * wid_class = window_id.c_str();
524const char * fun_name = utf_B.c_str();
525
526#define gtk_event_def(ev_ename, ...)
527#define gtk_fun_def(glade_ID, gtk_class, gtk_function, _ZAname,_Z,_A,_help) \
528 if (!(strncmp(wid_class, #glade_ID, wid_len) || \
529 strcmp(fun_name, #gtk_function))) \
530 return FNUM_ ## gtk_class ## _ ## gtk_function;
531
532#include "Gtk/Gtk_map.def"
533
534 MORE_ERROR() << "function string class=" << wid_class
535 << ", function=" << fun_name << " could not be resolved";
536 return FNUM_INVALID;
537}

544//----------------------------------------------------------------------------
545Value_P
546Quad_GTK::close_window(int fd)
547{
548 loop(w, open_windows.size())
549 {
550 window_entry & we = open_windows[w];
551 if (fd == we.fd)
552 {
553 we = open_windows.back();
554 open_windows.pop_back();
555 if (write_TL0(fd, 5))
556 {
557 CERR << "write(close Tag) failed in ⎕GTK::close_window()";
558 }
559 const int err = Quad_FIO::close_handle(fd);
560 return IntScalar(err, LOC);
561 }
562 }
563
564 MORE_ERROR() << "Invalid ⎕GTK handle " << fd;
565 DOMAIN_ERROR;
566}
567//----------------------------------------------------------------------------
568int
569Quad_GTK::write_TL0(int fd, int tag)
570{
571unsigned char TLV[8];
572 memset(TLV, 0, 8);
573 TLV[0] = tag >> 24;
574 TLV[1] = tag >> 16;
575 TLV[2] = tag >> 8;
576 TLV[3] = tag;
577
578 errno = 0;
579const size_t tx_len = write(fd, TLV, 8);
580 if (tx_len != 8)
581 {
582 CERR << "write(Tag = " << tag << ") failed in ⎕GTK::write_TL0()"
583 << endl << " tx_len = " << tx_len << endl
584 << " errno says: " << strerror(errno) << endl;
585 return -1;
586 }
587
588 return 0;
589}
590//----------------------------------------------------------------------------
591int
592Quad_GTK::write_TLV(int fd, int tag, const UTF8_string & value)
593{
594const int TLV_len = 8 + value.size();
595unsigned char TLV[TLV_len];
596 TLV[0] = tag >> 24 & 0xFF;
597 TLV[1] = tag >> 16 & 0xFF;
598 TLV[2] = tag >> 8 & 0xFF;
599 TLV[3] = tag & 0xFF;
600 TLV[4] = value.size() >> 24;
601 TLV[5] = value.size() >> 16;
602 TLV[6] = value.size() >> 8;
603 TLV[7] = value.size();
604 loop(s, value.size()) TLV[8 + s] = value[s];
605
606 if (write(fd, TLV, TLV_len) != TLV_len)
607 {
608 CERR << "write(Tag = " << tag << ") failed in ⎕GTK::write_TLV()";
609 return -1;
610 }
611
612 return 0;
613}
614//----------------------------------------------------------------------------
615int
616Quad_GTK::open_window(const UCS_string & gui_filename,
617 const UCS_string * css_filename)
618{
619 // locate the Gtk_server. It should live in one of two places:
620 //
621 // 1, for an installed apl: in the same directory as apl, or
622 // 2. during development in subdir Gtk of the src directory
623 //
624char path1[APL_PATH_MAX + 1 + 8];
625char path2[APL_PATH_MAX + 1 + 8];
626const char * bin_dir = LibPaths::get_APL_bin_path();
627int slen = snprintf(path1, APL_PATH_MAX, "%s/Gtk/Gtk_server", bin_dir);
628 if (slen >= APL_PATH_MAX) path1[APL_PATH_MAX] = 0;
629 slen = snprintf(path2, APL_PATH_MAX, "%s/Gtk_server", bin_dir);
630 if (slen >= APL_PATH_MAX) path2[APL_PATH_MAX] = 0;
631
632const char * path = 0;
633 if (!access(path1, X_OK)) path = path1;
634 else if (!access(path2, X_OK)) path = path2;
635 else
636 {
637 MORE_ERROR() << "No Gtk_server found in " << path1
638 << "\nor " << path2;
639 DOMAIN_ERROR;
640 }
641
642const char * evars[] = { "DISPLAY", "XAUTHORITY", "XAUTHLOCALHOST" };
643enum { evar_count = sizeof(evars) / sizeof(*evars) };
644
645char * envp[evar_count + 1] = { 0 };
646int envp_idx = 0;
647 loop(c, evar_count)
648 {
649 const char * var = evars[c];
650 if (char * val = getenv(var))
651 {
652 char * env = new char[strlen(var) + 2 + strlen(val)];
653 sprintf(env, "%s=%s", var, val);
654 envp[envp_idx++] = env;
655 envp[envp_idx] = 0;
656 }
657 }
658const int fd = Quad_FIO::do_FIO_57(path, envp);
659
660 // write TLVs 1 and 3 or 1, 2, and 3 to Gtk_server...
661 //
662UTF8_string gui_utf8(gui_filename);
663 slen = snprintf(path1 + 8, APL_PATH_MAX, "%s", gui_utf8.c_str());
664 if (slen >= APL_PATH_MAX) path1[APL_PATH_MAX] = 0;
665 memset(path1, 0, 8);
666 path1[3] = 1; // tag = 1: gui filename
667 path1[6] = gui_utf8.size() >> 8;
668 path1[7] = gui_utf8.size() & 0xFF;
669 slen = write(fd, path1, 8 + gui_utf8.size());
670 if (slen == -1)
671 {
672 Quad_FIO::close_handle(fd);
673 MORE_ERROR() << "write(Tag 1) failed in ⎕GTK";
674 DOMAIN_ERROR;
675 }
676
677 if (css_filename)
678 {
679 UTF8_string css_utf8(*css_filename);
680 memset(path2, 0, 8);
681 slen = snprintf(path2 + 8, APL_PATH_MAX, "%s", css_utf8.c_str());
682 path2[3] = 2; // tag = 2: css filename
683 path2[6] = css_utf8.size() >> 8;
684 path2[7] = css_utf8.size() & 0xFF;
685 slen = write(fd, path2, 8 + css_utf8.size());
686 if (slen == -1)
687 {
688 Quad_FIO::close_handle(fd);
689 MORE_ERROR() << "write(Tag 2) failed in ⎕GTK";
690 DOMAIN_ERROR;
691 }
692 }
693
694 if (write_TL0(fd, 3))
695 {
696 Quad_FIO::close_handle(fd);
697 MORE_ERROR() << "write(Tag 3) failed in ⎕GTK";
698 DOMAIN_ERROR;
699 }
700
701window_entry we = { fd };
702 open_windows.push_back(we);
703 return fd;
704}

38//----------------------------------------------------------------------------
40static bool
41symcomp(const Symbol * const & s1, const Symbol * const & s2, const void *)
42{
43 return s2->get_name().compare(s1->get_name()) == COMP_LT;
44}
45//----------------------------------------------------------------------------
46Doxy::Doxy(ostream & cout, const UCS_string & dest_dir)
47 : out(cout),
48 root_dir(dest_dir),
49 errors(0)
50{
51 ws_name = Workspace::get_WS_name();
52 if (ws_name.compare(UCS_string("CLEAR WS")) == 0)
53 ws_name = UCS_string("CLEAR-WS");
54 else if (Avec::is_digit(ws_name[0]))
55 {
56 // workspace name starts with a library reference number. Skip the
57 // library reference number and whitespace directly after it
58 //
59 int pos = 1; // first blank after the library reference number
60 while (pos < ws_name.size() && Avec::is_white(ws_name[pos])) ++pos;
61 ws_name = UCS_string(ws_name, pos, ws_name.size() - pos);
62 }
63
64 root_dir.append_ASCII("/");
65 root_dir.append_UTF8(UTF8_string(ws_name));
66
67 Log(LOG_command_DOXY)
68 out << "Creating output directory " << root_dir << endl;
69
70 errno = 0;
71 if (mkdir(root_dir.c_str(), 0777))
72 {
73 const char * why = strerror(errno);
74
75 // EEXIST says "File exists" which is a little misleading. We
76 // therefore say ""Directory already exists".
77 //
78 if (errno == EEXIST) why = "Directory already exists";
79
80 CERR << "Cannot create fresh destination directory "
81 << root_dir << ": " << why << endl;
82 ++errors;
83 DOMAIN_ERROR;
84 }
85 write_css();
86}

123//----------------------------------------------------------------------------
124void
125Doxy::gen()
126{
127const SymbolTable & symtab = Workspace::get_symbol_table();
128
129std::vector<const Symbol *> all_symbols = symtab.get_all_symbols();
130std::vector<const Symbol *> variables;
131
132 loop(a, all_symbols.size())
133 {
134 const Symbol * sym = all_symbols[a];
135 if (sym->is_erased()) continue;
136 if (sym->get_name()[0] == UNI_MUE) continue; // macro
137
138 bool is_function = false;
139 bool is_variable = false;
140 loop(si, sym->value_stack_size())
141 {
142 switch((*sym)[si].get_NC())
143 {
144 case NC_VARIABLE: is_variable = true; break;
145 case NC_FUNCTION: is_function = true; break;
146 case NC_OPERATOR: is_function = true; break;
147 default: ;
148 }
149 }
150 if (is_function)
151 all_functions.push_back(sym);
152
153
154 if (is_variable)
155 variables.push_back(sym);
156 }
157
158 if (all_functions.size() > 1)
159 Heapsort<const Symbol *>::
160 sort(&all_functions[0], all_functions.size(), 0, symcomp);
161
162 if (variables.size() > 1)
163 Heapsort<const Symbol *>::
164 sort(&variables[0], variables.size(), 0, symcomp);
165
166UTF8_string index_filename(root_dir);
167 index_filename.append_ASCII("/index.html");
168
169 Log(LOG_command_DOXY)
170 out << "Writing top-level HTML file " << index_filename << endl;
171
172ofstream page(index_filename.c_str());
173 page <<
174"<!DOCTYPE HTML PUBLIC \"-//W3C//DTD HTML 4.01//EN\"" CRLF
175" \"http://www.w3.org/TR/html4/strict.dtd\">" CRLF
176"<HTML>" CRLF
177" <HEAD>" CRLF
178" <TITLE>Documentation of " << ws_name << "</title> " CRLF
179" <META http-equiv=\"content-type\" " CRLF
180" content=\"text/html; charset=UTF-8\">" CRLF
181" <LINK rel='stylesheet' type='text/css' href='apl_doxy.css'>" CRLF
182" </HEAD>" CRLF
183" <BODY>" CRLF
184" <H1>Documentation of workspace " << ws_name << "</H1><HR>" CRLF;
185
186 make_call_graph(all_functions);
187 functions_table(all_functions, page);
188 variables_table(variables, page);
189 SI_table(page);
190
191 set_call_graph_root(0);
192const UCS_string alias = "all_functions";
193 if (write_call_graph(0, alias, false) == 0)
194 {
195 page <<
196" <H3>Global Function Call Graph</H3>" CRLF
197" <IMG class=cg1 usemap=#CG src=cg_" << alias << ".png>" CRLF;
198
199 // copy the cmap file for the IMG (generated by dot) into the page
200 {
201 UTF8_string cmapx_filename(root_dir);
202 cmapx_filename.append_ASCII("/cg_");
203 cmapx_filename.append_UTF8(UTF8_string(alias));
204 cmapx_filename.append_ASCII(".cmapx");
205 if (FILE * cmap = fopen(cmapx_filename.c_str(), "r"))
206 {
207 char buffer[400];
208 for (;;)
209 {
210 const char * s = fgets(buffer, sizeof(buffer), cmap);
211 if (s == 0) break;
212 buffer[sizeof(buffer) - 1] = 0;
213 page << " " << buffer;
214 }
215 fclose(cmap);
216 }
217 else
218 {
219 CERR << "*** Cannot open " << cmapx_filename
220 << ": " << strerror(errno) << endl;
221 }
222
223 Log(LOG_command_DOXY) {} else unlink(cmapx_filename.c_str());
224 }
225 }
226
227 page <<
228" </BODY>" CRLF
229"</HTML>" CRLF;
230 page.close();
231}
232//----------------------------------------------------------------------------
233void
234Doxy::functions_table(const std::vector<const Symbol *> & functions,
235 ofstream & page)
236{
237 if (functions.size() == 0) return;
238
239 page <<
240" <H3>Defined Functions</H3>" CRLF
241" <TABLE class=funtab>" CRLF
242" <TR>" CRLF
243" <TH>Function" CRLF
244" <TH>L" CRLF
245" <TH>SI" CRLF
246" <TH>Lines" CRLF
247" <TH>Header" CRLF
248" <TH class=doxy_comment>Doxy Comments" CRLF;
249
250int total_lines = 0;
251 loop(f, functions.size())
252 {
253 const Symbol * fun_sym = functions[f];
254 loop(si, fun_sym->value_stack_size())
255 {
256 Function_P fp = fun_sym->get_function(si);
257 const UserFunction * ufun = fp->get_func_ufun();
258 if (fp == 0) continue;
259
260 const int si_level = fun_sym->get_SI_level(fp);
261
262 // colunm 1: function name/link
263 //
264 page <<
265" <TR>" CRLF
266" <TD class=code>";
267
268 int line_count = 0;
269 if (fp->is_native())
270 {
271 native_page(fp, fun_sym->get_name());
272 page << fun_anchor(fun_sym->get_name()) << " (native)" CRLF;
273 }
274 else if (fp->is_lambda())
275 {
276 line_count = 1;
277 function_page(ufun, fun_sym->get_name());
278 page << fun_anchor(fun_sym->get_name()) << CRLF;
279 }
280 else if (ufun)
281 {
282 line_count = ufun->get_text_size();
283 const UserFunction * ufun = fp->get_func_ufun();
284 function_page(ufun, ufun->get_name());
285 page << fun_anchor(fun_sym->get_name()) << CRLF;
286 }
287 else
288 {
289 page << fun_sym->get_name() << CRLF;
290 }
291
292 // column 2: local stack depth
293 //
294 page <<
295" <TD class='code center'>" << si << CRLF;
296
297 // column 3: SI level
298 //
299 if (si_level) page <<
300" <TD class='code center'>" << si_level << CRLF;
301 else page <<
302" <TD class='code center'>G" CRLF;
303
304 // column 4: number of lines
305 //
306 page <<
307" <TD class='code center'>" << line_count << CRLF;
308
309 total_lines += line_count;
310 // column 5: function header (if any)
311 //
312 page <<
313" <TD class=code>";
314 if (fp->is_native())
315 page << reinterpret_cast<const NativeFunction *>
316 (fp)->get_so_path();
317 else if (fp->is_lambda())
318 {
319 Assert(ufun);
320 const UCS_string & text = ufun->get_text(1);
321 page << "{" << UCS_string(text, 2, text.size() - 2)
322 << "}" CRLF;
323 }
324 else if (ufun)
325 bold_name(page, ufun);
326 else
327 page << "-";
328 page << CRLF;
329
330 // column 6: Doxygen comments
331 //
332 page <<
333" <TD class=doxy_comment>";
334 if (fp->is_lambda())
335 page << "(named λ)" CRLF;
336 else if (ufun && !fp->is_native())
337 {
338 loop(l, ufun->get_text_size())
339 {
340 UCS_string line = ufun->get_text(l);
341 line.remove_leading_and_trailing_whitespaces();
342 if (line.size() >= 2 &&
343 line[0] == UNI_COMMENT &&
344 line[1] == UNI_COMMENT)
345 page << line.to_HTML(2, false) << CRLF;
346 }
347 }
348 else
349 page << "-" CRLF;
350 }
351 }
352
353 // summary line
354 //
355 page <<
356" <TR><TD><TD colspan=2 class='code center'>Total<TD class='code center'>"
357 << total_lines << CRLF
358" </TABLE>" CRLF;
359}
360//----------------------------------------------------------------------------
361void
362Doxy::variables_table(const std::vector<const Symbol *> & variables,
363 ofstream & page)
364{
365 if (variables.size() == 0) return;
366
367 page <<
368" <H3>Variables</H3>" CRLF
369" <TABLE class=vartab>" CRLF
370" <TR>" CRLF
371" <TH>Variable" CRLF
372" <TH>L" CRLF
373" <TH>SI" CRLF
374" <TH>⍴⍴" CRLF
375" <TH>⍴" CRLF
376" <TH>≡" CRLF
377" <TH>Type" CRLF
378" <TH>↑ϵ" CRLF;
379 loop(v, variables.size())
380 {
381 const Symbol & var_sym = *variables[v];
382 loop(si, var_sym.value_stack_size())
383 {
384 if (var_sym[si].get_NC() != NC_VARIABLE) continue;
385
386 const Value * value = var_sym[si].get_val_cptr();
387 const int si_level = var_sym.get_SI_level(*value);
388 Value_P elem = Bif_F12_ELEMENT::do_eval_B(value);
389 Value_P first = Bif_F12_TAKE::first(*elem.get());
390 page <<
391" <TR>" CRLF
392" <TD class=code>" << var_sym.get_name() << CRLF
393" <TD class='code center'>" << si << CRLF;
394 if (si_level) page <<
395" <TD class='code center'>" << si_level << CRLF;
396 else page <<
397" <TD class='code center'>G" CRLF;
398 page <<
399" <TD class='code center'>" << value->get_rank() << CRLF
400" <TD class='code center'>";
401 if (value->get_rank())
402 {
403 loop(r, value->get_rank())
404 page << " " << value->get_shape_item(r);
405 page << CRLF;
406 }
407 else
408 {
409 page << "⍬" CRLF;
410 }
411
412 page << CRLF
413" <TD class='code center'>" << value->compute_depth() << CRLF;
414
415 const CellType ct = value->deep_cell_types();
416 page <<
417" <TD class=code>";
418 if ((ct & CT_NUMERIC) && (ct & CT_CHAR)) page << "Mixed" CRLF;
419 else if (ct & CT_NUMERIC) page << "Numeric" CRLF;
420 else if (ct & CT_CHAR) page << "Character" CRLF;
421 else page << "? ? ?" CRLF;
422
423 page << CRLF
424" <TD class='code center'>" << *first << CRLF;
425 }
426 }
427 page <<
428" </TABLE>" CRLF;
429}
430//----------------------------------------------------------------------------
431void
432Doxy::SI_table(ofstream & page)
433{
434 // collect SI entries in reverse order...
435 //
436std::vector<const StateIndicator *> stack;
437
438 for (const StateIndicator * si = Workspace::SI_top();
439 si; si = si->get_parent())
440 {
441 stack.push_back(si);
442 }
443
444 if (stack.size() == 0) return;
445
446 page <<
447" <H3>SI Stack</H3>" CRLF
448" <TABLE class=sitab>" CRLF
449" <TR>" CRLF
450" <TH>SI" CRLF
451" <TH>T" CRLF
452" <TH>Fun[line]" CRLF
453" <TH>Statement" CRLF
454" <TH>Local" CRLF
455" <TH>Error" CRLF;
456
457 loop(d, stack.size())
458 {
459 const StateIndicator * si = stack[stack.size() - d - 1];
460 const Executable * exec = si->get_executable();
461 Assert(exec);
462 const UserFunction * ufun = exec->get_exec_ufun(); // possibly 0!
463
464 page <<
465" <TR>" CRLF
466" <TD class='code center'>" << si->get_level() << CRLF
467" <TD class='code center'>" << si->get_parse_mode_name() << CRLF;
468 if (si->get_parse_mode() == PM_FUNCTION)
469 {
470 Assert(ufun);
471 const Function_PC PC = si->get_prefix().get_error_PC();
472 if (const ErrorCode ec =
473 StateIndicator::get_error(si).get_error_code())
474 {
475 // get_error_line_2() is something like fun[line] statement.
476 // find the space after fun[line].
477 //
478 int spc2 = 0;
479 const UCS_string el2(UTF8_string(
480 StateIndicator::get_error(si).get_error_line_2()));
481 loop(e, el2.size())
482 {
483 if (el2[e] == ']')
484 {
485 spc2 = e + 1;
486 break;
487 }
488 }
489
490 int spe2 = spc2;
491 while (spe2 < el2.size() &&
492 el2[spe2] == UNI_SPACE) ++spe2;
493 const int spl2 = el2.size() - spe2;
494 const UCS_string & el3 =
495 StateIndicator::get_error(si).get_error_line_3();
496
497 // put fun[line] into one <TD> and the failed statement into
498 // the next <TD>
499 //
500 page <<
501" <TD class=code>" << UCS_string(el2, 0, spc2) << CRLF
502" <TD class=code>" << UCS_string(el2, spe2, spl2) << BRLF
503" " << el3.to_HTML(spe2, true) << CRLF
504" <TD class=code>>"; if (ufun) ufun->print_local_vars(page);
505 page << CRLF
506" <TD class=code>" << Error::error_name(ec) << CRLF;
507 }
508 else
509 {
510 page <<
511" <TD class=code>" << ufun->get_name_and_line(PC) << CRLF
512" <TD class=code>" << exec->statement_text(PC) << CRLF
513" <TD class=code>"; if (ufun) ufun->print_local_vars(page);
514 page << CRLF;
515 }
516 }
517 else if (si->get_parse_mode() == PM_STATEMENT_LIST)
518 {
519 page <<
520" <TD class='code center'>-" CRLF
521" <TD class=code>" << exec->get_text(0) << CRLF;
522 }
523 else // PM_EXECUTE
524 {
525 page <<
526" <TD class='code center'>-" CRLF
527" <TD class=code>" << exec->get_text(0) << CRLF;
528 }
529 }
530 page <<
531" </TABLE>" CRLF;
532}
533//----------------------------------------------------------------------------
534void
535Doxy::native_page(Function_P fun, const UCS_string & alias)
536{
537 Assert(fun->is_native());
538
539const NativeFunction * nat = reinterpret_cast<const NativeFunction *>(fun);
540UCS_string Z(" <TR><TD class=center>");
541 if (nat->has_result())
542 {
543 Z += UNI_Z;
544 Z += UNI_LEFT_ARROW;
545 }
546
547UTF8_string fun_filename(root_dir);
548 fun_filename.append_ASCII("/f_");
549 fun_filename.append_UTF8(UTF8_string(alias));
550 fun_filename.append_ASCII(".html");
551
552 Log(LOG_command_DOXY)
553 out << "Writing native function HTML file " << fun_filename << endl;
554
555ofstream page(fun_filename.c_str());
556
557 page <<
558"<!DOCTYPE HTML PUBLIC \"-//W3C//DTD HTML 4.01//EN\"" CRLF
559" \"http://www.w3.org/TR/html4/strict.dtd\">" CRLF
560"<HTML>" CRLF
561" <HEAD>" CRLF
562" <TITLE>Documentation of function" << alias << "</TITLE> " CRLF
563" <META http-equiv=\"content-type\" " CRLF
564" content=\"text/html; charset=UTF-8\">" CRLF
565" <LINK rel='stylesheet' type='text/css' href='apl_doxy.css'>" CRLF
566" </HEAD>" CRLF
567" <BODY>" CRLF
568" <TABLE class=h1tab><TR>" CRLF
569" <TD class=h1tab width=10%><H3>↑ <A href=index.html>UP</A></H3>" CRLF
570" <TD class=h1tab width=80%><H1>Native Function " << alias << "</H1>" CRLF
571" <TD class=h1tab width=10%>" CRLF
572" </TABLE><HR>" CRLF
573" <H3>Definition</H3>" CRLF
574"<P class=center>shared library: " << nat->get_so_path() << BRLF
575 BRLF
576" <H3>Supported signatures</H3>" CRLF
577 BRLF
578" <TABLE class=funtab>" CRLF;
579
580 if (nat->f_eval_) page << Z << alias << CRLF;
581 if (nat->f_eval_B) page << Z << alias << " B" CRLF;
582 if (nat->f_eval_AB) page << Z << "A " << alias << " B" CRLF;
583 if (nat->f_eval_LB) page << Z << "(LO " << alias << ") B" CRLF;
584 if (nat->f_eval_ALB) page << Z << "A (LO " << alias << ") B" CRLF;
585 if (nat->f_eval_LRB) page << Z << "(LO " << alias << " RO) B" CRLF;
586 if (nat->f_eval_ALRB) page << Z << "A (LO " << alias << " RO) B" CRLF;
587 if (nat->f_eval_XB) page << Z << alias << "[X] B" CRLF;
588 if (nat->f_eval_AXB) page << Z << "A " << alias << "[X] B" CRLF;
589 if (nat->f_eval_LXB) page << Z << "(LO " << alias << ")[X] B" CRLF;
590 if (nat->f_eval_ALXB) page << Z << "A (LO " << alias << ") B" CRLF;
591 if (nat->f_eval_LRXB) page << Z << "(LO " << alias << " RO)[X] B" CRLF;
592 if (nat->f_eval_ALRXB) page << Z << "A (LO " << alias << " RO)[X] B" CRLF;
593
594 page <<
595" </TABLE>" CRLF;
596
597 swap_caller_calee();
598 set_call_graph_root(fun);
599 if (write_call_graph(fun, alias, true) == 0)
600 {
601 page <<
602" <H3>Caller Graph (defined functions calling native function "
603 << alias << ")</H3>" CRLF
604" <IMG class=cg2 usemap=#GC src=gc_" << alias << ".png>" CRLF;
605
606 // copy the cmap file for the IMG (generated by dot) into the page
607 {
608 UTF8_string cmapx_filename(root_dir);
609 cmapx_filename.append_ASCII("/gc_");
610 cmapx_filename.append_UTF8(UTF8_string(alias));
611 cmapx_filename.append_ASCII(".cmapx");
612 FILE * cmap = fopen(cmapx_filename.c_str(), "r");
613 Assert(cmap);
614 char buffer[400];
615 for (;;)
616 {
617 const char * s = fgets(buffer, sizeof(buffer), cmap);
618 if (s == 0) break;
619 buffer[sizeof(buffer) - 1] = 0;
620 page << " " << buffer;
621 }
622 fclose(cmap);
623 Log(LOG_command_DOXY) {} else unlink(cmapx_filename.c_str());
624 }
625 }
626 swap_caller_calee(); // restore
627
628 page <<
629" </BODY>" CRLF
630" </HTML>" CRLF;
631
632 page.close();
633}
634//----------------------------------------------------------------------------
635void
636Doxy::function_page(const UserFunction * ufun, const UCS_string & alias)
637{
638UTF8_string fun_filename(root_dir);
639 fun_filename.append_ASCII("/f_");
640 fun_filename.append_UTF8(UTF8_string(alias));
641 fun_filename.append_ASCII(".html");
642
643 Log(LOG_command_DOXY)
644 out << "Writing function HTML file " << fun_filename << endl;
645
646ofstream page(fun_filename.c_str());
647 page <<
648"<!DOCTYPE HTML PUBLIC \"-//W3C//DTD HTML 4.01//EN\"" CRLF
649" \"http://www.w3.org/TR/html4/strict.dtd\">" CRLF
650"<HTML>" CRLF
651" <HEAD>" CRLF
652" <TITLE>Documentation of function" << alias << "</TITLE> " CRLF
653" <META http-equiv=\"content-type\" " CRLF
654" content=\"text/html; charset=UTF-8\">" CRLF
655" <LINK rel='stylesheet' type='text/css' href='apl_doxy.css'>" CRLF
656" </HEAD>" CRLF
657" <BODY>" CRLF
658" <TABLE class=h1tab><TR>" CRLF
659" <TD class=h1tab width=10%><H3>↑ <A href=index.html>UP</A></H3>" CRLF
660" <TD class=h1tab width=80%><H1>Function " << alias << "</H1>" CRLF
661" <TD class=h1tab width=10%>" CRLF
662" </TABLE><HR>" CRLF;
663
664 page <<
665" <H3>Definition</H3>" CRLF
666" <TABLE class=onefuntab><TR><TD><PRE>" CRLF
667" ∇ ";
668 loop(l, ufun->get_text_size())
669 {
670 const UCS_string & line(ufun->get_text(l));
671 if (l > 99) page << "[" << l << "]";
672 else if (l > 9) page << "[" << l << "] ";
673 else if (l > 0) page << "[" << l << "] ";
674 page << line.to_HTML(0, false) << CRLF;
675 }
676 page <<
677" ∇</PRE>" CRLF
678" </TABLE>" CRLF;
679
680 set_call_graph_root(ufun);
681 if (write_call_graph(ufun, alias, false) == 0)
682 {
683 page << " <H3>Call Graph (defined functions called from function "
684 << alias << ")</H3>" CRLF
685 " <IMG class=cg1 usemap=#CG src=cg_"

791//----------------------------------------------------------------------------
792void
793Doxy::add_fun_to_call_graph(const Symbol * caller_sym,
794 const UserFunction * ufun)
795{
796 Log(LOG_command_DOXY)
797 out << " add (caller-) Symbol " << caller_sym->get_name() << endl;
798
799const Token_string & body = ufun->get_body();
800
801 // scan the body of ufun for symbol Token and filter those that are
802 // functions. This is rather heuriostic and therefore not reliable.
803 //
804 // It can fail if the same name is used for both functions and variables.
805 //
806 loop(b, body.size())
807 {
808 const Token & tok = body[b];
809 if (tok.get_Class() != TC_SYMBOL) continue;
810
811 const Symbol * callee_ptr = tok.get_sym_ptr();
812 Assert(callee_ptr);
813
814 // assume (maybe incorrectly) that local symbol of ufun are variables.
815 //
816 if (ufun->localizes(callee_ptr)) continue;
817
818 const Symbol & callee_sym = *callee_ptr;
819 loop(si, callee_sym.value_stack_size())
820 {
821 if (Function_P fp = callee_sym.get_function(si))
822 {
823 Assert(fp);
824 const UserFunction * callee = fp->get_func_ufun();
825 if (!(callee || fp->is_native())) continue;
826
827 // ignore multiple calls of the same callee from the
828 // same caller...
829 //
830 bool have_call = false;
831 loop(cg, call_graph.size())
832 {
833 if (call_graph[cg].caller == ufun &&
834 call_graph[cg].callee == fp)
835 {
836 have_call = true;
837 break; // loop(cg, ...
838 }
839 }
840
841 if (!have_call)
842 {
843 const fcall_edge edge(ufun, caller_sym->get_name(),
844 fp, callee_sym.get_name());
845
846 call_graph.push_back(edge);
847 Log(LOG_command_DOXY)
848 out << " " << caller_sym->get_name()
849 << " calls " << fp->get_name() << endl;
850 }
851 }
852 }
853 }

855//----------------------------------------------------------------------------
856void
857Doxy::set_call_graph_root(Function_P fun)
858{
859const int MAX = 2 * call_graph.size(); // MAX means unreachable
860UCS_string root_name("all");
861 if (fun) root_name = fun->get_name();
862
863 if (call_graph.size() == 0)
864 {
865 // there are no calls (all functions are leafs). If the call graph
866 // root is the top-level (i.e. fun == 0), then add all nodes
867 // to nodes.
868 //
869 if (fun == 0)
870 {
871 nodes.clear();
872 aliases.clear();
873 loop(f, all_functions.size())
874 {
875 const Symbol * sym = all_functions[f];
876 Function_P fun = sym->get_function();
877 nodes.push_back(fun);
878 aliases.push_back(fun->get_name());
879 }
880 return;
881 }
882 }
883
884 // 1. set all edges starting at fun to 0 and all others unreachable
885 //
886 loop(cg, call_graph.size())
887 {
888 fcall_edge & edge = call_graph[cg];
889 if (edge.caller == fun) edge.value = 0; // root
890 else if (fun == 0) edge.value = 1; // no root, all edges
891 else edge.value = MAX; // unreachable
892 }
893
894bool progress = true;
895 for (int depth = 0; progress; ++depth)
896 {
897 progress = false;
898 loop(f, call_graph.size())
899 {
900 const fcall_edge & from = call_graph[f];
901 if (from.value != depth) continue;
902
903 // 'from' starts at level depth.
904 // Mark all edges 'to'at level depth + 1
905 //
906 loop(t, call_graph.size())
907 {
908 fcall_edge & to = call_graph[t];
909 if (from.callee != to.caller) continue;
910 if (to.value < (depth + 1)) continue;
911 to.value = depth + 1;
912 progress = true;
913 }
914 }
915 }
916
917 Log(LOG_command_DOXY)
918 loop(e, call_graph.size())
919 {
920 const fcall_edge & edge = call_graph[e];
921 if (edge.value == MAX) continue; // not reachable
922
923 out << "edge " << edge.caller->get_name() << " → "
924 << edge.callee->get_name()
925 << " has distance " << edge.value
926 << " from " << root_name << endl;
927 }
928
929 // create a list of nodes that are reachable
930 //
931 nodes.clear();
932 aliases.clear();
933 loop(e, call_graph.size())
934 {
935 const fcall_edge & edge = call_graph[e];
936 if (edge.value == MAX) continue; // not reachable
937
938 bool caller_is_new = true;
939 bool callee_is_new = true;
940 loop(n, nodes.size())
941 {
942 if (nodes[n] == edge.caller) caller_is_new = false;
943 if (nodes[n] == edge.callee) callee_is_new = false;
944 }
945
946 if (caller_is_new)
947 {
948 nodes.push_back(edge.caller);
949 aliases.push_back(edge.caller_name);
950 }
951 if (callee_is_new)
952 {
953 nodes.push_back(edge.callee);
954 aliases.push_back(edge.callee_name);
955 }
956 }
957
958 Log(LOG_command_DOXY)
959 {
960 loop(n, nodes.size())
961 out << "Node: " << nodes[n]->get_name()
962 << " alias: " << aliases[n] << endl;
963 }
964}
965//----------------------------------------------------------------------------
966int
967Doxy::write_call_graph(Function_P fun, const UCS_string & alias, bool caller)
968{
969UTF8_string cg_filename(root_dir);
970 if (caller) cg_filename.append_ASCII("/gc_");
971 else cg_filename.append_ASCII("/cg_");
972 cg_filename.append_UTF8(UTF8_string(alias));
973UTF8_string png_filename(cg_filename);
974UTF8_string cmapx_filename(cg_filename);
975 cg_filename.append_ASCII(".gv");
976 png_filename.append_ASCII(".png");
977 cmapx_filename.append_ASCII(".cmapx");
978
979 Log(LOG_command_DOXY)
980 out << "Writing function call graph .gv file " << cg_filename << endl;
981
982ofstream gv(cg_filename.c_str());
983
984const char * node_attributes = " shape=rect"
985 " fontcolor=blue"
986 " fillcolor=\"#D0D0D0\"";
987const char * node0_attributes = " shape=rect"
988 " style=filled"
989 " fillcolor=\"#D0D0D0\"";
990
991 if (caller) gv << "digraph GC" << endl;
992 else gv << "digraph CG" << endl;
993
994 gv << "{" << endl
995 << " graph [rankdir=\"LR\"];" << endl;
996
997 if (nodes.size() == 0) // no edges, single node
998 {
999 gv << "n0 [label=" << alias << node0_attributes << "];" << endl;
1000 }
1001 else // normal graph, > 0 edges
1002 {
1003 // create nodes
1004 //
1005 loop(nrev, nodes.size())
1006 {
1007 // graphviz allocates nodes in reverse order, so we write them
1008 // in reverse order (as to display them ordered alphabetically).
1009 //
1010 const ShapeItem n = nodes.size() - nrev - 1;
1011 const UCS_string & alias = aliases[n];
1012 if (nodes[n] == fun) // the root
1013 gv << "n" << n << " [label= " << alias
1014 << node0_attributes << "];" << endl;
1015 else
1016 gv << "n" << n << " [label= <<u>" << alias << "</u>>"
1017 << node_attributes << " URL=\"f_" << alias << ".html\"];"
1018 << endl;
1019 }
1020
1021 // create edges
1022 //
1023 loop(e, call_graph.size())
1024 {
1025 const fcall_edge & edge = call_graph[e];
1026 if (edge.value >= int(call_graph.size())) // not reachable
1027 continue;
1028 const int n0 = node_ID(edge.caller);
1029 const int n1 = node_ID(edge.callee);
1030 Assert(n0 != -1);
1031 Assert(n1 != -1);
1032 if (caller) gv << "n" << n1 << " -> n" << n0 << endl;
1033 else gv << "n" << n0 << " -> n" << n1 << endl;
1034 }
1035 }
1036
1037 gv << "}" << endl;
1038
1039 gv.close();
1040
1041int ret = gv_to_png(cg_filename.c_str(), png_filename.c_str(), false);
1042 if (ret == 0)
1043 ret = gv_to_png(cg_filename.c_str(), cmapx_filename.c_str(), true);
1044 return ret;
1045}

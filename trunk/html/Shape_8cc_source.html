25Shape::Shape(const Value & A, int qio_A)
26 : rho_rho(0),
27 volume(1)
28{
29 // check that A is a shape value, like the left argument of A⍴B
30 //
31 if (A.get_rank() > 1) RANK_ERROR;
32const ShapeItem len_A = A.element_count();
33 if (len_A > MAX_RANK) LIMIT_ERROR_RANK; // of A
34
35 loop(a, len_A)
36 {
37 add_shape_item(A.get_cravel(a).get_near_int() - qio_A);
38 }
39}

41Shape Shape::abs() const
42{
43Shape ret;
44 loop(r, rho_rho)
45 {
46 const ShapeItem len = rho[r];
47 if (len < 0) ret.add_shape_item(- len);

49 }
50
51 return ret;

65void
66Shape::expand(const Shape & B)
67{
68 // increase rank as necessary
69 //
70 expand_rank(B.get_rank());
71
72 // increase axes as necessary
73 //
74 volume = 1;
75 loop(r, rho_rho)
76 {
77 if (rho[r] < B.rho[r]) rho[r] = B.rho[r];
78 volume *= rho[r];
79 }
80}

82Shape
83Shape::insert_axis(sAxis axis, ShapeItem len) const
84{
85 if (get_rank() >= MAX_RANK) LIMIT_ERROR_RANK;
86
87Shape ret;
88
89 if (axis <= 0) // insert before first axis
90 {
91 ret.add_shape_item(len);
92 loop(a, get_rank()) ret.add_shape_item(get_shape_item(a));
93 }
94 else if (uAxis(axis) >= get_rank()) // insert after last axis
95 {
96 loop(a, get_rank()) ret.add_shape_item(get_shape_item(a));
97 ret.add_shape_item(len);
98 }
99 else
100 {
101 loop(a, axis) ret.add_shape_item(get_shape_item(a));
102 ret.add_shape_item(len);
103 loop(a, get_rank() - axis) ret.add_shape_item(get_shape_item(a + axis));
104 }
105
106 return ret;
107}

109ShapeItem
110Shape::ravel_pos(const Shape & idx) const
111{
112ShapeItem p = 0;
113ShapeItem w = 1;
114
115 for (sRank r = get_rank(); r-- > 0;)
116 {
117 p += w*idx.get_shape_item(r);
118 w *= get_shape_item(r);
119 }
120
121 return p;
122}
123//----------------------------------------------------------------------------
124Shape
125Shape::offset_to_index(ShapeItem offset, int quad_io) const
126{
127Shape ret;
128 ret.rho_rho = rho_rho;
129 for (int r = rho_rho; r > 0;)
130 {
131 --r;
132 if (rho[r] == 0)
133 {
134 Assert(offset == 0);
135 ret.rho[r] = 0;
136 continue;
137 }
138
139 ret.rho[r] = quad_io + offset % rho[r];
140 offset /= rho[r];
141 }
142
143 return ret;
144}

158//----------------------------------------------------------------------------
159bool
160Shape::is_permutation() const
161{
162ShapeItem axes = 0; // a bitmap of axes
163
164 loop(r, get_rank())
165 {
166 const ShapeItem ax = get_shape_item(r);
167 if (ax < 0) return false;
168 if (ax >= get_rank()) return false;
169 if (axes & 1 << ax) return false;
170 axes |= 1 << ax;
171 }
172
173 return true;
174}

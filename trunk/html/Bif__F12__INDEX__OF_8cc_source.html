29Token
30Bif_F12_INDEX_OF::eval_B(Value_P B) const
31{
32 if (B->get_rank() > 1) RANK_ERROR;
33
34const APL_Integer qio = Workspace::get_IO();
35const ShapeItem ec = B->element_count();
36
37 if (ec == 1)
38 {
39 // interval (standard ⍳B with scalar or 1-element B)
40 //
41 const APL_Integer len = B->get_cfirst().get_near_int();
42 if (len < 0) DOMAIN_ERROR;
43
44 Value_P Z(len, LOC);
45
46 loop(z, len) Z->next_ravel_Int(qio + z);
47
48 Z->check_value(LOC);
49 return Token(TOK_APL_VALUE1, Z);
50 }
51
52 // generalized ⍳B a la Dyalog APL...
53 //
54 // ⍴B ←→ ⍴⍴Z, B = ⍴Z
55 if (ec == 0)
56 {
57 Value_P Z(LOC);
58 Z->next_ravel_Pointer(Idx0(LOC).get());
59 Z->check_value(LOC);
60 return Token(TOK_APL_VALUE1, Z);
61 }
62
63Shape sh_Z(*B, 0);
64 loop(b, ec) if (sh_Z.get_shape_item(b) < 0) DOMAIN_ERROR;
65
66 // at this point sh is correct and ⍳ cannot fail.
67 //
68Value_P Z(sh_Z, LOC);
69const sRank rk_Z = Z->get_rank();
70 loop(z, Z->element_count())
71 {
72 Value_P ZZ(rk_Z, LOC);
73 ShapeItem N = z;
74 ShapeItem zz[rk_Z];
75 loop(r, rk_Z)
76 {
77 const ShapeItem q = sh_Z.get_shape_item(ec - r - 1);
78 zz[ec - r - 1] = N % q + qio;
79 N /= q;
80 }
81
82 loop(r, rk_Z) ZZ->next_ravel_Int(zz[r]);
83
84 ZZ->check_value(LOC);
85 Z->next_ravel_Pointer(ZZ.get());
86 }
87
88 if (Z->element_count() == 0) // empty result
89 {
90 Value_P ZZ(rk_Z, LOC); // ZZ←(⍴⍴Z)⍴0...
91 while (ZZ->more()) ZZ->next_ravel_0();
92 ZZ->check_value(LOC);
93
94 new (&Z->get_wproto()) PointerCell(ZZ.get(), *Z); // ⊂(⍴⍴Z)⍴0
95 }
96
97 Z->check_value(LOC);
98 return Token(TOK_APL_VALUE1, Z);
99}

102Token
103Bif_F12_INDEX_OF::eval_AB(Value_P A, Value_P B) const
104{
105 // A⍳B (aka. Index of)
106 //
107const bool simple_result = A->is_scalar_or_vector();
108const double qct = Workspace::get_CT();
109const APL_Integer qio = Workspace::get_IO();
110
111const ShapeItem len_A = A->element_count();
112const ShapeItem len_BZ = B->element_count();
113
114Value_P Z(B->get_shape(), LOC);
115
116#if 1
117 // ⎕RL←42 ◊ D←?100 100⍴100 ◊ T←⎕FIO ¯1 ◊ ⊣(⍳100) ⍳ D ◊ -T-⎕FIO ¯1
118
119 if (len_A >= 64 && len_BZ > 5)
120 {
121 // array A is being searched len_BZ times. We therefore reduce the
122 // per-item search time from O(len_A) to O(log(len_A)). That costs us
123 // O(len_A × log(len_A)) for sorting A, but hopefully pays off if
124 // len_BZ > log(len_A).
125 //
126 // We don't do that for too small A though, as to compensate for the
127 // start-up cost of the sorting.
128 //
129 const ShapeItem * Idx_A = Cell::sorted_indices(&A->get_cfirst(), len_A,
130 SORT_ASCENDING, 1);
131 loop(bz, len_BZ)
132 {
133 const APL_Integer z = find_B_in_sorted_A(&A->get_cfirst(),
134 len_A, Idx_A,
135 B->get_cravel(bz), qct);
136
137 if (simple_result) Z->next_ravel_Int(qio + z);
138 else if (z == len_A) // not found: set result item to ⍬
139 {
140 Value_P zilde(ShapeItem(0), LOC);
141 Z->next_ravel_Pointer(zilde.get());
142 }
143 else // element found (first at z (+⎕IO)
144 {
145 const Shape Sz = A->get_shape().offset_to_index(z, qio);
146 Value_P Vz(LOC, &Sz);
147 Z->next_ravel_Pointer(Vz.get());
148 }
149 }
150 delete[] Idx_A;
151 }
152 else
153#endif
154 {
155 loop(bz, len_BZ)
156 {
157 const APL_Integer z = find_B_in_A(&A->get_cfirst(), len_A,
158 B->get_cravel(bz), qct);
159
160 if (simple_result) Z->next_ravel_Int(qio + z);
161 else if (z == len_A) // not found: set result item to ⍬
162 {
163 Value_P zilde(ShapeItem(0), LOC);
164 Z->next_ravel_Pointer(zilde.get());
165 }
166 else // element found (first at z (+⎕IO)
167 {
168 const Shape Sz = A->get_shape().offset_to_index(z, qio);
169 Value_P Vz(LOC, &Sz);
170 Z->next_ravel_Pointer(Vz.get());
171 }
172 }
173 }
174
175 Z->check_value(LOC);
176 return Token(TOK_APL_VALUE1, Z);
177}

179int
180Bif_F12_INDEX_OF::bs_cmp(const Cell & cell, const ShapeItem & A,
181 const void * ctx)
182{
183const Cell * cells_A = reinterpret_cast<const Cell *>(ctx);
184const Cell & cell_A = cells_A[A];
185
186 if (cell_A.is_pointer_cell() && !cell.is_pointer_cell()) return COMP_LT;
187
188const int ret = cell.compare(cell_A);
189 return ret;
190}

192ShapeItem
193Bif_F12_INDEX_OF::find_B_in_sorted_A(const Cell * ravel_A, ShapeItem len_A,
194 const ShapeItem * Idx_A,
195 const Cell & cell_B, double qct)
196{
197const ShapeItem * const posp = Heapsort<ShapeItem>::search<const Cell &>(
198 cell_B, Idx_A, len_A, &bs_cmp, ravel_A);
199 if (!posp) return len_A; // cell_B was not found in ravel A
200
201ShapeItem pos = Idx_A[posp - Idx_A]; // A[pos] = cell_B within qct
202 Assert(cell_B.equal(ravel_A[pos], qct));
203
204 // A[pos] = cell_B, but there could be predecessors of pos that also
205 // satisfy A[pos] = cell_B. Search neighbor with smallest index in A.
206 //
207ShapeItem ret = pos;
208 for (const ShapeItem * posp1 = posp - 1; posp1 >= Idx_A; --posp1)
209 {
210 ShapeItem pos1 = Idx_A[posp1 - Idx_A];
211 const Cell & C1 = ravel_A[pos1];
212 if (!cell_B.equal(C1, qct)) break;
213 if (ret > pos1) ret = pos1;
214 }
215
216 for (const ShapeItem * posp2 = posp + 1; posp2 < (Idx_A + len_A); ++posp2)
217 {
218 ShapeItem pos2 = Idx_A[posp2 - Idx_A];
219 const Cell & C2 = ravel_A[pos2];
220 if (!cell_B.equal(C2, qct)) break;
221 if (ret > pos2) ret = pos2;
222 }
223
224 return ret;
225}

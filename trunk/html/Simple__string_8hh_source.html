130  void insert_before(ShapeItem pos, const T & t)
131  {
132  Assert(pos <= items_valid);
133  if (pos == items_valid)
134  {
135  append(t);
136  return;
137  }
138 
139  extend(items_valid + 1);
140 
141  for (ShapeItem s = items_valid; s > pos; --s) items[s] = items[s - 1];
142  items[pos] = t;
143  ++items_valid;
144  }

167  void erase(ShapeItem pos)
168  {
169  Assert(pos < items_valid);
170 
171  const ShapeItem rest = items_valid - (pos + 1);
172 
173  loop(r, rest)
174  {
175  T * t = items + pos + r;
176  t->~T();
177  new (t) T(items[pos + 1 + r]); // copy the item above
178  }
179  --items_valid;

181  }

223  void allocate(ShapeItem min_size)
224  {
225  Assert1(min_size >= 0);
226  items_valid = min_size;
227  items_allocated = items_valid + ADD_ALLOC; // and a few more
228  if (items_allocated < MIN_ALLOC) items_allocated = MIN_ALLOC;
229  items = new T[items_allocated];
230  }

233  void extend(ShapeItem new_size)
234  {
235  if ((items_allocated - new_size) >= 0) return; // large enough
236 
237  T * old_items = items;
238  items_allocated = new_size + ADD_ALLOC;
239  items = new T[items_allocated];
240  loop(c, items_valid)
241  {
242  new (items + c) T(old_items[c]);
243  }
244  delete [] old_items;
245  }

36CollatingCache::CollatingCache(const Value & A, const Cell * base,
37 ShapeItem clen)
38 : rank(A.get_rank()),
39 base_B1(base),
40 comp_len(clen)
41{
42const ShapeItem ec_A = A.element_count();
43UCS_string UA;
44 UA.reserve(ec_A);
45 loop(a, ec_A) UA.append(A.get_cravel(a).get_char_value());
46
47UCS_string UA1 = UA.unique();
48
49 reserve(UA1.size());
50
51 // create CollatingCacheEntry for every char in UA1. At this point, all
52 // entries are located at the end of A.
53 //
54 loop(a, UA1.size())
55 {
56 const Unicode uni = UA1[a];
57 const CollatingCacheEntry entry(uni, A.get_shape());
58 push_back(entry);
59 }
60
61 // move entries back
62 //
63 loop(a, ec_A)
64 {
65 const Unicode uni = A.get_cravel(a).get_char_value();
66 CollatingCacheEntry & entry = at(find_entry(uni));
67
68 ShapeItem aq = a;
69 loop(r, A.get_rank())
70 {
71 const sAxis axis = entry.ce_shape.get_rank() - r - 1;
72 const ShapeItem ar = aq % A.get_shape_item(axis);
73 Assert(ar <= A.get_shape_item(axis));
74 if (entry.ce_shape.get_shape_item(axis) > ar)
75 entry.ce_shape.set_shape_item(axis, ar);
76 aq /= A.get_shape_item(axis);
77 }
78 }
79
80 // add one entry for all characters in B that are not in A
81 //
82CollatingCacheEntry others(Invalid_Unicode, A.get_shape());
83 push_back(others);
84}

86bool
87CollatingCache::greater_vec(const IntCell & Za, const IntCell & Zb,
88 const void * comp_arg)
89{
90const CollatingCache & cache =
91 *reinterpret_cast<const CollatingCache *>(comp_arg);
92const Cell * ca = cache.base_B1 + cache.comp_len * Za.get_int_value();
93const Cell * cb = cache.base_B1 + cache.comp_len * Zb.get_int_value();
94
95const sRank rank = cache.get_rank();
96
97 loop(r, rank)
98 {
99 loop(c, cache.get_comp_len())
100 {
101 const APL_Integer a = ca[c].get_int_value();
102 const APL_Integer b = cb[c].get_int_value();
103 const int diff = cache[a].compare_axis(cache[b], rank - r - 1);
104
105 if (diff) return diff > 0;
106 }
107 }
108
109 return ca > cb;
110}

112bool
113CollatingCache::smaller_vec(const IntCell & Za, const IntCell & Zb,
114 const void * comp_arg)
115{
116const CollatingCache & cache =
117 *reinterpret_cast<const CollatingCache *>(comp_arg);
118const Cell * ca = cache.base_B1 + cache.comp_len * Za.get_int_value();
119const Cell * cb = cache.base_B1 + cache.comp_len * Zb.get_int_value();
120const sRank rank = cache.get_rank();
121
122 loop(r, rank)
123 {
124 loop(c, cache.get_comp_len())
125 {
126 const APL_Integer a = ca[c].get_int_value();
127 const APL_Integer b = cb[c].get_int_value();
128 const int diff = cache[a].compare_axis(cache[b], rank - r - 1);
129
130 if (diff) return diff < 0;
131 }
132 }
133
134 return ca > cb;
135}

137ShapeItem
138CollatingCache::find_entry(Unicode uni) const
139{
140const CollatingCacheEntry * entries = &at(0);
141const CollatingCacheEntry * entry =
142
143 Heapsort<CollatingCacheEntry>::search<Unicode>(uni, entries, size(),
144 CollatingCacheEntry::compare_chars, 0);
145
146 if (entry) return entry - entries;
147 return size() - 1; // the entry for characters not in A
148}

150Token
151Bif_F12_SORT::sort(Value_P B, Sort_order order)
152{
153 if (B->is_scalar()) return Token(TOK_ERROR, E_RANK_ERROR);
154 if (!B->can_be_compared()) return Token(TOK_ERROR, E_DOMAIN_ERROR);
155
156const ShapeItem len_BZ = B->get_shape_item(0);
157 if (len_BZ == 0) return Token(TOK_APL_VALUE1, Idx0(LOC));
158const ShapeItem comp_len = B->element_count()/len_BZ;
159
160const ShapeItem * indices = Cell::sorted_indices(&B->get_cfirst(),
161 len_BZ, order, comp_len);
162 if (indices == 0) WS_FULL;
163
164Value_P Z(len_BZ, LOC);
165const int qio = Workspace::get_IO();
166
167 loop(a, len_BZ) Z->next_ravel_Int(indices[a] + qio);
168 delete[] indices;
169
170 Z->check_value(LOC);
171 return Token(TOK_APL_VALUE1, Z);
172}
173//----------------------------------------------------------------------------
174Token
175Bif_F12_SORT::sort_collating(Value_P A, Value_P B, Sort_order order)
176{
177 if (A->is_scalar()) RANK_ERROR;
178 if (A->NOTCHAR()) DOMAIN_ERROR;
179
180const APL_Integer qio = Workspace::get_IO();
181 if (B->NOTCHAR()) DOMAIN_ERROR;
182 if (B->is_scalar()) return Token(TOK_APL_VALUE1, IntScalar(qio, LOC));
183
184const ShapeItem len_BZ = B->get_shape_item(0);
185 if (len_BZ == 0) return Token(TOK_APL_VALUE1, Idx0(LOC));
186
187 // first set Z←⍳len_BZ
188 //
189Value_P Z(len_BZ, LOC);
190 loop(l, len_BZ) Z->next_ravel_Int(l + qio);
191 Z->check_value(LOC);
192 if (len_BZ == 1) return Token(TOK_APL_VALUE1, Z);
193
194const ShapeItem ec_B = B->element_count();
195const ShapeItem comp_len = ec_B/len_BZ;
196
197 // create a vector B1 which has the same shape as B, but instead of
198 // B's characters, it has the index of the corresponding CollatingCache
199 // index for each character in B.
200 //
201Value_P B1(B->get_shape(), LOC);
202const Cell * base_B1 = &B1->get_cfirst() - qio*comp_len;
203CollatingCache cache(*A, base_B1, comp_len);
204 loop(b, ec_B)
205 {
206 const Unicode uni = B->get_cravel(b).get_char_value();
207 const APL_Integer b1 = cache.find_entry(uni);
208 B1->next_ravel_Int(b1);
209 }
210 B1->check_value(LOC);
211
212 // then sort Z (actually re-arrange Z so that B[Z] is sorted)
213 //
214IntCell * Z0 = reinterpret_cast<IntCell *>(&Z->get_wfirst());
215 if (order == SORT_ASCENDING)
216 Heapsort<IntCell>::sort(Z0, len_BZ, &cache, &CollatingCache::greater_vec);
217 else
218 Heapsort<IntCell>::sort(Z0, len_BZ, &cache, &CollatingCache::smaller_vec);
219
220 Z->check_value(LOC);
221 return Token(TOK_APL_VALUE1, Z);
222}

36Token
37Bif_SCAN::expand(Value_P A, Value_P B, uAxis axis)
38{
39 // turn scalar B into ,B
40 //
41Shape shape_B = B->get_shape();
42 if (shape_B.get_rank() == 0)
43 {
44 shape_B.add_shape_item(1);
45 axis = 0;
46 }
47 if (axis >= shape_B.get_rank()) INDEX_ERROR;
48
49 if (A->get_rank() > 1) RANK_ERROR;
50 if (shape_B.get_rank() <= axis) RANK_ERROR;
51
52const ShapeItem ec_A = A->element_count();
53ShapeItem ones_A = 0;
54std::vector<ShapeItem> rep_counts;
55 rep_counts.reserve(ec_A);
56 loop(a, ec_A)
57 {
58 APL_Integer rep_A = A->get_cravel(a).get_near_int();
59 rep_counts.push_back(rep_A);
60 if (rep_A == 0) ;
61 else if (rep_A == 1) ++ones_A;
62 else DOMAIN_ERROR;
63 }
64
65Shape shape_Z(shape_B);
66 shape_Z.set_shape_item(axis, ec_A);
67Value_P Z(shape_Z, LOC);
68
69 if (ec_A == 0) // (⍳0)/B :
70 {
71 if (shape_B.get_shape_item(axis) > 1) LENGTH_ERROR;
72
73 Z->set_default(*B.get(), LOC);
74 Z->check_value(LOC);
75 return Token(TOK_APL_VALUE1, Z);
76 }
77
78const Shape3 shape_Z3(shape_Z, axis);
79
80const Cell * cB = &B->get_cfirst();
81const bool lval = cB->is_lval_cell();
82
83ShapeItem inc_1 = shape_Z3.l(); // increment after result l items
84ShapeItem inc_2 = 0; // increment after result m*l items
85
86 if (B->is_scalar() || (shape_B.get_shape_item(axis) == 1
87 && (shape_Z3.l() != 1)))
88 {
89 inc_1 = 0;
90 inc_2 = shape_Z3.l();
91 }
92 else if (ones_A != shape_B.get_shape_item(axis)) LENGTH_ERROR;
93
94 loop(h, shape_Z3.h())
95 {
96 const Cell * fill = cB;
97 loop(m, rep_counts.size())
98 {
99 if (rep_counts[m] == 1) // copy items from B
100 {
101 loop(l, shape_Z3.l()) Z->next_ravel_Cell(cB[l]);
102 cB += inc_1;
103 }
104 else // init items
105 {
106 if (lval)
107 {
108 loop(l, shape_Z3.l()) Z->next_ravel_Lval(0, 0);
109 }
110 else
111 {
112 loop(l, shape_Z3.l()) Z->next_ravel_Proto(fill[l]);
113 }
114 }
115 }
116
117 cB += inc_2;
118 }
119
120 Z->set_default(*B.get(), LOC);
121
122 Z->check_value(LOC);
123 return Token(TOK_APL_VALUE1, Z);
124}

126Token
127Bif_SCAN::scan(Token & tok_LO, Value_P B, uAxis axis) const
128{
129 // if B is a scalar, then Z is B.
130 //
131 if (B->get_rank() == 0) return Token(TOK_APL_VALUE1, CLONE_P(B, LOC));
132
133 if (!tok_LO.is_function())
134 {
135 MORE_ERROR() << "The left argument of operator A /"
136 << get_name() << " is not a function";
137 DOMAIN_ERROR;
138 }
139
140Function_P LO = tok_LO.get_function();
141
142 if (!LO->has_result())
143 {
144 MORE_ERROR() << "The left argument of operator "
145 << get_name() << " is a function that returns no result";
146 DOMAIN_ERROR;
147 }
148
149 if (LO->get_fun_valence() != 2)
150 {
151 MORE_ERROR() << "The left argument of operator "
152 << get_name() << " is a function that is not dyadic";
153 SYNTAX_ERROR;
154 }
155
156 if (axis >= B->get_rank()) AXIS_ERROR;
157
158const ShapeItem m_len = B->get_shape_item(axis);
159
160 if (m_len == 0) return Token(TOK_APL_VALUE1, CLONE_P(B, LOC));
161
162 if (m_len == 1)
163 {
164 const Shape shape_Z = B->get_shape().without_axis(axis);
165 return Bif_F12_RHO::do_reshape(shape_Z, *B);
166 }
167
168const Shape3 shape_Z3(B->get_shape(), axis);
169
170Value_P Z(B->get_shape(), LOC);
171ErrorCode (Cell::*assoc_f2)(Cell *, const Cell *) const = LO->get_assoc();
172
173 if (assoc_f2)
174 {
175 // LO is an associative primitive scalar function.
176 //
177 const Cell * cB = &B->get_cfirst();
178 ShapeItem z = 0;
179 loop(h, shape_Z3.h())
180 loop(m, shape_Z3.m())
181 loop(l, shape_Z3.l())
182 {
183 if (m == 0) // first item in scanned vector
184 {
185 Z->next_ravel_Cell(*cB++);
186 }
187 else // subsequent item in scanned vector
188 {
189 const Cell & prev_Z = Z->get_cravel(z - shape_Z3.l());
190
191 Value_P AA(prev_Z, LOC); // AA is Z[h; m-1; l]
192 Value_P BB(*cB++, LOC); // BB is B[h; m ; l]
193
194 Token tok = LO->eval_AB(AA, BB);
195 if (!tok.is_apl_val()) return tok; // error in AA LO BB
196
197 Z->next_ravel_Cell(tok.get_apl_val()->get_cscalar());
198 }
199 ++z;
200 }
201
202 Z->check_value(LOC);
203 return Token(TOK_APL_VALUE1, Z);
204 }
205
206 // non-trivial reduce (len > 1)
207 //
208const Shape3 Z3(B->get_shape(), axis);
209 if (LO->may_push_SI()) // user defined LO
210 {
211 Value_P X4(4, LOC);
212 X4->next_ravel_Int(axis + Workspace::get_IO());
213 X4->next_ravel_Int(Z3.h());
214 X4->next_ravel_Int(Z3.m());
215 X4->next_ravel_Int(Z3.l());
216 X4->check_value(LOC);
217 return Macro::get_macro(Macro::MAC_Z__LO_SCAN_X4_B)
218 ->eval_LXB(tok_LO, X4, B);
219 }
220
221 if (B->get_shape().is_empty()) return LO->eval_identity_fun(B, axis);
222
223 return Bif_REDUCE::do_reduce(B->get_shape(), Z3, -1, LO, B, m_len);
224}

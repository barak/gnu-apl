210//----------------------------------------------------------------------------
211Token
212Quad_PLOT::eval_AB(Value_P A, Value_P B) const
213{
214 CHECK_SECURITY(disable_Quad_PLOT);
215
216 if (B->get_rank() > 3) RANK_ERROR;
217 if (B->element_count() < 2) LENGTH_ERROR;
218
219 // plot window with default attributes
220 //
221Plot_data * data = setup_data(*B);
222 if (data == 0) DOMAIN_ERROR;
223
224Plot_window_properties * w_props = new Plot_window_properties(data, verbosity);
225 if (w_props == 0)
226 {
227 delete data;
228 WS_FULL;
229 }
230
231 Log(LOG_Quad_PLOT)
232 CERR << "wprops = " << w_props << " created." << endl;
233
234 // from here on 'data' is owned by 'w_props' (whose destructor
235 // will delete it).
236 //
237 if (const ErrorCode ec = parse_attributes(*A, w_props))
238 {
239 delete w_props;
240 throw_apl_error(ec, LOC);
241 }
242 if (w_props->update(verbosity)) { delete w_props; DOMAIN_ERROR; }
243
244 // do_plot_data takes ownership of w_props and will delete w_props
245 //
246 return Token(TOK_APL_VALUE1, do_plot_data(w_props, data));
247}
248//----------------------------------------------------------------------------
249ErrorCode
250Quad_PLOT::parse_attributes(const Value & A, Plot_window_properties * w_props)
251{
252 if (A.is_member()) // new-style attributes
253 {
254 loop(row, A.get_rows())
255 {
256 const Cell & att_name = A.get_cravel(2*row);
257 if (att_name.is_pointer_cell()) // used entry in A
258 {
259 const UCS_string ucs = att_name.get_pointer_value()
260 ->get_UCS_ravel();
261 if (const char * error = w_props->set_attribute(ucs,
262 A.get_cravel(2*row + 1)))
263 {
264 MORE_ERROR() << error << " in ⎕PLOT attribute ." << ucs;
265 return E_DOMAIN_ERROR;
266 }
267 }
268 }
269 return E_NO_ERROR;
270 }
271
272 // old-style attributes
273 //
274 if (A.get_rank() > 1) { delete w_props; RANK_ERROR; }
275
276const ShapeItem len_A = A.element_count();
277 if (len_A < 1) return E_LENGTH_ERROR;
278
279const APL_Integer qio = Workspace::get_IO();
280
281 loop(a, len_A)
282 {
283 const Cell & cell_A = A.get_cravel(a);
284 if (!cell_A.is_pointer_cell())
285 {
286 MORE_ERROR() << "A[" << (a + qio)
287 << "] is not a string in A ⎕PLOT B";
288 return E_DOMAIN_ERROR;
289 }
290
291 const Value * attr = cell_A.get_pointer_value().get();
292 if (!attr->is_char_string())
293 {
294 MORE_ERROR() << "A[" << (a + qio)
295 << "] is not a string in A ⎕PLOT B";
296 return E_DOMAIN_ERROR;
297 }
298
299 UCS_string ucs = attr->get_UCS_ravel();
300 ucs.remove_leading_and_trailing_whitespaces();
301 if (ucs.size() == 0) continue;
302 if (Avec::is_comment(ucs[0])) continue;
303 UTF8_string utf(ucs);
304 if (const char * error = w_props->set_attribute(utf.c_str()))
305 {
306 MORE_ERROR() << error << " in ⎕PLOT attribute '" << ucs << "'";
307 return E_DOMAIN_ERROR;
308 }
309 }
310
311 return E_NO_ERROR;
312}
313//----------------------------------------------------------------------------
314Token
315Quad_PLOT::eval_B(Value_P B) const
316{
317 CHECK_SECURITY(disable_Quad_PLOT);
318
319 if (B->get_rank() == 0 && !B->get_cfirst().is_pointer_cell())
320 {
321 // scalar (integer) argument: plot window control
322 union
323 {
324 APL_Integer B0; // APL
325 const Plot_window_properties * w_props; // gtk
326 const void * vp; // plot_stop() result
327 pthread_t thread; // xcb
328 } u;
329
330 u.B0 = B->get_cscalar().get_int_value();
331 if (u.B0 == 0) // reset plot verbosity
332 {
333 verbosity = 0;
334 CERR << "⎕PLOT verbosity turned off" << endl;
335 return Token(TOK_APL_VALUE1, Idx0(LOC));
336 }
337
338 if (u.B0 == -1) // enable SHOW_EVENTS
339 {
340 verbosity |= SHOW_EVENTS;
341 CERR << "⎕PLOT will show X events " << endl;
342 return Token(TOK_APL_VALUE1, Idx0(LOC));
343 }
344
345 if (u.B0 == -2) // enable SHOW_DATA
346 {
347 verbosity |= SHOW_DATA;
348 CERR << "⎕PLOT will show APL data " << endl;
349 return Token(TOK_APL_VALUE1, Idx0(LOC));
350 }
351
352 if (u.B0 == -3) // close all windows
353 {
354#if HAVE_GTK3
355 Value_P Z(plot_threads.size(), LOC);
356 loop(p, plot_threads.size())
357 {
358 u.B0 = plot_threads[p];
359 Z->next_ravel_Int(u.B0);
360
361 plot_stop(u.w_props);
362 }
363 plot_threads.clear();
364 Z->check_value(LOC);
365 return Token(TOK_APL_VALUE1, Z);
366#else // XCB
367 while (plot_threads.size())
368 {
369 plot_threads.pop_back();
370 }
371
372 // continue below...
373#endif
374 }
375
376 if (u.B0 == -4) // enable SHOW_DRAW
377 {
378 verbosity |= SHOW_DRAW;
379 CERR << "⎕PLOT will show rendering details " << endl;
380 return Token(TOK_APL_VALUE1, Idx0(LOC));
381 }
382
383#if HAVE_GTK3
384
385 u.vp = plot_stop(u.w_props);
386 return Token(TOK_APL_VALUE1, IntScalar(u.B0, LOC));
387
388#else // XCB
389
390 bool found = false;
391 sem_wait(plot_threads_sema);
392 loop(pt, Quad_PLOT::plot_threads.size())
393 {
394 if (Quad_PLOT::plot_threads[pt] != u.thread) continue;
395
396 plot_threads[pt] = plot_threads[plot_threads.size() - 1];
397 plot_threads.pop_back();
398 found = true;
399 break;
400 }
401 sem_post(plot_threads_sema);
402 return Token(TOK_APL_VALUE1, IntScalar(found ? u.B0 : 0, LOC));
403
404#endif // GTK vs. XCB
405 }
406
407 if (B->get_rank() == 1 && B->element_count() == 0)
408 {
409 help();
410 return Token(TOK_APL_VALUE1, Idx0(LOC));
411 }
412
413 if (B->get_rank() > 3) RANK_ERROR;
414
415 // plot window with default attributes
416 //
417Plot_data * data = setup_data(*B);
418 if (data == 0) DOMAIN_ERROR;
419
420Plot_window_properties * w_props = new Plot_window_properties(data, verbosity);
421 if (w_props == 0)
422 {
423 delete data;
424 WS_FULL;
425 }
426
427 return Token(TOK_APL_VALUE1, do_plot_data(w_props, data));
428}
429//----------------------------------------------------------------------------
430Plot_data *
431Quad_PLOT::setup_data(const Value & B)
432{
442const Value * pB = &B;
443 if (B.is_scalar()) // case 2a. → 1a. by disclosing scalar B
444 {
445 if (!B.get_cfirst().is_pointer_cell()) DOMAIN_ERROR;
446 pB = B.get_cfirst().get_pointer_value().get(); // B ← ⊃ B
447 }
448
449 if (pB->get_rank() == 3) return setup_data_3D(*pB);
450 if (!pB->get_cfirst().is_pointer_cell()) return setup_data_2D(*pB);
451 return setup_data_2D_2b(*pB);
452}
453//----------------------------------------------------------------------------
454Plot_data *
455Quad_PLOT::setup_data_2D(const Value & B)
456{
469const ShapeItem cols_B = B.get_cols();
470const ShapeItem rows_B = B.get_rows();
471const ShapeItem len_B = rows_B * cols_B;
472
473 // all items of B shall be simple numbers (integer, real, or complex).
474 loop(b, len_B)
475 {
476 if (!B.get_cravel(b).is_numeric()) return 0;
477 }
478
479 // split B into X=real B, Y=imag Y
480
481double * X = new double[3*len_B];
482double * Y = X + len_B;
483double * Z = Y + len_B;
484 if (!X) WS_FULL;
485
486const APL_Integer qio = Workspace::get_IO();
487 loop(b, len_B)
488 {
489 const Cell & cB = B.get_cravel(b);
490 if (cB.is_complex_cell()) // (x, y)
491 {
492 X[b] = cB.get_real_value();
493 Z[b] = Y[b] = cB.get_imag_value();
494 }
495 else // (N, y)
496 {
497 X[b] = qio + b % cols_B;
498 Z[b] = Y[b] = cB.get_real_value();
499 }
500 }
501
502Plot_data * data = new Plot_data(rows_B);
503 loop(r, rows_B)
504 {
505 const double * pX = X + r*cols_B;
506 const double * pY = Y + r*cols_B;
507 const double * pZ = Z + r*cols_B;
508 const Plot_data_row * pdr = new Plot_data_row(pX, pY, pZ, r, cols_B);
509 data->add_row(pdr);
510 }
511
512 return data;
513}
514//----------------------------------------------------------------------------
515Plot_data *
516Quad_PLOT::setup_data_2D_2b(const Value & B)
517{
525ShapeItem data_points = 0;
526 if (B.get_rank() > 1) RANK_ERROR;
527const ShapeItem rows = B.element_count(); // number of plot rows
528 loop(r, rows)
529 {
530 const Value * vrow = B.get_cravel(r).get_pointer_value().get();
531 if (vrow->get_rank() > 1) RANK_ERROR;
532 const ShapeItem row_len = vrow->element_count();
533 data_points += row_len;
534 loop(rb, row_len)
535 {
536 if (!vrow->get_cravel(rb).is_numeric()) DOMAIN_ERROR;
537 }
538 }
539
540double * X = new double[3*data_points];
541double * Y = X + data_points;
542double * Z = Y + data_points;
543 if (!X) WS_FULL;
544
545ShapeItem idx = 0;
546const APL_Integer qio = Workspace::get_IO();
547
548 loop(r, rows)
549 {
550 const Value * vrow = B.get_cravel(r).get_pointer_value().get();
551 loop(v, vrow->element_count())
552 {
553 const Cell & cB = vrow->get_cravel(v);
554 if (cB.is_complex_cell())
555 {
556 X[idx] = cB.get_real_value();
557 Z[idx] = Y[idx] = cB.get_imag_value();
558 }
559 else
560 {
561 X[idx] = qio + v;
562 Z[idx] = Y[idx] = cB.get_real_value();
563 }
564 ++idx;
565 }
566 }
567
568Plot_data * data = new Plot_data(rows);
569 idx = 0;
570 loop(r, rows)
571 {
572 const double * pX = X + idx;
573 const double * pY = Y + idx;
574 const double * pZ = Z + idx;
575 const Value * vrow = B.get_cravel(r).get_pointer_value().get();
576 const ShapeItem row_len = vrow->element_count();
577 const Plot_data_row * pdr = new Plot_data_row(pX, pY, pZ, r,
578 row_len);
579 data->add_row(pdr);
580 idx += row_len;
581 }
582
583 return data;
584}
585//----------------------------------------------------------------------------
586Plot_data *
587Quad_PLOT::setup_data_3D(const Value & B)
588{
633const ShapeItem planes = B.get_shape_item(0); // (X), Y, and (Z)
634const ShapeItem rows = B.get_shape_item(1);
635const ShapeItem cols = B.get_shape_item(2);
636
637 if (planes < 1 || planes > 3) LENGTH_ERROR;
638 if (rows < 1) LENGTH_ERROR;
639 if (cols < 1) LENGTH_ERROR;
640
641Plot_data * data = 0;
642
643const ShapeItem data_points = rows * cols;
644
645 if (planes == 3) // X, Y, and Z
646 {
647 double * X = new double[3*data_points];
648 double * Y = X + data_points;
649 double * Z = Y + data_points;
650 if (!X) WS_FULL;
651
652 data = new Plot_data(rows);
653 loop(r, rows)
654 {
655 loop(c, cols)
656 {
657 const ShapeItem p = c + r*cols;
658 const Cell & cX = B.get_cravel(p);
659 const Cell & cY = B.get_cravel(p + data_points);
660 const Cell & cZ = B.get_cravel(p + 2*data_points);
661
662 if (!(cX.is_integer_cell() ||
663 cX.is_real_cell())) DOMAIN_ERROR;
664 if (!(cY.is_integer_cell() ||
665 cY.is_real_cell())) DOMAIN_ERROR;
666 if (!(cZ.is_integer_cell() ||
667 cZ.is_real_cell())) DOMAIN_ERROR;
668
669 X[p] = cX.get_real_value();
670 Y[p] = cY.get_real_value();
671 Z[p] = cZ.get_real_value();
672 }
673 const double * pX = X + r*cols;
674 const double * pY = Y + r*cols;
675 const double * pZ = Z + r*cols;
676 const Plot_data_row * pdr = new Plot_data_row(pX, pY, pZ,
677 r, cols);
678 data->add_row(pdr);
679 }
680 }
681 else if (planes == 2) // X and Y, but no Z
682 {
683 double * X = new double[2*data_points];
684 double * Y = X + data_points;
685 if (!X) WS_FULL;
686
687 data = new Plot_data(rows);
688 loop(r, rows)
689 {
690 loop(c, cols)
691 {
692 const ShapeItem p = c + r*cols;
693 const Cell & cX = B.get_cravel(p);
694 const Cell & cY = B.get_cravel(p + data_points);
695
696 if (!(cX.is_integer_cell() ||
697 cX.is_real_cell())) DOMAIN_ERROR;
698 if (!(cY.is_integer_cell() ||
699 cY.is_real_cell())) DOMAIN_ERROR;
700
701 X[p] = cX.get_real_value();
702 Y[p] = cY.get_real_value();
703 }
704 const double * pX = X + r*cols;
705 const double * pY = Y + r*cols;
706 const Plot_data_row * pdr = new Plot_data_row(pX, pY, 0, r, cols);
707 data->add_row(pdr);
708 }
709 }
710 else // Y, but no X and no Z
711 {
712 double * Y = new double[data_points];
713 if (!Y) WS_FULL;
714
715 data = new Plot_data(rows);
716 loop(r, rows)
717 {
718 loop(c, cols)
719 {
720 const ShapeItem p = c + r*cols;
721 const Cell & cY = B.get_cravel(p);
722
723 if (!(cY.is_integer_cell() ||
724 cY.is_real_cell())) DOMAIN_ERROR;
725
726 Y[p] = cY.get_real_value();
727 }
728 const double * pY = Y + r*cols;
729 const Plot_data_row * pdr = new Plot_data_row(0, pY, 0, r, cols);
730 data->add_row(pdr);
731 }
732 }
733
734 data->surface = true;
735 return data;
736}
737//----------------------------------------------------------------------------
738Value_P
739Quad_PLOT::do_plot_data(Plot_window_properties * w_props,
740 const Plot_data * data)
741{
742 w_props->set_verbosity(verbosity);
743 verbosity > 0 && w_props->print(CERR);
744
745 // check (possibly again) for empty plot ranges which could be caused
746 // by bad plot data but also by bad properties.
747 //
748 if (w_props->get_min_X() >= w_props->get_max_X())
749 {
750 MORE_ERROR() << "empty X range in A ⎕PLOT B";
751 DOMAIN_ERROR;
752 }
753 if (w_props->get_min_Y() >= w_props->get_max_Y())
754 {
755 MORE_ERROR() << "empty Y range in A ⎕PLOT B";
756 DOMAIN_ERROR;
757 }
758
759 if (w_props->get_min_Z() >= w_props->get_max_Z())
760 {
761 MORE_ERROR() << "empty Z range in A ⎕PLOT B";
762 DOMAIN_ERROR;
763 }
764
765union
766{
767 pthread_t thread; // xcb
768 const Plot_window_properties * w_props; // gtk
769 APL_Integer ret; // APL
770} u;
771 u.ret = 0;
772
773#if HAVE_GTK3 // GTK
774 u.w_props = w_props;
775 plot_main(w_props);
776#else // XCB
777 pthread_create(&u.thread, 0, plot_main, w_props);
778#endif
779
780 sem_wait(plot_threads_sema);
781 plot_threads.push_back(u.thread);
782 sem_post(plot_threads_sema);
783
784 sem_wait(plot_window_sema); // blocks until window shown
785 return IntScalar(u.ret, LOC);
786}
787//----------------------------------------------------------------------------
788static UCS_string
789fill_14(const std::string & name)
790{
791UTF8_string name_utf(name.c_str());
792UCS_string ret(name_utf);
793 while (ret.size() < 14) ret += UNI_SPACE;
794 return ret;
795}

797void
798Quad_PLOT::help()
799{
800 CERR <<
801"\n"
802" ⎕PLOT Usage:\n"
803"\n"
804" ⎕PLOT B plot B with default attribute values\n"
805" A ⎕PLOT B plot B with attributes specified by A\n"
806"\n"
807" A is a nested vector of strings.\n"
808" Each string A[i] has the form \"Attribute: Value\"\n"
809" Colors are specified either as #RGB or as #RRGGBB or as RR GG BB)\n"
810"\n"
811" The attributes understood by ⎕PLOT and their default values are:\n"
812"\n"
813" 1. Global (plot window) Attributes:\n"
814"\n";
815
816 CERR << left;

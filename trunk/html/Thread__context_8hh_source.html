99 while (atomic_read(busy_worker_count) != 0) /* busy wait */ ;

103 void PF_join()
104 {
105 atomic_add(busy_worker_count, -1); // we are ready
106 ++job_number; // we reached master job_number
107
108 // wait until all workers finished or new job from master
109 while (atomic_read(busy_worker_count) != 0 &&
110 get_master().job_number == job_number)
111 /* busy wait */ ;
112 }

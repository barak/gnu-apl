33Token
34Bif_OPER1_EACH::eval_ALB(Value_P A, Token & _LO, Value_P B) const
35{
36 // dyadic EACH: call _LO for corresponding items of A and B
37
38 if (!A->same_shape(*B))
39 {
40 // if the shapes differ then either A or B must be a scalar or
41 // 1-element vector.
42 //
43 if (A->get_rank() != B->get_rank())
44 {
45 if (A->is_scalar_or_len1_vector()) ; // OK
46 else if (B->is_scalar_or_len1_vector()) ; // OK
47 else if (A->get_rank() != B->get_rank()) RANK_ERROR;
48 else LENGTH_ERROR;
49 }
50 }
51
52Function_P LO = _LO.get_function();
53 Assert1(LO);
54
55 // for the ambiguous operators /. ⌿, \, and ⍀ is_operator() returns true,
56 // which is incorrect in this context. We use get_signature() instead.
57 //
58 if ((LO->get_signature() & SIG_DYA) != SIG_DYA) VALENCE_ERROR;
59
60 if (A->is_empty() || B->is_empty())
61 {
62 if (!LO->has_result()) return Token(TOK_VOID);
63
64 Value_P First_A = Bif_F12_TAKE::first(*A);
65 Value_P First_B = Bif_F12_TAKE::first(*B);
66 Shape shape_Z; // will be ⍴A or ⍴B and therefore empty
67
68 if (A->is_empty()) shape_Z = A->get_shape();
69 else if (!A->is_scalar()) DOMAIN_ERROR;
70
71 if (B->is_empty()) shape_Z = B->get_shape();
72 else if (!B->is_scalar()) DOMAIN_ERROR;
73
74 Value_P Z1 = LO->eval_fill_AB(First_A, First_B).get_apl_val();
75 Value_P Z(shape_Z, LOC);
76 if (Z1->is_simple_scalar())
77 Z->set_ravel_Cell(0, Z1->get_cscalar());
78 else // need to encose Z1
79 Z->set_ravel_Pointer(0, Z1.get());
80 Z->check_value(LOC);
81 return Token(TOK_APL_VALUE1, Z);
82 }
83
84 if (LO->may_push_SI()) // user defined LO
85 {
86 const bool extend_A = A->is_scalar_or_len1_vector() && !B->is_scalar();
87 const bool extend_B = B->is_scalar_or_len1_vector();
88
89 Macro * macro = 0;
90 if (LO->has_result())
91 {
92 if (extend_A)
93 {
94 macro = Macro::get_macro(extend_B
95 ? Macro::MAC_Z__sA_LO_EACH_sB
96 : Macro::MAC_Z__sA_LO_EACH_vB);
97 }
98 else
99 {
100 macro = Macro::get_macro(extend_B
101 ? Macro::MAC_Z__vA_LO_EACH_sB
102 : Macro::MAC_Z__vA_LO_EACH_vB);
103 }
104 }
105 else // LO has no result, so we can ignore the shape of the result
106 {
107 if (extend_A && extend_B)
108 {
109 macro = Macro::get_macro(Macro::MAC_sA_LO_EACH_sB);
110 }
111
112 if (extend_B)
113 {
114 macro = Macro::get_macro(Macro::MAC_vA_LO_EACH_sB);
115 }
116 else if (extend_A)
117 {
118 macro = Macro::get_macro(Macro::MAC_sA_LO_EACH_vB);
119 }
120 else
121 {
122 macro = Macro::get_macro(Macro::MAC_vA_LO_EACH_vB);
123 }
124 }
125
126 if (extend_A && !A->is_scalar()) // 1-element non-scalar A
127 {
128 if (extend_B && !B->is_scalar()) // 1-element non-scalar B
129 {
130 Value_P A1(LOC); // A1 ← , A
131 A1->get_wscalar().init(A->get_cscalar(), *A1, LOC);
132 A1->check_value(LOC);
133
134 Value_P B1(LOC); // B1 ← , B
135 B1->get_wscalar().init(B->get_cscalar(), *B1, LOC);
136 B1->check_value(LOC);
137
138 return macro->eval_ALB(A1, _LO, B1);
139 }
140 else
141 {
142 Value_P A1(LOC);
143 A1->get_wscalar().init(A->get_cfirst(), *A1, LOC);
144 A1->check_value(LOC);
145
146 return macro->eval_ALB(A1, _LO, B);
147 }
148 }
149 else if (extend_B && !B->is_scalar()) // 1-element non-scalar B
150 {
151 Value_P B1(LOC);
152 B1->get_wscalar().init(B->get_cfirst(), *B1, LOC);
153 B1->check_value(LOC);
154
155 return macro->eval_ALB(A, _LO, B1);
156 }
157 else
158 {
159 return macro->eval_ALB(A, _LO, B);
160 }
161 }
162
163
164 // use the same scheme as ScalarFunction::do_scalar_AB() to determine
165 // the shape of the result. In order to detect conformity errors we compute
166 // shape_Z even if no result is returned.
167 //
168const int inc_A = A->get_increment();
169const int inc_B = B->get_increment();
170
171const Shape * shape_Z = 0;
172 if (A->is_scalar()) shape_Z = &B->get_shape();
173 else if (B->is_scalar()) shape_Z = &A->get_shape();
174 else if (inc_A == 0) shape_Z = &B->get_shape();
175 else if (inc_B == 0) shape_Z = &A->get_shape();
176 else if (A->same_shape(*B)) shape_Z = &B->get_shape();
177 else // error
178 {
179 if (!A->same_rank(*B)) RANK_ERROR;
180 else LENGTH_ERROR;
181 }
182
183const ShapeItem len_Z = shape_Z->get_volume();
184Value_P Z;
185 if (LO->has_result()) Z = Value_P(*shape_Z, LOC);
186
187 loop(z, len_Z)
188 {
189 const Cell * cA = &A->get_cravel(inc_A * z);
190 const Cell * cB = &B->get_cravel(inc_B * z);
191 const bool left_val = cB->is_lval_cell();
192 Value_P LO_A = cA->to_value(LOC); // left argument of LO
193 Value_P LO_B = cB->to_value(LOC); // right argument of LO;
194 if (left_val)
195 {
196 Cell * dest = cB->get_lval_value();
197 if (dest->is_pointer_cell())
198 {
199 Value_P sub = dest->get_pointer_value();
200 LO_B = sub->get_cellrefs(LOC);
201 }
202 }
203
204 Token result = LO->eval_AB(LO_A, LO_B);
205
206 // if LO was a primitive function, then result may be a value.
207 // if LO was a user defined function then result may be TOK_SI_PUSHED.
208 // in both cases result could be TOK_ERROR.
209 //
210 if (result.get_Class() == TC_VALUE)
211 {
212 Value_P vZ = result.get_apl_val();
213
214 if (vZ->is_simple_scalar() || (left_val && vZ->is_scalar()))
215 Z->next_ravel_Cell(vZ->get_cfirst());
216 else
217 Z->next_ravel_Pointer(vZ.get());
218
219 continue; // next z
220 }
221
222 if (result.get_tag() == TOK_ERROR) return result;
223
224 Q1(result); FIXME;
225 }
226
227 if (!Z) return Token(TOK_VOID); // LO without result
228
229 Z->set_default(*B.get(), LOC);
230 Z->check_value(LOC);
231 return Token(TOK_APL_VALUE1, Z);
232}

234Token
235Bif_OPER1_EACH::do_eval_LB(Token & _LO, Value_P B)
236{
237 // monadic EACH: call _LO for every item of B
238
239Function_P LO = _LO.get_function();
240 Assert1(LO);
241
242 if (LO->is_operator() && !is_SLASH_or_BACKSLASH(_LO.get_tag())) SYNTAX_ERROR;
243 if (!(LO->get_signature() & SIG_B)) VALENCE_ERROR;
244
245 if (B->is_empty())
246 {
247 if (!LO->has_result()) return Token(TOK_VOID);
248
249 Value_P First_B = Bif_F12_TAKE::first(*B);
250 Token tZ = LO->eval_fill_B(First_B);
251 Value_P Z1 = tZ.get_apl_val();
252
253 if (Z1->is_simple_scalar())
254 {
255 Z1->set_shape(B->get_shape());
256 return Token(TOK_APL_VALUE1, Z1);
257 }
258
259 Value_P Z(B->get_shape(), LOC);
260 Z->set_ravel_Pointer(0, Z1.get());
261 Z->check_value(LOC);
262 return Token(TOK_APL_VALUE1, Z);
263 }
264
265 if (LO->may_push_SI()) // user defined LO
266 {
267 if (!LO->has_result())
268 return Macro::get_macro(Macro::MAC_LO_EACH_B)->eval_LB(_LO, B);
269
270 if (LO == Bif_F1_EXECUTE::fun)
271 return Macro::get_macro(Macro::MAC_Z__EXEC_EACH_B)->eval_B(B);
272
273 return Macro::get_macro(Macro::MAC_Z__LO_EACH_B)->eval_LB(_LO, B);
274 }
275
276const ShapeItem len_Z = B->element_count();
277Value_P Z;
278 if (LO->has_result()) Z = Value_P(B->get_shape(), LOC);
279
280 loop (z, len_Z)
281 {
282 if (LO->get_fun_valence() == 0)
283 {
284 // we allow niladic functions N so that one can loop over them with
285 // N ¨ 1 2 3 4
286 //
287 Token result = LO->eval_();
288
289 if (result.get_Class() == TC_VALUE)
290 {
291 Value_P vZ = result.get_apl_val();
292 if (vZ->is_simple_scalar())
293 Z->next_ravel_Cell(vZ->get_cfirst());
294 else
295 Z->next_ravel_Pointer(vZ.get());
296
297 continue; // next z
298 }
299
300 if (result.get_tag() == TOK_VOID) continue; // next z
301
302 if (result.get_tag() == TOK_ERROR) return result;
303
304 Q1(result); FIXME;
305 }
306 else
307 {
308 const Cell * cB = &B->get_cravel(z);
309 const bool left_val = cB->is_lval_cell();
310 Value_P LO_B = cB->to_value(LOC); // right argument of LO
311
312 if (left_val)
313 {
314 Cell * dest = cB->get_lval_value();
315 if (dest->is_pointer_cell())
316 {
317 Value_P sub = dest->get_pointer_value();
318 LO_B = sub->get_cellrefs(LOC);
319 }
320 }
321
322 Token result = LO->eval_B(LO_B);
323 if (result.get_Class() == TC_VALUE)
324 {
325 Value * vZ = result.get_apl_val().get();
326
327 if (vZ->is_simple_scalar() || (left_val && vZ->is_scalar()))
328 Z->next_ravel_Cell(vZ->get_cfirst());
329 else
330 Z->next_ravel_Pointer(vZ);
331
332 continue; // next z
333 }
334
335 if (result.get_tag() == TOK_VOID) continue; // next z
336
337 if (result.get_tag() == TOK_ERROR) return result;
338
339 Q1(result); Q1(*LO) FIXME;
340 }
341 }
342
343 if (!Z) return Token(TOK_VOID); // LO without result
344
345 Z->set_default(*B.get(), LOC);
346 Z->check_value(LOC);
347 return Token(TOK_APL_VALUE1, Z);
348}

56{
57public:
59 UTF8_string()

63 UTF8_string(const char * str)
64 { while (*str) *this += *str++; }
65
67 UTF8_string(const UTF8 * str, size_t len)
68 {
69 loop(l, len)
70 if (*str) *this += *str++;
71 else break;
72 }
73
76 UTF8_string(const UCS_string & ucs);
77
79 bool operator ==(const UTF8_string & other) const
80 {
81 if (size() != other.size()) return false;
82 loop(c, size()) if (at(c) != other.at(c)) return false;
83 return true;
84 }
85
87 const char * c_str() const
88 { return reinterpret_cast<const char *>

106 { while (*ascii) *this += *ascii++; }

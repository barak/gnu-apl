65 Shape(const Shape & other)
66 : rho_rho(other.rho_rho),
67 volume(other.volume)
68 {
69 /* this may trigger a somewhat bogus -Wmaybe-uninitialized warning but
70 makes some workspaces 3 times faster. The slower code with no warning
71 would be:
72
73 loop(r, rhorho) rho[r] = other.rho[r];
74
75 but that prevents loop-unrolling.
76 */
77 loop(r, MAX_RANK) rho[r] = other.rho[r];

139 ShapeItem count = 1;

144 void set_shape_item(sAxis r, ShapeItem sh)
145 { Assert(r < rho_rho);
146 if (rho[r]) { volume /= rho[r]; rho[r] = sh; volume *= rho[r]; }
147 else { rho[r] = sh; recompute_volume(); } }
148
150 void recompute_volume()
151 { volume = 1; loop(r, rho_rho) volume *= rho[r]; }
152
154 void increment_shape_item(sRank r)
155 { set_shape_item(r, rho[r] + 1); }
156
158 void set_last_shape_item(ShapeItem sh)
159 { set_shape_item(rho_rho - 1, sh); }
160
162 void add_shape_item(ShapeItem len)
163 { if (rho_rho >= MAX_RANK) LIMIT_ERROR_RANK;
164 rho[rho_rho++] = len; volume *= len; }
165
167 void expand_rank(sRank new_rank)
168 { if (rho_rho >= new_rank) return; // no need to expand
169 const int diff = new_rank - rho_rho;
170 for (sRank r = rho_rho - 1; r >= 0; --r) rho[r + diff] = rho[r];
171 loop(r, diff) rho[r] = 1;
172 rho_rho = new_rank;
173 }

219 { Assert(rho_rho == value.rho_rho);

224 Shape get_weights() const
225 { Shape ret; ret.rho_rho = rho_rho;
226 ShapeItem weight = 1;
227 ret.volume = 1;
228 for (sRank r = rho_rho - 1; r >= 0; --r)
229 {
230 ret.rho[r] = weight;
231 ret.volume *= weight;

233 }
234 return ret;

266class Shape3 : public Shape
267{
268public:
270 Shape3(const Shape & shape, ShapeItem axis)
271 : Shape(1, 1, 1)
272 {
273 loop(r, shape.get_rank())
274 {
275 if (r < axis) rho[0] *= shape.get_shape_item(r);
276 else if (r > axis) rho[2] *= shape.get_shape_item(r);
277 else rho[1] = shape.get_shape_item(r);
278 }
279 }
280
282 Shape3(ShapeItem _h, ShapeItem _m, ShapeItem _l)
283 : Shape(_h, _m, _l)
284 {}
285
287 ShapeItem h() const
288 { return rho[0]; }
289
291 ShapeItem hml(ShapeItem hh, ShapeItem mm, ShapeItem ll) const
292 { return ll + l()*(mm + m()*hh); }
293
295 ShapeItem m() const
296 { return rho[1]; }
297
299 ShapeItem l() const
300 { return rho[2]; }

38void
39divide_matrix(Value & Z, bool need_complex,
40 ShapeItem rows, ShapeItem cols_A, const Cell * cA,
41 ShapeItem cols_B, const Cell * cB)
42{
43 // the following has been checked by the caller:
44 //
45 // rows_B >= cols_B and
46 // rows_A == rows_B (aka. rows below)
47 //
48 loop(c, cols_A)
49 {
50 if (need_complex)
51 {
52 APL_Float * ad = reinterpret_cast<APL_Float *>(
53 malloc(rows * cols_A * 2*sizeof(APL_Float)));
54 if (ad == 0) WS_FULL
55 ZZ * const a = reinterpret_cast<ZZ *>(ad);
56 loop(r, rows)
57 {
58 new (a + r) ZZ(cA[r*cols_A + c].get_real_value(),
59 cA[r*cols_A + c].get_imag_value());
60 }
61
62 APL_Float * bd = reinterpret_cast<APL_Float *>(
63 malloc(rows * cols_B * 2*sizeof(APL_Float)));
64 if (bd == 0) { free(a); WS_FULL }
65
66 ZZ * const b = reinterpret_cast<ZZ *>(bd);
67 ZZ * bb = b;
68 loop(rr, cols_B)
69 loop(cc, rows)
70 {
71 new (bb++) ZZ(cB[rr + cc*cols_B].get_real_value(),
72 cB[rr + cc*cols_B].get_imag_value());
73 }
74
75 {
76 const APL_Float rcond = Workspace::get_CT();
77
78 Matrix<ZZ> B(b, rows, cols_B, /* LDB */ rows);
79 Matrix<ZZ> A(a, rows, 1, /* LDA */ rows);
80 const ShapeItem rk = gelsy<ZZ>(B, A, rcond);
81 free(bd);
82 if (rk != cols_B)
83 {
84 free(ad);
85 DOMAIN_ERROR;
86 }
87 }
88
89 // cols_A = rows_Z. We have computed the result for col c of A
90 // which is row c of Z.
91 //
92 loop(r, cols_B)
93 Z.set_ravel_Complex(r*cols_A + c, a[r].real(), a[r].imag());
94 free(ad);
95 }
96 else // real
97 {
98 APL_Float * a = reinterpret_cast<APL_Float *>(
99 malloc(rows * cols_A * sizeof(APL_Float)));
100 if (a == 0) WS_FULL
101 loop(r, rows)
102 {
103 a[r] = cA[r*cols_A + c].get_real_value();
104 }
105
106 APL_Float * b = reinterpret_cast<APL_Float *>(
107 malloc(rows * cols_B * sizeof(APL_Float)));
108 if (b == 0) { free(a); WS_FULL }
109
110 APL_Float * bb = b;
111 loop(rr, cols_B)
112 loop(cc, rows)
113 {
114 *bb++ = cB[rr + cc*cols_B].get_real_value();
115 }
116
117 {
118 const APL_Float rcond = Workspace::get_CT();
119
120 Matrix<DD> B(b, rows, cols_B, /* LDB */ rows);
121 Matrix<DD> A(a, rows, 1, /* LDA */ rows);
122 const ShapeItem rk = gelsy<DD>(B, A, rcond);
123 free(b);
124 if (rk != cols_B)
125 {
126 free(a);
127 DOMAIN_ERROR;
128 }
129 }
130
131 // cols_A = rows_Z. We have computed the result for col c of A
132 // which is row c of Z.
133 //
134
135 loop(r, cols_B) Z.set_ravel_Float(r*cols_A + c, a[r]);
136 free(a);
137 }
138 }
139}

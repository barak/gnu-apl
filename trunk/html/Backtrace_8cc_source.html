60const char *
61Backtrace::find_src(int64_t pc)
62{
63 if (pc == NO_PC) return 0; // no pc
64
65 if (pc_2_src.size()) // database was properly set up.
66 {
67 if (const PC_src * posp = Heapsort<Backtrace::PC_src>
68 ::search<const int64_t &>(pc, pc_2_src, &pc_cmp, 0))
69 return posp->src_loc; // found
70 }
71
72 0 && cerr << "PC=" << hex << pc << dec << " not found in apl.lines" << endl;
73 return 0; // not found
74}
75//----------------------------------------------------------------------------
76void
77Backtrace::open_lines_file()
78{
79extern int main(int argc, const char *argv[]);
80
81 if (lines_status != LINES_not_checked) return;
82
83 // we are here for the first time.
84 // Assume that apl.lines is outdated until proven otherwise.
85 //
86 lines_status = LINES_outdated;
87
88struct stat st;
89 if (stat("apl", &st)) return; // stat apl failed
90
91const time_t apl_time = st.st_mtime;
92
93const int fd = open("apl.lines", O_RDONLY);
94 if (fd == -1) return;
95
96 if (fstat(fd, &st)) // stat apl.lines failed
97 {
98 close(fd);
99 return;
100 }
101
102const time_t apl_lines_time = st.st_mtime;
103
104 if (apl_lines_time < apl_time)
105 {
106#ifndef Makefile__srcdir
107# define Makefile__srcdir "src"
108#endif
109 close(fd);
110 get_CERR() << endl
111 << "Cannot show line numbers, since apl.lines is older "
112 "than apl." << endl
113 << "Consider running 'make apl.lines' in directory " << endl
114 << Makefile__srcdir " to fix this" << endl;
115 return;
116 }
117
118 pc_2_src.reserve(100000);
119char buffer[1000];
120FILE * file = fdopen(fd, "r");
121size_t file_lines = 0;
122size_t asm_lines = 0;
123
124 assert(file);
125
126const char * src_line = 0;
127bool new_line = false;
128int64_t prev_pc = NO_PC;
129
130 for (;;)
131 {
132 const char * s = fgets(buffer, sizeof(buffer) - 1, file);
133 if (s == 0) break; // end of file.
134 ++file_lines;
135
136 buffer[sizeof(buffer) - 1] = 0;
137 int slen = strlen(buffer);
138 if (slen && buffer[slen - 1] == '\n') buffer[--slen] = 0;
139 if (slen && buffer[slen - 1] == '\r') buffer[--slen] = 0;
140
141 /* we look for 3 types of lines:
142
143 the (single) <main> line,
144 absolute source file path lines, and
145 code lines.
146
147 For example:
148
149000000000014c7a9 <main>: [<main> line]
150main():
151/home/eedjsa/apl-1.8/src/main.cc:611 [source file path]
152 14c7a9:»······55 »··push %rbp [code line]
153 14c7aa:»······48 89 e5 »··mov %rsp,%rbp [code line]
154 ...
155
156 NOTE: this only works with:
157
158 CXXFLAGS='-rdynamic -gdwarf-2' ./configure ...
159
160 since newer compilers strip location information by default.
161 */
162
163 if (strstr(s, "<main>:")) // <main line>
164 {
165 const int64_t main_funct = reinterpret_cast<long long>(main);
166 const int64_t main_loc = strtoll(s, 0, 16);
167 main_offset = main_funct - main_loc;
168
169 0 && cerr << hex << setfill(' ')
170 << "main() in apl.lines " << setw(16) << main_loc << endl
171 << "main() offset: +" << setw(16) << main_offset << endl
172 << "main() start: =" << setw(16) << main_funct << endl
173 << dec;
174 }
175
176 if (s[0] == '/') // source file path
177 {
178 src_line = strdup(strrchr(s, '/') + 1);
179 new_line = true;
180 continue;
181 }
182
183 if (!new_line) continue;
184
185 if (s[8] == ':') // opcode address
186 {
187 ++asm_lines;
188 long long pc = NO_PC;
189 if (1 == sscanf(s, " %llx:", &pc))
190 {
191 if (pc < prev_pc)
192 {
193 assert(0 && "file apl.lines is not ordered by PC");
194 break;
195 }
196
197 PC_src pcs = { pc, src_line };
198 0 && cerr << "adding " << uhex << pcs.pc << dec
199 << " aka. " << pcs.src_loc << endl;
200 pc_2_src.push_back(pcs);
201 prev_pc = pc;
202
203 // new_line = false; causes some lines to not be found
204 }
205 continue;
206 }
207 }
208
209 fclose(file); // also closes fd
210
211 cerr << "total_lines in apl.lines: " << file_lines << endl
212 << "assembler lines in apl.lines: " << asm_lines << endl
213 << "source line numbers found: " << pc_2_src.size() << endl;
214 lines_status = LINES_valid;
215}

217void
218Backtrace::show_item(int idx, char * s)
219{
220#ifdef HAVE_EXECINFO_H
221 //
222 // s looks like this:
223 //
224 // ./apl(_ZN10APL_parser9nextTokenEv+0x1dc) [0x80778dc]
225 //
226
227 // split off address from s.
228 //
229const char * addr = "????????";
230int64_t pc = NO_PC;
231 {
232 char * space = strchr(s, ' ');
233 if (space)
234 {
235 *space = 0;
236 addr = space + 2;
237 char * e = strchr(space + 2, ']');
238 if (e) *e = 0;
239 pc = strtoll(addr, 0, 16);
240 pc -= main_offset;
241 }
242 }
243
244const char * src_loc = find_src(pc);
245
246 // split off function from s.
247 //
248char * fun = 0;
249 {
250 char * opar = strchr(s, '(');
251 if (opar)
252 {
253 *opar = 0;
254 fun = opar + 1;
255 char * e = strchr(opar + 1, ')');
256 if (e) *e = 0;
257 }
258 }
259
260 // split off offset from fun.
261 //
262int offs = 0;
263 if (fun)
264 {
265 char * plus = strchr(fun, '+');
266 if (plus)
267 {
268 *plus++ = 0;
269 sscanf(plus, "%X", &offs);
270 }
271 }
272
273char obuf[200] = "@@@@";
274 if (fun)
275 {
276 strncpy(obuf, fun, sizeof(obuf) - 1);
277 obuf[sizeof(obuf) - 1] = 0;
278
279 size_t obuflen = sizeof(obuf) - 1;
280 int status = 0;
281// cerr << "mangled fun is: " << fun << endl;
282 __cxxabiv1::__cxa_demangle(fun, obuf, &obuflen, &status);
283 switch(status)
284 {
285 case 0: // demangling succeeded
286 break;
287
288 case -2: // not a valid name under the C++ ABI mangling rules.
289 break;
290
291 default:
292 cerr << "__cxa_demangle() returned " << status << endl;
293 break;
294 }
295 }
296
297// cerr << setw(2) << idx << ": ";
298
299 cerr << HEX(pc);
300
301// cerr << left << setw(20) << s << right << " ";
302
303 // indent.
304 for (int i = -1; i < idx; ++i) cerr << " ";
305
306 cerr << obuf;
307
308// if (offs) cerr << " +" << offs;
309
310 if (src_loc) cerr << " at " << src_loc;
311 cerr << endl;
312#endif
313}

315void
316Backtrace::show(const char * file, int line)
317{
318 // CYGWIN, for example, has no execinfo.h and the functions declared there
319 //
320#ifndef HAVE_EXECINFO_H
321 cerr << "Cannot show function call stack since execinfo.h seems not"
322 " to exist on this OS (WINDOWs ?)." << endl;
323 return;
324
325#else
326
327 open_lines_file();
328
329void * buffer[200];
330const int size = backtrace(buffer, sizeof(buffer)/sizeof(*buffer));
331
332char ** strings = backtrace_symbols(buffer, size);
333
334 cerr << endl
335 << "----------------------------------------" << endl
336 << "-- Stack trace at " << file << ":" << line << endl
337 << "----------------------------------------" << endl;
338
339 if (strings == 0)
340 {
341 cerr << "backtrace_symbols() failed. Using backtrace_symbols_fd()"
342 " instead..." << endl << endl;
343 // backtrace_symbols_fd(buffer, size, STDERR_FILENO);
344 for (int b = size - 1; b > 0; --b)
345 {
346 for (int s = b + 1; s < size; ++s) cerr << " ";
347 backtrace_symbols_fd(buffer + b, 1, STDERR_FILENO);
348 }
349 cerr << "========================================" << endl;
350 return;
351 }
352
353 for (int i = 1; i < size - 1; ++i)
354 {
355 // make a copy of strings[i] that can be
356 // messed up in show_item().
357 //
358 const char * si = strings[size - i - 1];
359 char cc[strlen(si) + 1];
360 strcpy(cc, si);
361 cc[sizeof(cc) - 1] = 0;
362 show_item(i - 1, cc);
363 }
364 cerr << "========================================" << endl;
365
366#if 0
367 // crashes at times
368 free(strings); // but not strings[x] !
369#endif
370
371#endif
372}

374#ifdef HAVE_EXECINFO_H
375int
376Backtrace::demangle_line(char * result, size_t result_max, const char * buf)
377{
378std::string tmp;
379 tmp.reserve(result_max + 1);
380 for (const char * b = buf; *b && b < (buf + result_max); ++b)
381 tmp += *b;
382 tmp.append(0);
383
384char * e = 0;
385int status = 3;
386char * p = strchr(&tmp[0], '(');
387 if (p == 0) goto error;
388 else ++p;
389
390 e = strchr(p, '+');
391 if (e == 0) goto error;
392 else *e = 0;
393
394// cerr << "mangled fun is: " << p << endl;
395 __cxxabiv1::__cxa_demangle(p, result, &result_max, &status);
396 if (status) goto error;
397 return 0;
398
399error:
400 strncpy(result, buf, result_max);
401 result[result_max - 1] = 0;
402 return status;
403}

405const char *
406Backtrace::caller(int offset)
407{
408void * buffer[200];
409const int size = backtrace(buffer, sizeof(buffer)/sizeof(*buffer));
410char ** strings = backtrace_symbols(buffer, size);
411
412char * demangled = static_cast<char *>(malloc(200));
413 if (demangled)
414 {
415 *demangled = 0;
416 demangle_line(demangled, 200, strings[offset]);
417 return demangled;
418 }
419 return strings[offset];
420}

22/********
23
24Usage:
25
26 m4 -D Svar_signals=def_file < tcp_signal.m4 > $@
27
28This reads a Svar_signals specification from file def_file.def and prints
29a header file on stdout that defines (the serialization of) signals according
30to def_file.def.
31
32In a Makefile, you would use it, for example, like this:
33
34 my_signal.hh: tcp_signal.m4 my_signal.def
35 m4 -D Svar_signals=tcp_signal $< > $@
36
37to produce my_signal.hh from my_signal.def. After that, my_signal.hh
38can be used to send the signals defined in tcp_signal.def from one process
39to another process.
40
41That is:
42
43 my_signal.def
44 |
45 |
46 | tcp_signal.m4
47 |
48 V
49 my_signal.hh
50 |
51 |
52 |
53 |
54 V
55 my_client_program.cc
56 my_server_program.cc
57 |
58 |
59 | compile
60 | & link
61 V
62 my_client_program
63 my_server_program
64
65
66and then:
67
68 my_client_program --- signals defined in ---> my_server_program
69 my_server_program --- in my_signal.def ---> my_client_program
70

90template<typename T, int bytes>
91class Sig_item_int
92{
93public:
95 Sig_item_int(T v) : value(v & get_mask()) {}
96
98 Sig_item_int(const uint8_t * & buffer)
99 {
100 value = 0;
101 for (int b = 0; b < bytes; ++b)
102 value = value << 8 | *buffer++;
103
104 if (bytes == 6 && (value & 0x000080000000000ULL))
105 value |= 0xFFFF000000000000ULL;
106 }
107
109 T get_svalue() const { return value; }
110
112 void store(string & buffer) const
113 {
114 for (int b = bytes; b > 0;) buffer += char(value >> (8*--b));
115 }
116
118 ostream & print(ostream & out) const
119 {
120 return out << value;
121 }
122
124 static uint64_t get_mask()
125 { uint64_t mask = 0;
126 for (int b = 0; b < bytes; ++b) mask = mask << 8 | 0x00FF;
127 return mask;
128 }
129
131 T value;
132};
133//-----------------------------------------------------------------------------
135template<typename T, int bytes>
136class Sig_item_xint : public Sig_item_int<T, bytes>
137{
138public:
140 Sig_item_xint(T v) : Sig_item_int<T, bytes>(v) {}
141
143 Sig_item_xint(const uint8_t * & buffer) : Sig_item_int<T, bytes>(buffer) {}
144
146 ostream & print(ostream & out) const
147 {
148 return out << "0x" << hex << setfill('0') << setw(bytes)
149 << Sig_item_int<T, bytes>::value
150 << setfill(' ') << dec;
151 }
152};

174class Sig_item_string
175{
176public:
178 Sig_item_string(const string & str)
179 : value(str)
180 {}
181
183 Sig_item_string(const uint8_t * & buffer)
184 {
185 Sig_item_u16 len (buffer);
186 value.reserve(len.get_svalue() + 2);
187 for (int b = 0; b < len.get_svalue(); ++b) value += *buffer++;
188 }
189
191 const string get_svalue() const { return value; }
192
194 void store(string & buffer) const
195 {
196 const Sig_item_u16 len (value.size());
197 len.store(buffer);
198 for (size_t b = 0; b < value.size(); ++b) buffer += value[b];
199 }
200
202 ostream & print(ostream & out) const
203 {
204 bool printable = true;
205 for (size_t b = 0; b < value.size(); ++b)
206 {
207 const int v = value[b] & 0xFF;
208 if (v < ' ' || v > '~') { printable = false; break; }
209 }
210
211 if (printable) return out << "\"" << value << "\"";
212
213 out << hex << setfill('0');
214 for (size_t b = 0; b < value.size(); ++b)
215 {
216 if (b > 16) { out << "..."; break; }
217 out << " " << setw(2) << (value[b] & 0xFF);
218 }
219 return out << dec << setfill(' ');
220 }
221
222protected:
224 string value;
225};

664 {
665 string buffer;

667
668 const uint32_t ll = htonl(buffer.size());
669 const char * ccp_ll = reinterpret_cast<const char *>(&ll);
670
671// MAC OS has no MSG_MORE
672#ifndef MSG_MORE
673# define MSG_MORE 0
674#endif
675 if (sizeof(ll) == send(tcp_sock, ccp_ll, sizeof(ll), MSG_MORE))
676 {
677 return send(tcp_sock, buffer.data(), buffer.size(), 0);
678 }
679 cerr << "*** send_TCP() : " << errno << strerror(errno) << endl;
680 return -errno;

51const UCS_string &
52id_name(Id id)
53{
54 switch(id)
55 {
56#define av(x) case ID_ ## x: return id_ ## x;
57#define pp(x) case ID_ ## x: return id_ ## x;
58#define qf(x) case ID_Quad_ ## x: return id_Quad_ ## x;
59#define qv(x) case ID_Quad_ ## x: return id_Quad_ ## x;
60#define st(x) case ID_ ## x: return id_ ## x;
61
62#define id_def(_id, _uni, _val, _mac) _mac(_id)
63#include "Id.def"
64 }
65
66 CERR << "Unknown Id " << HEX(id);
67 Assert(0 && "Bad Id");
68}

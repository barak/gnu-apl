33class ScalarFunction : public PrimitiveFunction
34{
35 // ScalarFunction is only a helper class whose function shall not be
36 // called directly but only via their virtual counterparts in base
37 // class Function. We therefore delare all functions protected:
38 //
39protected:
41 ScalarFunction(TokenTag tag, CellFunctionStatistics * stat_AB,
42 CellFunctionStatistics * stat_B,
43 ShapeItem thresh_AB,
44 ShapeItem thresh_B)
45 : PrimitiveFunction(tag, stat_AB, stat_B)
46 {
47 set_dyadic_threshold(thresh_AB);
48 set_monadic_threshold(thresh_B);
49 }
50
52 Token eval_scalar_B(Value_P B, prim_f1 fun) const;
53
55 Token eval_scalar_AB(Value_P A, Value_P B, prim_f2 fun) const;
56
58 virtual prim_f2 get_scalar_f2() const = 0;
59
61 virtual Token eval_fill_AB(Value_P A, Value_P B) const
62 { return do_eval_fill_AB(A, B); }
63
65 virtual Token eval_fill_B(Value_P B) const
66 { return do_eval_fill_B(B); }
67
69 virtual bool has_result() const { return true; }
70
72 virtual bool may_parallel() const { return true; }
73
75 Value_P do_scalar_B(ErrorCode & ec, Value_P B, prim_f1 fun) const;
76
78 Value_P do_scalar_AB(ErrorCode & ec, Value_P A,
79 Value_P B, prim_f2 fun) const;
80
82 void expand_nested(Value * Z, const Cell * cell_A,
83 const Cell * cell_B, prim_f2 fun) const;
84
86 Token eval_scalar_AXB(Value_P A, Value_P X, Value_P B, prim_f2 fun) const;
87
89 Value_P eval_scalar_AXB(Value_P A, AxesBitmap axes_in_X,
90 Value_P B, prim_f2 fun, bool reversed) const;
91
93 Token do_eval_fill_AB(const Value_P A, const Value_P B) const;
94
96 Token do_eval_fill_B(const Value_P B) const;
97
99 static Token eval_scalar_identity_fun(Value_P B, sAxis axis,
100 const Cell & FI0);
101
103 static Thread_context::PoolFunction PF_scalar_AB;
104
106 static Thread_context::PoolFunction PF_scalar_B;
107
109 static const IntCell integer_0;
110
112 static const IntCell integer_1;
113
115 static const FloatCell float_min;
116
118 static const FloatCell float_max;
119
123 static const Shape * conforming_shape(ErrorCode & ec, const Shape & sh_A,
124 const Shape & sh_B);
125};

127#define PERF_A(x) TOK_F2_ ## x, \
128 & Performance::cfs_F2_ ## x ## _AB, 0, \
129 Performance::thresh_F2_ ## x ## _AB, -1
130
131#define PERF_AB(x) TOK_F12_ ## x, \
132 & Performance::cfs_F12_ ## x ## _AB, \
133 & Performance::cfs_F12_ ## x ## _B, \
134 Performance::thresh_F12_ ## x ## _AB, \
135 Performance::thresh_F12_ ## x ## _B
136
137#define PERF_B(x) TOK_F12_ ## x, \
138 0, & Performance::cfs_F12_ ## x ## _B, \
139 -1, Performance::thresh_F12_ ## x ## _B

950#if 1 // enable/disable deleted values chain for faster memory allocation

955 static void * operator new(size_t sz)
956 {
957 if (deleted_values) // we have deleted values: recycle one
958 {
959 --deleted_values_count;
960 void * ret = deleted_values;
961 deleted_values = deleted_values->next;
962 ++fast_new_count;
963 return ret;
964 }
965
966 ++slow_new_count;
967 return ::operator new(sz);
968 }
969
971 static void operator delete(void * ptr)
972 {
973 if (deleted_values_count < deleted_values_MAX) // we have space
974 {
975 ++deleted_values_count;
976 reinterpret_cast<_deleted_value *>(ptr)->next = deleted_values;
977 deleted_values = reinterpret_cast<_deleted_value *>(ptr);
978 }
979 else // no more space
980 {
981 free(ptr);
982 }
983 }
984
985#endif

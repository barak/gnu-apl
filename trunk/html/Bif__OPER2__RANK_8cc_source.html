60//----------------------------------------------------------------------------
61Token
62Bif_OPER2_RANK::do_LyXB(Token & _LO, Value_P X, Value_P B, sRank rank_chunk_B)
63{
64Function_P LO = _LO.get_function();
65 Assert(LO);
66 if (!LO->has_result()) DOMAIN_ERROR;
67
68 // split shape of B into high (=frame) and low (= chunk) shapes.
69 //
70const Shape shape_Z = B->get_shape().high_shape(B->get_rank() - rank_chunk_B);
71 if (shape_Z.is_empty())
72 {
73 Value_P Fill_B = Bif_F12_TAKE::first(*B);
74 Token tZ = LO->eval_fill_B(Fill_B);
75 Value_P Z = tZ.get_apl_val();
76 Z->set_shape(B->get_shape());
77 Z->check_value(LOC);
78 return Token(TOK_APL_VALUE1, Z);
79 }
80
81const Shape shape_B = B->get_shape().low_shape(rank_chunk_B);
82
83Value_P vsh_B(shape_B.get_rank(), LOC);
84 vsh_B->set_proto_Int(); // prototype
85 loop(sh, shape_B.get_rank())
86 vsh_B->next_ravel_Int(shape_B.get_shape_item(sh));
87 vsh_B->check_value(LOC);
88
89Value_P vsh_Z(shape_Z.get_rank(), LOC);
90 loop(sh, shape_Z.get_rank())
91 vsh_Z->next_ravel_Int(shape_Z.get_shape_item(sh));
92 vsh_Z->check_value(LOC);
93
94Value_P X5(5, LOC);
95 if (!X) X5->next_ravel_Int(-1); // ¯1 means no X
96 else X5->next_ravel_Pointer(X.get());
97
98 X5->next_ravel_Int(shape_B.get_volume()); // LB
99 X5->next_ravel_Pointer(vsh_B.get()); // rho_B
100 X5->next_ravel_Int(shape_Z.get_volume()); // N_max
101 X5->next_ravel_Pointer(vsh_Z.get()); // rho_Z
102 X5->check_value(LOC);
103
104 return Macro::get_macro(Macro::MAC_Z__LO_RANK_X5_B)->eval_LXB(_LO, X5, B);
105}

134//----------------------------------------------------------------------------
135Token
136Bif_OPER2_RANK::do_ALyXB(Value_P A, sRank rank_chunk_A, Token & _LO,
137 Value_P X, Value_P B, sRank rank_chunk_B)
138{
139Function_P LO = _LO.get_function();
140 Assert(LO);
141 if (!LO->has_result()) DOMAIN_ERROR;
142
143sRank rk_A_frame = A->get_rank() - rank_chunk_A; // rk_A_frame is y8
144sRank rk_B_frame = B->get_rank() - rank_chunk_B; // rk_B_frame is y9
145
146 // if both high-ranks are 0, then return A LO B.
147 //
148 if (rk_A_frame == 0 && rk_B_frame == 0) return LO->eval_AB(A, B);
149
150 // split shapes of A1 and B1 into high (frame) and low (chunk) shapes.
151 // Even though A and B have the same shape, rk_A_frame and rk_B_frame
152 // could be different, leading to different split shapes for A and B
153 //
154const Shape shape_Z = rk_B_frame ? B->get_shape().high_shape(rk_B_frame)
155 : A->get_shape().high_shape(rk_A_frame);
156
157 if (rk_A_frame && rk_B_frame) // A and B frames non-scalar
158 {
159 if (rk_A_frame != rk_B_frame) RANK_ERROR;
160 if (shape_Z != A->get_shape().high_shape(rk_A_frame)) LENGTH_ERROR;
161 }
162
163 if (shape_Z.is_empty())
164 {
165 Value_P Fill_A = Bif_F12_TAKE::first(*A);
166 Value_P Fill_B = Bif_F12_TAKE::first(*B);
167 Shape shape_Z;
168
169 if (A->is_empty()) shape_Z = A->get_shape();
170 else if (!A->is_scalar()) DOMAIN_ERROR;
171
172 if (B->is_empty()) shape_Z = B->get_shape();
173 else if (!B->is_scalar()) DOMAIN_ERROR;
174
175 Value_P Z1 = LO->eval_fill_AB(Fill_A, Fill_B).get_apl_val();
176
177 Value_P Z(shape_Z, LOC);
178 Z->set_ravel_Value(0, Z1.get());
179 Z->check_value(LOC);
180 return Token(TOK_APL_VALUE1, Z);
181 }
182
183const Shape low_A = A->get_shape().low_shape(rank_chunk_A);
184const Shape low_B = B->get_shape().low_shape(rank_chunk_B);
185
186Value_P vsh_A(LOC, &low_A);
187Value_P vsh_B(LOC, &low_B);
188Value_P vsh_Z(LOC, &shape_Z);
189
190Value_P X7(7, LOC);
191 if (!X) X7->next_ravel_Int(-1); // no X
192 else X7->next_ravel_Value(X.get());
193
194 X7->next_ravel_Int(low_A.get_volume()); // LA
195 X7->next_ravel_Value(vsh_A.get());
196 X7->next_ravel_Int(low_B.get_volume()); // LB
197 X7->next_ravel_Value(vsh_B.get());
198 X7->next_ravel_Int(shape_Z.get_volume()); // N_max
199 X7->next_ravel_Value(vsh_Z.get());
200 X7->check_value(LOC);
201 return Macro::get_macro(Macro::MAC_Z__A_LO_RANK_X7_B)
202 ->eval_ALXB(A, _LO, X7, B);
203}
204//----------------------------------------------------------------------------
205void
206Bif_OPER2_RANK::y123_to_B(Value_P y123, sRank & rank_B)
207{
208 // y123_to_AB() splits the ranks of A and B into a (higher-dimensions)
209 // "frame" and a (lower-dimensions) "chunk" as specified by y123.
210
211 // 1. on entry rank_B is the rank of B.
212 //
213 // Remember the rank of B to limit rank_B
214 // if values in y123 should exceed them.
215 //
216const sRank rk_B = rank_B;
217
218 if (!y123) VALUE_ERROR;
219 if ( y123->get_rank() > 1) DOMAIN_ERROR;
220
221 // 2. the number of elements in y determine how rank_B shall be computed:
222 //
223 // -- monadic f⍤ -- -- dyadic f⍤ --
224 // rank_A rank_B rank_A rank_B
225 // ---------------------------------------------------------
226 // y : N/A y y y
227 // yA yB : N/A yB yA yB
228 // yM yA yB : N/A yM yA yB
229 // ---------------------------------------------------------
230
231 switch(y123->element_count())
232 {
233 case 1: rank_B = y123->get_cfirst().get_near_int(); break;
234
235 case 2: y123->get_cfirst().get_near_int();
236 rank_B = y123->get_cravel(1).get_near_int(); break;
237
238 case 3: rank_B = y123->get_cfirst().get_near_int();
239 y123->get_cravel(1).get_near_int();
240 y123->get_cravel(2).get_near_int(); break;
241
242 default: LENGTH_ERROR;
243 }
244
245 // 3. adjust rank_B if they exceed its initial value or
246 // if it is negative
247 //
248 if (rank_B > rk_B) rank_B = rk_B;
249 if (rank_B < 0) rank_B += rk_B;
250 if (rank_B < 0) rank_B = 0;
251}
252//----------------------------------------------------------------------------
253void
254Bif_OPER2_RANK::y123_to_AB(Value_P y123, sRank & rank_A, sRank & rank_B)
255{
256 // y123_to_AB() splits the ranks of A and B into a (higher-dimensions)
257 // "frame" and a (lower-dimensions) "chunk" as specified by y123.
258
259 // 1. on entry rank_A and rank_B are the ranks of A and B.
260 //
261 // Remember the ranks of A and B to limit rank_A and rank_B
262 // if values in y123 should exceed them.
263 //
264const sRank rk_A = rank_A;
265const sRank rk_B = rank_B;
266
267 if (!y123) VALUE_ERROR;
268 if ( y123->get_rank() > 1) DOMAIN_ERROR;
269
270 // 2. the number of elements in y determine how rank_A and rank_B
271 // shall be computed:
272 //
273 // -- monadic f⍤ -- -- dyadic f⍤ --
274 // rank_A rank_B rank_A rank_B
275 // ---------------------------------------------------------
276 // y : N/A y y y
277 // yA yB : N/A yB yA yB
278 // yM yA yB : N/A yM yA yB
279 // ---------------------------------------------------------
280
281 switch(y123->element_count())
282 {
283 case 1: rank_A = y123->get_cfirst().get_near_int();
284 rank_B = rank_A; break;
285
286 case 2: rank_A = y123->get_cfirst().get_near_int();
287 rank_B = y123->get_cravel(1).get_near_int(); break;
288
289 case 3: y123->get_cfirst().get_near_int();
290 rank_A = y123->get_cravel(1).get_near_int();
291 rank_B = y123->get_cravel(2).get_near_int(); break;
292
293 default: LENGTH_ERROR;
294 }
295
296 // 3. adjust rank_A and rank_B if they exceed their initial value or
297 // if they are negative
298 //
299 if (rank_A > rk_A) rank_A = rk_A;
300 if (rank_A < 0) rank_A += rk_A;
301 if (rank_A < 0) rank_A = 0;
302
303 if (rank_B > rk_B) rank_B = rk_B;
304 if (rank_B < 0) rank_B += rk_B;
305 if (rank_B < 0) rank_B = 0;
306}
307//----------------------------------------------------------------------------
308void
309Bif_OPER2_RANK::split_y123_B(Value_P y123_B, Value_P & y123, Value_P & B)
310{
311 // The ISO standard and NARS define the reduction pattern for the RANK
312 // operator ⍤ as:
313 //
314 // Z ← A f ⍤ y B (ISO)
315 // Z ← f ⍤ y B (ISO)
316 // Z ← A f ⍤ [X] y B (NARS)
317 // Z ← f ⍤ [X] y B (NARS)
318 //
319 // GNU APL may bind y to B at tokenization time if y and B are constants
320 // This function tries to "unbind" its argument y123_B into the original
321 // components y123 (= y in the standard) and B. The tokenization time
322 // binding is shown as y123:B
323 //
324 // Usage y123 : B Result: j123 B
325 //-------------------------------------------------------------------------
326 // 1. f ⍤ (y123):B... nested any y123 B
327 // 2. (f ⍤ y123:⍬) simple empty y123 -
328 // 3. f ⍤ y123:(B) simple nested skalar y123 B
329 // 4a. f ⍤ y123:B... simple any y123 B...
330 //
331
332 // y123_B shall be a scalar or vector
333 //
334 if (y123_B->get_rank() > 1) RANK_ERROR;
335
336const ShapeItem length = y123_B->element_count();
337 if (length == 0) LENGTH_ERROR;
338
339 // check for case 1 (the only one with nested first element)
340 //
341 if (y123_B->get_cfirst().is_pointer_cell()) // (y123)
342 {
343 y123 = y123_B->get_cfirst().get_pointer_value();
344 if (length == 1) // empty B
345 {
346 }
347 else if (length == 2) // skalar B
348 {
349 const Cell & B0 = y123_B->get_cravel(1);
350 if (B0.is_pointer_cell()) // (B)
351 {
352 B = B0.get_pointer_value();
353 }
354 else
355 {
356 B = Value_P(LOC);
357 B->next_ravel_Cell(B0);
358 }
359 }
360 else // vector B
361 {
362 B = Value_P(length - 1, LOC);
363 loop(l, length - 1)
364 B->next_ravel_Cell(y123_B->get_cravel(l + 1));
365 }
366 y123->check_value(LOC);
367 B->check_value(LOC);
368 return;
369 }
370
371 // case 1. ruled out, so the first 1, 2, or 3 cells are j123.
372 // see how many (at most)
373 //
374int y123_len = 0;
375 loop(yy, 3)
376 {
377 if (yy >= length) break;
378 const Cell & cy = y123_B->get_cravel(yy);
379 if (cy.is_near_int()) ++y123_len;
380 else break;
381 }
382 if (y123_len == 0) LENGTH_ERROR; // at least y1 is needed
383
384 // cases 2.-4. start with integers of length 1, 2, or 3
385 //
386 if (length == y123_len) // case 2: y123:⍬
387 {
388 y123 = y123_B;
389 //
390 // NOTE: B is NOT assigned so that Prefix::reduce_F_D_B_() can detect
391 // that y123_ was only y123 !
392 //
393 return;
394 }
395
396 if (length == (y123_len + 1) &&
397 y123_B->get_cravel(y123_len).is_pointer_cell()) // case 3. y123:⊂B
398 {
399 y123 = Value_P(y123_len, LOC);
400 loop(yy, y123_len) y123->next_ravel_Cell(y123_B->get_cravel(yy));
401 B = y123_B->get_cravel(y123_len).get_pointer_value();
402 y123->check_value(LOC);
403 B->check_value(LOC);
404 return;
405 }
406
407 // case 4: y123:B...
408 //
409 y123 = Value_P(y123_len, LOC);
410 loop(yy, y123_len) y123->next_ravel_Cell(y123_B->get_cravel(yy));
411
412const ShapeItem B_len = length - y123_len;
413 B = Value_P(B_len, LOC);
414 loop(bb, B_len) B->next_ravel_Cell(y123_B->get_cravel(bb + y123_len));
415 B->check_value(LOC);
416}

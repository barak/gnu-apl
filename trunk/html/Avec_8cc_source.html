54//----------------------------------------------------------------------------
55int
56Avec::show_error_pos(int i, int line, bool cond, int def_line)
57{
58 if (!cond)
59 {
60 CERR << "Error index (line " << line << ", Avec.def line "
61 << def_line << ") = " << i << " = " << HEX(i) << endl;
62 Assert(0);
63 return 1;
64 }
65
66 return 0; // OK
67}

71void
72Avec::check_file(const char * filename)
73{
74const int fd = open(filename, O_RDONLY);
75 if (fd == -1) return;
76
77 Log(LOG_startup) CERR << "Checking " << filename << endl;
78
79uint32_t datalen;
80 {
81 struct stat s;
82 const int res = fstat(fd, &s);
83 Assert(res == 0);
84 datalen = s.st_size;
85 }
86
87const void * data = mmap(NULL, datalen, PROT_READ, MAP_PRIVATE, fd, 0);
88 Assert(data != (void *)-1);
89
90const UTF8_string utf = UTF8_string((const UTF8 *)data, datalen);
91UCS_string ucs(utf);
92
93 loop(i, ucs.size())
94 {
95 if (!is_known_char(ucs[i]))
96 CERR << "APL char " << UNI(ucs[i]) << " is missing in AV ("
97 << i << ")" << endl;
98 }
99
100 close(fd);
101}

108//----------------------------------------------------------------------------
109void
110Avec::check_av_table()
111{
112int errors = 0;
113
114 if (MAX_AV != 256)
115 {
116 CERR << "AV has " << MAX_AV << " entries (should be 256)" << endl;
117 ++errors;
118 }
119
120 Assert(sizeof(character_table) / sizeof(*character_table) == MAX_AV);
121
122
123 // check that character_table.unicode is sorted by increasing UNICODE
124 //
125 for (int i = 1; i < MAX_AV; ++i)
126 {
127 errors += show_error_pos(i, __LINE__,
128 character_table[i].def_line,
129 character_table[i].unicode >
130 character_table[i - 1].unicode);
131 }
132
133 // check that ASCII chars map to themselves.
134 //
135 loop(i, 0x80)
136 {
137 errors += show_error_pos(i, __LINE__,
138 character_table[i].def_line,
139 character_table[i].unicode == i);
140 }
141
142 // check for holes and duplicate AV positions
143 {
144 int count = 0;
145 int positions[MAX_AV];
146 loop(p, MAX_AV) positions[p] = 0;
147
148#define char_def(_name, _uni, _tag, _flags, av_pos) \
149 ++count; ++positions[0x ## av_pos];
150#define char_uni(_name, _uni, _tag, _flags)
151#include "Avec.def"
152
153// CERR << "checking ⎕AV..." << endl;
154 loop(p, MAX_AV)
155 {
156 if (positions[p] > 1) // duplicate
157 {
158 ++errors;
159 CERR << "In Avec.def: duplicate av_pos " << HEX2(p)
160 << " (" << positions[p] << " entries)" << endl;
161 }
162 if (positions[p] < 1) // hole
163 {
164 ++errors;
165 CERR << "In Avec.def: no av_pos " << HEX2(p) << endl;
166 }
167 }
168 }
169
170 // check that find_char() works
171 //
172 loop(i, MAX_AV)
173 {
174 errors += show_error_pos(i, __LINE__, character_table[i].def_line,
175 i == find_char(character_table[i].unicode));
176 }
177
178 Log(LOG_SHOW_AV)
179 {
180 for (int i = 0x80; i < MAX_AV; ++i)
181 {
182 CERR << character_table[i].unicode << " is AV[" << i << "] AV_"
183 << character_table[i].char_name << endl;
184 }
185 }
186
187// check_file("../keyboard");
188// check_file("../keyboard1");
189
190 if (errors) exit(1);
191}

214//----------------------------------------------------------------------------
215Token
216Avec::uni_to_token(Unicode & uni, const char * loc)
217{
218 {
219 const CHT_Index idx = find_char(uni); // search uni in character_table
220 if (idx != Invalid_CHT) return Token(character_table[idx].token_tag, uni);
221 }
222
223 // not found: try alternative characters.
224const CHT_Index idx = map_alternative_char(uni);
225 if (idx != Invalid_CHT)
226 {
227 uni = character_table[idx].unicode;
228 return Token(character_table[idx].token_tag, uni);
229 }
230
231 // uni was neither the standard character nor an alternative character
232 // for a token.
233 Log(LOG_verbose_error)
234 {
235 CERR << endl << "Avec::uni_to_token() : Char " << UNI(uni)
236 << " (" << uni << ") not found in ⎕AV! (called from "
237 << loc << ")" << endl;
238
239 BACKTRACE
240 }
241
242 return Token();
243}
244//----------------------------------------------------------------------------
245uint32_t Avec::find_av_pos(Unicode av)
246{
247const CHT_Index pos = find_char(av);
248
249 if (pos < 0) return MAX_AV; // not found
250
251 return character_table[pos].av_pos;
252}
253//----------------------------------------------------------------------------
254Avec::CHT_Index
255Avec::find_char(Unicode av)
256{
257int l = 0;
258int h = sizeof(character_table)/sizeof(*character_table) - 1;
259
260 for (;;)
261 {
262 if (l > h) return Invalid_CHT; // not in table.
263
264 const int m((h + l)/2);
265 const Unicode um = character_table[m].unicode;
266 if (av < um) h = m - 1;
267 else if (av > um) l = m + 1;
268 else return CHT_Index(m);
269 }
270}
271//----------------------------------------------------------------------------
272Avec::CHT_Index
273Avec::map_alternative_char(Unicode alt_av)
274{
275 // map Unicodes that look similar to characters used in GNU APL's ⎕AV
276 // to the corresponding character in GNU APL's ⎕AV.
277 //
278 switch(int(alt_av))
279 {
280 case 0x005E: return AV_AND; // map ^ to ∧
281 case 0x007C: return AV_DIVIDES; // map | to ∣
282 case 0x007E: return AV_TILDE_OPERATOR; // map ~ to ∼
283 case 0x03B1: return AV_ALPHA; // map α to ⍺
284 case 0x03B5: return AV_ELEMENT; // map ϵ to itself
285 case 0x03B9: return AV_IOTA; // map ι to ⍳
286 case 0x03C1: return AV_RHO; // map ρ to ⍴
287 case 0x03C9: return AV_OMEGA; // map ω to ⍵
288 case 0x03F5: return AV_ELEMENT; // map ϵ to ϵ
289 case 0x2018: return AV_SINGLE_QUOTE; // map ‘ to '
290 case 0x2019: return AV_SINGLE_QUOTE; // map ’ to '
291 case 0x2208: return AV_ELEMENT; // map ∈ to ϵ
292 case 0x220A: return AV_ELEMENT; // map ∊ to ϵ
293 case 0x2212: return AV_MINUS; // map − to -
294 case 0x22BC: return AV_NAND; // map ⊼ to ⍲
295 case 0x22BD: return AV_NOR; // map ⊽ to ⍱
296 case 0x22C4: return AV_DIAMOND; // map ⋄ to ◊
297 case 0x22F8: return AV_EPSILON_UNDERBAR; // map ⋸ to ⍷
298 case 0x25AF: return AV_Quad_Quad; // map ▯ to ⎕
299 case 0x25E6: return AV_RING_OPERATOR; // map ◦ to ∘
300 case 0x2662: return AV_DIAMOND; // map ♢ to ◊
301 case 0x26AA: return AV_CIRCLE; // map ⚪ to ○
302 case 0x2A7D: return AV_LESS_OR_EQUAL; // map ⩽ to ≤
303 case 0x2A7E: return AV_MORE_OR_EQUAL; // map ⩾ to ≥
304 case 0x2B25: return AV_DIAMOND; // map ⬥ to ◊
305 case 0x2B26: return AV_DIAMOND; // map ⬦ to ◊
306 case 0x2B27: return AV_DIAMOND; // map ⬧ to ◊
307
308 default: return Invalid_CHT; // unknown alt_av
309 }
310}
311//----------------------------------------------------------------------------
312Unicode
313Avec::make_standard(Unicode uni)
314{
315const CHT_Index idx = map_alternative_char(uni);
316 return idx == Invalid_CHT ? uni : character_table[idx].unicode;
317}
318//----------------------------------------------------------------------------
319bool
320Avec::is_known_char(Unicode av)
321{
322 // a "known" character is either a standard APL character or an alternative
323 // (= similar looking) character.
324 switch(av)
325 {
326#define char_def(_name, uni, _tag, _flags, _av_pos) case uni:
327#define char_uni(_name, uni, _tag, _flags) case uni:
328#include "Avec.def"
329 return true;
330
331 default: break;
332 }
333
334 return false; // not found
335}
336//----------------------------------------------------------------------------
337int
338Avec::digit_value(Unicode uni, bool hex)
339{
340 if (uni < UNI_0) return -1;
341 if (uni <= UNI_9) return uni - '0';
342 if (hex)
343 {
344 if (uni < UNI_A) return -1;
345 if (uni <= UNI_F) return 10 + uni - 'A';
346 if (uni < UNI_a) return -1;
347 if (uni <= UNI_f) return 10 + uni - 'a';
348 }
349 return -1;
350}
351//----------------------------------------------------------------------------
352bool
353Avec::need_UCS(Unicode uni)
354{
355 if (is_control(uni)) return true; // ASCII control
356 if (uni >= 0 && uni < UNI_DELETE) return false; // printable ASCII
357
358const CHT_Index idx = find_char(uni);
359 if (idx == Invalid_CHT) return true; // char not in GNU APL's ⎕AV
360 if (unicode_to_cp(uni) == 0xB0) return true; // not in IBM's ⎕AV
361
362 return false;
363}
364//----------------------------------------------------------------------------
365bool
366Avec::is_symbol_char(Unicode av)
367{
368const int32_t idx = find_char(av);
369 if (idx == -1) return false;
370 return (character_table[idx].flags & FLG_SYMBOL) != 0;
371}
372//----------------------------------------------------------------------------
373bool
374Avec::is_first_symbol_char(Unicode av)
375{
376 return is_symbol_char(av) && ! is_digit(av);
377}

571void
572Avec::print_inverse_IBM_quad_AV()
573{
574 // a helper function that sorts ibm_av by Unicode and prints it on CERR
575 //
576 // To use it change #if 0 to #if 1 in Command.cc function
577 // transfer_context::add(), recompile and start apl, and then
578 // )IN <file> or so.
579 //
580 loop(c, 256)
581 {
582 inverse_ibm_av[c].uni = -1;
583 inverse_ibm_av[c].cp = -1;
584 }
585
586int current_max = -1;
587Unicode_to_IBM_codepoint * map = inverse_ibm_av;
588 loop(c, 256)
589 {
590 // find next Unicode after current_max
591 //
592 int next_idx = -1;
593 loop(n, 256)
594 {
595 const int uni = ibm_av[n];
596 if (uni <= current_max) continue; // already done
597 if (next_idx == -1) next_idx = n;
598 else if (uni < ibm_av[next_idx]) next_idx = n;
599 }
600 current_max = map->uni = ibm_av[next_idx];
601 map->cp = next_idx;
602 ++map;
603 }
604
605 // print the IBM ⎕AV...
606 //
607 for (int row = 0x20; row < 0x100; row += 0x10)
608 {
609 CERR << HEX2(row) << " ";
610 loop(col, 0x10) CERR << " " << Unicode(ibm_av[row + col]);
611 CERR << endl;
612 }
613
614 loop(row, 64)
615 {
616 CERR << " ";
617 loop(col, 4)
618 {
619 const int pos = col + 4*row;
620 CERR << " { " << HEX4(inverse_ibm_av[pos].uni) << ", "
621 << setw(3) << inverse_ibm_av[pos].cp << " }";
622 if (pos < 255) CERR << ",";
623 }
624 CERR << endl;
625 }
626}

636unsigned char
637Avec::unicode_to_cp(Unicode uni)
638{
639 if (uni <= 0x80) return uni;
640 if (uni == UNI_STAR_OPERATOR) return '*'; // ⋆ → *
641 if (uni == UNI_AND) return '^'; // ∧ → ^
642 if (uni == UNI_TILDE_OPERATOR) return 126; // ∼ → ~
643
644 // search in uni_to_cp_map table
645 //
646const void * where = bsearch(&uni, inverse_ibm_av, 256,
647 sizeof(Unicode_to_IBM_codepoint), compare_uni);
648
649 if (where == 0)
650 {
651 // the workspace being )OUT'ed can contain characters that are not
652 // in IBM's APL character set. We replace such characters by 0xB0
653 //
654 return 0xB0;
655 }
656
657 Assert(where);
658 return reinterpret_cast<const Unicode_to_IBM_codepoint *>(where)->cp;
659}

116//----------------------------------------------------------------------------
117void
118Thread_context::PF_lock_unlock_pool(Thread_context & tctx)
119{
120 Log(LOG_Parallel)
121 {
122 PRINT_LOCKED(CERR << "worker #" << tctx.get_N()
123 << " will now block itself on its pool_sema" << endl)
124 }
125
126 tctx.do_join = false;
127 tctx.PF_join();
128
129 tctx.blocked = true;
130 sem_wait(&tctx.pool_sema);
131 tctx.blocked = false;
132
133 Log(LOG_Parallel)
134 {
135 PRINT_LOCKED(CERR << "thread #" << tctx.get_N()
136 << " was unblocked from pool_sema" << endl)
137 }
138}

172//----------------------------------------------------------------------------
173void
174Thread_context::bind_to_cpu(CPU_Number core, bool logit)
175{
176#if ! HAVE_AFFINITY_NP
177
178 CPU = CPU_0;
179
180#else
181
182 CPU = core;
183
184 Log(LOG_Parallel || logit)
185 {
186 PRINT_LOCKED(CERR << "Binding thread #" << N
187 << " to core " << core << endl;);
188 }
189
190cpu_set_t cpus;
191 CPU_ZERO(&cpus);
192
193 if (active_core_count == CCNT_1)
194 {
195 // there is only one core in total (which is the master).
196 // restore its affinity to all cores.
197 //
198 loop(a, CPU_pool::get_count())
199 CPU_SET(CPU_pool::get_CPU(a), &cpus);
200 }
201 else
202 {
203 CPU_SET(CPU, &cpus);
204 }
205
206const int err = pthread_setaffinity_np(thread, sizeof(cpu_set_t), &cpus);
207 if (err)
208 {
209 cerr << "pthread_setaffinity_np() failed with error "
210 << err << endl;
211 }
212#endif // HAVE_AFFINITY_NP
213}

222}

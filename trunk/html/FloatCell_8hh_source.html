219 static APL_Integer gcd(APL_Integer m, APL_Integer n)
220 {
221 if (m == 0) return n;
222 if (n == 0) return m;
223
224 if (m < 0) m = -m;
225 if (n < 0) n = -n;
226
227 // E1: Initialize
228 APL_Integer a_ = 1;
229 APL_Integer b = 1;
230 APL_Integer a = 0;
231 APL_Integer b_ = 0;
232 APL_Integer c = m;
233 APL_Integer d = n;
234 for (;;)
235 {
236 // E2: divide
237 const APL_Integer r = c%d;
238
239 // E3: remainder zero?
240 if (r == 0) return d;
241
242 const APL_Integer q = c/d; // from E2
243
244 // E4: recycle
245 const APL_Integer ta_ = a_;
246 const APL_Integer tb_ = b_;
247 c = d;
248 d = r;
249 a_ = a;
250 a = ta_ - q*a;
251 b_ = b;
252 b = tb_ - q*b;
253 }
254 }

84void
85init_1(const char * argv0, bool log_startup)
86{
87 // init the workspace memory limits
88 //
89 Quad_WA::init(log_startup);
90
91 enum { VALUE_HEADER = sizeof(Value)
92 - sizeof(Shape)
93 - SHORT_VALUE_LENGTH_WANTED * sizeof(Cell)
94 };
95
96 if (log_startup)
97 CERR << endl
98 << "sizeof(int) is " << sizeof(int) << endl
99 << "sizeof(long) is " << sizeof(long) << endl
100 << "sizeof(void *) is " << sizeof(void *) << endl
101 << endl
102 << "sizeof(Cell) is " << sizeof(Cell) << endl
103 << "sizeof(SI stack item) is " << sizeof(StateIndicator) << endl
104 << "sizeof(Svar_partner) is " << sizeof(Svar_partner) << endl
105 << "sizeof(Svar_record) is " << sizeof(Svar_record) << endl
106 << "sizeof(Symbol) is " << sizeof(Symbol) << endl
107 << "sizeof(Token) is " << sizeof(Token) << endl
108 << "sizeof(Value) is " << sizeof(Value)
109 << " (" << VALUE_HEADER << " byte header + "
110 << sizeof(Shape) << " byte shape + "
111 << SHORT_VALUE_LENGTH_WANTED << " Cells)" << endl
112 << "sizeof(ValueStackItem) is " << sizeof(ValueStackItem) << endl
113 << "sizeof(UCS_string) is " << sizeof(UCS_string) << endl
114 << "sizeof(UserFunction) is " << sizeof(UserFunction) << endl
115 << endl
116 << "⎕WA total memory is " << Quad_WA::total_memory
117 << " bytes (" << (Quad_WA::total_memory/1000000) << " MB, 0x"
118 << hex << Quad_WA::total_memory << ")" << dec << endl
119 << endl
120 << "configure command: " << CONFIGURE_ARGS << endl
121 << endl;
122
123 // CYGWIN does not have RLIMIT_NPROC
124 //
125#ifdef RLIMIT_NPROC
126
127 // unlimit the number of threads and processes...
128 //
129rlimit rl;
130 getrlimit(RLIMIT_NPROC, &rl);
131 if (log_startup)
132 CERR << "increasing rlimit RLIMIT_NPROC from " << rl.rlim_cur
133 << " to infinity" << endl;
134 rl.rlim_cur = RLIM_INFINITY;
135 setrlimit(RLIMIT_NPROC, &rl);
136
137 // limit the virtual memory size to avoid new() problems with large values
138 //
139#endif
140
141 Avec::init();
142 LibPaths::init(argv0, log_startup);
143 Value::init();
144 VH_entry::init();
145}

188//----------------------------------------------------------------------------
189void
190control_C(int)
191{
192APL_time_us when = now();
193
194 CIN << "^C";
195
196 attention_raised = true;
197 ++attention_count;
198 if ((when - interrupt_when) < 1000000) // second ^C within 1 second
199 {
200 interrupt_raised = true;
201 ++interrupt_count;
202 }
203
204 interrupt_when = when;
205}

237//----------------------------------------------------------------------------
238APL_time_us
239now()
240{
241timeval tv_now;
242 gettimeofday(&tv_now, 0);
243
244APL_time_us ret = tv_now.tv_sec;
245 ret *= 1000000;
246 ret += tv_now.tv_usec;
247 return ret;
248}
249//----------------------------------------------------------------------------
250YMDhmsu::YMDhmsu(APL_time_us at)
251 : micro(at % 1000000)
252{
253const time_t secs = at/1000000;
254struct tm * t = gmtime(&secs);
255
256 year = t->tm_year + 1900;
257 month = t->tm_mon + 1;
258 day = t->tm_mday;
259 hour = t->tm_hour;
260 minute = t->tm_min;
261 second = t->tm_sec;
262}
263//----------------------------------------------------------------------------
264APL_time_us
265YMDhmsu::get() const
266{
267tm t;
268 t.tm_year = year - 1900;
269 t.tm_mon = month - 1;
270 t.tm_mday = day;
271 t.tm_hour = hour;
272 t.tm_min = minute;
273 t.tm_sec = second;
274 t.tm_isdst = 0;
275
276APL_time_us ret = mktime(&t);
277 ret *= 1000000;
278 ret += micro;
279 return ret;
280}

294//----------------------------------------------------------------------------
295int
296nibble(Unicode uni)
297{
298 switch(uni)
299 {
300 case UNI_0 ... UNI_9: return (uni - UNI_0);
301 case UNI_A ... UNI_F: return 10 + (uni - UNI_A);
302 case UNI_a ... UNI_f: return 10 + (uni - UNI_a);
303 default: break;
304 }
305
306 return -1; // uni is not a hex digit
307}
308//----------------------------------------------------------------------------
309int
310sixbit(Unicode uni)
311{
312 switch(uni)
313 {
314 case UNI_A ... UNI_Z: return (uni - UNI_A);
315 case UNI_a ... UNI_z: return 26 + (uni - UNI_a);
316 case UNI_0 ... UNI_9: return 52 + (uni - UNI_0);
317
318 case UNI_PLUS: // standard 62
319 case UNI_MINUS: return 62; // alternative 62
320
321 case UNI_SLASH: // standard 63
322 case UNI_UNDERSCORE: return 63; // alternative 63
323
324 case UNI_EQUAL: return 64; // fill character
325
326 default: break;
327 }
328
329 return -1; // uni is not a hex digit
330}

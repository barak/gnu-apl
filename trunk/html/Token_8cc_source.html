42//----------------------------------------------------------------------------
43Token::Token(TokenTag t, IndexExpr & idx)
44{
45 /* construct a token for an index expressions. There are 3 cases which
46 differ by tag and by value type:
47
48 1. t indicates a partial index. Partial indices are used after the
49 trailing ] but before the leading [ was parsed (remember that we
50 parse right to left). Tag is TOK_PINDEX / TV_INDEX
51
52 2. t indicates a complete token (after parsing [). There are 2 sub-cases:
53
54 2a. idx has rank 1: [ axis ]. IndexExpr is converted to a (single) value,
55 possibly 0 to indicate an elided index []. Tag is TOK_AXIS / TV_VAL
56
57 2b. idx is [i1;i2...]: Tag is TOK_INDEX / TV_INDEX.
58
59 In case 1. and 2b. the tag == t; only in case 2a. is the caller's
60 t == TOK_INDEX replaced with tok == TOK_AXIS.
61 */
62
63 if (t == TOK_PINDEX) // case 1. (partial index ...] )
64 {
65 // this token is a partial index in the prefix parser.
66 // use it as is.
67 //
68 tag = TOK_PINDEX;
69 value.index_val = &idx; // type is TV_INDEX
70 }
71 else // case 2. (complete index [ ... ] )
72 {
73 Assert(t == TOK_INDEX);
74 if (idx.get_rank() < 2) // case 2a. [idx] or []
75 {
76 tag = TOK_AXIS;
77 value.apl_val = idx.extract_axis(); // type is TV_VAL
78 }
79 else // [idx; ...]
80 {
81 tag = TOK_INDEX;
82 value.index_val = &idx; // type is TV_INDEX
83 }
84 }
85}

136//----------------------------------------------------------------------------
137ostream &
138operator << (ostream & out, const Token & token)
139{
140 if (token.get_tag() == TOK_CHARACTER)
141 {
142 out << "CHAR«" << token.get_char_val() << "»";
143 return out;
144 }
145
146 if (token.get_tag() == TOK_APL_VALUE1 ||
147 token.get_tag() == TOK_APL_VALUE2 ||
148 token.get_tag() == TOK_APL_VALUE3)
149 {
150 if (token.get_tag() == TOK_APL_VALUE1) out << "VALUE1";
151 else if (token.get_tag() == TOK_APL_VALUE2) out << "VALUE2";
152 else if (token.get_tag() == TOK_APL_VALUE3) out << "VALUE3";
153 else out << "VALUE???";
154 Value_P value = token.get_apl_val();
155 Assert(+value);
156 const APL_types::Depth depth = value->compute_depth();
157 out << "«";
158 for (APL_types::Depth d = 0; d < depth; ++d) out << "≡";
159
160 if (value->get_rank()) out << value->get_shape();
161
162 const PrintContext pctx(PR_APL);
163 PrintBuffer pb(*value, pctx, 0);
164 bool more = pb.get_row_count() > 1;
165 if (pb.get_row_count() > 0)
166 {
167 UCS_string ucs = pb.get_line(0).no_pad();
168 if (ucs.size() > 20)
169 {
170 ucs.resize(20);
171 more = true;
172 }
173 out << ucs;
174 }
175 if (more) out << "...";
176 return out << "»";
177 }
178
179 if (token.get_tag() == TOK_ERROR)
180 {
181 return out << Error::error_name(ErrorCode(token.get_int_val()));
182 }
183
184 if (token.get_tag() == TOK_BRANCH)
185 {
186 return out << token.get_Id() << token.get_int_val();
187 }
188
189 if (token.get_tag() == TOK_NOBRANCH)
190 {
191 return out << token.get_Id() << token.get_int_val();
192 }
193
194 if (token.get_Id() > ID_No_ID2) return out << token.get_Id();
195
196 switch(token.get_Class())
197 {
198 case TC_END: return out << "END";
199 case TC_RETURN:
200 if (token.get_tag() == TOK_RETURN_EXEC)
201 return out << "RETURN ⍎";
202 if (token.get_tag() == TOK_RETURN_STATS)
203 return out << "RETURN ◊";
204 if (token.get_tag() == TOK_RETURN_VOID)
205 return out << "RETURN ∇FUN";
206 if (token.get_tag() == TOK_RETURN_SYMBOL)
207 return out << "RETURN Z←FUN";
208 return out << "RETURN ???";
209
210 case TC_VALUE: return token.print_value(out);
211
212 case TC_INDEX:
213 case TC_PINDEX:
214 if (token.get_tag() == TOK_INDEX)
215 return out << token.get_index_val();
216 else if (token.get_tag() == TOK_AXIS)
217 return out << *token.get_axes();
218 else
219 FIXME;
220
221 case TC_SYMBOL:
222 if (token.get_tag() == TOK_LSYMB)
223 {
224 token.get_sym_ptr()->print(out << "'");
225 out << UNI_LEFT_ARROW;
226 }
227 else if (token.get_tag() == TOK_LSYMB2)
228 {
229 token.get_sym_ptr()->print(out << "'(... ");
230 out << ")←";
231 }
232 else
233 {
234 token.get_sym_ptr()->print(out << "'");
235 }
236 return out;
237
238 case TC_FUN0:
239 case TC_FUN12:
240 case TC_OPER1:
241 case TC_OPER2: return token.print_function(out);
242 case TC_LINE: return out << "\n[" << token.get_fun_line() << "] ";
243 case TC_NUMERIC: return out << "Numeric";
244 case TC_NEWLINE: return out << "{LFeed}";
245
246 default: break;
247 }
248
249 return out << "{-unknown Token " << token.get_tag() << "-}";
250}

402//----------------------------------------------------------------------------
403ostream &
404Token::print_value(ostream & out) const
405{
406 switch(tag)
407 {
408 case TOK_VARIABLE: return value.sym_ptr->print(out);
409 case TOK_FUN0: return out << "{ fun/0 }";
410 case TOK_CHARACTER: return out << value.char_val;
411 case TOK_INTEGER: return out << value.int_vals[0];
412 case TOK_REAL: return out << value.float_vals[0];
413
414 case TOK_COMPLEX: return out << value.float_vals[0] << "J"
415 << value.float_vals[1];
416
417 case TOK_APL_VALUE1:
418 case TOK_APL_VALUE3:
419 {
420 const Value * v = value.apl_val.get();
421 if (v->get_rank() == 0) out << "''";
422 loop(r, v->get_rank())
423 {
424 if (r) out << " ";
425 out << v->get_shape_item(r);
426 }
427 out << UNI_RHO;
428 loop(e, v->element_count())
429 {
430 if (e == 4)
431 {
432 out << "...";
433 break;
434 }
435
436 const PrintContext pctx(PR_APL, 2, 80);
437 out << UCS_string(v->get_cravel(e)
438 .character_representation(pctx), 0, 80)
439 << " ";
440 }
441 return out;
442 }
443
444 case TOK_Quad_Quad: return out << UNI_Quad_Quad;
445
446 case TOK_Quad_AI:
447 case TOK_Quad_AV:
448 case TOK_Quad_CT:
449 case TOK_Quad_EM:
450 case TOK_Quad_ET:
451 case TOK_Quad_FC:
452 case TOK_Quad_IO:
453 case TOK_Quad_LC:
454 case TOK_Quad_LX:
455 case TOK_Quad_L:
456 case TOK_Quad_PP:
457 case TOK_Quad_PR:
458 case TOK_Quad_PS:
459 case TOK_Quad_PW:
460 case TOK_Quad_R:
461 case TOK_Quad_RL:

472 {
473 out << ", Id = " << get_Id();
474 }
475
476 return out << ", }";

478//----------------------------------------------------------------------------
479void
480Token::show_trace(ostream & out, const UCS_string & fun_name,
481 Function_Line line) const
482{
483UCS_string fn = fun_name;
484 fn.append_UTF8("[");
485 fn.append_number(line);
486 fn.append_UTF8("] ");
487
488 out << fn;
489
490 switch(get_tag())
491 {
492 case TOK_APL_VALUE1:
493 case TOK_APL_VALUE2:
494 case TOK_APL_VALUE3:
495 break; // continue below
496
497 case TOK_BRANCH:
498 out << "→" << get_int_val() << endl;
499 return;
500
501 case TOK_NOBRANCH:
502 out << "→⍬" << endl;
503 return;
504
505 case TOK_ESCAPE:
506 out << "→" << endl;
507 return;
508
509 case TOK_VOID:
510 case TOK_NO_VALUE:
511 out << endl;
512 return;
513
514 default: Q1(*this)
515 FIXME;
516 }
517
518 // print a value
519
520PrintContext pctx = Workspace::get_PrintContext(PR_APL);
521const Value & val = *get_apl_val();
522 if (val.get_rank() == 0) // scalar
523 {
524 pctx.set_style(PR_APL_MIN);
525 }
526 else if (val.get_rank() == 1) // vector
527 {
528 if (val.element_count() == 0 && // empty vector
529 val.get_cfirst().is_simple_cell())
530 {
531 out << endl;
532 return;
533 }
534
535 pctx.set_style(PR_APL_MIN);
536 }
537 else // matrix or higher
538 {
539 pctx.set_style(PrintStyle(pctx.get_style() | PST_NO_FRACT_0));
540 }
541
542PrintBuffer pb(val, pctx, 0);
543const UCS_string indent(fn.size(), UNI_SPACE);
544 loop(l, pb.get_row_count())
545 {
546 if (l) out << indent;
547 out << pb.get_line(l).no_pad() << endl;
548 }
549
550 if (pb.get_row_count() == 0) out << endl;
551}

558//----------------------------------------------------------------------------
559UCS_string
560Token::canonical(PrintStyle style) const
561{
562UCS_string ucs;
563
564 switch(get_Class())
565 {
566 case TC_ASSIGN:
567 ucs.append(UNI_LEFT_ARROW);
568 break;
569
570 case TC_R_ARROW:
571 ucs.append(UNI_RIGHT_ARROW);
572 break;
573
574 case TC_L_BRACK:
575 if (get_tag() == TOK_L_BRACK)
576 ucs.append(UNI_L_BRACK);
577 else if (get_tag() == TOK_SEMICOL)
578 ucs.append(UNI_SEMICOLON);
579 else
580 Assert(0);
581 break;
582
583 case TC_R_BRACK:
584 ucs.append(UNI_R_BRACK);
585 break;
586
587 case TC_END:
588 ucs.append(UNI_DIAMOND);
589 break;
590
591 case TC_RETURN: break;
592 case TC_LINE:
593 ucs.append(UNI_LF);
594 break;
595
596 case TC_VALUE:
597 {
598 PrintContext pctx(style, DEFAULT_Quad_PP, DEFAULT_Quad_PW);
599 PrintBuffer pbuf(*get_apl_val(), pctx, 0);
600 if (pbuf.get_row_count() == 0) return ucs;
601 return pbuf.l1();
602 }
603
604 case TC_SYMBOL:
605 ucs.append(get_sym_ptr()->get_name());
606 break;
607
608
609 case TC_R_PARENT:
610 case TC_L_PARENT:
611 case TC_R_CURLY:
612 case TC_L_CURLY:
613 return ID::get_name_UCS(get_Id());
614
615 case TC_FUN0:
616 case TC_FUN12:
617 case TC_OPER1:
618 case TC_OPER2:
619 if (get_Id() == ID_No_ID) return get_function()->get_name();
620 return ID::get_name_UCS(get_Id());
621
622 default:
623 CERR << "Token: " << HEX4(tag) << " " << *this
624 << " at " << LOC << endl;
625 Q1(get_Class())
626 BACKTRACE
627 FIXME;
628 }
629
630 return ucs;
631}
632//----------------------------------------------------------------------------
633UCS_string
634Token::tag_name() const
635{
636 switch(get_tag())
637 {
638#define TD(tag, _tc, _tv, _id) case tag: return UCS_string( #tag );
639#include "Token.def"
640 case TOK_NONE: return UCS_string("???Bad-Tah???");
641 }
642
643char cc[40];
644 snprintf(cc, sizeof(cc), "0x%X", get_tag());
645UCS_string ucs(cc);
646 return ucs;
647}
648//----------------------------------------------------------------------------
649int
650Token::error_info(UCS_string & ucs) const
651{
652UCS_string can = canonical(PR_APL_FUN).remove_pad();
653
654const Unicode c1 = ucs.back();
655const Unicode c2 = can.size() ? can[0] : Invalid_Unicode;
656
657 // conditions when we don't need a space
658 //
659bool need_space = ! (Avec::no_space_after(c1) || Avec::no_space_before(c2));
660
661 if (need_space) ucs.append(UNI_SPACE);
662
663 ucs.append(can);
664 return need_space ? -can.size() : can.size();
665}

719//----------------------------------------------------------------------------
720const char *
721Token::class_name(TokenClass tc)
722{
723#define tcn(x) case x: return #x;
724
725 switch(tc)
726 {
727 tcn( TC_ASSIGN )

844Token_string::reverse_from_to(ShapeItem from, ShapeItem to)
845{
846Token * t1 = &at(from);
847Token * t2 = &at(to);
848 Assert(0 <= from);
849 Assert(from <= to);
850 Assert(to <= ShapeItem(size()));
851
852 while (t1 < t2) t1++->Hswap(*t2--);
853}

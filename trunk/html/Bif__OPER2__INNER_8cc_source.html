35Token
36Bif_OPER2_INNER::fill(const Shape shape_Z, Value_P A, Function_P fun,
37 Value_P B, const char * loc)
38{
39 // this function is called from A f.g B when A fun B is called with an
40 // empty A or B. In this case shape_Z is empty since A->get_shape() or
41 // B->get_shape() (or both) contain axes of length 0.
42
43Value_P Fill_A; // argument A of the fill function
44Value_P Fill_B; // argument B of the fill function
45
46 if (A->is_empty()) Fill_A = A->prototype(LOC);
47 else Fill_A = Bif_F12_TAKE::first(*A);
48
49 if (B->is_empty()) Fill_B = B->prototype(LOC);
50 else Fill_B = Bif_F12_TAKE::first(*B);
51
52Token tok = fun->eval_fill_AB(Fill_A, Fill_B);
53
54 if (tok.get_Class() != TC_VALUE) return tok;
55
56Value * Z = tok.get_apl_val().get();
57
58Value_P Z1(shape_Z, LOC); // shape_Z is empty
59 Z1->get_wproto().init_from_value(Z, *Z1, loc);
60
61 Z1->check_value(LOC);
62 return Token(TOK_APL_VALUE1, Z1);
63}

65Token
66Bif_OPER2_INNER::eval_ALRB(Value_P A, Token & _LO, Token & _RO, Value_P B) const
67{
68 if (!_LO.is_function() || !_RO.is_function()) SYNTAX_ERROR;
69
70Function_P LO = _LO.get_function();
71Function_P RO = _RO.get_function();
72 Assert1(LO);
73 Assert1(RO);
74
75 if (LO->get_fun_valence() != 2 || RO->get_fun_valence() != 2) SYNTAX_ERROR;
76 if (!LO->has_result() || !RO->has_result()) DOMAIN_ERROR;
77
78 if (!A->is_scalar_extensible() && !B->is_scalar_extensible() &&
79 A->get_rank() > 1 && B->get_rank() > 1 &&
80 A->get_shape().get_last_shape_item() !=
81 B->get_shape().get_shape_item(0)) LENGTH_ERROR;
82
83const Shape shape_A1 =A->get_shape().without_last_axis();
84const ShapeItem len_A = A->get_last_shape_item();
85
86const Shape shape_B1 = B->get_shape().without_first_axis();
87const ShapeItem len_B = B->get_first_shape_item();
88
89 // we do not check len_A == len_B here, since a non-scalar LO may
90 // accept different lengths of its left and right arguments
91
92const ShapeItem items_A1 = shape_A1.get_volume();
93const ShapeItem items_B1 = shape_B1.get_volume();
94
95 if (items_A1 == 0 || items_B1 == 0) // empty result
96 {
97 // the outer product portion of LO.RO is empty.
98 // Apply the fill function of RO
99 //
100 const Shape shape_Z = shape_A1 + shape_B1;
101 return fill(shape_Z, A, RO, B, LOC);
102 }
103
104 if (LO->may_push_SI() || RO->may_push_SI()) // user defined LO or RO
105 {
106 // ISO: if A1 and B1 are both vectors, return f/A1 g B1.
107 //
108 if (A->get_rank() <= 1 && B->get_rank() <= 1)
109 return Macro::get_macro(Macro::MAC_Z__vA_LO_INNER_RO_vB)
110 ->eval_ALRB(A, _LO, _RO, B);
111 else
112 return Macro::get_macro(Macro::MAC_Z__A_LO_INNER_RO_B)
113 ->eval_ALRB(A, _LO, _RO, B);
114 }
115
116Value_P Z(shape_A1 + shape_B1, LOC);
117
118 // an important (and the most likely) special case is LO and RO being scalar
119 // functions. This case can be implemented in a far simpler fashion than
120 // the general case.
121 //
122 job.LO = LO->get_scalar_f2();
123 job.RO = RO->get_scalar_f2();
124 if (job.LO && job.RO && A->is_simple() && B->is_simple() && len_A)
125 {
126 job.incA = (A->element_count() == 1) ? 0 : 1;
127 job.incB = (B->element_count() == 1) ? 0 : 1;
128
129 // len_A must be len_B, unless at least one length is 1
130 //
131 if (len_A != len_B && job.incA && job.incB) LENGTH_ERROR;
132
133 job.cZ = &Z->get_wfirst();
134 job.cA = &A->get_cfirst();
135 job.ZAh = items_A1;
136 job.LO_len = A->is_scalar() ? len_B : len_A;
137 job.cB = &B->get_cfirst();
138 job.ZBl = items_B1;
139 job.ec = E_NO_ERROR;
140
141 scalar_inner_product();
142 if (job.ec != E_NO_ERROR) throw_apl_error(job.ec, LOC);
143
144 Z->set_default(*B.get(), LOC);
145
146 Z->check_value(LOC);
147 return Token(TOK_APL_VALUE1, Z);
148 }
149
150const bool A_enclosed = A->get_rank() > 1;
151const bool B_enclosed = B->get_rank() > 1;
152
153 // enclose last axis of A if necessary
154 //
155 if (A_enclosed)
156 {
157 const Shape last_axis(A->get_rank() - 1);
158 A = Bif_F12_PARTITION::enclose_with_axes(last_axis, A);
159 }
160
161 // enclose first axis of B if necessary
162 //
163 if (B_enclosed)
164 {
165 const Shape first_axis(0);
166 B = Bif_F12_PARTITION::enclose_with_axes(first_axis, B);
167 }
168
169 loop (a, items_A1)
170 loop (b, items_B1)
171 {
172 Value_P RO_A(A, LOC);
173 if (A_enclosed) RO_A = A->get_cravel(a).get_pointer_value();
174
175 Value_P RO_B(B, LOC);
176 if (B_enclosed) RO_B = B->get_cravel(b).get_pointer_value();
177
178 const Token T1 = RO->eval_AB(RO_A, RO_B);
179
180 if (T1.get_tag() == TOK_ERROR) return T1;
181
182 Value_P A_RO_B = T1.get_apl_val();
183
184 if (A_RO_B->is_simple_scalar()) // A_RO_B is A RO B
185 {
186 // A RO B has returned a scalar, so LO/A_RO_B is A_RO_B
187 //
188 Z->next_ravel_Cell(A_RO_B->get_cfirst());
189 }
190 else
191 {
192 // A RO B has returned vector A_RO_B, compute LO/A_RO_B
193 //
194 const Token T2 = Bif_OPER1_REDUCE::reduce(_LO, A_RO_B,
195 A_RO_B->get_rank() - 1);
196 if (T2.get_tag() == TOK_ERROR) return T2;
197
198 Value_P V2 = T2.get_apl_val();
199 if (V2->is_simple_scalar()) Z->next_ravel_Cell(V2->get_cfirst());
200 else Z->next_ravel_Pointer(V2.get());
201 }
202 }
203
204 Z->set_default(*B.get(), LOC);
205 Z->check_value(LOC);
206 return Token(TOK_APL_VALUE1, Z);
207}

209void
210Bif_OPER2_INNER::scalar_inner_product() const
211{
212#ifdef PERFORMANCE_COUNTERS_WANTED
213const uint64_t start_1 = cycle_counter();
214#endif
215
216 // the empty cases have been ruled out already in inner_product()
217
218 job.ec = E_NO_ERROR;
219
220#if PARALLEL_ENABLED
221 if ( Parallel::run_parallel
222 && Thread_context::get_active_core_count() > 1
223 && job.ZAh * job.ZBl > get_dyadic_threshold())
224 {
225 job.cores = Thread_context::get_active_core_count();
226 Thread_context::do_work = PF_scalar_inner_product;
227 Thread_context::M_fork("scalar_inner_product"); // start pool
228 PF_scalar_inner_product(Thread_context::get_master());
229 Thread_context::M_join();
230 }
231 else
232#endif // PARALLEL_ENABLED
233 {
234 job.cores = CCNT_1;
235 PF_scalar_inner_product(Thread_context::get_master());
236 }
237
238#ifdef PERFORMANCE_COUNTERS_WANTED
239const uint64_t end_1 = cycle_counter();
240 Performance::fs_OPER2_INNER_AB.add_sample(end_1-start_1, job.ZAh * job.ZBl);
241#endif
242}
243//----------------------------------------------------------------------------
244void
245Bif_OPER2_INNER::PF_scalar_inner_product(Thread_context & tctx)
246{
247const ShapeItem Z_len = job.ZAh * job.ZBl;
248
249const ShapeItem slice_len = (Z_len + job.cores - 1)/job.cores;
250ShapeItem z = tctx.get_N() * slice_len;
251ShapeItem end_z = z + slice_len;
252 if (end_z > Z_len) end_z = Z_len;
253
254 for (; z < end_z; ++z)
255 {
256 const ShapeItem zah = z/job.ZBl; // z row = A row
257 const ShapeItem zbl = z - zah*job.ZBl; // z column = B column
258 const Cell * row_A = job.cA + job.incA*((zah + 1) * job.LO_len);
259 const Cell * col_B = job.cB + job.incB*(zbl + job.LO_len*job.ZBl);
260
261 // compute Z[z] ← LO / (row_A RO colB)
262 // e.g. Z[z] ← +/ (row_A × colB)
263 //
264 // we use Z[z] as accumulator for LO /
265 //
266 // we use the terms sum and product as if Z←A LO.RO B were A +.* B
267 //
268 Cell * sum = job.cZ + z;
269 loop(l, job.LO_len)
270 {
271 row_A -= job.incA;
272 col_B -= job.incB*job.ZBl;
273
274 if (l == 0) // store first product in Z[z]
275 {
276 job.ec = (col_B->*job.RO)(sum, row_A);
277 if (job.ec != E_NO_ERROR) return;
278 }
279 else // add subsequent product to Z[z]
280 {
281 Cell product; // the result of RO, e.g. of × in +/×
282 job.ec = (col_B->*job.RO)(&product, row_A);
283 if (job.ec != E_NO_ERROR) return;
284
285 job.ec = (sum->*job.LO)(sum, &product);
286 if (job.ec != E_NO_ERROR) return;
287 }
288 }
289 }

36void
37DynamicObject::print_chain(ostream & out) const
38{
39int pos = 0;
40 for (const DynamicObject * p = this; ;)
41 {
42 out << " Chain[" << setw(2) << pos++ << "] "
43 << voidP(p->prev) << " --> "
44 << voidP(p) << " --> "
45 << voidP(p->next) << " "
46 << p->where_allocated() << endl;
47
48 p = p->next;
49 if (p == this) break;
50 }
51}

71Value &
72DynamicObject::rValue()
73{
74 /* static_cast<> is inherently unsafe, but we cannot used dynamic_cast<>()
75 because DynamicObject is non-virtual. The caller must therefore be
76 careful and we allow only class Value to use rValue().

79}

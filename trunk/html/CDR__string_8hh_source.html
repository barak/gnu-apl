29struct CDR_header
30{
31 uint32_t be_ptr;
32 uint32_t be_nb;
33 int32_t be_nelm;
34 uint8_t type;
35 uint8_t rank;
36 uint8_t fill[2];
37
38 // some compilers do not allow zero -sized arrays like dim[].
39 // Since dim is not used we just ignore it.
40 //
41// uint32_t dim[0]; ///< shape of the value
42
44 uint32_t get_nb() const
45 { return get_be32(utf8P(&be_nb)); }
46
48 uint32_t get_nelm() const
49 { return get_be32(utf8P(&be_nelm)); }
50
52 static uint32_t get_be32(const uint8_t * data)
53 {
54 uint32_t ret = *data++; ret <<= 8;
55 ret |= *data++; ret <<= 8;
56 ret |= *data++; ret <<= 8;
57 ret |= *data++;
58 return ret;
59 }
60};

94 const uint8_t * ravel() const
95 { return &(*this)[0] + 16 + 4*get_rank(); }
96
98 bool is_integer() const
99 { return get_type() == CDR_BOOL1 || get_type() == CDR_INT32; }
100
102 bool is_character() const
103 { return get_type() == CDR_CHAR8 || get_type() == CDR_CHAR32; }
104
106 int check() const
107 {
108 uint32_t cnt = 1;
109 for (int r = 0; r < get_rank(); ++r) cnt *= get_4(16 + 4*r);
110 if (size() < 32) return 2;
111 if (get_ptr() != 0x00002020) return 3;
112 if (header().get_nb() != size()) return 4;
113 if (header().get_nelm() != cnt) return 5;
114 if (uint32_t(get_type()) > CDR_NEST32) return 6;
115 return 0;
116 }
117
119 void debug(ostream & out, const char * loc = 0) const
120 {
121 const int error = check();
122 out << "CDR record";
123 if (loc) out << " at " << loc;
124 if (error) out << "(error " << error << ")";
125 out << ":" << endl;
126 loop(i, size()) cerr << " " << hex << setfill('0')
127 << setw(2) << (at(i) & 0xFF);
128 cerr << dec << setfill(' ') << endl;
129 }
130
131protected:
133 uint32_t get_4(unsigned int offset) const
134 { return CDR_header::get_be32(&(*this)[0] + offset); }
135};

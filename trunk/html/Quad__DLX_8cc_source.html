33//============================================================================
35class DLX_Node
36{
37public:
39 DLX_Node(bool hdr)
40 : row(-1),
41 col(-1),
42 up(this),
43 down(this),
44 left(this),
45 right(this),
46 is_header(hdr)
47 {}
48
50 DLX_Node(bool hdr, ShapeItem R, ShapeItem C,
51 DLX_Node * u, DLX_Node * d, DLX_Node * l, DLX_Node * r)
52 : row(R),
53 col(C),
54 up(u),
55 down(d),
56 left(l),
57 right(r),
58 is_header(hdr)
59 { insert_lr(); insert_ud(); }
60
62 void operator = (const DLX_Node & other)
63 { new (this) DLX_Node(other); }
64
66 void insert_lr() { left->right = this; right->left = this; }
67
69 void insert_ud() { up->down = this; down->up = this; }
70
72 void remove_lr() { left->right = right; right->left = left; }
73
75 void remove_ud() { up->down = down; down->up = up; }
76
78 void check() const
79 {
80 Assert(up); Assert(up->col == col);
81 Assert(down); Assert(down->col == col);
82 Assert(left); Assert(left->row == row);
83 Assert(right); Assert(right->row == row);
84 }
85
87 ostream & print_RC(ostream & out) const
88 {
89 return out << std::right << setw(2) << row << ":"
90 << std::left << setw(2) << col << std::right;
91 }
92
94 void print(ostream & out) const
95 {
96 print_RC(out);
97 up->print_RC(out << " ↑");
98 down->print_RC(out << " ↓");
99 left->print_RC(out << " ←");
100 right->print_RC(out << " →");
101 out << endl;
102 check();
103 }
104
106 enum Col_Type
107 {
108 Col_INVALID = -1,
109 Col_NULL = 0,
110 Col_PRIMARY = 1,
111 Col_SECONDARY = 2,
112 Col_UNKNOWN = Col_NULL
113 };
114
116 static Col_Type get_col_type(const Cell & cell);
117
119 const ShapeItem row;
120
122 const ShapeItem col;
123
125 DLX_Node * up;
126
128 DLX_Node * down;
129
131 DLX_Node * left;
132
134 DLX_Node * right;
135
137 const bool is_header;
138};

166//----------------------------------------------------------------------------
167DLX_Node::Col_Type
168DLX_Node::get_col_type(const Cell & cell)
169{
170 if (cell.is_character_cell())
171 {
172 // blank and '0' shall mean 0, '1' means 1, and '2' means 2
173 //
174 switch(cell.get_char_value())
175 {
176 case ' ':
177 case '0': return Col_NULL;
178 case '1': return Col_PRIMARY;
179 case '2': return Col_SECONDARY;
180 default: return Col_INVALID;
181 }
182 }
183 if (cell.is_numeric())
184 {
185 switch(cell.get_int_value())
186 {
187 case 0: return Col_NULL;
188 case 1: return Col_PRIMARY;
189 case 2: return Col_SECONDARY;
190 default: return Col_INVALID;
191 }
192 }
193
194 return Col_INVALID;
195}
196//============================================================================
198class DLX_Root_Node : public DLX_Node
199{
200public:
202 DLX_Root_Node(ShapeItem rows, ShapeItem cols, ShapeItem max_sol,
203 const Value & B);
204
206 void deep_check() const;
207
209 ostream & indent(ostream & out)
210 {
211 loop(s, level) out << " ";
212 return out;
213 }
214
216 void cover(ShapeItem j)
217 {
218 ++cover_count;
219
220 DLX_Header_Node & c = headers[j];
221 Assert(c.row == -1);
222 Assert(c.col == j);
223 c.remove_lr();
224 if (c.col_type == Col_PRIMARY) --primary_count;
225
226 for (DLX_Node * i = c.down; i != &c; i = i->down)
227 {
228 Assert(i->col == j);
229 for (DLX_Node * j = i->right; j != i; j = j->right)
230 {
231 j->remove_ud();
232 --headers[j->col].count;
233 }
234 }
235 }
236
238 void uncover(ShapeItem j)
239 {
240 DLX_Header_Node & c = headers[j];
241 Assert(c.row == -1);
242 Assert(c.col == j);
243 for (DLX_Node * i = c.up; i != &c; i = i->up)
244 {
245 Assert(i->col == j);
246 for (DLX_Node * j = i->left; j != i; j = j->left)
247 {
248 j->insert_ud();
249 headers[j->col].count++;
250 }
251 }
252 c.insert_lr();
253 if (c.col_type == Col_PRIMARY) ++primary_count;
254 c.check();
255 }
256
258 ShapeItem get_column_count() const;
259
261 void display(ostream & out) const;
262
264 void solve();
265
267 Token preset(const Cell * steps, ShapeItem count, const Cell * cB);
268
270 ShapeItem get_solution_count() const { return solution_count; }
271
273 ShapeItem get_pick_count() const { return pick_count; }
274
276 ShapeItem get_cover_count() const { return cover_count; }
277
279 std::vector<ShapeItem> all_solutions;
280
281protected:
283 const ShapeItem max_solutions;
284
286 const ShapeItem rows;
287
289 const ShapeItem cols;
290
292 std::vector<DLX_Header_Node> headers;
293
295 std::vector<DLX_Node> nodes;
296
298 ShapeItem primary_count;
299
301 ShapeItem secondary_count;
302
304 ShapeItem solution_count;
305
307 ShapeItem level;
308
310 ShapeItem pick_count;
311
313 ShapeItem cover_count;
314
316 const Cell * first_0;
317};
318//----------------------------------------------------------------------------
319DLX_Root_Node::DLX_Root_Node(ShapeItem rs, ShapeItem cs, ShapeItem max_sol,
320 const Value & B)
321 : DLX_Node(true, -1, -1, this, this, this, this),
322 max_solutions(max_sol),
323 rows(rs),
324 cols(cs),
325 primary_count(0),
326 secondary_count(0),
327 solution_count(0),
328 level(0),
329 pick_count(0),
330 cover_count(0),
331 first_0(0)
332{
333 // count the number of non-zero elements in the matrix
334 //
335const ShapeItem ec_B = B.element_count();
336const int qio = Workspace::get_IO();
337ShapeItem ones = 0;
338const Cell * b = &B.get_cfirst();
339 loop(e, ec_B)
340 {
341 const Cell & cell = *b++;
342 const Col_Type ct = get_col_type(cell);
343 if (ct == Col_INVALID)
344 {
345 const ShapeItem row = qio + e/B.get_cols();
346 const ShapeItem col = qio + e%B.get_cols();
347 UCS_string & more = MORE_ERROR() << "Bad B[" << row << ";"
348 << col << "]: ";
349 if (cell.is_integer_cell())
350 more << cell.get_int_value();
351 else if (cell.is_character_cell())
352 more << "'" << cell.get_char_value() << "'";
353 else
354 more << "neither integer nor character";
355 more << " in ⎕DLX B";
356 DOMAIN_ERROR;
357 }
358 if (ct != Col_UNKNOWN) ++ones;
359 else if (!first_0) first_0 = &cell;
360 }
361
362 Log(LOG_Quad_DLX) CERR << "Matrix has " << ones << " ones" << endl;
363
364 // set up column headers. std::vector.push_back() may move the headers so
365 // we first append all headers and initialize then
366 //
367 headers.reserve(cols);
368 loop (c, cols) headers.push_back(DLX_Header_Node());
369 loop (c, cols)
370 {
371 if (c) new (&headers[c]) DLX_Header_Node(c, &headers[c - 1], this);
372 else new (&headers[c]) DLX_Header_Node(c, this, this);
373 }
374
375 // set up non-header nodes. std::vector.push_back() may move the headers so
376 // we first append all nodes and initialize then.
377 //
378 b = &B.get_cfirst();
379 nodes.reserve(ones);
380 loop(o, ones) nodes.push_back(DLX_Node(false));
381
382 DLX_Node * n = &nodes[0];
383 loop (r, rows)
384 {
385 DLX_Node * const lm = n; // leftmost item in this row
386 DLX_Node * rm = n; // rightmost item in this row
387 loop (c, cols)
388 {
389 DLX_Header_Node & hdr = headers[c];
390 const Col_Type ct = get_col_type(*b++);
391 if (ct < Col_PRIMARY) continue; // most likely: not '1' or '2'
392
393 if (hdr.col_type == Col_UNKNOWN) // first '1' or '2' in this col
394 {
395 hdr.col_type = ct;
396 if (ct == Col_PRIMARY) ++primary_count;
397 else ++secondary_count;
398 }
399 else if (ct != hdr.col_type) // col type mismatch
400 {
401 MORE_ERROR() <<
402 "the columns of B in ⎕DLX B must either be primary ("
403 "containing 1s) or secondary (containing 2s), but not both";
404 DOMAIN_ERROR;
405 }
406
407
408 if (rm == n)
409 rm = new (n++) DLX_Node(false, r, c, hdr.up, &hdr, rm, lm);
410 else
411 rm = new (n++) DLX_Node(false, r, c, hdr.up, &hdr, rm, lm);
412 ++headers[c].count;
413 }
414 }
415
416 // check that all columns have a 1 or 2
417 //
418 loop (c, cols)
419 {
420 if (headers[c].col_type == Col_UNKNOWN)
421 {
422 MORE_ERROR() << "⎕DLX B with empty column B[;"
423 << (c + qio) << "]";
424 DOMAIN_ERROR;
425 }
426 }
427
428 Log(LOG_Quad_DLX) { loop(n, ones) nodes[n].print(CERR); }
429}
430//----------------------------------------------------------------------------
431void
432DLX_Root_Node::display(ostream & out) const
433{
434int w = 1;
435 if (cols >= 10) w = 2;
436 if (cols >= 100) w = 3;
437 if (cols >= 1000) w = 4;
438
439ShapeItem pcnt = 0;
440
441 out << endl << " N:";
442 for (const DLX_Node * x = right; x != this; x = x->right)
443 {
444 const DLX_Header_Node & hdr =
445 *reinterpret_cast<const DLX_Header_Node *>(x);
446 Assert(hdr.row == -1);
447 if (hdr.col_type == Col_PRIMARY) ++pcnt;
448 out << " " << setw(w) << hdr.count;
449 }
450 out << endl << "Col:";
451 for (const DLX_Node * x = right; x != this; x = x->right)
452 {
453 const DLX_Header_Node & hdr =
454 *reinterpret_cast<const DLX_Header_Node *>(x);
455 Assert(hdr.row == -1);
456 out << " " << setw(w) << hdr.col;
457 }
458 out << " (" << pcnt << ")" << endl << "----";
459 for (const DLX_Node * x = right; x != this; x = x->right)
460 {
461 for (int ww = -1; ww < w; ++ww) out << "-";
462 }
463 out << endl;
464
465char rows_used[rows];
466 memset(rows_used, 0, sizeof(rows_used));
467
468 for (const DLX_Node * x = right; x != this; x = x->right)
469 for (const DLX_Node * y = x->down; y != x; y = y->down)
470 {
471 rows_used[y->row] = 1; // mark row used
472 }
473
474 for (ShapeItem r = 0; r < rows; ++r)
475 {
476 if (!rows_used[r]) continue;
477
478 int row[cols];
479 memset(row, 0, sizeof(row));
480 for (const DLX_Node * x = right; x != this; x = x->right)
481 for (const DLX_Node * y = x->down; y != x; y = y->down)
482 {
483 if (y->row == r) row[y->col] = 1;
484 }
485
486 out << setw(3) << r << ":";
487 for (const DLX_Node * x = right; x != this; x = x->right)
488 {
489 out << " " << setw(w) << row[x->col];
490 }
491 out << endl;
492 }
493
494 out << "----";
495 for (const DLX_Node * x = right; x != this; x = x->right)
496 {
497 for (int ww = -1; ww < w; ++ww) out << "-";
498 }
499 out << endl;
500
501}
502//----------------------------------------------------------------------------
503void
504DLX_Root_Node::deep_check() const
505{
506 for (const DLX_Node * x = right; x != this; x = x->right)
507 {
508 const DLX_Header_Node & hdr =
509 *reinterpret_cast<const DLX_Header_Node *>(x);
510 Assert(hdr.row == -1);
511 hdr.check();
512 ShapeItem ones = 0;
513 for (const DLX_Node * y = x->down; y != x; y = y->down)
514 {
515 y->check();
516 ++ones;
517 }
518 Assert(ones == hdr.count);
519 }
520}
521//----------------------------------------------------------------------------
522void
523DLX_Root_Node::solve()
524{
525new_level:
526 Assert(level <= cols);
527
528 deep_check();
529
530 if (LOG_Quad_DLX || attention_is_raised())
531 {
532 CERR << "⎕DLX[" << level << "]";
533 loop(s, level)
534 CERR << " " << (headers[s].item_r->row + Workspace::get_IO());
535 CERR << endl;
536 clear_attention_raised(LOC);
537 }
538
539 if (interrupt_is_raised())
540 {
541 clear_interrupt_raised(LOC);
542 return;
543 }
544
545 // the problem is solved if there are no primary columns left
546 //
547DLX_Node * item = right;
548
549 Log(LOG_Quad_DLX) item->print_RC(CERR << "At header item ") << endl;
550
551 if (primary_count == 0)
552 {
553 all_solutions.push_back(level);
554 loop(s, level)
555 all_solutions.push_back(headers[s].item_r->row);
556
557 Log(LOG_Quad_DLX)
558 {
559 CERR << "!!!!! solution " << solution_count << ": rows are";
560 loop(s, level) CERR << " " << setw(2) << headers[s].item_r->row;
561 CERR << endl;
562 }
563
564 ++solution_count;
565 if (max_solutions && solution_count >= max_solutions) return;
566
567 goto level_done;
568 }
569
570 if (item != this) // matrix not empty: select shortest column
571 {
572 ShapeItem col_size = 2*cols;
573 for (DLX_Node * i = right; i != this; i = i->right)
574 {
575 if (headers[i->col].col_type != Col_PRIMARY) continue;
576 const ShapeItem i_size = headers[i->col].count;
577 if (i_size < col_size)
578 {
579 item = i;
580 col_size = i_size;
581 }
582 }
583
584 if (col_size == 0)
585 {
586 Log(LOG_Quad_DLX) CERR << "column " << item-> col
587 << " is empty" << endl;
588 goto level_done; // empty column: no solution
589 }
590 }
591
592 Assert(item->row == -1);
593 Log(LOG_Quad_DLX) indent(CERR) << "Choose and cover column c="
594 << item->col << endl;
595 cover(item->col);
596
597 headers[level].item_r = item;
598
599rloop: // running ↓
600 {
601 DLX_Node * r = headers[level].item_r->down;
602
603 Log(LOG_Quad_DLX) r->print_RC(indent(CERR) << "rloop ↓ at r= ") << endl;
604
605 for (DLX_Node * j = r->right; j != r; j = j->right)
606 {
607 Log(LOG_Quad_DLX)
608 {
609 indent(CERR) << "Covering column j=" << j->col << endl;
610 }
611 cover(j->col);
612 }
613
614 Log(LOG_Quad_DLX)
615 {
616 r->print_RC(indent(CERR) << "Picking item ") << endl;
617 }
618 headers[level].item_r = r;
619 ++level;
620 ++pick_count;
621 goto new_level;
622 }
623
624level_done:
625 if (level > 0)
626 {
627 --level;
628 DLX_Node * r = headers[level].item_r;
629 Log(LOG_Quad_DLX)
630 {
631 r->print_RC(indent(CERR) << "Backtracking") << endl;
632 }
633
634 for (DLX_Node * j = r->left; j != r; j = j->left)
635 {
636 Log(LOG_Quad_DLX)
637 {
638 indent(CERR) << "Uncovering column j=" << j->col << endl;
639 }
640 uncover(j->col);
641 }
642
643 r = r->down;
644 if (r->row == -1) // end of rloop reached
645 {
646 Log(LOG_Quad_DLX)
647 {
648 indent(CERR) << "Uncovering column c=" << r->col << endl;
649 }
650 uncover(r->col);
651 goto level_done;
652 }
653
654 goto rloop;
655 }
656}
657//----------------------------------------------------------------------------
658ShapeItem
659DLX_Root_Node::get_column_count() const
660{
661ShapeItem cols_Z = 0;
662 for (DLX_Node * h = right; h != this; h = h->right) ++cols_Z;
663 return cols_Z;
664}
665//----------------------------------------------------------------------------
666Token
667DLX_Root_Node::preset(const Cell * steps, ShapeItem step_count, const Cell * cB)
668{
669const int qio = Workspace::get_IO();
670 loop(a, step_count)
671 {
672 const APL_Integer row = steps++->get_int_value() - qio;
673 if (row < 0 || row >= rows)
674 {
675 MORE_ERROR() << "bad row: " << row;
676 DOMAIN_ERROR;
677 }
678
679 bool found_a = false;
680
681 // find a '1' or '2' item in row
682 //
683 loop(b, nodes.size())
684 {
685 const DLX_Node & n = nodes[b];
686 if (n.row == row) // found '1' or '2'
687 {
688 found_a = true;
689 Log(LOG_Quad_DLX)
690 n.print_RC(CERR << "Covering column " << n.col
691 << " of rightmost item ") << endl;
692 cover(n.col);
693
694 for (DLX_Node * j = n.right; j != &n; j = j->right)
695 {
696 Log(LOG_Quad_DLX)
697 {
698 CERR << "Covering column j=" << j->col << endl;
699 }
700 cover(j->col);
701 }
702
703 break; // next row a
704 }
705 }
706 Assert(found_a);
707 }
708
709 // construct return value
710 //
711const ShapeItem cols_Z = get_column_count();
712
713const Shape shape_Z(rows, cols_Z);
714Value_P Z(shape_Z, LOC);
715
716 // first init all cells in Z to whatever means 0
717 //
718 loop(z, (rows*cols_Z))
719 {
720 if (first_0) Z->next_ravel_Cell(*first_0);
721 else if (cB->is_integer_cell()) Z->next_ravel_0();
722 else Z->next_ravel_Char(UNI_0);
723 }
724
725 // then override those cells in the linked structure to whatever means 1
726 //
727 Assert(is_header);
728
729DLX_Node * h = right;
730 loop(col, cols_Z)
731 {
732 Assert(h->is_header); // therefore for (v...) stops
733 for (DLX_Node * v = h->down; !v->is_header; v = v->down)
734 {
735 Assert(!v->is_header);
736 Assert(h->col == v->col);
737 const Cell & src = cB[h->col + cols*v->row];
738 Z->get_wravel(col + cols_Z*v->row).init(src, *Z, LOC);
739 }
740 h = h->right;
741 }
742
743 Z->check_value(LOC);
744 return Token(TOK_APL_VALUE1, Z);
745}
746//============================================================================
747Token
748Quad_DLX::eval_AB(Value_P A, Value_P B) const
749{
750 if (A->get_rank() > 1) RANK_ERROR;
751 if (A->element_count() < 1) LENGTH_ERROR;
752
753 if (B->get_rank() != 2) RANK_ERROR;
754const APL_Integer a0 = A->get_cfirst().get_int_value();
755
756 if (a0 < -4) DOMAIN_ERROR;
757 if (a0 == -4) // perform some dance steps with the constraints matrix
758 {
759 const ShapeItem rows = B->get_rows();
760 const ShapeItem cols = B->get_cols();
761 DLX_Root_Node root(rows, cols, 0, *B);
762 return root.preset(&A->get_cravel(1), A->element_count() - 1,
763 &B->get_cfirst());
764 }
765
766 if (A->element_count() != 1) LENGTH_ERROR;
767 return do_DLX(a0, *B);
768}
769//----------------------------------------------------------------------------
770Token
771Quad_DLX::eval_B(Value_P B) const
772{
773 if (B->get_rank() != 2) RANK_ERROR;
774 return do_DLX(0, *B);
775};
776//----------------------------------------------------------------------------
777Token
778Quad_DLX::do_DLX(ShapeItem how, const Value & B)
779{
780
781const ShapeItem rows = B.get_rows();
782const ShapeItem cols = B.get_cols();
783
784ShapeItem result_count = how;
785 if (how == 0) result_count = 1; // first solution
786 else if (how == -1) result_count = 0; // all solution
787 else if (how == -2) result_count = 1; // statistics: first solution
788 else if (how == -3) result_count = 0; // statistics: all solutions
789
790DLX_Root_Node root(rows, cols, result_count, B);
791 root.solve();
792
793 if (how <= -2) // return statistics
794 {
795 Value_P Z(3, LOC);
796 Z->next_ravel_Int(root.get_solution_count());
797 Z->next_ravel_Int(root.get_pick_count());
798 Z->next_ravel_Int(root.get_cover_count());
799 Z->check_value(LOC);
800 return Token(TOK_APL_VALUE1, Z);
801 }
802
803const APL_Integer qio = Workspace::get_IO();
804 if (how == 0) // return first (flat) solution
805 {
806 if (root.get_solution_count() == 0) // but no solution found
807 {
808 return Token(TOK_APL_VALUE1, Idx0(LOC));
809 }
810
811 const ShapeItem len = root.all_solutions[0];
812 Value_P Z(len, LOC);
813 loop(z, len)
814 {
815 Z->next_ravel_Int(qio + root.all_solutions[z + 1]);
816 }
817 Z->check_value(LOC);
818 return Token(TOK_APL_VALUE1, Z);
819 }
820
821 // return all solutions
822 //
823Value_P Z(root.get_solution_count(), LOC);
824 if (root.get_solution_count() == 0) // empty result
825 {
826 Value_P Z1 = Idx0(LOC); // Z1 ← ⍬
827 new (&Z->get_wproto()) PointerCell(Z1.get(), *Z);
828 }
829 else
830 {
831 ShapeItem a = 0;
832 loop(s, root.get_solution_count())
833 {
834 const ShapeItem len = root.all_solutions[a++];
835 Value_P Zs(len, LOC);
836 loop(z, len)
837 {
838 Zs->next_ravel_Int(qio + root.all_solutions[a++]);
839 }
840 Zs->check_value(LOC);
841 Z->next_ravel_Pointer(Zs.get());
842 }
843 }

57 ty get_ ## na() const { return na; } \
58 \
59 void set_ ## na(ty val) \
60 { na = val; \
61 if (!strcmp(#na, "rangeX_min")) \
62 rangeX_type = Plot_Range_type(rangeX_type | PLOT_RANGE_MIN); \
63 else if (!strcmp(#na, "rangeX_max")) \

65 else if (!strcmp(#na, "rangeY_min")) \

113static void
114signal_WINCH_handler(int)
115{
116 // fgets() returns EOF when the WINCH signal is received. We remember
117 // this fact and repeat fgets() once after a WINCH signal
118 //
119 got_WINCH = true;
120
121struct winsize wsize;
122 // TIOCGWINSZ is 0x5413 on GNU/Linux. We use 0x5413 instead of TIOCGWINSZ
123 // because TIOCGWINSZ may not exist on all platforms
124 //
125 if (0 != ioctl(STDIN_FILENO, 0x5413, &wsize)) return;
126 if (wsize.ws_col < MIN_Quad_PW) return;
127 if (wsize.ws_col > MAX_Quad_PW) return;
128
129 Workspace::set_PW(wsize.ws_col, LOC);
130}

273void
274remap_stdio()
275{
276 if (uprefs.tcp_port <= 0) return;
277
278const int listen_socket = socket(AF_INET, SOCK_STREAM, 0);
279 if (listen_socket == -1)
280 {
281 perror("socket() failed");
282 exit(1);
283 }
284
285sockaddr_in local;
286 memset(&local, 0, sizeof(local));
287 local.sin_family = AF_INET;
288 local.sin_addr.s_addr = htonl(0x7F000001); // localhost (127.0.0.1)
289 local.sin_port = htons(uprefs.tcp_port);
290
291 // fix bind() error when listening socket is openend too quickly
292 {
293 const int yes = 1;
294 if (setsockopt(listen_socket, SOL_SOCKET, SO_REUSEADDR,
295 &yes, sizeof(yes)) < 0)
296 {
297 perror("setsockopt(SO_REUSEADDR) failed");
298 }
299
300 // continue, since a failed setsockopt() is sort of OK here.
301 }
302
303 if (::bind(listen_socket, (const sockaddr *)&local, sizeof(local)))
304 {
305 perror("bind() failed");
306 exit(1);
307 }
308
309 if (listen(listen_socket, 10))
310 {
311 perror("listen() failed");
312 exit(1);
313 }
314
315 0 && CERR << "The GNU APL server is listening on TCP port "
316 << uprefs.tcp_port << endl;
317
318 for (;;) // connection server loop
319 {
320 sockaddr_in remote;
321 socklen_t remote_len = sizeof(remote);
322 const int connection = ::accept(listen_socket,
323 reinterpret_cast<sockaddr *>(&remote),
324 &remote_len);
325 if (connection == -1)
326 {
327 perror("accept() failed");
328 exit(1);
329 }
330
331 0 && CERR << "GNU APL server got TCP connction from "
332 << (ntohl(remote.sin_addr.s_addr) >> 24 & 0xFF) << "."
333 << (ntohl(remote.sin_addr.s_addr) >> 16 & 0xFF) << "."
334 << (ntohl(remote.sin_addr.s_addr) >> 8 & 0xFF) << "."
335 << (ntohl(remote.sin_addr.s_addr) >> 0 & 0xFF) << " port "
336 << ntohs(remote.sin_port) << endl;
337
338 // fork() and let the client return while the server remains in this
339 // server loop for the next connection.
340 //
341 const pid_t fork_result = fork();
342 if (fork_result == -1) // fork() failed
343 {
344 close(connection);
345 perror("fork() failed");
346 exit(1);
347 }
348
349 if (fork_result) // parent (server)
350 {
351 close(connection);
352 continue;
353 }
354
355 // child (client)
356 //
357 close(listen_socket);
358 dup2(connection, STDIN_FILENO);
359 dup2(connection, STDOUT_FILENO);
360 dup2(connection, STDERR_FILENO);
361 close(connection);
362 return;
363 }

368int
369init_apl(int argc, const char * argv[])
370{
371 {
372 // make curses happy
373 //
374 const char * term = getenv("TERM");
375 if (term == 0 || *term == 0) setenv("TERM", "dumb", 1);
376 }
377
378const bool log_startup0 = uprefs.parse_argv_0(argc, argv);
379 if (LOG_argc_argv || log_startup0)
380 {
381 CERR << "argc/argv before expansion:\n";
382 show_argv(argc, argv);
383 }
384
385 uprefs.expand_argv(argc, argv);
386
387const bool log_startup = uprefs.parse_argv_1() || log_startup0;
388 if (LOG_argc_argv || log_startup)
389 {
390 CERR << "argc/argv after expansion:\n";
391 show_argv(uprefs.expanded_argv.size(), &uprefs.expanded_argv[0]);
392 }
393
394
395#ifdef DYNAMIC_LOG_WANTED
396 if (log_startup) Log_control(LID_startup, true);
397#endif // DYNAMIC_LOG_WANTED
398
399 init_1(argv[0], log_startup);
400
401 uprefs.read_config_file(true, log_startup); // in /etc/gnu-apl.d/
402 uprefs.read_config_file(false, log_startup); // in $HOME/.config/gnu_apl/
403 uprefs.read_threshold_file(true, log_startup); // dito parallel_thresholds
404 uprefs.read_threshold_file(false, log_startup); // dito parallel_thresholds
405
406 // NOTE: struct sigaction differs between GNU/Linux and other systems,
407 // which causes compile errors for direct curly bracket assignment on
408 // some systems.
409 //
410 // We therefore memset everything to 0 and then set the handler (which
411 // should compile on GNU/Linux and also on other systems.
412 //
413 memset(&new_control_C_action, 0, sizeof(struct sigaction));
414 memset(&new_WINCH_action, 0, sizeof(struct sigaction));
415 memset(&new_USR1_action, 0, sizeof(struct sigaction));
416 memset(&new_SEGV_action, 0, sizeof(struct sigaction));
417 memset(&new_TERM_action, 0, sizeof(struct sigaction));
418 memset(&new_HUP_action, 0, sizeof(struct sigaction));
419
420 new_control_C_action.sa_handler = &control_C;
421 new_WINCH_action .sa_handler = &signal_WINCH_handler;
422 new_USR1_action .sa_handler = &signal_USR1_handler;
423 new_SEGV_action .sa_handler = &signal_SEGV_handler;
424 new_TERM_action .sa_handler = &signal_TERM_handler;
425 new_HUP_action .sa_handler = &signal_HUP_handler;
426
427 sigaction(SIGINT, &new_control_C_action, &old_control_C_action);
428 sigaction(SIGUSR1, &new_USR1_action, &old_USR1_action);
429 sigaction(SIGSEGV, &new_SEGV_action, &old_SEGV_action);
430 sigaction(SIGTERM, &new_TERM_action, &old_TERM_action);
431 sigaction(SIGHUP, &new_HUP_action, &old_HUP_action);
432 signal(SIGCHLD, SIG_IGN); // do not create zombies
433 if (uprefs.WINCH_sets_pw)
434 {
435 sigaction(SIGWINCH, &new_WINCH_action, &old_WINCH_action);
436 signal_WINCH_handler(0); // pretend window size change
437 }
438 else
439 {
440 Workspace::set_PW(uprefs.initial_pw, LOC);
441 }
442
443#if PARALLEL_ENABLED
444 memset(&new_control_BSL_action, 0, sizeof(struct sigaction));
445 new_control_BSL_action.sa_handler = &control_BSL;
446 sigaction(SIGQUIT, &new_control_BSL_action, &old_control_BSL_action);
447#endif
448
449 uprefs.parse_argv_2(log_startup);
450
451 // maybe use TCP connection instead of stdin/stderr. This function blocks
452 // until a TCP connections was received.
453 //
454 remap_stdio();
455
456 if (uprefs.CPU_limit_secs)
457 {
458 rlimit rl;
459 getrlimit(RLIMIT_CPU, &rl);
460 rl.rlim_cur = uprefs.CPU_limit_secs;
461 setrlimit(RLIMIT_CPU, &rl);
462 }
463
464 if (uprefs.emacs_mode)
465 {
466 UCS_string info;
467 if (uprefs.emacs_arg)
468 {
469 info = NativeFunction::load_emacs_library(uprefs.emacs_arg);
470 }
471
472 if (info.size()) // problems loading library
473 {
474 CERR << info << endl;
475 }
476 else
477 {
478 // CIN = U+F00C0 = UTF8 F3 B0 83 80 ...
479 Output::color_CIN[0] = 0xF3;
480 Output::color_CIN[1] = 0xB0;
481 Output::color_CIN[2] = 0x83;
482 Output::color_CIN[3] = 0x80;
483 Output::color_CIN[4] = 0;
484
485 // COUT = U+F00C1 = UTF8 F3 B0 83 81 ...
486 Output::color_COUT[0] = 0xF3;
487 Output::color_COUT[1] = 0xB0;
488 Output::color_COUT[2] = 0x83;
489 Output::color_COUT[3] = 0x81;
490 Output::color_COUT[4] = 0;
491
492 // CERR = U+F00C2 = UTF8 F3 B0 83 82 ...
493 Output::color_CERR[0] = 0xF3;
494 Output::color_CERR[1] = 0xB0;
495 Output::color_CERR[2] = 0x83;
496 Output::color_CERR[3] = 0x82;
497 Output::color_CERR[4] = 0;
498
499 // UERR = U+F00C3 = UTF8 F3 B0 83 83 ...
500 Output::color_UERR[0] = 0xF3;
501 Output::color_UERR[1] = 0xB0;
502 Output::color_UERR[2] = 0x83;
503 Output::color_UERR[3] = 0x83;
504 Output::color_UERR[4] = 0;
505
506 // no clear_EOL
507 Output::clear_EOL[0] = 0;
508 }
509 }
510
511 if (uprefs.daemon)
512 {
513 const pid_t pid = fork();
514 if (pid) // parent
515 {
516 Log(LOG_startup)
517 CERR << "parent pid = " << getpid()
518 << " child pid = " << pid << endl;
519
520 exit(0); // parent returns
521 }
522
523 Log(LOG_startup)
524 CERR << "child forked (pid" << getpid() << ")" << endl;
525 }
526
527 if (uprefs.wait_ms) usleep(1000*uprefs.wait_ms);
528
529 init_2(log_startup);
530
531 if (!uprefs.silent) show_welcome(cout, argv[0]);
532
533 if (log_startup) CERR << "PID is " << getpid() << endl;
534 Log(LOG_argc_argv || log_startup) show_argv(argc, argv);
535
536 if (ProcessorID::init(log_startup))
537 {
538 // error message printed in ProcessorID::init()
539 return 8;
540 }
541
542 if (uprefs.do_Color) Output::toggle_color("ON");
543
544 if (uprefs.latent_expression.size())
545 {
546 // there was a --LX expression on the command line
547 //
548 UCS_string lx(uprefs.latent_expression);
549
550 if (log_startup)
551 CERR << "executing --LX '" << lx << "'" << endl;
552
553 Command::process_line(lx);
554 }
555
556 // maybe )LOAD the CONTINUE or SETUP workspace. Do that unless the user
557 // has given
558 //
559 // (1) --noCONT, or
560 // (2) --script (which implies --noCONT), or
561 // (3) -L wsname
562 //
563 if (uprefs.do_CONT && !uprefs.initial_workspace.size())
564 {
565 UCS_string cont("CONTINUE");
566 UTF8_string filename =
567 LibPaths::get_lib_filename(LIB0, cont, true, ".xml", ".apl");
568
569 if (access(filename.c_str(), F_OK) == 0)
570 {
571 // CONTINUE workspace exists and was not inhibited by --noCONT
572 //
573 UCS_string load_cmd(")LOAD CONTINUE");
574 Command::process_line(load_cmd);
575 return 0;
576 }
577
578 // no CONTINUE workspace but maybe SETUP
579 //
580 cont = UCS_string("SETUP");
581 filename =
582 LibPaths::get_lib_filename(LIB0, cont, true, ".xml", ".apl");
583
584 if (access(filename.c_str(), F_OK) == 0)
585 {
586 // SETUP workspace exists and was not inhibited by --noCONT
587 //
588 UCS_string load_cmd(")LOAD SETUP");
589 Command::process_line(load_cmd);
590 return 0;
591 }
592 }
593
594 if (uprefs.initial_workspace.size())
595 {
596 // the user has provided a workspace name via -L
597 //
598 UCS_string init_ws(uprefs.initial_workspace);
599 const char * cmd = uprefs.silent ? ")QLOAD " : ")LOAD ";
600 UCS_string load_cmd(cmd);
601 load_cmd.append(init_ws);
602 Command::process_line(load_cmd);
603 }
604
605 Quad_TZ::compute_offset();
606 return 0;
607}
608//----------------------------------------------------------------------------
610int
611main(int argc, const char *argv[])
612{
613 if (const int ret = init_apl(argc, argv)) return ret;
614
615 if (uprefs.eval_exprs.size())
616 {
617 loop(e, uprefs.eval_exprs.size())
618 {
619 const char * expr = uprefs.eval_exprs[e];
620 const UTF8_string expr_utf(expr);
621 UCS_string expr_ucs(expr_utf);
622 Command::process_line(expr_ucs);
623 }
624 Command::cmd_OFF(0);
625 return 0;
626 }
627
628#if HAVE_PTHREAD_SETNAME_NP
629 pthread_setname_np(pthread_self(), "apl/main");
630#endif
631
632 for (;;)
633 {
634 Token t = Workspace::immediate_execution(
635 IO_Files::test_mode == IO_Files::TM_EXIT_AFTER_FILE_ERROR);
636 if (t.get_tag() == TOK_OFF) Command::cmd_OFF(0);
637 }
638
639 return 0;
640}

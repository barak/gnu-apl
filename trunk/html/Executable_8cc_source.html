30//----------------------------------------------------------------------------
31Executable::Executable(const UCS_string & ucs, bool multi_line,
32 ParseMode pm, const char * loc)
33 : alloc_loc(loc),
34 pmode(pm),
35 refcount(0)
36{
37// { cerr << "Executable " << (void *)this << " created at " << loc << endl; }
38
39 // initialize text, stripping carriage returns and line feeds.
40 //
41 if (multi_line)
42 {
43 UCS_string line;
44
45 loop(t, ucs.size())
46 {
47 const Unicode uni = ucs[t];
48 switch (uni)
49 {
50 case UNI_CR: break;
51 case UNI_LF: text.push_back(line);
52 line.clear(); break;
53 default: line.append(uni);
54 }
55 }
56
57 if (line.size()) text.push_back(line);
58 }
59 else
60 {
61 text.push_back(ucs);
62 }
63}
64//----------------------------------------------------------------------------
65Executable::Executable(Fun_signature sig, int lambda_num,
66 const UCS_string & lambda_text, const char * loc)
67 : alloc_loc(loc),
68 pmode(PM_FUNCTION),
69 refcount(0)
70{
71UCS_string header = UserFunction_header::lambda_header(sig, lambda_num);
72
73 // remove local vars from lambda_text and add them to the header
74 //
75ShapeItem last_semi = -1;
76 for (ShapeItem t = lambda_text.size() - 1; t >= 0; --t)
77 {
78 const Unicode cc = lambda_text[t];
79 if (cc == UNI_SPACE) continue;
80 if (Avec::is_symbol_char(cc)) continue;
81 if (Avec::is_quad(cc)) continue;
82 if (cc == UNI_SEMICOLON) last_semi = t;
83 else break;
84 }
85
86 if (last_semi != -1)
87 {
88 for (ShapeItem t = last_semi; t < lambda_text.size(); ++t)
89 header.append(lambda_text[t]);
90 text.push_back(header);
91
92 UCS_string new_body(lambda_text);
93 new_body.resize(last_semi);
94 new_body.remove_trailing_whitespaces();
95 text.push_back(new_body);
96 }
97 else
98 {
99 text.push_back(header);
100 text.push_back(lambda_text);
101 }
102}

114//----------------------------------------------------------------------------
115void
116Executable::clear_body()
117{
118 loop(b, body.size())
119 {
120 body[b].extract_apl_val(LOC);
121
122 if (body[b].is_function())
123 {
124 Function_P fun = body[b].get_function();
125 const UserFunction * ufun = fun->get_func_ufun();
126 if (ufun && ufun->is_lambda())
127 const_cast<UserFunction *>(ufun)->decrement_refcount(LOC);
128 new (&body[b]) Token();
129 }
130 }
131
132 body.clear();
133}
134//----------------------------------------------------------------------------
135ErrorCode
136Executable::parse_body_line(Function_Line line, const UCS_string & ucs_line,
137 bool trace, bool tolerant, const char * loc,
138 bool macro)
139{
140 Log(LOG_UserFunction__set_line)
141 CERR << "[" << line << "]" << ucs_line << endl;
142
143Token_string in;
144const Parser parser(get_parse_mode(), loc, macro);
145ErrorCode ec = parser.parse(ucs_line, in);
146 if (ec)
147 {
148 if (tolerant) return ec;
149
150 if (ec == E_NO_TOKEN)
151 {
152 Error error(ec, LOC);
153 throw error;
154 }
155
156 if (ec != E_NO_ERROR) Error::throw_parse_error(ec, LOC, LOC);
157 }
158
159 return parse_body_line(line, in, trace, tolerant, loc);
160}
161//----------------------------------------------------------------------------
162ErrorCode
163Executable::parse_body_line(Function_Line line, const Token_string & input,
164 bool trace, bool tolerant, const char * loc)
165{
166ShapeItem idx = 0;
167const ShapeItem end = input.size();
168
169 // handle labels (if any)
170 //
171 if (get_parse_mode() == PM_FUNCTION &&
172 end > 1 &&
173 input[0].get_tag() == TOK_SYMBOL &&
174 input[1].get_tag() == TOK_COLON)
175 {
176 Token tok_sym = input[0]; // get the label symbol
177 idx = 2; // skip the :
178
179 UserFunction * ufun = get_exec_ufun();
180 Assert(ufun);
181 ufun->add_label(tok_sym.get_sym_ptr(), line);
182 }
183
184Token_string out;
185 while ((end - idx))
186 {
187 // find next diamond. If there is one, copy statement in reverse order,
188 // append TOK_ENDL, and continue.
189
190 // 1. determine statement length
191 //
192 ShapeItem stat_len = 0; // statement length, not counting ◊
193 int diamond = 0;
194 loop(t, (end - idx))
195 {
196 if (input[idx + t].get_tag() == TOK_DIAMOND)
197 {
198 diamond = 1;
199 break;
200 }
201 else
202 {
203 ++stat_len;
204 }
205 }
206
207 // 2. check for bad token and append statement to out
208 //
209 loop(s, stat_len)
210 {
211 const Token & tok = input[idx + stat_len - s - 1];
212 if (tok.get_Class() >= TC_MAX_PERM &&
213 tok.get_tag() != TOK_L_CURLY &&
214 tok.get_tag() != TOK_R_CURLY)
215 {
216 if (tolerant) return E_SYNTAX_ERROR;
217
218 CERR << "Line " << line << endl
219 << "Offending token: (tag > TC_MAX_PERM) "
220 << tok.get_tag() << " " << tok << endl
221 << "Statement: ";
222 loop(t, input.size()) CERR << "`" << input[t] << " ";
223 CERR << endl;
224 SYNTAX_ERROR;
225 }
226 out.push_back(tok);
227 }
228 idx += stat_len + diamond; // skip statement and maybe ◊
229
230 // 3. maybe add a TOK_END or TOK_ENDL
231 //
232 if (stat_len && (diamond || get_parse_mode() != PM_EXECUTE))
233 {
234 const int64_t tr = trace ? 1 : 0;
235 out.push_back(Token(TOK_END, tr));
236 }
237 }
238
239 // replace the trailing TOK_END (if any) into TOK_ENDL
240 //
241 if (out.size() && out.back().get_tag() == TOK_END)
242 out.back().ChangeTag(TOK_ENDL);
243
244 Log(LOG_UserFunction__set_line)
245 {
246 CERR << "[final line " << line << "] ";
247 out.print(CERR, false);
248 }
249
250 loop(t, out.size()) body.push_back(out[t]);
251 return E_NO_ERROR;
252
253#undef get
254}

284//----------------------------------------------------------------------------
285UCS_string
286Executable::statement_text(Function_PC pc) const
287{
288 if (pc >= Function_PC(body.size()))
289 {
290 Assert(pc == Function_PC(body.size()));
291 pc = Function_PC(body.size() - 1);
292 }
293
294 if (pc > 0 && body[pc].get_Class() == TC_RETURN) pc = Function_PC(pc - 1);
295 if (pc > 0 && body[pc].get_Class() == TC_END) pc = Function_PC(pc - 1);
296
297const Function_Line line = get_line(pc);
298const UCS_string & line_txt = text[line];
299
300 // count statement ends before pc
301 //
302int stats_before = 0;
303 for (Function_PC p = line_start(line); p < pc; ++p)
304 {
305 // a TOK_STOP_LINE is immediately followd by a TC_END, but the
306 // TC_END has no ◊ in line_txt. We therefore discount stats_before.
307 //
308 if (body[p].get_tag() == TOK_STOP_LINE) --stats_before;
309
310 if (body[p].get_Class() == TC_END) ++stats_before;
311 else if (body[p].get_tag() == TOK_RETURN_EXEC) ++stats_before;
312 }
313
314int tidx = 0;
315 while (stats_before)
316 {
317 Assert(tidx < line_txt.size());
318 if (Avec::is_diamond(line_txt[tidx++])) --stats_before;
319 }
320
321 // skip leading spaces
322 //
323 while (line_txt[tidx] == ' ') ++tidx;
324
325UCS_string ret;
326 while (tidx < line_txt.size())
327 {
328 if (Avec::is_diamond(line_txt[tidx])) break;
329 ret.append(line_txt[tidx++]);
330 }
331
332 // skip trailing spaces
333 //
334 while (ret.size() && ret.back() <= ' ') ret.pop_back();
335
336 return ret;
337}
338//----------------------------------------------------------------------------
339void
340Executable::set_error_info(Error & error, Function_PC2 pc_from_to) const
341{
342const ErrorCode ec = error.get_error_code();
343UCS_string message_2(UTF8_string(error.get_error_line_2()));
344
345 // for value errors we point to the failed symbol itself.
346 //
347 if (ec == E_VALUE_ERROR) pc_from_to.low = pc_from_to.high;
348
349 Log(LOG_prefix__location_info)
350 {
351 Q1(body.size())
352 Q1(pc_from_to.low)
353 Q1(pc_from_to.high)
354 }
355
356 if (body[pc_from_to.high].get_Class() == TC_RETURN &&
357 pc_from_to.high > pc_from_to.low)
358 pc_from_to.high = Function_PC(pc_from_to.high - 1);
359
360/*
361 The statement is stored in reverse order in the body:
362
363 Body:
364 start ... pc_low ... pc_high ... end
365 ^ ^
366 | |
367 +--- failed action ---+
368
369 Display:
370 end ... pc_high ... pc_low ... start ◊ next statement
371 ^ ^
372 | |
373 +--- failed action ---+
374 */
375
376const Function_PC start = get_statement_start(pc_from_to.low);
377Function_PC end = start;
378 while (body[end].get_Class() != TC_END &&
379 body[end].get_tag() != TOK_RETURN_EXEC) ++end;
380
381 Assert(start <= pc_from_to.low);
382 Assert(pc_from_to.low <= pc_from_to.high);
383 if (pc_from_to.high > end) pc_from_to.high = end;
384
385 // if the error was caused by some function being executed then
386 // the right arg of the function was OK and we skip it
387 //
388 if (body[pc_from_to.low].get_Class() == TC_VALUE)
389 {
390 ++pc_from_to.low;
391 }
392
393 Log(LOG_prefix__location_info)
394 {
395 Q1(start)
396 Q1(pc_from_to.low)
397 Q1(pc_from_to.high)
398 Q1(end)
399 }
400
401 // Line 2: statement
402 //
403int len_left = 0; // characters before the first caret
404int len_between = 0; // distance between the carets
405 for (int q = end - 1; q >= start; --q)
406 {
407 // avoid duplicate ∘ in ∘.f
408 //
409 if (body[q].get_tag() == TOK_JOT && q > 0 &&
410 body[q - 1].get_tag() == TOK_OPER2_OUTER) continue;
411
412 // Note: Token::error_info returns -len if it inserts a space.
413 // Such an inserted space counts for the previous token.
414 // The previous token is q + 1 since we count down.
415 //
416 int len = body[q].error_info(message_2);
417
418 if (len < 0) // space inserted, len is negative
419 {
420 if ((q + 1) > pc_from_to.high) len_left++;
421 else if ((q + 1) > pc_from_to.low) len_between++;
422 len = -len;
423 }
424
425 if (q > pc_from_to.high) len_left += len;
426 else if (q > pc_from_to.low) len_between += len;
427 }
428
429 {
430 UTF8_string utf(message_2);
431 error.set_error_line_2(utf.c_str());
432 }
433
434 // Line 3: carets
435 //
436 error.set_left_caret(error.get_left_caret() + len_left);
437
438 if (pc_from_to.high != pc_from_to.low) // two carets
439 {
440 error.set_right_caret(error.get_left_caret() + len_between);
441 }
442}
443//----------------------------------------------------------------------------
444Function_PC
445Executable::get_statement_start(int pc) const
446{
447 // this function is used in error reporting so it should
448 // not Assert() and the like to avoid infinite recursion.
449
450 if (pc >= int(body.size())) pc = body.size() - 1;
451
452 // if we are at the end of the statement, move back.
453 //
454 if (pc < 2) return Function_PC_0;
455 if (body[pc].get_Class() == TC_RETURN) --pc;
456 if (body[pc].get_Class() == TC_END) --pc;
457
458 for (; pc > 0; --pc)
459 {
460 if (body[pc].get_Class() == TC_END) return Function_PC(pc + 1);
461 if (body[pc].get_tag() == TOK_RETURN_EXEC) return Function_PC(pc + 1);
462 }
463
464 return Function_PC_0;
465}
466//----------------------------------------------------------------------------
467void
468Executable::unmark_all_values() const
469{
470 loop (b, body.size())
471 {
472 const Token & tok = body[b];
473 if (tok.get_ValueType() == TV_VAL)
474 {
475 Value_P value = tok.get_apl_val();
476 if (+value) value->unmark();
477 }
478
479 if (tok.get_ValueType() == TV_FUN)
480 {
481 Function_P fun = tok.get_function();
482 const UserFunction * ufun = fun->get_func_ufun();
483 if (ufun && ufun->is_lambda())
484 {
485 ufun->unmark_all_values();
486 }
487 }
488 }
489}

510//----------------------------------------------------------------------------
511void
512Executable::setup_lambdas()
513{
514 // quick check if this body contains any lambdas. Parser::match_par_bra()
515 // should have checked for unbalanced { } so we simply look for } (which
516 // is the first to occur since the body is reversed.
517 //
518bool have_curly = false;
519 loop(b, body.size())
520 {
521 if (body[b].get_tag() == TOK_R_CURLY)
522 {
523 have_curly = true;
524 break;
525 }
526 }
527
528 if (!have_curly) return; // no { ... } in this body
529
530 // undo the token reversion of the body so that the body token and the
531 // function text run in the same direction.
532 //
533 reverse_statement_token(body);
534
535int lambda_num = 0;
536 loop(b, body.size())
537 {
538 if (body[b].get_tag() != TOK_L_CURLY) continue; // not {
539
540 ShapeItem end = -1;
541 int level = 1; // since body[b] is {
542 for (ShapeItem b1 = b + 1; b1 < ShapeItem(body.size()); ++b1)
543 {
544 switch(body[b1].get_tag())
545 {
546 default: continue;
547 case TOK_L_CURLY: ++level; continue;
548 case TOK_R_CURLY: if (--level) continue;
549 end = b1;
550 b1 = body.size();
551 break;
552 case TOK_END: MORE_ERROR() << "Illegal ◊ in {}";
553 DEFN_ERROR;
554 }
555 break; // for b1...
556 }
557 Assert(end != -1);
558 b = setup_one_lambda(b, end, ++lambda_num) - 1; // -1 due to ++b in loop(b) above
559 }
560
561 // redo the token reversion of the body so that the body token and the
562 // text run in the same direction.
563 //
564 reverse_statement_token(body);
565
566 Parser::match_par_bra(body, true);
567 adjust_line_starts();
568 Parser::remove_void_token(body); // do this AFTER adjust_line_starts() !!!
569}
570//----------------------------------------------------------------------------
571ShapeItem
572Executable::setup_one_lambda(ShapeItem b, ShapeItem bend, int lambda_num)
573{
574 Assert(body[bend].get_tag() == TOK_R_CURLY);
575
576 body[b++].clear(LOC); // invalidate {
577 body[bend].clear(LOC); // invalidate }
578
579Token_string lambda_body;
580
581const Fun_signature signature =
582 compute_lambda_body(lambda_body, b, bend);
583
584 // at this point { ... } was copied from body to lambda_body and
585 // bend is at the (now invalidated) { token.
586 //
587const UCS_string lambda_text = extract_lambda_text(signature, lambda_num - 1);
588
589 reverse_statement_token(lambda_body);
590 reverse_all_token(lambda_body);
591
592#if 0 // not yet working
593
594UCS_string ufun_text = UserFunction_header::lambda_header(signature,lambda_num);
595 ufun_text.append(UNI_LF);
596 ufun_text.append(lambda_text);
597 ufun_text.append(UNI_LF);
598
599Q1(ufun_text)
600
601int error_line = -1;
602const char * error_loc = 0;
603const UTF8_string creator("Executable::setup_one_lambda()");
604
605UserFunction * ufun = new UserFunction(ufun_text, LOC, creator, false, false);
606
607 ufun->increment_refcount(LOC);
608
609#else
610
611UserFunction * ufun = new UserFunction(signature, lambda_num,
612 lambda_text, lambda_body);
613 ufun->increment_refcount(LOC);
614
615#endif
616 // put a token for the lambda at the place where the { was.
617 // That replaces, for example, (in forward notation):
618 //
619 // A←{ ... } by:
620 // A←UFUN UFUN being a user-defined function with body { ... }
621 //
622 body[bend].move_2(ufun->get_token(), LOC);
623
624 return bend + 1;
625}
626//----------------------------------------------------------------------------
627Fun_signature
628Executable::compute_lambda_body(Token_string & lambda_body,
629 ShapeItem b, ShapeItem bend)
630{
631int signature = SIG_FUN;
632int level = 0;
633
634 for (; b < bend; ++b)
635 {
636 Token t;
637 t.move_1(body[b], LOC);
638 body[b].clear(LOC); // invalidate in main body
639
640 // figure the signature by looking for ⍺, ⍶, ⍵, ⍹, and χ
641 // and complain about ◊ and →
642 switch(t.get_tag())
643 {
644 case TOK_ALPHA: if (!level) signature |= SIG_A; // no break
645 case TOK_OMEGA: if (!level) signature |= SIG_B; break;
646 case TOK_CHI: if (!level) signature |= SIG_X; break;
647 case TOK_OMEGA_U: if (!level) signature |= SIG_RO; // no break
648 case TOK_ALPHA_U: if (!level) signature |= SIG_LO; break;
649
650 case TOK_DIAMOND: DEFN_ERROR;
651 case TOK_BRANCH: DEFN_ERROR;
652 case TOK_ESCAPE: DEFN_ERROR;
653
654 case TOK_L_CURLY: ++level; break;
655 case TOK_R_CURLY: --level; break;
656 default: break;
657 }
658
659 if (lambda_body.size() == 0)
660 {
661 // first token: prepend λ ← tokens
662 //
663 Symbol * sym_Z = &Workspace::get_v_LAMBDA();
664 lambda_body.push_back(Token(TOK_LAMBDA, sym_Z));
665 lambda_body.push_back(Token(TOK_ASSIGN1));
666
667 signature |= SIG_Z;
668 }
669
670 lambda_body.push_back(t);
671 }
672
673 if ((signature & SIG_B) == 0 && // niladic
674 (signature & (SIG_A | SIG_LO | SIG_RO | SIG_X)))
675 {
676 MORE_ERROR() <<
677 "niladic lambda with axis. left argument, or function argument(s)";
678 DEFN_ERROR;
679 }
680
681 if ((signature & SIG_LO) && (signature & SIG_X)) // dyadic oper with axis
682 {
683 MORE_ERROR() << "invalid lambda operator (dyadic with axis)";
684 DEFN_ERROR;
685 }
686
687 // if the lambda has at least one token, then it is supposedly returns λ.
688 // Otherwise the lambda is empty (and result-less)
689 //
690 if (signature & SIG_Z)
691 {
692 const int64_t trace = 0;
693 lambda_body.push_back(Token(TOK_ENDL, trace));
694
695 Token ret_lambda(TOK_RETURN_SYMBOL, &Workspace::get_v_LAMBDA());
696 lambda_body.push_back(ret_lambda);
697 }
698 else
699 {
700 Token ret_void(TOK_RETURN_VOID);
701 lambda_body.push_back(ret_void);
702 }
703
704 return Fun_signature(signature);
705}
706//----------------------------------------------------------------------------
707UCS_string
708Executable::extract_lambda_text(Fun_signature signature, int skip) const
709{
710UCS_string lambda_text;
711 if (signature & SIG_Z) lambda_text.append_UTF8("λ←");
712
713int level = 0; // {/} nesting level
714bool copying = false;
715size_t tidx = 0; // the current line in text[]
716size_t tcol = 0; // the current column in text[tidx];
717bool in_single_quotes = false;
718bool in_double_quotes = false;
719
720 // skip over the first skip lambdas and copy the next one to lambda_text
721 //
722 for (;;)
723 {
724 if (tidx >= text.size())
725 {
726 Q1(copying) Q1(tidx) Q1(skip) FIXME;
727 }
728
729 const UCS_string & line = text[tidx];
730
731 if (tcol >= size_t(line.size())) // end of line: wrap to next line
732 {
733 ++tidx; // next line
734 tcol = 0; // first column
735 in_single_quotes = false;
736 in_double_quotes = false;
737 continue;
738 }
739
740 const Unicode uni = line[tcol++];
741 if (in_single_quotes)
742 {
743 if (uni == UNI_SINGLE_QUOTE) in_single_quotes = false;
744 if (copying) lambda_text.append(uni);
745 continue;
746 }
747
748 if (in_double_quotes)
749 {
750 if (uni == UNI_DOUBLE_QUOTE // either \" or string end
751 && !(tcol >= 2 && line[tcol - 2] == UNI_BACKSLASH))
752 in_double_quotes = false;
753 if (copying) lambda_text.append(uni);
754 continue;
755 }
756
757 // at this point uni is outside strings
758 //
759 if (uni == UNI_COMMENT || uni == UNI_NUMBER_SIGN)
760 {
761 ++tidx;
762 tcol = 0;
763 in_single_quotes = false;
764 in_double_quotes = false;
765 continue;
766 }
767
768 if (copying) lambda_text.append(uni);
769
770 switch(uni)
771 {
772 case UNI_SINGLE_QUOTE: // start of a new '...' string
773 in_single_quotes = true;
774 continue;
775
776 case UNI_DOUBLE_QUOTE: // start of a new "..." string
777 in_double_quotes = true;
778 continue;
779
780 case UNI_L_CURLY: // start of a new { ... }
781 if (level++ == 0 && skip == 0) copying = true;
782 continue;
783

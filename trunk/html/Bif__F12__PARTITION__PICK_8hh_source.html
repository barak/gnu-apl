30class Bif_F12_PARTITION : public NonscalarFunction_default_identity
31{
32public:
34 Bif_F12_PARTITION()
35 : NonscalarFunction_default_identity(TOK_F12_PARTITION)
36 {}
37
39 virtual Token eval_B(Value_P B) const
40 { return Token(TOK_APL_VALUE1, do_eval_B(B)); }
41
43 static Value_P do_eval_B(Value_P B);
44
46 virtual Token eval_AB(Value_P A, Value_P B) const
47 { return Token(TOK_APL_VALUE1, partition(A, B, B->get_rank() - 1)); }
48
50 virtual Token eval_XB(Value_P X, Value_P B) const
51 {
52 X->to_bitmap("⊂[X] B", B->get_rank()); // check X
53 const Shape shape_X = Value::to_shape(X.get());
54 return Token(TOK_APL_VALUE1, enclose_with_axes(shape_X, B));
55 }
56
58 static Value_P do_eval_XB(Value_P X, Value_P B);
59
61 virtual Token eval_AXB(Value_P A, Value_P X, Value_P B) const;
62
63 static Bif_F12_PARTITION * fun;
64 static Bif_F12_PARTITION _fun;
65
67 static Value_P enclose_with_axes(const Shape & shape_X, Value_P B);
68
69protected:
71 struct Partition
72 {
73 ShapeItem start;
74 ShapeItem end;
75
77 ShapeItem length() const { return end - start; }
78 };
79
81 static Token enclose_with_axis(Value_P B, Value_P X);
82
84 static Value_P partition(Value_P A, Value_P B, sAxis axis);
85};

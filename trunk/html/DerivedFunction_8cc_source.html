126//----------------------------------------------------------------------------
127Token
128DerivedFunction::eval_B(Value_P B) const
129{
130 Log(LOG_FunOperX)
131 {
132 print(CERR << "entering DerivedFunction");
133 CERR << "::eval_B() , this = "
134 << voidP(this) << endl;
135 }
136
137 if (left_arg.get_tag() == TOK_VOID) // function bound to axis
138 {
139 return oper->eval_XB(axis, B);
140 }
141
142 if (right_fun.get_tag() != TOK_VOID) // dyadic operator
143 {
144 Token & left = const_cast<Token &>(left_arg);
145 Token & right = const_cast<Token &>(right_fun);
146 return oper->eval_LRB(left, right, B);
147 }
148 else // monadic operator
149 {
150 if (left_arg.is_function()) // normal operator
151 {
152 Token & left = const_cast<Token &>(left_arg);
153 if (!axis) return oper->eval_LB(left, B);
154 else return oper->eval_LXB(left, axis, B);
155 }
156 else // operator with left value operand
157 {
158 Value_P A = left_arg.get_apl_val();
159 if (!axis) return oper->eval_AB(A, B);
160 else return oper->eval_AXB(A, axis, B);
161 }
162 }
163}

165Token
166DerivedFunction::eval_XB(Value_P X, Value_P B) const
167{
168 Log(LOG_FunOperX)
169 {
170 print(CERR << "entering DerivedFunction");
171 CERR << "::eval_XB() , this = "
172 << voidP(this) << endl;
173 }
174
175 if (right_fun.get_tag() != TOK_VOID) // dyadic operator
176 {
177 Token & left = const_cast<Token &>(left_arg);
178 Token & right = const_cast<Token &>(right_fun);
179 return oper->eval_LRXB(left, right, X, B);
180 }
181 else // monadic operator
182 {
183 if (left_arg.is_function()) // normal operator
184 {
185 Token & left = const_cast<Token &>(left_arg);
186 if (!axis) return oper->eval_LB(left, B);
187 else return oper->eval_LXB(left, axis, B);
188 }
189 else // operator with left value operand
190 {
191 Value_P A = left_arg.get_apl_val();
192 if (!axis) return oper->eval_AB(A, B);
193 else return oper->eval_AXB(A, axis, B);
194 }
195 }

197//----------------------------------------------------------------------------
198Token
199DerivedFunction::eval_AB(Value_P A, Value_P B) const
200{
201 Log(LOG_FunOperX)
202 {
203 print(CERR << "entering DerivedFunction");
204 CERR << "::eval_AB()" << endl;
205 }
206
207 if (left_arg.get_tag() == TOK_VOID) // function bound to axis
208 {
209 return oper->eval_AXB(A, axis, B);
210 }
211
212 if (right_fun.get_tag() != TOK_VOID) // dyadic operator
213 {
214 Token & left = const_cast<Token &>(left_arg);
215 Token & right = const_cast<Token &>(right_fun);
216 if (!axis) return oper->eval_ALRB(A, left, right, B);
217 else return oper->eval_ALRXB(A, left, right, axis, B);
218 }
219 else // monadic operator
220 {
221 if (left_arg.is_function())
222 {
223 Token & left = const_cast<Token &>(left_arg);
224 if (!axis) return oper->eval_ALB(A, left, B);
225 else return oper->eval_ALXB(A, left, axis, B);
226 }
227 else
228 {
229 Value_P A = left_arg.get_apl_val();
230 if (!axis) return oper->eval_AB(A, B);
231 else return oper->eval_AXB(A, axis, B);
232 }
233 }
234}
235//----------------------------------------------------------------------------
236Token
237DerivedFunction::eval_AXB(Value_P A, Value_P X, Value_P B) const
238{
239 Log(LOG_FunOperX)
240 {
241 print(CERR << "entering DerivedFunction");
242 CERR << "::eval_AXB()" << endl;
243 }
244
245 if (right_fun.get_tag() != TOK_VOID) // dyadic operator
246 {
247 Token & left = const_cast<Token &>(left_arg);
248 Token & right = const_cast<Token &>(right_fun);
249 return oper->eval_ALRXB(A, left, right, X, B);
250 }
251 else // monadic operator
252 {
253 Token & left = const_cast<Token &>(left_arg);
254 return oper->eval_ALXB(A, left, X, B);
255 }

257//----------------------------------------------------------------------------
258ostream &
259DerivedFunction::print(ostream & out) const
260{
261 out << "(";
262 if (left_arg.is_function()) left_arg.get_function()->print(out);
263 else out << "VAL";
264 out << " ";
265
266 oper->print(out);
267 if (+axis) out << "[]";
268
269 if (right_fun.get_tag() != TOK_VOID) // dyadic operator
270 {
271 out << " ";
272 if (right_fun.is_function()) right_fun.get_function()->print(out);
273 else out << "VAL";
274 }
275
276 return out << ")";
277}

280DerivedFunction::has_result() const

284 return true; // operator bound to left value

339//----------------------------------------------------------------------------
340void
341DerivedFunctionCache::reset()
342{
343 while (idx)
344 {
345 --idx; // back to last item
346 get(LOC)->destroy_derived(LOC);
347 --idx; // undo idx++ by get(LOC)
348 }
349
350 Log(LOG_FunOperX)
351 {
352 CERR << "DerivedFunctionCache reset, cache at "
353 << voidP(cache) << "..."
354 << voidP(cache + MAX_FUN_OPER)
355 << endl;
356 }
357}
358//----------------------------------------------------------------------------
359DerivedFunction *
360DerivedFunctionCache::get(const char * loc)
361{
362 if (idx >= MAX_FUN_OPER) LIMIT_ERROR_FUNOPER;
363
364 Log(LOG_FunOperX)
365 {
366 CERR << "DerivedFunctionCache get( " << idx << " ), cache at "
367 << voidP(cache) << "..."
368 << voidP(cache + MAX_FUN_OPER)
369 << " at " << loc << endl;
370 }
371
372 return reinterpret_cast<DerivedFunction *>
373 (cache + idx++*sizeof(DerivedFunction));
374}

32/* from RFC 3629 / STD 63:
33
34 Char. number range | UTF-8 octet sequence
35 (hexadecimal) | (binary)
36 --------------------+---------------------------------------------
37 0000 0000-0000 007F | 0xxxxxxx
38 0000 0080-0000 07FF | 110xxxxx 10xxxxxx
39 0000 0800-0000 FFFF | 1110xxxx 10xxxxxx 10xxxxxx
40 0001 0000-0010 FFFF | 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx
41
42 RFC 2279, the predecessor of RFC 3629, also allowed these:
43
44 0000 0000-0000 007F 0xxxxxxx
45 0000 0080-0000 07FF 110xxxxx 10xxxxxx
46 0000 0800-0000 FFFF 1110xxxx 10xxxxxx 10xxxxxx
47 0001 0000-001F FFFF 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx
48 0020 0000-03FF FFFF 111110xx 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx
49 0400 0000-7FFF FFFF 1111110x 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx 10xxxxxx
50
51 In order to be more general, we follow RFC 2279 and also allow 5-byte and
52 6-byte encodings
53*/
54
55//----------------------------------------------------------------------------
56UTF8_string::UTF8_string(const UCS_string & ucs)
57{
58 Log(LOG_char_conversion)
59 CERR << "UTF8_string::UTF8_string(ucs = " << ucs << ")" << endl;
60
61 loop(i, ucs.size())
62 {
63 int uni = ucs[i];
64 if (uni < 0x80) // 1-byte unicode (ASCII)
65 {
66 *this += uni;
67 }
68 else if (uni < 0x800) // 2-byte unicode
69 {
70 const uint8_t b1 = uni & 0x3F; uni >>= 6;
71 *this += uni | 0xC0;
72 *this += b1 | 0x80;
73 }
74 else if (uni < 0x10000) // 3-byte unicode
75 {
76 const uint8_t b2 = uni & 0x3F; uni >>= 6;
77 const uint8_t b1 = uni & 0x3F; uni >>= 6;
78 *this += uni | 0xE0;
79 *this += b1 | 0x80;
80 *this += b2 | 0x80;
81 }
82 else if (uni < 0x200000) // 4-byte unicode
83 {
84 const uint8_t b3 = uni & 0x3F; uni >>= 6;
85 const uint8_t b2 = uni & 0x3F; uni >>= 6;
86 const uint8_t b1 = uni & 0x3F; uni >>= 6;
87 *this += uni | 0xF0;
88 *this += b1 | 0x80;
89 *this += b2 | 0x80;
90 *this += b3 | 0x80;
91 }
92 else if (uni < 0x4000000) // 5-byte unicode
93 {
94 const uint8_t b4 = uni & 0x3F; uni >>= 6;
95 const uint8_t b3 = uni & 0x3F; uni >>= 6;
96 const uint8_t b2 = uni & 0x3F; uni >>= 6;
97 const uint8_t b1 = uni & 0x3F; uni >>= 6;
98 *this += uni | 0xF8;
99 *this += b1 | 0x80;
100 *this += b2 | 0x80;
101 *this += b3 | 0x80;
102 *this += b4 | 0x80;
103 }
104 else if (size_t(uni) < 0x80000000) // 6-byte unicode
105 {
106 const uint8_t b5 = uni & 0x3F; uni >>= 6;
107 const uint8_t b4 = uni & 0x3F; uni >>= 6;
108 const uint8_t b3 = uni & 0x3F; uni >>= 6;
109 const uint8_t b2 = uni & 0x3F; uni >>= 6;
110 const uint8_t b1 = uni & 0x3F; uni >>= 6;
111 *this += uni | 0xFC;
112 *this += b1 | 0x80;
113 *this += b2 | 0x80;
114 *this += b3 | 0x80;
115 *this += b4 | 0x80;
116 *this += b5 | 0x80;
117 }
118 else
119 {
120 CERR << "Bad Unicode: " << UNI(uni) << endl
121 << "The offending ucs string is:";
122 loop(ii, ucs.size()) CERR << " " << HEX(ucs[ii]);
123 CERR << endl;
124
125 BACKTRACE
126 Assert(0 && "Error in UTF8_string::UTF8_string(ucs)");
127 }
128 }
129
130 Log(LOG_char_conversion)
131 CERR << "UTF8_string::UTF8_string(): utf = " << *this << endl;
132}

148Unicode
149UTF8_string::toUni(const UTF8 * string, int & len, bool verbose)
150{
151const uint32_t b0 = *string++;
152 if (b0 < 0x80) { len = 1; return Unicode(b0); }
153
154uint32_t bx = b0; // the "significant" bits in b0
155 if ((b0 & 0xE0) == 0xC0) { len = 2; bx &= 0x1F; }
156 else if ((b0 & 0xF0) == 0xE0) { len = 3; bx &= 0x0F; }
157 else if ((b0 & 0xF8) == 0xF0) { len = 4; bx &= 0x0E; }
158 else if ((b0 & 0xFC) == 0xF8) { len = 5; bx &= 0x0E; }
159 else if ((b0 & 0xFE) == 0xFC) { len = 6; bx &= 0x0E; }
160 else if (verbose)
161 {
162 CERR << "Bad UTF8 sequence: " << HEX(b0);
163 loop(j, 6)
164 {
165 const uint32_t bx = string[j];
166 if (bx & 0x80) CERR << " " << HEX(bx);
167 else break;
168 }
169
170 CERR << " at " LOC << endl;
171
172 BACKTRACE
173 Assert(0 && "Internal error in UTF8_string::toUni()");
174 }
175 else
176 {
177 len = 0;
178 return Invalid_Unicode;
179 }
180
181uint32_t uni = 0;
182 loop(l, len - 1)
183 {
184 const UTF8 subc = *string++;
185 if ((subc & 0xC0) != 0x80)
186 {
187 CERR << "Bad UTF8 sequence: " << HEX(b0) << "... at " LOC << endl;
188 Assert(0 && "Internal error in UTF8_string::toUni()");
189 }
190
191 bx <<= 6;
192 uni <<= 6;
193 uni |= subc & 0x3F;
194 }
195
196 return Unicode(bx | uni);
197}
198//----------------------------------------------------------------------------
199Unicode
200UTF8_string::getc(istream & in)
201{
202const uint32_t b0 = in.get() & 0xFF;
203 if (!in.good()) return Invalid_Unicode;
204 if (b0 < 0x80) { return Unicode(b0); }
205
206uint32_t bx = b0; // the "significant" bits in b0
207int len;
208
209 if ((b0 & 0xE0) == 0xC0) { len = 1; bx &= 0x1F; }
210 else if ((b0 & 0xF0) == 0xE0) { len = 2; bx &= 0x0F; }
211 else if ((b0 & 0xF8) == 0xF0) { len = 3; bx &= 0x0E; }
212 else
213 {
214 CERR << "Bad UTF8 sequence: " << HEX(b0) << "... at " LOC << endl;
215 BACKTRACE
216 return Invalid_Unicode;
217 }
218
219char cc[4];
220 in.get(cc, len + 1); // read subsequent characters + terminating 0
221 if (!in.good()) return Invalid_Unicode;
222
223uint32_t uni = 0;
224 loop(l, len)
225 {
226 const UTF8 subc = cc[l];
227 if ((subc & 0xC0) != 0x80)
228 {
229 CERR << "Bad UTF8 sequence: " << HEX(b0) << "... at " LOC << endl;
230 return Invalid_Unicode;
231 }
232
233 bx <<= 6;
234 uni <<= 6;
235 uni |= subc & 0x3F;
236 }
237
238 return Unicode(bx | uni);
239}

241bool
242UTF8_string::starts_with(const char * path) const
243{
244 if (path == 0) return false; // no path provided
245
246const size_t path_len = strlen(path);
247 if (path_len > size()) return false; // path_len longer than this string
248
249 // can't use strncmp() because this string may not be 0-terminated
250 //
251 loop(p, path_len) if (at(p) != path[p]) return false;
252 return true;
253}

261 if (ext_len > size()) return false; // ext longer than this string
262

264 //
265 loop(e, ext_len) if (at(size() - ext_len + e) != ext[e]) return false;
266 return true;

269int
270UTF8_string::un_HTML(int in_HTML)
271{
272int dest = 0;
273bool got_tag = false;
274 loop(src, size())
275 {
276 const char cc = at(src);
277 if (in_HTML == 2) // inside HTML tag, i.e. < seen
278 {
279 if (cc == '>') in_HTML = 1; // in HTML but not in HTML tag
280 continue;
281 }
282
283 if (cc == '<') // start of HTML tag
284 {
285 in_HTML = 2; // now inside HTML tag
286 got_tag = true;
287 continue;
288 }
289
290 if (cc != '&') // unless HTML-escaped char
291 {
292 at(dest++) = cc;
293 continue;
294 }
295
296 // at this point cc == '&' which is the start of an HTML-escaped
297 // character. This can be:
298 //
299 // &NN; with decimal digits NN..., or
300 // &#XX; with hexadecimal digits XX..., or (incomplete list)
301 // &gt; for >, or
302 // &lt; for <
303 //
304 // There exist more HTML escapes, but the function producing them
305 // (i.e. UCS_string::to_HTML()) does not emit them.
306 //
307 const int rest = size() - src;
308 if (rest > 3 && at(src + 1) == '#' &&
309 strchr("0123456789", at(src + 2)))
310 {
311 src += 2; // skip "&#"
312 int val = 0;
313 char * end = 0;
314 if (at(src) == 'x') // hex value
315 {
316 ++src; // skip "x"
317 val = strtoll(c_str() + src, &end, 16);
318 }
319 else // decimal
320 {
321 val = strtoll(c_str() + src, &end, 10);
322 }
323 at(dest++) = val;
324 src = end - c_str(); // skip hex or decimal digits
325 // ";" skipped by loop()
326 }
327 else if (rest > 3 && at(src + 1) == 'g' &&
328 at(src + 2) == 't' &&
329 at(src + 3) == ';')
330 {
331 at(dest++) = '>';
332 src += 3; // skip "gt;"
333 }
334 else if (rest > 3 && at(src + 1) == 'l' &&
335 at(src + 2) == 't' &&
336 at(src + 3) == ';')
337 {
338 at(dest++) = '<';
339 src += 3; // skip "lt;"
340 }
341 else if (rest > 4 && at(src + 1) == 'n' &&
342 at(src + 2) == 'b' &&
343 at(src + 3) == 's' &&
344 at(src + 4) == 'p' &&
345 at(src + 5) == ';')
346 {
347 at(dest++) = ' ';
348 src += 5; // skip "nbsp;"
349 }
350 else
351 {
352 at(dest++) = cc;
353 }
354 }
355
356 if (got_tag) dest = 0;
357 resize(dest);
358 return in_HTML;
359}
360//----------------------------------------------------------------------------
361bool
362UTF8_string::round_0_1()
363{
364 if (back() >= '5') // round up
365 {
366 // rounding up of the last digit creates a carry
367 //
368 for (int j = size() - 2; j >= 0; --j)
369 {
370 if (at(j) == '9') // propagate carry
371 {
372 at(j) = '0';
373 continue; // carry survived
374 }
375
376 // eat carry
377 //
378 ++at(j);
379 pop_back(); // discard last digit.
380 return false; // no carry
381 }
382 }
383
384 // if carry survived then digits were 0.999... and were then rounded
385 // up to 1.000...
386 //
387 at(0) = '1';
388 pop_back();
389 return true; // 1.0 → 0.1
390}

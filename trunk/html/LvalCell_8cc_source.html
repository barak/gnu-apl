47PrintBuffer
48LvalCell::character_representation(const PrintContext & pctx) const
49{
50 if (!value.lval)
51 {
52 UTF8_string utf("(Cell * 0)");
53 UCS_string ucs(utf);
54 ColInfo ci;
55 PrintBuffer pb(ucs, ci);
56 return pb;
57 }
58
59PrintBuffer pb = value.lval->character_representation(pctx);
60 pb.pad_l(Unicode('='), 1);
61 pb.pad_r(Unicode('='), 1);
62 return pb;
63}
64//----------------------------------------------------------------------------
65void
66LvalCell::check_consistency() const
67{
68 if (value.lval) // valid owner
69 {
70 const Cell * C0 = &value.pval.owner->get_cfirst();
71 const Cell * CN = C0 + value.pval.owner->nz_element_count();
72 if (value.lval < C0 || value.lval >= CN) // wrong owner
73 {
74 Q1(C0)
75 Q1(value.lval)
76 Q1(CN)
77 Assert(0 && "LvalCell::check_consistency() failed");
78 BACKTRACE;
79 }
80 }
81 else Assert(value.pval.owner == 0); // no owner
82}

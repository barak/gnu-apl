39struct Pixel_XY
40{
42 Pixel_XY(Pixel_X px, Pixel_Y py)
43 : x(px),
44 y(py)
45 {}
46
48 const double distance2(const Pixel_XY other) const
49 { const double dx = x - other.x;
50 const double dy = y - other.y;
51 return dx*dx + dy*dy;
52 }
53
54
56 const double distance(const Pixel_XY other) const
57 { return sqrt(distance2(other)); }
58
59 Pixel_X x;
60 Pixel_Y y;
61};

66class Plot_data_row
67{
68public:
74 Plot_data_row(const double * pX, const double * pY, const double * pZ,
75 uint32_t idx, uint32_t len)
76 : X(pX), Y(pY), Z(pZ), row_num(idx), N(len)
77 {
78 min_X = max_X = get_X(0);
79 min_Y = max_Y = get_Y(0);
80 min_Z = max_Z = get_Z(0);
81
82 for (unsigned int n = 1; n < N; ++n)
83 {
84 const double Xn = get_X(n);
85 if (min_X > Xn) min_X = Xn;
86 if (max_X < Xn) max_X = Xn;
87
88 const double Yn = get_Y(n);
89 if (min_Y > Yn) min_Y = Yn;
90 if (max_Y < Yn) max_Y = Yn;
91
92 const double Zn = get_Z(n);
93 if (min_Z > Zn) min_Z = Zn;
94 if (max_Z < Zn) max_Z = Zn;
95 }
96 }
97
99 ~Plot_data_row()
100 {
101 // all rows belong to the same double *, so we only delete row 0
102 //
103 if (row_num == 0) delete[] X;
104 }
105
107 uint32_t get_N() const
108 { return N; }
109
111 double get_X(uint32_t n) const
112 { return X ? X[n] : n + 1; }
113
115 double get_Y(uint32_t n) const
116 { return Y[n]; }
117
119 double get_Z(uint32_t n) const
120 { return Z ? Z[n] : row_num + 1; }
121
123 double get_min_X() const

190class Plot_data
191{
192public:
194 Plot_data(uint32_t rows)
195 : surface(false),
196 row_count(rows),
197 idx(0)
198 {
199 data_rows = new const Plot_data_row *[row_count];
200 }
201
203 ~Plot_data()
204 {
205 loop(r, idx) delete data_rows[r];
206 delete data_rows;
207 }
208
210 uint32_t get_row_count() const
211 { return row_count; }
212
214 void get_XY(double & X, double & Y, uint32_t row, uint32_t col) const
215 {
216 Assert(row < idx);
217 X = data_rows[row]->get_X(col);
218 Y = data_rows[row]->get_Y(col);
219 }

247 { return data_rows[row]->get_X(col); }
248
250 double get_Y(uint32_t row, uint32_t col) const
251 { return data_rows[row]->get_Y(col); }
252
254 double get_Z(uint32_t row, uint32_t col) const
255 { return data_rows[row]->get_Z(col); }
256
258 bool is_surface_plot() const { return surface; }
259
261 void add_row(const Plot_data_row * row)
262 {
263 Assert(idx < row_count);
264 data_rows[idx++] = row;
265 if (idx == 1) // first row
266 {
267 min_X = row->get_min_X();
268 max_X = row->get_max_X();
269 min_Y = row->get_min_Y();
270 max_Y = row->get_max_Y();
271 min_Z = row->get_min_Y();
272 max_Z = row->get_max_Y();
273 }
274 else
275 {
276 if (min_X > row->get_min_X()) min_X = row->get_min_X();
277 if (max_X < row->get_max_X()) max_X = row->get_max_X();
278 if (min_Y > row->get_min_Y()) min_Y = row->get_min_Y();
279 if (max_Y < row->get_max_Y()) max_Y = row->get_max_Y();
280 if (min_Z > row->get_min_Z()) min_Z = row->get_min_Z();
281 if (max_Z < row->get_max_Z()) max_Z = row->get_max_Z();
282 }
283 }

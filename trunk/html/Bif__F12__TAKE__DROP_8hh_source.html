97class TakeDropIterator
98{
99public:
101 TakeDropIterator(bool take, const Shape & sh_A, const Shape & sh_B)
102 : ref_B(sh_B),
103 current_offset(0),
104 outside_B(0),
105 has_overtake(false),
106 done(false)
107 {
108 // increase ⍴⍴B if necessary
109 //
110 ref_B.expand_rank(sh_A.get_rank());
111
112 ShapeItem _weight = 1;
113 loop(r, sh_A.get_rank())
114 {
115 const ShapeItem sA = sh_A.get_rank() == 0
116 ? 1 : sh_A.get_transposed_shape_item(r);
117 const ShapeItem sB = (sh_B.get_rank() == 0 ||
118 r >= sh_B.get_rank())
119 ? 1 : sh_B.get_transposed_shape_item(r);
120
121 ShapeItem _from, _to;
122 if (take) // sh_A ↑ B
123 {
124 if (sA < 0) // take from end
125 {
126 _to = sB;
127 _from = sB + sA; // + since sA < 0
128 }
129 else // take from start
130 {
131 _from = 0;
132 _to = sA;
133 }
134
135 if (_from < 0 || _from >= sB) outside_B |= 1ULL << r;
136 if (_from < 0 || _to >= sB) has_overtake = true;
137 }
138 else // sh_A ↓ B
139 {
140 if (sA < 0) // drop from end
141 {
142 _from = 0;
143 _to = sB + sA; // + since sA < 0
144 }
145 else // drop from start
146 {
147 _from = sA;
148 _to = sB;
149 }
150
151 if (_from >= _to) // over-drop
152 {
153 _from = 0;
154 _to = 0;
155 }
156 }
157
158 Assert(_from <= _to);
159
160 _ftwc & ftwc_r = ftwc[r];
161 ftwc_r.from = _from;
162 ftwc_r.to = _to;
163 ftwc_r.weight = _weight;
164 ftwc_r.current = _from;
165 current_offset += _from * _weight;
166
167 if (_from == _to) done = true; // empty array
168
169 _weight *= sB;
170 }
171 }
172
174 void operator++()
175 {
176 loop(r, ref_B.get_rank())
177 {
178 _ftwc & ftwc_r = ftwc[r];
179 ++ftwc_r.current;
180 current_offset += ftwc_r.weight;
181 if (has_overtake)
182 {
183 if (ftwc_r.current < 0 ||
184 ftwc_r.current >= ref_B.get_transposed_shape_item(r))
185 outside_B |= 1ULL << r;
186 else
187 outside_B &= ~(1ULL << r);
188 }
189
190 if (ftwc_r.current < ftwc_r.to) return;
191
192 // end of dimension reached: reset this dimension
193 // and increment next dimension
194 //
195 current_offset -= ftwc_r.weight * (ftwc_r.current - ftwc_r.from);
196 ftwc_r.current = ftwc_r.from;
197 if (has_overtake)
198 {
199 if (ftwc_r.current < 0 ||
200 ftwc_r.current >= ref_B.get_transposed_shape_item(r))
201 outside_B |= 1ULL << r;
202 else
203 outside_B &= ~(1ULL << r);
204 }
205 }
206 done = true;
207 }
208
210 ShapeItem axis_proto(AxesBitmap axes) const;
211
213 bool more() const { return !done; }
214
216 ShapeItem operator()() const
217 { return outside_B ? -1 : current_offset; }
218
219protected:
221 Shape ref_B;
222
224 struct _ftwc ftwc[MAX_RANK];
225
227 ShapeItem current_offset;
228
230 ShapeItem outside_B;
231
233 bool has_overtake;
234
236 bool done;
237};

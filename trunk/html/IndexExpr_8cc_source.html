44//----------------------------------------------------------------------------
45Value_P
46IndexExpr::extract_axis()
47{
48 if (rank == 0 || !values[0]) return Value_P(); // no index or [ ]
49
50 Assert1(rank == 1); // must only be called for [ axis ]
51 --value_count;
52Value_P ret = values[0];
53 values[0].reset(); // decrement owner count
54 return ret;
55}
56//----------------------------------------------------------------------------
57void
58IndexExpr::check_index_range(const Shape & shape) const
59{
60 loop(r, rank)
61 {
62 if (const Value * ival = get_axis_value(r)) // unless elided index
63 {
64 const ShapeItem max_idx = shape.get_shape_item(r) + quad_io;
65 loop(i, ival->element_count())
66 {
67 const APL_Integer idx = ival->get_cravel(i).get_near_int();
68 if (idx < quad_io || idx >= max_idx) // invalid index
69 {
70 UCS_string & ucs = MORE_ERROR();
71 ucs << "Offending index: " << idx
72 << " (with ⎕IO: " << quad_io << ")\n"
73 << "offending axis: " << (r + quad_io)
74 << " of some";
75 loop(s, shape.get_rank())
76 ucs << " " << shape.get_shape_item(s);
77 ucs << "⍴...";
78 if (idx >= max_idx)
79 ucs << "\nthe max index for axis "
80 << (r + quad_io)<< " is: "
81 << (max_idx - quad_io);
82 INDEX_ERROR;
83 }
84 }
85 }
86 }
87}
88//----------------------------------------------------------------------------
89sRank
90IndexExpr::get_axis(sRank max_axis) const
91{
92 if (rank != 1) INDEX_ERROR;
93
94const APL_Integer qio = Workspace::get_IO();
95
96Value_P I = values[0];
97 if (!I->is_scalar_or_len1_vector()) INDEX_ERROR;
98
99 if (!I->get_cfirst().is_near_int()) INDEX_ERROR;
100
101 // if axis becomes (signed) negative then it will be (unsigned) too big.
102 // Therefore we need not test for < 0.
103 //
104sRank axis = I->get_cfirst().get_near_int() - qio;
105 if (axis >= max_axis) INDEX_ERROR;
106
107 return axis;
108}
109//----------------------------------------------------------------------------
110int
111IndexExpr::print_stale(ostream & out)
112{
113int count = 0;
114
115 for (const DynamicObject * dob = all_index_exprs.get_next();
116 dob != &all_index_exprs; dob = dob->get_next())
117 {
118 const IndexExpr * idx = dob->pIndexExpr();
119
120 out << dob->where_allocated();
121
122 try { out << *idx; }
123 catch (...) { out << " *** corrupt ***"; }
124
125 out << endl;
126
127 ++count;
128 }
129
130 return count;
131}
132//----------------------------------------------------------------------------
133void
134IndexExpr::erase_stale(const char * loc)
135{
136 Log(LOG_Value__erase_stale)
137 CERR << endl << endl << "erase_stale() called from " << loc << endl;
138
139 for (DynamicObject * dob = all_index_exprs.get_next();
140 dob != &all_index_exprs; dob = dob->get_next())
141 {
142 IndexExpr * idx = dob->pIndexExpr();
143
144 Log(LOG_Value__erase_stale)
145 {
146 CERR << "Erasing stale IndexExpr:" << endl
147 << " Allocated by " << idx->alloc_loc << endl;
148 }
149
150 dob = dob->get_prev();
151 idx->unlink();
152 delete idx;
153 }
154}
155//----------------------------------------------------------------------------
156ostream &
157operator <<(ostream & out, const IndexExpr & idx)
158{
159 out << "[";
160 loop(i, idx.get_rank())
161 {
162 if (i) out << ";";
163 if (const Value * ival = idx.get_axis_value(i))
164 {
165 // value::print() may print a trailing LF that we don't want here.
166 // We therefore print the index values ourselves.
167 const PrintContext pctx = Workspace::get_PrintContext(PR_APL_MIN);
168 PrintBuffer pb(*ival, pctx, 0);
169
170 UCS_string ucs(pb, ival->get_rank(), Workspace::get_PW());
171 out << ucs;
172 }
173 }
174 return out << "]";
175}

85//----------------------------------------------------------------------------
86void
87Quad_SVx::start_AP(AP_num ap)
88{
89 if (!uprefs.system_do_svars) // something went wrong
90 {
91 if (uprefs.user_do_svars) // user wanted APs
92 {
93 // user wanted APs, but something went wrong
94 //
95 CERR << "*** Not starting AP << ap because the connecton to "
96 " APserver has failed earlier." << endl;
97 }
98 else // user gave --noSV
99 {
100 CERR << "*** Not starting AP " << ap
101 << " because --noSV (or equivalent) was given." << endl
102 << " That conflicts with the use of ⎕SVxxx functions "
103 "and variables." << endl;
104 }
105 ATTENTION;
106 }
107
108const char * dirs[] = { "", "/APs" };
109 enum { dircount = sizeof(dirs) / sizeof(*dirs) };
110
111const AP_num own_ID = ProcessorID::get_own_ID();
112const AP_num par_ID = ProcessorID::get_parent_ID();
113
114const char * verbose = "";
115 Log(LOG_shared_variables) verbose = " -v";
116
117char filename[APL_PATH_MAX + 1];
118
119bool found_executable = false;
120 for (int d = 0; d < dircount; ++d)
121 {
122 const int slen = snprintf(filename, APL_PATH_MAX,
123 "%s%s/AP%u --id %u --par %u --gra %u --auto%s",
124 LibPaths::get_APL_bin_path(), dirs[d], ap, ap,
125 own_ID, par_ID, verbose);
126 if (slen >= APL_PATH_MAX) filename[APL_PATH_MAX] = 0;
127
128 found_executable = is_executable(filename);
129 if (found_executable) break;
130 }
131
132 if (!found_executable)
133 {
134 CERR << "No binary found for AP " << ap << " (interpreter path = "
135 << LibPaths::get_APL_bin_path() << ")" << endl;
136 return;
137 }
138
139FILE * fp = popen(filename, "r");
140 if (fp == 0)
141 {
142 CERR << "popen(" << filename << " failed: " << strerror(errno)
143 << endl;
144 return;
145 }
146
147 for (int cc; (cc = getc(fp)) != EOF;) CERR << char(cc);
148 CERR << endl;
149
150 pclose(fp);
151
152 usleep(100000); // give new AP time to register with APserver
153}

155Token
156Quad_SVC::eval_AB(Value_P A, Value_P B) const
157{
158 if (A->get_rank() > 2) RANK_ERROR;
159 if (B->get_rank() > 2) RANK_ERROR;
160
161const ShapeItem var_count = B->get_rows();
162 if (A->get_rows() != var_count) LENGTH_ERROR;
163 if (A->get_rank() > 0 && A->get_cols() != 4) LENGTH_ERROR;
164const UCS_string_vector vars(*B, false);
165
166const Cell * cA = &A->get_cfirst();
167
168Shape sh_Z;
169 if (var_count > 1) sh_Z.add_shape_item(var_count);
170 sh_Z.add_shape_item(4);
171Value_P Z(sh_Z, LOC);
172
173 loop(z, var_count)
174 {
175 // construct control value
176 //
177 int ctl = 0;
178 if (A->is_scalar())
179 {
180 const bool val = A->get_cfirst().get_near_bool();
181 ctl = val ? ALL_SVAR_CONTROLS : NO_SVAR_CONTROL;
182 }
183 else if (A->is_vector())
184 {
185 if (A->get_cfirst().get_near_bool()) ctl |= SET_BY_1;
186 if (A->get_cravel(1).get_near_bool()) ctl |= SET_BY_2;
187 if (A->get_cravel(2).get_near_bool()) ctl |= USE_BY_1;
188 if (A->get_cravel(3).get_near_bool()) ctl |= USE_BY_2;
189 }
190 else // matrix
191 {
192 if (cA++->get_near_bool()) ctl |= SET_BY_1;
193 if (cA++->get_near_bool()) ctl |= SET_BY_2;
194 if (cA++->get_near_bool()) ctl |= USE_BY_1;
195 if (cA++->get_near_bool()) ctl |= USE_BY_2;
196 }
197
198 // set control.
199 //
200 Symbol * sym = Workspace::lookup_existing_symbol(vars[z]);
201 if (sym == 0) Error::throw_symbol_error(vars[z], LOC);
202
203 const SV_key key = sym->get_SV_key();
204 Svar_DB::set_control(key, Svar_Control(ctl));
205 ctl = Svar_DB::get_control(key);
206
207 Z->next_ravel_Int(ctl & SET_BY_1 ? 1 : 0);
208 Z->next_ravel_Int(ctl & SET_BY_2 ? 1 : 0);
209 Z->next_ravel_Int(ctl & USE_BY_1 ? 1 : 0);
210 Z->next_ravel_Int(ctl & USE_BY_2 ? 1 : 0);
211 }
212
213 Z->check_value(LOC);
214 return Token(TOK_APL_VALUE1, Z);
215}

253//----------------------------------------------------------------------------
254void
255Quad_SVE::assign(Value_P value, bool clone, const char * loc)
256{
257 // ⎕SVE←X merely remembers the timer expiration time. The timer is only
258 // started if ⎕SVE is referenced (in Quad_SVE::get_apl_value())
259 //
260 if (!value->is_scalar()) RANK_ERROR;
261
262const APL_time_us duration = 1000000 * value->get_cfirst().get_real_value();
263 if (duration < 0) DOMAIN_ERROR;
264
265 if (duration == 0.0)
266 {
267 // ⎕SVE←0 causes get_apl_value() to return immediately. We achieve
268 // that by setting the end time to the past.
269 //
270 timer_end = now() - 1; // end time in the past
271 return;
272 }
273
274 // set timer end
275 //
276 timer_end = now() + duration;
277}
278//----------------------------------------------------------------------------
279Value_P
280Quad_SVE::get_apl_value() const
281{
282 for (;;)
283 {
284 Svar_event events = SVE_NO_EVENTS;
285 SV_key key = Svar_DB::get_events(events, ProcessorID::get_id());
286
287 if (key || (events != SVE_NO_EVENTS)) break; // something happend
288
289 if (timer_end < now()) // timer expired: return 0
290 {
291 Svar_DB::clear_all_events(ProcessorID::get_id());
292 return IntScalar(0, LOC);
293 }
294
295 usleep(50000); // wait 50 ms
296 }
297
298 // referencing ⎕SVE always clears all events
299 //
300 Svar_DB::clear_all_events(ProcessorID::get_id());
301
302 // we have got an event; return remaining time.
303 //
304const APL_Float remaining = timer_end - now();
305
306 if (remaining < 0.0) return IntScalar(0, LOC);
307
308 return FloatScalar(0.000001 * remaining, LOC);
309}

314Token
315Quad_SVO::eval_AB(Value_P A, Value_P B) const
316{
317 if (A->get_rank() > 1) RANK_ERROR;
318 if (B->get_rank() > 2) RANK_ERROR;
319
320const ShapeItem var_count = B->get_rows();
321 if (A->get_rank() == 1 && A->element_count() != var_count) LENGTH_ERROR;
322
323const UCS_string_vector apl_vars(*B, false);
324const UCS_string_vector surrogates(*B, true);
325
326Shape sh_Z;
327 if (var_count > 1) sh_Z.add_shape_item(var_count);
328Value_P Z(sh_Z, LOC);
329
330 loop(z, var_count)
331 {
332 ShapeItem a = A->is_scalar() ? 0 : z;
333 uint32_t vname[MAX_SVAR_NAMELEN + 1];
334 const int vlen = surrogates[z].size();
335 if (vlen > MAX_SVAR_NAMELEN) DOMAIN_ERROR;
336 bool bad_name = false;
337
338 loop(v, vlen)
339 {
340 const Unicode uni = surrogates[z][v];
341 if (!Avec::is_symbol_char(uni)) bad_name = true;
342 vname[v] = uni;
343 }
344 vname[vlen] = 0;
345
346 if (bad_name || vlen == 0)
347 {
348 Z->next_ravel_Int(NO_COUPLING);
349 continue;
350 }
351
352 // make sure current name class allows ⎕SVO
353 //
354 Symbol * sym = Workspace::lookup_symbol(apl_vars[z]);
355 assert(sym);
356
357 const AP_num proc = AP_num(A->get_cravel(a).get_int_value());
358
359 ValueStackItem * vsp = sym->top_of_stack();
360 Assert(vsp);
361
362 switch(sym->get_NC())
363 {
364 case NC_UNUSED_USER_NAME:
365 case NC_VARIABLE:
366 break;
367
368 case NC_SYSTEM_VAR:
369 {
370 // CERR << "Shared variable " << apl_vars[z]
371 // << " is already shared" << endl;
372
373 const SV_key key = sym->get_SV_key();
374 Z->next_ravel_Int(Svar_DB::get_coupling(key));
375 }
376 continue; // next z
377
378 default: // function etc.
379 Q1(sym->get_NC()) DOMAIN_ERROR;
380 }
381
382 SV_Coupling coupling = NO_COUPLING;
383 const SV_key key = share_one_variable(proc, vname, coupling);
384 if (coupling >= SV_OFFERED) // offered or coupled
385 {
386 sym->share_var(key);
387 }
388
389 Z->next_ravel_Int(coupling);
390 }
391
392 Z->check_value(LOC);
393 return Token(TOK_APL_VALUE1, Z);
394}
395//----------------------------------------------------------------------------
396SV_key
397Quad_SVO::share_one_variable(AP_num to_ap, const uint32_t * vname,
398 SV_Coupling & coupling)
399{
400 // if to_ap is a (dependent) AP then start it if necessary...
401 //
402AP_num3 to_proc(to_ap, ProcessorID::get_own_ID(), ProcessorID::get_parent_ID());
403bool auto_started = false;
404
405 if (to_ap && to_ap < AP_FIRST_USER) // to_proc is an AP
406 {
407 if (!Svar_DB::is_registered_id(to_proc))
408 {
409 start_AP(to_ap);
410 auto_started = true;
411 }
412 }
413 else // to_proc is an independent processor
414 {
415 new (&to_proc) AP_num3(to_ap);
416 }
417
418Svar_partner from(ProcessorID::get_id(), NO_TCP_SOCKET);
419
420const SV_key key = Svar_DB::match_or_make(vname, to_proc, from);
421 coupling = Svar_DB::get_coupling(key);
422
423 if (coupling == SV_COUPLED)
424 {
425 // if an AP was started automatically, then coupling could be
426 // SV_COUPLED or SV_OFFERED, depending on how fast the AP did start
427 // up. We pretend here that it started slowly so that auto-started
428 // APs always return SV_OFFERED
429 //
430 if (auto_started) coupling = SV_OFFERED;
431
432 return key;
433 }
434
435 // if something went wrong, complain.
436 //
437 if (coupling == NO_COUPLING)
438 {
439 Log(LOG_shared_variables)
440 CERR << "Svar_DB::match_or_store() returned unexpected 0" << endl;
441 return 0;
442 }
443
444 Assert(coupling == SV_OFFERED);
445
446 return key;
447}

513//----------------------------------------------------------------------------
514Value_P
515Quad_SVQ::get_processors()
516{
517 // shared variable processors are identified by one of the following:
518 //
519 // 1. auxiliary processors that can be started on demand
520 // (e.g. AP100, AP210). These APs are collocated with the apl binary,
521 // either in the same directory or in the 'APs' subdirectory,
522 //
523 // 2. running processors that have offered variables in Svar_DB.
524 //
525std::vector<AP_num> processors;
526
527 // case 1...
528 //
529const char * dirs[] = { "", "/APs" };
530 enum { dircount = sizeof(dirs) / sizeof(*dirs) };
531
532 for (int d = 0; d < dircount; ++d)
533 {
534 char dirname[APL_PATH_MAX + 1];
535 int slen = snprintf(dirname, APL_PATH_MAX, "%s%s", LibPaths::get_APL_bin_path(),
536 dirs[d]);
537 if (slen >= APL_PATH_MAX) dirname[APL_PATH_MAX] = 0;
538
539 dirname[APL_PATH_MAX] = 0;
540 DIR * dir = opendir(dirname);
541
542 // the APs directory only exists below the src directory (i.e when
543 // apl is started locally without installing it. We do not complain
544 // if the APs directory does not exist.
545 //
546 if (dir == 0)
547 {
548 if (*dirs[d] == 0)
549 {
550 CERR << "Could not open directory " << dirname << " : "
551 << strerror(errno) << " at " << LOC << endl;
552 }
553 continue;
554 }
555
556 for (;;) // scan files in directory
557 {
558 dirent * entry = readdir(dir);
559 if (entry == 0) break; // directory done
560
561#ifdef _DIRENT_HAVE_D_TYPE
562 if (entry->d_type != DT_REG) continue; // not a regular file
563#endif
564
565 char filename[APL_PATH_MAX + 1];
566 int slen = snprintf(filename, APL_PATH_MAX, "%s/%s",
567 dirname, entry->d_name);
568 if (slen >= APL_PATH_MAX) filename[APL_PATH_MAX] = 0;
569
570 if (!is_executable(filename)) continue;
571
572 int apnum;
573 if (sscanf(entry->d_name, "AP%u", &apnum) != 1) continue;
574
575 char expected[APL_PATH_MAX + 1];
576 slen = snprintf(expected, sizeof(expected), "AP%u", apnum);
577 if (slen >= APL_PATH_MAX) expected[APL_PATH_MAX] = 0;
578 if (strcmp(entry->d_name, expected)) continue;
579
580 processors.push_back(AP_num(apnum));
581 }
582
583 closedir(dir);
584 }
585
586 // case 2...
587 //
588 Svar_DB::get_offering_processors(ProcessorID::get_own_ID(), processors);
589
590 // sort and remove duplicates
591 //
592std::vector<int32_t> sorted;
593 while (processors.size())
594 {
595 // find smallest
596 //
597 int smallest = processors[0];
598 for (size_t s = 1; s < processors.size(); ++s)
599 if (smallest > processors[s]) smallest = processors[s];
600
601 // add smallest to sorted
602 //
603 sorted.push_back(smallest);
604
605 // remove smallest from processors
606 //
607 for (size_t s = 0; s < processors.size();)
608 {
609 if (processors[s] != smallest)
610 {
611 ++s;
612 }
613 else
614 {
615 processors[s] = processors.back();
616 processors.pop_back();
617 }
618 }
619 }
620
621Value_P Z(sorted.size(), LOC);
622 loop(z, sorted.size()) Z->next_ravel_Int(sorted[z]);
623 return Z;
624}
625//----------------------------------------------------------------------------
626Value_P
627Quad_SVQ::get_variables(AP_num proc)
628{
629std::vector<uint32_t> varnames;
630 Svar_DB::get_offered_variables(ProcessorID::get_own_ID(), proc, varnames);
631
632 // varnames is a sequence of 0-terminated Unicodes
633 //
634std::vector<int> var_lengths; // including terminating 0
635int last_zero = -1;
636 loop(v, varnames.size())
637 {
638 if (varnames[v] == 0)
639 {
640 var_lengths.push_back(v + 1 - last_zero);
641 last_zero = v;
642 }
643 }
644
645const int count = var_lengths.size();
646ShapeItem max_len = 0;
647 loop(z, count)
648 if (max_len < (var_lengths[z] - 1)) max_len = var_lengths[z] - 1;
649
650const Shape shZ(count, max_len);
651Value_P Z(shZ, LOC);
652int v = 0;
653
654 loop(z, count)
655 loop(c, max_len)
656 {
657 if (c < var_lengths[z])
658 {
659 if (varnames[v] == 0) ++v;
660 Z->next_ravel_Char(Unicode(varnames[v++]));
661 }
662 else
663 {
664 Z->next_ravel_Char(UNI_SPACE);
665 }
666 }
667
668 Z->set_proto_Spc(); // prototype: character
669 return Z;
670}

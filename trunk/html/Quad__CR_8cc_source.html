141//----------------------------------------------------------------------------
142Token
143Quad_CR::do_eval_B(const Value * B, bool remove_extra_spaces)
144{
145UCS_string symbol_name(*B);
146
147 // remove trailing whitespaces in B
148 //
149 while (symbol_name.back() <= ' ') symbol_name.pop_back();
150
151 /* return an empty character matrix, if:
152 * 1) symbol_name is not known, or
153 * 2) symbol_name is not user defined, or
154 * 3) symbol_name is a function, or
155 * 4) symbol_name is not displayable
156 */
157
158 if (symbol_name.size() == 0) return Token(TOK_APL_VALUE1, Str0_0(LOC));
159
160const Function * function = 0;
161 if (symbol_name[0] == UNI_MUE) // macro
162 {
163 loop(m, Macro::MAC_COUNT)
164 {
165 const Macro * macro =
166 Macro::get_macro(Macro::Macro_num(m));
167 if (symbol_name == macro->get_name())
168 {
169 function = macro;
170 break;
171 }
172 }
173
174 }
175 else // maybe user defined function
176 {
177 const NamedObject * obj = Workspace::lookup_existing_name(symbol_name);
178 if (obj && obj->is_user_defined())
179 {
180 function = obj->get_function();
181 if (function && function->get_exec_properties()[0]) function = 0;
182 }
183 }
184 if (function == 0) return Token(TOK_APL_VALUE1, Str0_0(LOC));
185
186 // show the function...
187 //
188const UCS_string ucs = function->canonical(false);
189UCS_string_vector tlines;
190 ucs.to_vector(tlines);
191int max_len = 0;
192 loop(row, tlines.size())
193 {
194 if (remove_extra_spaces)
195 tlines[row].remove_leading_and_trailing_whitespaces();
196
197 if (max_len < tlines[row].size()) max_len = tlines[row].size();
198 }
199
200Shape shape_Z;
201 shape_Z.add_shape_item(tlines.size());
202 shape_Z.add_shape_item(max_len);
203
204Value_P Z(shape_Z, LOC);
205 loop(row, tlines.size())
206 {
207 const UCS_string & line = tlines[row];
208 loop(col, line.size())
209 Z->next_ravel_Char(line[col]);
210
211 loop(col, max_len - line.size())
212 Z->next_ravel_Char(UNI_SPACE);
213 }
214
215 Z->check_value(LOC);
216 return Token(TOK_APL_VALUE1, Z);
217}
218//----------------------------------------------------------------------------
219sAxis
220Quad_CR::subfun_to_axis(const UCS_string & name) const
221{
222UTF8_string name_utf(name);
223const char * function_name = name_utf.c_str();
224
225 enum { SF_SIZE = sizeof(_sub_fun),
226 SF_COUNT = sizeof(sub_functions) / SF_SIZE };
227
228 if (const void * vp = bsearch(function_name, sub_functions,
229 SF_COUNT, SF_SIZE, fun_compare))
230 return reinterpret_cast<const _sub_fun *>(vp)->val;
231
232 return -1; // not found
233
234}
235//----------------------------------------------------------------------------
236Token
237Quad_CR::eval_AB(Value_P A, Value_P B) const
238{
239int function_number = -1;
240
241 if (A->get_rank() > 1) RANK_ERROR;
242 if (A->is_char_array()) // function name, e.g. "APL_expression"
243 {
244 UCS_string ucs_A(*A);
245 UTF8_string utf_A(ucs_A);
246 function_number = subfun_to_axis(utf_A.c_str());
247 if (function_number == -1)
248 {
249 MORE_ERROR() << "Bad function name X in ⎕FIO[X]B (X is '"
250 << ucs_A << "')";
251 DOMAIN_ERROR;
252 }
253 }
254 else
255 {
256 if (!A->is_scalar_or_len1_vector()) LENGTH_ERROR;
257 function_number = A->get_cfirst().get_int_value();
258 }
259
260PrintContext pctx = Workspace::get_PrintContext(PST_NONE);
261
262Value_P Z = do_CR(function_number, B.get(), pctx);
263 Z->check_value(LOC);
264 return Token(TOK_APL_VALUE1, Z);
265}
266//----------------------------------------------------------------------------
267Value_P
268Quad_CR::do_CR(APL_Integer a, const Value * B, PrintContext pctx)
269{
270 // some functions a have an inverse (which has its own number, but can
271 // also be specified as -a
272 //
273 if (a < 0) switch(a)
274 {
275 case -5: a = 13; break;
276 case -6: a = 13; break;
277 case -13: a = 6; break;
278 case -12: a = 11; break;
279 case -11: a = 12; break;
280 case -14: a = 15; break;
281 case -15: a = 14; break;
282 case -16: a = 17; break;
283 case -17: a = 16; break;
284 case -18: a = 19; break;
285 case -19: a = 18; break;
286 case -33: a = 34; break;
287 case -34: a = 33; break;
288 case -35: a = 36; break;
289 case -36: a = 35; break;
290 case -38: a = 39; break;
291 case -39: a = 38; break;
292 case -40: a = 41; break;
293 case -41: a = 40; break;
294 default: MORE_ERROR() << "A ⎕CR B with invalid A < 0";
295 DOMAIN_ERROR;
296 }
297
298bool extra_frame = false;
299 switch(a)
300 {
301 case 4:
302 case 8:
303 case 23:
304 case 24:
305 case 25:
306 case 29: extra_frame = true;
307
308 }
309
310 switch(a)
311 {
312
314#define FRAME(x) pctx.set_style(x); break;
315
316 case 0: FRAME(PR_APL)
317 case 1: FRAME(PR_APL_FUN)
318 case 2: FRAME(PR_BOXED_CHAR)
319 case 3: FRAME(PR_BOXED_GRAPHIC)
320 case 4: FRAME(PR_BOXED_GRAPHIC)
321 case 5: // byte-vector → HEX
322 case 6: return do_CR5_6(a, B); // byte-vector → hex
323 case 7: FRAME(PR_BOXED_GRAPHIC1)
324 case 8: FRAME(PR_BOXED_GRAPHIC1)
325 case 9: FRAME(PR_BOXED_GRAPHIC2)
326 case 10: return do_CR10(B);
327 case 11: return do_CR11(B); // Value → CDR conversion
328 case 12: return do_CR12(B); // CDR → Value conversion
329 case 13: return do_CR13(B); // hex → byte-vector
330 case 14: return do_CR14(B); // Value → CDR → hex conversion
331 case 15: return do_CR15(B); // hex → CDR → Value conversion
332 case 16: return do_CR16(B); // byte vector → base64, RFC 4648
333 case 17: return do_CR17(B); // base64 → byte vector, RFC 4648
334 case 18: return do_CR18(B); // UCS → UTF8 byte vector
335 case 19: return do_CR19(B); // UTF8 byte vector → UCS string
336 case 20: FRAME(PR_NARS)
337 case 21: FRAME(PR_NARS1)
338 case 22: FRAME(PR_NARS2)
339 case 23: FRAME(PR_NARS)
340 case 24: FRAME(PR_NARS1)
341 case 25: FRAME(PR_NARS2)
342 case 26: return do_CR26(B); // Cell types
343 case 27: // value as int
344 case 28: return do_CR27_28(a, B); // value2 as int
345 case 29: FRAME(PR_BOXED_GRAPHIC3)
346 case 30: return do_CR30(B); // conform B (for ⍤ macro)
347 case 31: // ⎕INP helper
348 case 32: return do_CR31_32(a, B); // ⎕INP helper
349 case 33: return do_CR33(B); // TV to TLV byte vector
350 case 34: return do_CR34(B); // TLV byte vector to TV
351 case 35: return do_CR35(B); // lines to nested strings
352 case 36: return do_CR36(B); // nested strings to lines
353 case 37: return do_CR37(B); // ⎕CR B but extra spaces kept
354 case 38: return do_CR38(B); // plain → structure
355 case 39: return do_CR39(B); // structure → plain
356 case 40: return do_CR40(B); // boolean → packed
357 case 41: return do_CR41(B); // packed → boolean
358
359 default: MORE_ERROR() << "A ⎕CR B with invalid A (=" << a << ")";
360 DOMAIN_ERROR;
361#undef FRAME
362 }
363
364 // common code for ⎕CR variants that only differ by print style...
365 //
366 if (extra_frame && !B->is_simple_scalar())
367 {
368 Value_P Z(LOC); // a nested scalar
369 Value * Zsub = const_cast<Value *>(B); // will die at } below
370 Z->next_ravel_Pointer(Zsub); // Z ← ⊂ B
371 Z->check_value(LOC);
372 PrintBuffer pb(*Z, pctx, 0);
373 return Value_P(pb, LOC);
374 }
375 else // no frame
376 {
377 PrintBuffer pb(*B, pctx, 0);
378 return Value_P(pb, LOC);
379 }
380}
381//----------------------------------------------------------------------------
382Value_P
383Quad_CR::do_CR5_6(int A_5_6, const Value * B)
384{
385const char * alpha = (A_5_6 == 5) ? "0123456789ABCDEF" : "0123456789abcdef";
386Shape shape_Z(B->get_shape());
387 if (shape_Z.get_rank() == 0) // scalar B
388 {
389 shape_Z.add_shape_item(2);
390 }
391 else
392 {
393 shape_Z.set_shape_item(B->get_rank() - 1, B->get_cols()*2);
394 }
395
396Value_P Z(shape_Z, LOC);
397
398const Cell * cB = &B->get_cfirst();
399 loop(b, B->element_count())
400 {
401 const int val = cB++->get_byte_value() & 0x00FF;
402 const int h = alpha[val >> 4];
403 const int l = alpha[val & 0x0F];
404 Z->next_ravel_Char(Unicode(h));
405 Z->next_ravel_Char(Unicode(l));
406 }
407
408 Z->set_proto_Spc();
409 Z->check_value(LOC);
410 return Z;
411}

433//----------------------------------------------------------------------------
434void
435Quad_CR::do_CR10(UCS_string_vector & result, const Value * B)
436{
437const UCS_string symbol_name(*B);
438const Symbol * symbol = Workspace::lookup_existing_symbol(symbol_name);
439 if (symbol == 0) DOMAIN_ERROR;
440
441 switch(symbol->get_NC())
442 {
443 case NC_VARIABLE:
444 {
445 const Value * value = symbol->get_apl_value().get();
446 do_CR10_variable(result, symbol_name, value);
447 return;
448 }
449
450 case NC_FUNCTION:
451 case NC_OPERATOR:
452 {
453 const Function & ufun = *symbol->get_function();
454 const UCS_string text = ufun.canonical(false);
455 if (ufun.is_lambda())
456 {
457 UCS_string res = symbol->get_name();
458 res.append(UNI_LEFT_ARROW);
459 res.append(UNI_L_CURLY);
460 int t = 0;
461 while (t < text.size()) // skip λ header
462 {
463 const Unicode uni = text[t++];
464 if (uni == UNI_LF) break;
465 }
466
467
468 while (t < text.size()) // copy body
469 {
470 const Unicode uni = text[t++];
471 if (uni == UNI_LF) break;
472 res.append(uni);
473 }
474 res.append(UNI_R_CURLY);
475
476 result.push_back(res);
477 }
478 else
479 {
480 UCS_string res("∇");
481
482 loop(u, text.size())
483 {
484 if (text[u] == '\n')
485 {
486 result.push_back(res);
487 res.clear();
488 UCS_string next(text, u+1, text.size()-(u+1));
489 if (!next.is_comment_or_label() &&
490 u < (text.size() - 1))
491 res.append(UNI_SPACE);
492 }
493 else
494 {
495 res.append(text[u]);
496 }
497 }
498 res.append(ufun.get_exec_properties()[0]
499 ? UNI_DEL_TILDE : UNI_NABLA);
500 result.push_back(res);
501 }
502 return;
503 }
504
505 default: DOMAIN_ERROR;
506 }
507}

509void
510Quad_CR::do_CR10_variable(UCS_string_vector & result,
511 const UCS_string & var_name,
512 const Value * value)
513{
514 if (value->is_member()) // normal variable
515 {
516 if (const char * error = do_CR10_structured(result, var_name, value))
517 {
518 CERR << "could not )DUMP structured variable " << var_name

522 }

529Picker picker(var_name);

531}
532//----------------------------------------------------------------------------
533void
534Quad_CR::do_CR10_value(UCS_string_vector & result, const Value * value,
535 Picker & picker, ShapeItem pidx)
536{
537 /*
538 recursively encode var_name with value.
539
540 we emit one of 2 formats:
541
542 short format: dest ← (⍴ value) ⍴ value
543
544 long format: dest ← (⍴ value) ⍴ 0 " prolog "
545 dest[] ← partial value ...
546
547 short format (the default) requires a reasonably short value
548
549 If value is nested then the short or long format is followed
550 by recursive do_CR10_value() calls for the sub-values.
551 */
552
553 picker.push(value->get_shape(), pidx);
554
555UCS_string left;
556 picker.get(left);
557
558 // compute shape_rho which is "" for scalars and true vectors
559 // of length > 1, "," for true vectors of length 1, and
560 // "n1 n2 ... ⍴" otherwise
561 //
562UCS_string shape_rho;
563 {
564 if (value->get_rank() == 1) // true vector
565 {
566 if (value->element_count() == 0) shape_rho.append_UTF8("0⍴");
567 else if (value->element_count() == 1) shape_rho.append_UTF8(",");
568 }
569 else if (value->get_rank() > 1) // matrix or higher
570 {
571 shape_rho.append_shape(value->get_shape());
572 shape_rho.append_UTF8("⍴");
573 }
574 }
575
576 if (value->element_count() == 0) // empty value
577 {
578 UCS_string reshape(2*(picker.get_level() - 1), UNI_SPACE);
579 Value_P proto = value->prototype(LOC);
580
581 // emit one line for the prototype
582 UCS_string proto_name;
583 picker.get(proto_name);
584 picker.pop();
585 do_CR10_value(result, proto.get(), picker, pidx);
586 result.back().append_UTF8(" ⍝ proto 1");
587
588 // and then another line to reshape the prototype
589 //
590 reshape.append(proto_name);
591 reshape.append_UTF8("←");
592 reshape.append(shape_rho);
593 reshape.append(proto_name);
594 result.push_back(reshape);
595 result.back().append_UTF8(" ⍝ proto 2");
596 return;
597 }
598
599bool nested = false;
600
601 // step 1: top-level ravel. Try short (one-line) ravel and use long
602 // (multi-line ravel) if that fails.
603 //
604 if (short_ravel(result, nested, value, picker, left, shape_rho))
605 {
606 result.push_back(compute_prolog(picker.get_level(), left, value));
607
608 ShapeItem pos = 0;
609 V_mode mode = Vm_NONE;
610 UCS_string rhs;
611 ShapeItem count = 0; // the number of items on this line
612 ShapeItem todo = value->nz_element_count(); // the number of items to do
613
614 loop(p, todo)
615 {
616 // recompute line (which changes because count changes)
617 //
618 UCS_string line(2*(picker.get_level() - 1), UNI_SPACE);
619 picker.get_indexed(line, pos, count);
620 line.append_UTF8("←");
621
622 // compute next item
623 //
624 UCS_string item;
625 const Cell & cell = value->get_cravel(p);
626 if (cell.is_pointer_cell()) nested = true;
627 const bool may_quote = use_quote(mode, value, p);
628 const V_mode next_mode = do_CR10_item(item, cell, mode, may_quote);
629
630 if (count && (rhs.size() + item.size() + line.size()) > 76)
631 {
632 // next item does not fit in this line. Close the line,
633 // append it to result, and start a new line.
634 //
635 close_mode(rhs, mode);
636
637 line.append(rhs);
638 pos += count;
639 count = 0;
640
641 result.push_back(line);
642 rhs.clear();
643 mode = Vm_NONE;
644 }
645
646 item_separator(rhs, mode, next_mode);
647
648 mode = next_mode;
649 rhs.append(item);
650 ++count;
651 }
652
653 // all items done: close mode
654 //
655 close_mode(rhs, mode);
656
657UCS_string line(2*(picker.get_level() - 1), UNI_SPACE);
658 picker.get_indexed(line, pos, count);
659 line.append_UTF8("←");
660 line.append(rhs);
661
662 result.push_back(line);
663 }
664
665 // step 2: nested items
666 //
667 if (nested)
668 {
669 loop(p, value->element_count())
670 {

681//----------------------------------------------------------------------------
682const char *
683Quad_CR::do_CR10_structured(UCS_string_vector & result,
684 const UCS_string & var_name,
685 const Value * value)
686{
687 if (value->get_rank() != 2) return "bad rank";
688 if (value->get_cols() != 2) return "bad shape";
689
690 loop(r, value->get_rows())
691 {
692 const Cell & member_cell = value->get_cravel(2*r);
693 if (!member_cell.is_pointer_cell()) continue;
694 const Value * member_name = member_cell.get_pointer_value().get();
695 if (!member_name->is_char_string()) continue;

698 UCS_string path = var_name;
699 path += UNI_FULLSTOP;
700 path.append(member_ucs);
701
702 const Cell & data_cell = value->get_cravel(2*r + 1);
703 if (data_cell.is_member_anchor()) // non-leaf
704 {
705 const Value * sub_value = data_cell.get_pointer_value().get();
706 do_CR10_structured(result, path, sub_value);
707 }
708 else // leaf
709 {
710 Picker picker(path);
711 if (data_cell.is_pointer_cell()) // nested data_cell
712 {
713 const Value * sub_value =
714 data_cell.get_pointer_value().get();
715 do_CR10_value(result, sub_value, picker, -99);
716 }
717 else // simple data_cell
718 {
719 Value_P sub_value(LOC);
720 sub_value->next_ravel_Cell(data_cell);
721 do_CR10_value(result, sub_value.get(), picker, -99);
722 }
723 }
724 }
725
726 return 0;
727}
728//----------------------------------------------------------------------------
729bool
730Quad_CR::short_ravel(UCS_string_vector & result, bool & nested,
731 const Value * value, const Picker & picker,
732 const UCS_string & left, const UCS_string & shape_rho)
733{
734 // some quick checks beforehand to avoid wasting time attempting to
735 // create short ravel
736 //
737 enum { MAX_SHORT_RAVEL_LEN = 70 };
738
739const ShapeItem value_len = value->nz_element_count();
740 if (value_len >= MAX_SHORT_RAVEL_LEN) return true; // too long
741int len = 0;
742 loop(v, value_len)
743 {
744 const Cell & cell = value->get_cravel(v);
745 if (cell.is_integer_cell()) len += 2; // min. integer (+ separator)
746 else if (cell.is_character_cell()) len += 1;
747 else if (cell.is_pointer_cell()) len += 1;
748 else if (cell.is_complex_cell()) len += 3;
749 else len += 2;
750
751 if (len >= MAX_SHORT_RAVEL_LEN) return true;
752 }
753
754 // at this point, a short ravel MAY be possible. Try it
755 //
756UCS_string line(2*(picker.get_level() - 1), UNI_SPACE); // indent
757
758 line.append(left);
759 line.append_UTF8("←");
760 line.append(shape_rho);
761
762V_mode mode = Vm_NONE;
763 loop(v, value_len)
764 {
765 // compute next item
766 //
767 UCS_string item;
768 const Cell & cell = value->get_cravel(v);
769 if (cell.is_pointer_cell()) nested = true;
770 const bool may_quote = use_quote(mode, value, v);
771 const V_mode next_mode = do_CR10_item(item, cell, mode, may_quote);
772
773 item_separator(line, mode, next_mode);
774
775 mode = next_mode;
776 line.append(item);
777 if ((line.size() + item.size()) >= MAX_SHORT_RAVEL_LEN) return true;
778 }
779
780 // all items done: close mode
781 //
782 close_mode(line, mode);
783
784 result.push_back(line);
785 return false;
786}

894//----------------------------------------------------------------------------
895Quad_CR::V_mode
896Quad_CR::do_CR10_item(UCS_string & item, const Cell & cell, V_mode mode,
897 bool may_quote)
898{
899 if (cell.is_character_cell())
900 {
901 if (may_quote) // use ''
902 {
903 const Unicode uni = cell.get_char_value();
904 item.append(uni);
905 if (uni == UNI_SINGLE_QUOTE) item.append(uni);
906 return Vm_QUOT;
907 }
908 else // use UCS()
909 {
910 item.append_number(cell.get_char_value());
911 return Vm_UCS;
912 }
913 }
914
915 if (cell.is_integer_cell())
916 {
917 APL_Integer aint = cell.get_int_value();
918 if (aint < 0)
919 {
920 item.append(UNI_OVERBAR);
921 aint = -aint;
922 }
923 item.append_number(aint);
924 return Vm_NUM;
925 }
926
927 if (cell.is_float_cell())
928 {
929 bool scaled = false;
930 PrintContext pctx(PR_APL_MIN, MAX_Quad_PP, MAX_Quad_PW);
931 item = UCS_string(cell.get_real_value(), scaled, pctx);
932 return Vm_NUM;
933 }
934
935 if (cell.is_complex_cell())
936 {
937 bool scaled = false;
938 PrintContext pctx(PR_APL_MIN, MAX_Quad_PP, MAX_Quad_PW);
939 item = UCS_string(cell.get_real_value(), scaled, pctx);
940 item.append_UTF8("J");
941 scaled = false;
942 item.append(UCS_string(cell.get_imag_value(), scaled, pctx));
943 return Vm_NUM;
944 }
945
946 if (cell.is_pointer_cell())
947 {
948 // cell is a pointer cell which will be replaced later.
949 // For now we add an item in the current mode as placeholder
950 //
951 if (mode == Vm_QUOT) { item.append_UTF8("∘"); return Vm_QUOT; }
952 if (mode == Vm_UCS) { item.append_UTF8("33"); return Vm_UCS; }
953 item.append_UTF8("00");
954 return Vm_NUM;
955 }
956
957 DOMAIN_ERROR;
958 return mode; // not reached
959}

1055//----------------------------------------------------------------------------
1056Value_P
1057Quad_CR::do_CR16(const Value * B)
1058{
1059 if (B->get_rank() > 1) RANK_ERROR;
1060
1061const ShapeItem full_quantums = B->element_count() / 3;
1062const ShapeItem len_Z = 4 * ((B->element_count() + 2) / 3);
1063Value_P Z(len_Z, LOC);
1064
1065const char *alpha = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
1066 "abcdefghijklmnopqrstuvwxyz"
1067 "0123456789+/";
1068
1069const Cell * cB = &B->get_cfirst();
1070 loop(b, full_quantums) // encode full quantums
1071 {
1072 /* -- b1 -- -- b2 -- -- b3 --
1073 z: 11111122 22223333 33444444
1074 */
1075 const int b1 = cB++->get_char_value() & 0x00FF;
1076 const int b2 = cB++->get_char_value() & 0x00FF;
1077 const int b3 = cB++->get_char_value() & 0x00FF;
1078
1079 const int z1 = b1 >> 2;
1080 const int z2 = (b1 & 0x03) << 4 | (b2 & 0xF0) >> 4;
1081 const int z3 = (b2 & 0x0F) << 2 | (b3 & 0xC0) >> 6;
1082 const int z4 = b3 & 0x3F;
1083
1084 Z->next_ravel_Char(Unicode(alpha[z1]));
1085 Z->next_ravel_Char(Unicode(alpha[z2]));
1086 Z->next_ravel_Char(Unicode(alpha[z3]));
1087 Z->next_ravel_Char(Unicode(alpha[z4]));
1088 }
1089
1090 // process final bytes
1091 //
1092 switch(B->element_count() - 3*full_quantums)
1093 {
1094 case 0: break; // length of B is 3 * N
1095
1096 case 1: // length of B is 3 * N + 1
1097 {
1098 const int b1 = cB++->get_char_value() & 0x00FF;
1099 const int b2 = 0;
1100
1101 const int z1 = b1 >> 2;
1102 const int z2 = (b1 & 0x03) << 4 | (b2 & 0xF0) >> 4;
1103
1104 Z->next_ravel_Char(Unicode(alpha[z1]));
1105 Z->next_ravel_Char(Unicode(alpha[z2]));
1106 Z->next_ravel_Char(Unicode('='));
1107 Z->next_ravel_Char(Unicode('='));
1108 }
1109 break;
1110
1111 case 2: // two bytes remaining
1112 {
1113 const int b1 = cB++->get_char_value() & 0x00FF;
1114 const int b2 = cB++->get_char_value() & 0x00FF;
1115 const int b3 = 0;
1116
1117 const int z1 = b1 >> 2;
1118 const int z2 = (b1 & 0x03) << 4 | (b2 & 0xF0) >> 4;
1119 const int z3 = (b2 & 0x0F) << 2 | (b3 & 0xC0) >> 6;
1120
1121 Z->next_ravel_Char(Unicode(alpha[z1]));
1122 Z->next_ravel_Char(Unicode(alpha[z2]));
1123 Z->next_ravel_Char(Unicode(alpha[z3]));
1124 Z->next_ravel_Char(Unicode('='));
1125 }
1126 break;
1127 }
1128
1129 Z->set_proto_Spc();
1130 Z->check_value(LOC);
1131 return Z;
1132}
1133//----------------------------------------------------------------------------
1134Value_P
1135Quad_CR::do_CR17(const Value * B)
1136{
1137 if (B->get_rank() != 1) RANK_ERROR;
1138
1139const int cols = B->get_cols();
1140 if (cols == 0) return Str0(LOC); // empty value
1141 if (cols & 3) LENGTH_ERROR; // length not 4*n
1142
1143 // figure number of missing chars in final quantum
1144 //
1145int missing = 0;
1146 if (B->get_cravel(cols - 2).get_char_value() == '=') missing = 2;
1147 else if (B->get_cravel(cols - 1).get_char_value() == '=') missing = 1;
1148
1149const ShapeItem len_Z = 3 * (B->element_count() / 4) - missing;
1150const ShapeItem quantums = B->element_count() / 4;
1151
1152Value_P Z(len_Z, LOC);
1153const Cell * cB = &B->get_cfirst();
1154 loop(q, quantums)
1155 {
1156 const int b1 = sixbit(cB++->get_char_value());
1157 const int b2 = sixbit(cB++->get_char_value());
1158 const int q3 = sixbit(cB++->get_char_value());
1159 const int q4 = sixbit(cB++->get_char_value());
1160 const int b3 = q3 & 0x3F;
1161 const int b4 = q4 & 0x3F;
1162
1163 /* b1 b2 b3 b4
1164 --zzzzzz --zzzzzz --zzzzzz --zzzzzz
1165 111111 112222 223333 333333
1166 */
1167 const int z1 = b1 << 2 | (b2 & 0x30) >> 4;
1168 const int z2 = (b2 & 0x0F) << 4 | b3 >> 2;
1169 const int z3 = (b3 & 0x03) << 6 | b4;
1170
1171 if (b1 < 0 || b2 < 0 || q3 == -1 || q4 == -1) DOMAIN_ERROR;
1172
1173 if (q < (quantums - 1) || missing == 0)
1174 {
1175 Z->next_ravel_Char(Unicode(z1));
1176 Z->next_ravel_Char(Unicode(z2));
1177 Z->next_ravel_Char(Unicode(z3));
1178 }
1179 else if (missing == 1) // k6 k2l4 l20000 =
1180 {
1181 Z->next_ravel_Char(Unicode(z1));
1182 Z->next_ravel_Char(Unicode(z2));
1183 }
1184 else // k6 k20000 = =
1185 {
1186 Z->next_ravel_Char(Unicode(z1));
1187 }
1188 }
1189
1190 Z->set_proto_Spc();
1191 Z->check_value(LOC);
1192 return Z;
1193}

1301//----------------------------------------------------------------------------
1302Value_P
1303Quad_CR::do_CR30(const Value * B)
1304{
1305 // Z is B with all items conformed to the same rank and shape. Primarily
1306 // an internal function used in macros Z__LO_RANK_X5_B and Z__A_LO_RANK_X7_B
1307 // but possibly useful elsewhere.
1308
1309const ShapeItem len_B = B->element_count();
1310 if (len_B == 0) return CLONE(B, LOC);
1311
1312 // we use 'ShapeItem max_shape[MAX_RANK] max_shape' instead of
1313 // 'Shape max_shape' to avoid multiple recompute_volume() in class Shape
1314 // while looping along the B ravel.
1315 //
1316ShapeItem max_shape[MAX_RANK];
1317 loop(r, MAX_RANK) max_shape[r] = 0;
1318sRank max_rank = 0;
1319
1320 loop(b, len_B)
1321 {
1322 const Cell & cB = B->get_cravel(b);
1323 if (cB.is_lval_cell()) DOMAIN_ERROR;
1324 if (!cB.is_pointer_cell()) continue; // simple scalar
1325
1326 const Shape sh = cB.get_pointer_value()->get_shape();
1327 const sRank rk = sh.get_rank();
1328 if (max_rank < rk) max_rank = rk;
1329 loop(s, rk)
1330 {
1331 const ShapeItem sh_s = sh.get_shape_item(s);
1332 if (max_shape[s] < sh_s) max_shape[s] = sh_s;
1333 }
1334 }
1335
1336Shape conformed;
1337 loop(r, max_rank) conformed.add_shape_item(max_shape[r]);
1338const ShapeItem conformed_len = conformed.get_volume();
1339
1340Shape shape_Z(B->get_shape() + conformed);
1341Value_P Z(shape_Z, LOC);
1342
1343 loop(b, len_B)
1344 {
1345 const Cell & cB = B->get_cravel(b);
1346 if (cB.is_pointer_cell())

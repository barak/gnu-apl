49//----------------------------------------------------------------------------
50void
51Quad_WA::init(bool log_startup)
52{
53 initial_sbrk = top_of_memory();
54
55rlimit rl;
56
57#ifndef RLIMIT_AS // BSD does not define RLIMIT_AS
58# define RLIMIT_AS RLIMIT_DATA
59#endif
60
61 rl.rlim_cur = 0; // suppress warning if getrlimit() is faked
62 getrlimit(RLIMIT_AS, &rl);
63 initial_rlimit = rl.rlim_cur;
64
65 if (log_startup)
66 {
67 CERR << "initial RLIMIT_AS (aka. virtual memory) is: ";
68 if (initial_rlimit == ~rlim_t(0)) CERR << "'unlimited'" << endl;
69 else CERR << initial_rlimit << endl;
70 }
71
72 // if the user has set a memory rlimit (and hopefully knowing what she is
73 // doing) then leave it as is; otherwise set the limit to 80 % of the
74 // avaiable memory
75 //
76 total_memory = get_free_memory();
77 if (log_startup)
78 CERR << "estimated available memory: " << total_memory
79 << " bytes (" << (total_memory/1000000) << " MB)" << endl;
80
81 if (rl.rlim_cur == RLIM_INFINITY)
82 {
83 // the user has not set a memory limit, so we may do that. However,
84 // limits > 2GB are treated as RLIM_INFINITY, so setting larger
85 // limits may have undesirable effects
86 //
87 if (total_memory < 0x8000000)
88 {
89 rl.rlim_cur = total_memory;
90 setrlimit(rl.rlim_cur, &rl);
91 getrlimit(RLIMIT_AS, &rl);
92 if (log_startup)
93 CERR << "decreasing RLIMIT_AS to: " << rl.rlim_cur
94 << " bytes (" << int64_t(rl.rlim_cur/1000000)
95 << " MB)" << endl;
96 }
97 }
98 else if (uint64_t(rl.rlim_cur) > total_memory)
99 {
100 // the user has set a memory limit, but it is above the available
101 // memory
102 //
103 CERR <<
104"*** Warning: the process memory limit (RLIMIT_AS) of " << rl.rlim_cur << endl
105<< " is more than the estimated total_memory of " << total_memory << "." << endl
106<< " This could cause improper WS FULL handling." << endl;
107 }
108}
109//----------------------------------------------------------------------------
110int
111Quad_WA::read_meminfo()
112{
113int ret = 0;
114
115 meminfo.Available = 0;
116 meminfo.Cached = 0;
117 meminfo.MemFree = 0;
118
119FILE * pm = fopen("/proc/meminfo", "r");
120 if (pm == 0) return 0; // error
121
122const struct _tag
123 {
124 const char * name;
125 uint64_t & value;
126 } tags[] = { { "MemAvailable:", meminfo.Available },
127 { "Cached:" , meminfo.Cached },
128 { "MemFree:" , meminfo.MemFree }
129 };
130
131 for (;;) // read all lines of /proc.meminfo
132 {
133 char buffer[2000];
134 if (fgets(buffer, sizeof(buffer) - 1, pm) == 0) break;
135 buffer[sizeof(buffer) - 1] = 0; // to allow strncmp()
136
137 for (size_t t = 0; t < sizeof(tags) / sizeof(*tags); ++t)
138 {
139 const _tag & tag = tags[t];
140 const size_t name_len = strlen(tag.name);
141 if (!strncmp(buffer, tag.name, name_len))
142 {
143 // proc/meminfo valuesa re in kbytes, so we * 1024
144 tag.value = 1024 * strtoll(buffer + name_len, 0, 10);
145 ++ret;
146 }
147 }
148 }
149 fclose(pm);
150
151 return ret; // success
152}
153//----------------------------------------------------------------------------
154int64_t
155Quad_WA::read_procfile(const char * filename)
156{
157 if (FILE * pm = fopen(filename, "r"))
158 {
159 long long ret = -1;
160 const int count = fscanf(pm, "%lld", &ret);
161 fclose(pm);
162 if (count == 1) return ret;
163 }
164
165 return -1; // something went wrong
166}
167//----------------------------------------------------------------------------
168uint64_t
169Quad_WA::get_free_memory()
170{
171uint64_t result = 0xC0000000; // assume ~3 GB on error
172
173 if (read_meminfo() == 3)
174 {
175 result = meminfo.Available ? meminfo.Available
176 : meminfo.Cached + meminfo.MemFree;
177 }
178
179 // limit size on 32-bit machines to 2.95 < 3 GB
180 if (sizeof(const void *) == 4 && result > 0xB0000000) result = 0xB0000000;
181
182 return result;
183}
184//----------------------------------------------------------------------------
185void
186Quad_WA::parse_mem(bool log_startup)
187{
188const char * mem_arg = uprefs.mem_arg;
189
190 if (mem_arg == 0) // no --mem option given
191 {
192 if (log_startup)
193 CERR << "--mem option not used" << endl;
194 return;
195 }
196
197 if (*mem_arg == 0) // -mem option without rgument
198 {
199 if (log_startup)
200 CERR << "using --mem with default value: 50%" << endl;
201
202 mem_arg = "50%";
203 }
204 else
205 {
206 if (log_startup)
207 CERR << "using --mem with user's value " << mem_arg << endl;
208 }
209
210 if (3 != read_meminfo())
211 {
212 CERR << "problem with the --mem option: "
213 "no readable /proc/meminfo: " << strerror(errno) << endl;
214 exit(3);
215 }
216
217char mem_unit = mem_arg[strlen(mem_arg) -1];
218 if (mem_unit == 'b' || mem_unit == 'B') // optional trailing B
219 mem_unit = mem_arg[strlen(mem_arg) -2];
220
221uint64_t mem_number = strtoll(mem_arg, 0, 0);
222uint64_t mem_val = 0;
223 if (mem_unit == '%')
224 {
225 if (mem_number < 5)
226 {
227 CERR << "*** FATAL ERROR: the --mem value '" << mem_arg
228 << "' is too small (minimum is: 5%)" << endl;
229 exit(3);
230 }
231
232 if (mem_number > 95)
233 {
234 CERR << "*** FATAL ERROR: the --mem value '" << mem_arg
235 << "' is too large (maximum is 95%)" << endl;
236 exit(3);
237 }
238
239 mem_val = (meminfo.MemFree / 100) * mem_number;
240 }
241 else if (mem_unit == 'k')
242 {
243 mem_val = mem_number*1000;
244 if (mem_val < 100000000)
245 {
246 CERR << "*** FATAL ERROR: the --mem value '" << mem_arg
247 << "' is too small (minimum is: 100,000k)" << endl;
248 exit(3);
249 }
250
251 if (mem_number*1000 > meminfo.MemFree) // more than we have
252 {
253 CERR << "*** FATAL ERROR: the --mem value '" << mem_arg
254 << "' is too large (max. is MemFree = "
255 << meminfo.MemFree/1000 << " kB)" << endl;
256 exit(3);
257 }
258
259 }
260 else if (mem_unit == 'M')
261 {
262 mem_val = mem_number*1000000;
263 if (mem_val < 100000000)
264 {
265 CERR << "*** FATAL ERROR: the --mem value '" << mem_arg
266 << "' is too small (minimum is: 100M)" << endl;
267 exit(3);
268 }
269
270 if (mem_number*1000*1000 > meminfo.MemFree) // more than we have
271 {
272 CERR << "*** FATAL ERROR: the --mem value '" << mem_arg
273 << "' is too large (max. is MemFree = "
274 << meminfo.MemFree/1000/1000 << " MB)" << endl;
275 exit(3);
276 }
277 }
278 else if (mem_unit == 'G')
279 {
280 mem_val = mem_number*1000000000;
281 if (mem_val < 100000000)
282 {
283 CERR << "*** FATAL ERROR: the --mem value '" << mem_arg
284 << "' is too small (minimum is: 100M)" << endl;
285 exit(3);
286 }
287
288 if (mem_number*1000*1000*1000 > meminfo.MemFree) // more than we have
289 {
290 CERR << "*** FATAL ERROR: the --mem value '" << mem_arg
291 << "' is too large (max. is MemFree = "
292 << meminfo.MemFree/1000/1000/1000 << " GB)" << endl;
293 exit(3);
294 }
295 }
296 else
297 {
298 CERR << "*** FATAL ERROR: invalid unit in --mem value '" << mem_arg
299 << "' (not k, M, G, kB, MB, or GB)" << endl;
300 exit(3);
301 }
302
303 // first check if --mem can be used at all
304 //
305const int64_t overcommit = read_procfile("/proc/sys/vm/overcommit_memory");
306 if (overcommit == -1)
307 {
308 CERR << "*** FATAL ERROR: --mem option used, "
309 "but /proc/sys/vm/overcommit_memory is not readable" << endl;
310 exit(3);
311 }
312
313 if (overcommit != 2)
314 {
315 CERR << "*** FATAL ERROR: the --mem option used with "
316 "memory overcommit enabled." << endl
317 << " I.e. /proc/sys/vm/overcommit_memory says '" << overcommit
318 <<"' while --mem requires '2'" << endl;
319 exit(3);
320 }
321
322 CERR << "!!! the estimated total_memory of " << total_memory/1000 << "kB"
323 " = " << total_memory/1000000 << "MB"
324 " = " << total_memory/1000000000 << "GB" << endl
325 << "!!! was overridden with " << mem_val/1000 << "kB"
326 " = " << mem_val/1000000 << "MB"
327 " = " << mem_val/1000000000 << "GB"
328 " by the --mem option !!!" << endl;
329
330 total_memory = mem_val;
331}

76UCS_string::UCS_string(const char * cstring)
77{
78 // calling this constructor with and utf8-encoded C string is usually wrong.
79 //
80 // Instead of: UCS_string(const char * utf8_string)
81 // one should: UCS_string(UTF8_string(utf8_string))
82 //
83 // so that the constructor below takes care of the utf8-decoding.
84 //
85 // For ASCII C strings this constuctor is fine.
86 //
87 create(LOC);
88
89 for (const char * str = cstring; *str; ++str)
90 {
91 if (0x80 & *cstring) // ASCII
92 {
93 CERR << "non-ASCII char in C-String '" << cstring << "'" << endl;
94 Assert(0 && "Bad C-string");
95 }
96 *this += Unicode(*str);
97 }
98}
99//----------------------------------------------------------------------------
100UCS_string::UCS_string(const UTF8_string & utf)
101{
102 create(LOC);
103
104 Log(LOG_char_conversion)
105 CERR << "UCS_string::UCS_string(): utf = " << utf << endl;
106
107size_t from = 0;
108
109 for (size_t i = 0; i < utf.size();)
110 {
111start_of_sequence:
112
113 const uint32_t b0 = utf[i++];
114 uint32_t bx = b0;
115 uint32_t more;
116 if ((b0 & 0x80) == 0x00) { more = 0; }
117 else if ((b0 & 0xE0) == 0xC0) { more = 1; bx &= 0x1F; }
118 else if ((b0 & 0xF0) == 0xE0) { more = 2; bx &= 0x0F; }
119 else if ((b0 & 0xF8) == 0xF0) { more = 3; bx &= 0x0E; }
120 else if ((b0 & 0xFC) == 0xF8) { more = 4; bx &= 0x07; }
121 else if ((b0 & 0xFE) == 0xFC) { more = 5; bx &= 0x03; }
122 else // invalid UTF start byte
123 {
124 Log(LOG_char_conversion)
125 {
126 utf.dump_hex(CERR << "Bad UTF8 string: ", 40)
127 << " at " << LOC << endl;
128 BACKTRACE
129 }
130
131 // map this string to the "supplementary private use area B" so
132 // that its hex code becomes (sort of) visible
133 //
134 append(Unicode(utf[from] | 0x100000));
135 i = ++from; // retry, starting at next char
136 goto start_of_sequence;
137 }
138
139 uint32_t uni = 0;
140 for (; more; --more)
141 {
142 if (i >= utf.size())
143 {
144 Log(LOG_char_conversion)
145 {
146 utf.dump_hex(CERR << "Truncated UTF8 string: ", 40)
147 << " len " << utf.size()
148 << " at " << LOC << endl;
149 if (utf.size() >= 40)
150 {
151 const UTF8_string end(&utf[utf.size() - 10], 10);
152 end.dump_hex(CERR << endl << "(ending with : ", 20)
153 << ")" << endl;
154 }
155 }
156
157 append(Unicode(utf[from] | 0x100000));
158 i = ++from; // retry, starting at next char
159 goto start_of_sequence;
160 }
161
162 const UTF8 subc = utf[i++];
163 if ((subc & 0xC0) != 0x80) // invalid UTF continuation byte
164 {
165 Log(LOG_char_conversion)
166 {
167 utf.dump_hex(CERR << "Bad UTF8 string: ", 40)
168 << " len " << utf.size()
169 << " at " << LOC << endl;
170 if (utf.size() >= 40)
171 {
172 const UTF8_string end(&utf[utf.size() - 10], 10);
173 end.dump_hex(CERR << endl << "(ending with : ", 20)
174 << ")" << endl;
175 }
176 BACKTRACE
177 }
178
179 append(Unicode(utf[from] | 0x100000));
180 i = ++from; // retry, starting at next char
181 goto start_of_sequence;
182 }
183
184 bx <<= 6;
185 uni <<= 6;
186 uni |= subc & 0x3F;
187 }
188
189 append(Unicode(bx | uni));
190 from = i;
191 }
192
193 Log(LOG_char_conversion)
194 CERR << "UCS_string::UCS_string(): ucs = " << *this << endl;
195}
196//----------------------------------------------------------------------------
197UCS_string::UCS_string(APL_Float value, bool & scaled,
198 const PrintContext & pctx)
199{
200 create(LOC);
201
202int quad_pp = pctx.get_PP();
203 if (quad_pp > MAX_Quad_PP) quad_pp = MAX_Quad_PP;
204 if (quad_pp < MIN_Quad_PP) quad_pp = MIN_Quad_PP;
205
206const bool negative = (value < 0.0);
207 if (negative) value = -value;
208
209int expo = 0;
210
211 if (value >= 10.0) // large number, positive exponent
212 {
213 if (value > BIG_FLOAT || !isnormal(value)) // something odd
214 {
215 if (isnormal(value) || isinf(value)) // rather large
216 {
217 if (negative) append_UTF8("¯∞");
218 else append_UTF8("∞");
219 FloatCell::map_FC(*this);
220 }
221 else
222 {
223 append_UTF8("-nan-");
224 }
225 return;
226 }
227
228 while (value >= 1e16) { value *= 1e-16; expo += 16; }
229 while (value >= 1e4) { value *= 1e-4; expo += 4; }
230 while (value >= 1e1) { value *= 1e-1; ++expo; }
231 }
232 else if (value < 1.0) // small number, negative exponent
233 {
234 if (value < 1e-305) // very small number: make it 0
235 {
236 append(UNI_0);
237 return;
238 }
239
240 while (value < 1e-16) { value *= 1e16; expo -= 16; }
241 while (value < 1e-4) { value *= 1e4; expo -= 4; }
242 while (value < 1.0) { value *= 10.0; --expo; }
243 }
244
245 // In theory, at this point, 1.0 ≤ value < 10.0. In reality value can
246 // be outside, though, due to rounding errors.
247
248 // create a string with quad_pp + 1 significant digits.
249 // The last digit is used for rounding and then discarded.
250 //
251UCS_string digits;
252 loop(d, (quad_pp + 2))
253 {
254 if (value >= 10.0)
255 {
256 // 10.0 or more is a rounding error from 9,999...
257 digits.append(Unicode(10 + '0'));
258 while (digits.size() < (quad_pp + 2)) digits.append(UNI_0);
259 break;
260 }
261 else if (value < 0.0)
262 {
263 // less than 0.0 is a rounding error from 0.000...
264 while (digits.size() < (quad_pp + 2)) digits.append(UNI_0);
265 break;
266 digits.append(UNI_0);
267 value = 0.0;
268 }
269 else
270 {
271 const int dig = int(value);
272 value -= dig;
273 value *= 10.0;
274 digits.append(Unicode(dig + '0'));
275 }
276 }
277
278 if (digits[0] != '0') digits.pop_back();
279
280 // round last digit
281 //
282const Unicode last = digits.back();
283 digits.pop_back();
284
285 if (last >= '5') digits.back() = Unicode(digits.back() + 1);
286
287 // adjust carries of 2nd to last digit
288 //
289 for (int d = digits.size() - 1; d > 0; --d) // all but first
290 {
291 if (digits[d] > '9')
292 {
293 digits[d] = Unicode(digits[d] - 10);
294 digits[d - 1] = Unicode(digits[d - 1] + 1);
295 }
296 }
297
298 // adjust carry of 1st digit
299 //
300 if (digits[0] > '9')
301 {
302 digits[0] = Unicode(digits[0] - 10);
303 digits.insert(0, UNI_1);
304 ++expo;
305 digits.pop_back();
306 }
307
308 // remove trailing zeros
309 //
310 while (digits.size() > 1 && digits.back() == UNI_0) digits.pop_back();
311
312 // force scaled format if:
313 //
314 // value < .00001 ( value has ≥ 5 leading zeroes)
315 // value > 10⋆quad_pp ( integer larger than ⎕PP)
316 //
317 if ((expo < -6) || (expo > quad_pp)) scaled = true;
318
319 if (negative) append(UNI_OVERBAR);
320
321 if (scaled)
322 {
323 append(digits[0]); // integer part
324 if (digits.size() > 1) // fractional part
325 {
326 append(UNI_FULLSTOP);
327 loop(d, (digits.size() - 1)) append(digits[d + 1]);
328 }
329 if (expo < 0)
330 {
331 append(UNI_E);
332 append(UNI_OVERBAR);
333 append_number(-expo);
334 }
335 else if (expo > 0)
336 {
337 append(UNI_E);
338 append_number(expo);
339 }
340 else if (!(pctx.get_style() & PST_NO_EXPO_0)) // expo == 0
341 {
342 append(UNI_E);
343 append(UNI_0);
344 }
345 }
346 else
347 {
348 if (expo < 0) // 0.000...
349 {
350 append(UNI_0);
351 append(UNI_FULLSTOP);
352 loop(e, (-(expo + 1))) append(UNI_0);
353 append(digits);
354 }
355 else // expo >= 0
356 {
357 loop(e, expo + 1)
358 {
359 if (e < digits.size()) append(digits[e]);
360 else append(UNI_0);
361 }
362
363 if ((expo + 1) < digits.size()) // there are fractional digits
364 {
365 append(UNI_FULLSTOP);
366 for (int e = expo + 1; e < digits.size(); ++e)
367 {
368 if (e < digits.size()) append(digits[e]);
369 else break;
370 }
371 }
372 }
373 }
374
375 FloatCell::map_FC(*this);
376}
377//----------------------------------------------------------------------------
378UCS_string::UCS_string(const PrintBuffer & pb, sRank rank, int quad_PW)
379{
380 create(LOC);
381
382 if (pb.get_row_count() == 0) return; // empty PrintBuffer
383
384const int total_width = pb.get_column_count();
385
386 // initialize chunk_lengths, based on the first row of the PrintBuffer.
387 // All subsequent rows are aligned to the first row, therefore the first
388 // row can be taken as a prototype for all rows.
389size_t chunk_len = 0;
390std::vector<int> chunk_lengths;
391 chunk_lengths.reserve(2*total_width/quad_PW);
392 for (int col = 0; col < total_width; col += chunk_len)
393 {
394 chunk_len = pb.get_line(0).compute_chunk_length(quad_PW,col);
395 chunk_lengths.push_back(chunk_len);
396 }
397
398 // print rows, breaking at chunk_lengths
399 //
400 loop(row, pb.get_row_count())
401 {
402 if (row) append(UNI_LF); // end previous row
403 int brk_idx = 0;
404
405 for (int col = 0; col < total_width; col += chunk_len)
406 {
407 chunk_len = chunk_lengths[brk_idx++];
408
409 if (col) append_UTF8("\n ");
410 UCS_string trow(pb.get_line(row), col, chunk_len);
411 trow.remove_trailing_padchars();
412 append(trow);
413 }
414 }
415
416 // replace pad chars with blanks.
417 //
418 loop(u, size())
419 {
420 if (is_iPAD_char(at(u))) at(u) = UNI_SPACE;
421 }
422}

476//----------------------------------------------------------------------------
477int
478UCS_string::compute_chunk_length(int quad_PW, int col) const
479{
480 // the space available is ⎕PW for the first line or ⎕PW less the usual
481 // 6 space indentation for subsequent lines.
482 //
483const int chunk_len = col ? quad_PW - 6 : quad_PW;
484
485 // if the rest of the line (from col to the end of the line) is small,
486 // then return the length of the rest.
487 //
488int pos = col + chunk_len;
489 if (pos >= size()) return size() - col;
490
491 // otherwise the rest does not fit into ⎕PW. Find a whitespace
492 // at which the line should be broken.
493 //
494 while (--pos > col)
495 {
496 const Unicode uni = at(pos);
497 if (uni == UNI_SPACE || uni == UNI_iPAD_U2 || uni == UNI_iPAD_U3)
498 {
499 return pos - col + 1;
500 }
501 }
502
503 // no whitespace found (the entire string may not have any).
504 // Break at chunk_len.
505 //
506 return chunk_len;
507}
508//----------------------------------------------------------------------------
509void
510UCS_string::remove_trailing_padchars()
511{
512 // remove trailing pad chars from align() and append_string(),
513 // but leave other pad chars intact.
514 // But only if the line has no frame (vert).
515 //
516again:
517 if (size() > 2 && contains(UNI_iPAD_L0))
518 {
519 Unicode first = Unicode_0;
520 if (back() == UNI_iPAD_U9) first = UNI_iPAD_U8;
521 else if (back() == UNI_iPAD_U5) first = UNI_iPAD_U4;
522
523 if (first && at(size() - 2) == UNI_iPAD_L0)
524 {
525 // last column is a nested dimension separator row
526 //
527 ShapeItem len = size() - 2;
528 while (len && (at(len - 1) == UNI_iPAD_L0 ||
529 at(len - 1) == UNI_iPAD_U0 ||
530 at(len - 1) == UNI_iPAD_U2 ||
531 at(len - 1) == UNI_iPAD_U6 ||
532 at(len - 1) == UNI_iPAD_U7)) --len;
533 if (len && at(len - 1) == first)
534 {
535 resize(len - 1);
536 goto again;
537 }
538 }
539 }
540
541 // If the line contains UNI_iPAD_L0 (higher dimension separator)
542 // then discard all chars (unless the line is framed)..
543 //
544 while (size())
545 {
546 const Unicode last = back();
547 if (last == UNI_iPAD_L0 || // ₀: higher dimension separator
548 last == UNI_iPAD_L1 || // ₁: pad line to PrintBuffer width
549 last == UNI_iPAD_L2 || // ₂: pad PrintBuffer width to line
550 last == UNI_iPAD_L3 || // ₃: pad integer part to the left
551 last == UNI_iPAD_L4 || // ₄: pad fractional part to the right
552 last == UNI_iPAD_U2 || // ² column separator (after notchar)
553 last == UNI_iPAD_U2 || // ² column separator (after notchar)
554 last == UNI_iPAD_U7) // ⁷ pad final to the right
555 pop_back();
556 else
557 break;
558 }
559}
560//----------------------------------------------------------------------------
561void
562UCS_string::remove_trailing_whitespaces()
563{
564 while (size() && back() <= UNI_SPACE) pop_back();
565}
566//----------------------------------------------------------------------------
567void
568UCS_string::remove_leading_whitespaces()
569{
570int count = 0;
571 loop(s, size())
572 {
573 if (at(s) <= UNI_SPACE) ++count;
574 else break;
575 }
576
577 if (count == 0) return; // no leading whitspaces
578 if (count == size()) clear(); // only whitespaces
579 else basic_string::erase(0, count);
580}
581//----------------------------------------------------------------------------
582void
583UCS_string::split_ws(UCS_string & rest)
584{
585 remove_leading_and_trailing_whitespaces();
586
587 loop(clen, size())
588 {
589 if (Avec::is_white(at(clen))) // whilespace: end of command
590 {
591 ShapeItem arg = clen;
592 while (arg < size() && Avec::is_white(at(arg))) ++arg;
593 while (arg < size()) rest.append(at(arg++));
594 resize(clen);
595 return;
596 }
597 }
598}

617UCS_string::substr_pos(const UCS_string & sub) const
618{
619const ShapeItem start_positions = 1 + size() - sub.size();
620 loop(start, start_positions)
621 {
622 bool mismatch = false;
623 loop(u, sub.size())
624 {
625 if (at(start + u) != sub[u])
626 {
627 mismatch = true;
628 break;
629 }
630 }
631
632 if (!mismatch) return start; // found sub at start
633 }
634
635 return -1; // not found
636}

665const ShapeItem s_len = strlen(suffix);
666 if (size() < s_len) return false;

670 return true;

682//----------------------------------------------------------------------------
683bool
684UCS_string::starts_iwith(const char * prefix) const
685{
686 loop(s, size())
687 {
688 char pc = *prefix++;
689 if (pc == 0) return true; // prefix matches this string.
690 if (pc >= 'a' && pc <= 'z') pc -= 'a' - 'A';
691
692 int uni = at(s);
693 if (uni >= 'a' && uni <= 'z') uni -= 'a' - 'A';
694
695 if (uni != Unicode(pc)) return false;
696 }
697
698 return *prefix == 0;
699}
700//----------------------------------------------------------------------------
701bool
702UCS_string::starts_iwith(const UCS_string & prefix) const
703{
704 if (prefix.size() > size()) return false;
705
706 loop(p, prefix.size())
707 {
708 int c1 = at(p);
709 int c2 = prefix[p];
710 if (c1 >= 'a' && c1 <= 'z') c1 -= 'a' - 'A';
711 if (c2 >= 'a' && c2 <= 'z') c2 -= 'a' - 'A';
712 if (c1 != c2) return false;
713 }
714
715 return true;
716}

764{
765 if (size() == 0) return false;

771 if (!Avec::is_symbol_char(at(t))) return false;
772 }
773
774 return false;
775}
776//----------------------------------------------------------------------------
777ShapeItem
778UCS_string::double_quote_count(bool in_quote2) const
779{
780ShapeItem count = 0;
781bool in_quote1 = false;
782 loop(s, size())
783 {
784 const Unicode uni = at(s);
785 switch(uni)
786 {
787 case UNI_SINGLE_QUOTE:
788 if (!in_quote2) in_quote1 = ! in_quote1;
789 break;
790
791 case UNI_DOUBLE_QUOTE:
792 if (!in_quote1)
793 {
794 ++count;
795 in_quote2 = ! in_quote2;
796 }
797 break;
798
799 case UNI_BACKSLASH:
800 if (in_quote2) ++s; // ignore next char inside ""
801 break;
802
803 case UNI_NUMBER_SIGN:
804 case UNI_COMMENT:
805 if (!(in_quote1 || in_quote2)) return count;
806
807 default: ;
808 }
809 }
810
811 return count;
812}

977//----------------------------------------------------------------------------
978UCS_string
979UCS_string::un_escape(bool double_quoted, bool keep_LF) const
980{
981const char * hex = "0123456789abcdef";
982UCS_string ret;
983 ret.reserve(size());
984
985 if (double_quoted)
986 {
987 loop(s, size())
988 {
989 const Unicode uni = at(s);
990 if (uni != UNI_BACKSLASH) // normal char
991 {
992 ret.append(uni);
993 continue;
994 }
995
996 if (s >= (size() - 1)) // \ at end of string
997 {
998 ret.append(UNI_BACKSLASH);
999 break;
1000 }
1001
1002 const Unicode uni1 = at(++s);
1003 switch(uni1)
1004 {
1005 case UNI_a: ret << UNI_BEL; continue;
1006 case UNI_b: ret << UNI_BS; continue;
1007 case UNI_f: ret << UNI_FF; continue;
1008 case UNI_n: if (keep_LF) break;
1009 ret << UNI_LF; continue;
1010 case UNI_r: ret << UNI_CR; continue;
1011 case UNI_t: ret << UNI_BS; continue;
1012 case UNI_v: ret << UNI_VT; continue;
1013 case UNI_DOUBLE_QUOTE:
1014 case UNI_BACKSLASH:
1015 ret << uni1; continue;
1016 default: break;
1017 }
1018
1019 int max_len = 0;
1020 if (uni1 == UNI_u)
1021 {
1022 max_len = 4;
1023 }
1024 else if (uni1 == UNI_x)
1025 {
1026 max_len = 2;
1027 }
1028 else // \n or \": keep them escaped
1029 {
1030 ret.append(uni);
1031 ret.append(uni1);
1032 continue;
1033 }
1034
1035 // \x or \u
1036 //
1037 int value = 0;
1038 loop(m, max_len)
1039 {
1040 if (s >= (size() - 1)) break;
1041 const int dig = at(s+1);
1042 const char * pos = strchr(hex, dig);
1043 if (pos == 0) break; // non-hex character
1044
1045 value = value << 4 | (pos - hex);
1046 ++s;
1047 }
1048 ret.append(Unicode(value));
1049 }
1050 }
1051 else
1052 {
1053 bool got_quote = false;
1054 loop(s, size())
1055 {
1056 const Unicode uni = at(s);
1057 if (uni == UNI_SINGLE_QUOTE)
1058 {
1059 if (got_quote)
1060 {
1061 ret.append(UNI_SINGLE_QUOTE);
1062 got_quote = false;
1063 }
1064 else
1065 {
1066 ret.append(UNI_SINGLE_QUOTE);
1067 got_quote = true;
1068 }
1069 }
1070 else
1071 {
1072 if (got_quote) ret.append(UNI_SINGLE_QUOTE); // mal-formed
1073 ret.append(uni);
1074 got_quote = false;
1075 }
1076 }
1077 }
1078
1079 return ret;
1080}
1081//----------------------------------------------------------------------------
1082UCS_string
1083UCS_string::do_escape(bool double_quoted) const
1084{
1085const char * hex = "0123456789abcdef";
1086UCS_string ret;
1087 ret.reserve(size());
1088
1089 if (double_quoted)
1090 {
1091 loop(s, size())
1092 {
1093 const Unicode uni = at(s);
1094 switch(uni)
1095 {
1096 case UNI_BEL: ret << "\\a"; continue;
1097 case UNI_BS: ret << "\\b"; continue;
1098 case UNI_HT: ret << "\\t"; continue;
1099 case UNI_LF: ret << "\\n"; continue;
1100 case UNI_VT: ret << "\\v"; continue;
1101 case UNI_FF: ret << "\\f"; continue;
1102 case UNI_CR: ret << "\\r"; continue;
1103 case UNI_DOUBLE_QUOTE: ret << "\\\""; continue;
1104 case UNI_BACKSLASH: ret << "\\\\"; continue;
1105 default: break;
1106 }
1107
1108 // none of the above
1109 //
1110 if (uni >= UNI_SPACE && uni < UNI_DELETE)
1111 {
1112 ret.append(uni);
1113 continue;
1114 }
1115
1116 if (uni <= 0x0F) // small ASCII
1117 {
1118 ret << "\\x0";
1119 ret << Unicode(hex[uni]);
1120 }
1121 else
1122 {
1123 ret.append(uni);
1124 }
1125 }
1126 }
1127 else // single-quoted
1128 {
1129 loop(s, size())
1130 {
1131 const Unicode uni = at(s);
1132 ret.append(uni);
1133 if (uni == UNI_SINGLE_QUOTE) ret.append(uni); // another '
1134 }
1135 }
1136
1137 return ret;
1138}
1139//----------------------------------------------------------------------------
1140size_t
1141UCS_string::to_vector(UCS_string_vector & result) const
1142{
1143size_t max_len = 0;
1144
1145 result.clear();
1146 if (size() == 0) return max_len;
1147
1148 result.push_back(UCS_string());
1149 loop(s, size())
1150 {
1151 const Unicode uni = at(s);
1152 if (uni == UNI_LF) // line done
1153 {
1154 const size_t len = result.back().size();
1155 if (max_len < len) max_len = len;
1156
1157 if (s < size() - 1) // more coming
1158 result.push_back(UCS_string());
1159 }
1160 else
1161 {
1162 if (uni != UNI_CR) // ignore \r.
1163 result.back().append(uni);
1164 }
1165 }
1166
1167 // if the last line lacked a \n we check max_len here again.
1168const size_t len = result.back().size();
1169 if (max_len < len) max_len = len;
1170
1171 return max_len;
1172}
1173//----------------------------------------------------------------------------
1174int
1175UCS_string::atoi() const
1176{
1177int ret = 0;
1178bool negative = false;
1179
1180 loop(s, size())
1181 {
1182 const Unicode uni = at(s);
1183
1184 if (!ret && Avec::is_white(uni)) continue; // leading whitespace
1185
1186 if (uni == UNI_MINUS || uni == UNI_OVERBAR)
1187 {
1188 negative = true;
1189 continue;
1190 }
1191
1192 if (uni < UNI_0) break; // non-digit
1193 if (uni > UNI_9) break; // non-digit
1194
1195 ret *= 10;
1196 ret += uni - UNI_0;
1197 }
1198
1199 return negative ? -ret : ret;
1200}

1287//----------------------------------------------------------------------------
1288UCS_string
1289UCS_string::from_uint(uint64_t value)
1290{
1291 if (value == 0) return UCS_string("0");
1292
1293int digits[40];
1294int * d = digits;
1295
1296 while (value)
1297 {
1298 const uint64_t v_10 = value / 10;
1299 *d++ = value - 10*v_10;
1300 value = v_10;
1301 }
1302
1303UCS_string ret;
1304 while (d > digits) ret.append(Unicode(UNI_0 + *--d));
1305 return ret;
1306}
1307//----------------------------------------------------------------------------
1308UCS_string
1309UCS_string::from_big(APL_Float & val)
1310{
1311 Assert(val >= 0.0);
1312
1313long double value = val;
1314int digits[320]; // DBL_MAX is 1.79769313486231470E+308
1315int * d = digits;
1316
1317const long double initial_fract = modf(value, &value);
1318long double fract;
1319 for (; value >= 1.0; ++d)
1320 {
1321 fract = modf(value / 10.0, &value); // U.x -> .U
1322 *d = int((fract + .02) * 10.0);
1323 fract -= 0.1 * *d;
1324 }
1325
1326 val = initial_fract;
1327
1328UCS_string ret;
1329 if (d == digits) ret.append(UNI_0); // 0.xxx
1330
1331 while (d > digits) ret.append(Unicode(UNI_0 + *--d));
1332 return ret;
1333}
1334//----------------------------------------------------------------------------
1335UCS_string
1336UCS_string::from_double_to_expo(APL_Float v, int fract_digits)
1337{
1338UCS_string ret;
1339
1340 if (v == 0.0)
1341 {
1342 ret.append(UNI_0);
1343 if (fract_digits) // unless integer only
1344 {
1345 ret.append(UNI_FULLSTOP);
1346 loop(f, fract_digits) ret.append(UNI_0);
1347 }
1348 ret.append(UNI_E);
1349 ret.append(UNI_0);
1350 return ret;
1351 }
1352
1353 if (v < 0.0) { ret.append(UNI_OVERBAR); v = -v; }
1354
1355int expo = 0;
1356 while (v >= 1.0E1)
1357 {
1358 if (v >= 1.0E9)
1359 if (v >= 1.0E13)
1360 if (v >= 1.0E15)
1361 if (v >= 1.0E16) { v = v * 1.0E-16; expo += 16; }
1362 else /* v >= 1.0E15 */ { v = v * 1.0E-15; expo += 15; }
1363 else
1364 if (v >= 1.0E14) { v = v * 1.0E-14; expo += 14; }
1365 else /* v >= 1.0E13 */ { v = v * 1.0E-13; expo += 13; }
1366 else
1367 if (v >= 1.0E11)
1368 if (v >= 1.0E12) { v = v * 1.0E-12; expo += 12; }
1369 else /* v >= 1.0E11 */ { v = v * 1.0E-11; expo += 11; }
1370 else
1371 if (v >= 1.0E10) { v = v * 1.0E-10; expo += 10; }
1372 else /* v >= 1.0E9 */ { v = v * 1.0E-9; expo += 9; }
1373 else
1374 if (v >= 1.0E5)
1375 if (v >= 1.0E7)
1376 if (v >= 1.0E8) { v = v * 1.0E-8; expo += 8; }
1377 else /* v >= 1.0E7 */ { v = v * 1.0E-7; expo += 7; }
1378 else
1379 if (v >= 1.0E6) { v = v * 1.0E-6; expo += 6; }
1380 else /* v >= 1.0E5 */ { v = v * 1.0E-5; expo += 5; }
1381 else
1382 if (v >= 1.0E3)
1383 if (v >= 1.0E4) { v = v * 1.0E-4; expo += 4; }
1384 else /* v >= 1.0E3 */ { v = v * 1.0E-3; expo += 3; }
1385 else
1386 if (v >= 1.0E2) { v = v * 1.0E-2; expo += 2; }
1387 else /* v >= 1.0E1 */ { v = v * 1.0E-1; expo += 1; }
1388 }
1389
1390 while (v < 1.0E0)
1391 {
1392 if (v < 1.0E-8)
1393 if (v < 1.0E-12)
1394 if (v < 1.0E-14)
1395 if (v < 1.0E-15) { v = v * 1.0E-16; expo += 16; }
1396 else /* v < 1.0E-14 */ { v = v * 1.0E-15; expo += 15; }
1397 else
1398 if (v < 1.0E-13) { v = v * 1.0E-14; expo += 14; }
1399 else /* v < 1.0E-12 */ { v = v * 1.0E-13; expo += 13; }
1400 else
1401 if (v < 1.0E-10)
1402 if (v < 1.0E-11) { v = v * 1.0E12; expo += -12; }
1403 else /* v < 1.0E-10 */ { v = v * 1.0E11; expo += -11; }
1404 else
1405 if (v < 1.0E-9 ) { v = v * 1.0E10; expo += -10; }
1406 else /* v < 1.0E-8 */ { v = v * 1.0E9; expo += -9; }
1407 else
1408 if (v < 1.0E-4)
1409 if (v < 1.0E-6)
1410 if (v < 1.0E-7) { v = v * 1.0E8; expo += -8; }
1411 else /* v < 1.0E-6 */ { v = v * 1.0E7; expo += -7; }
1412 else
1413 if (v < 1.0E-5) { v = v * 1.0E6; expo += -6; }
1414 else /* v < 1.0E-4 */ { v = v * 1.0E5; expo += -5; }
1415 else
1416 if (v < 1.0E-2)
1417 if (v < 1.0E-3) { v = v * 1.0E4; expo += -4; }
1418 else /* v < 1.0E-2 */ { v = v * 1.0E3; expo += -3; }
1419 else
1420 if (v < 1.0E-1) { v = v * 1.0E2; expo += -2; }
1421 else /* v < 1.0E0 */ { v = v * 1.0E1; expo += -1; }
1422 }
1423
1424 Assert(v >= 1.0);
1425 Assert(v < 10.0);
1426
1427 // print the mantissa in fixed format. Normally the mantissa is ≥ 1.0 and
1428 // < 10.0, i.e. excluding 10.0. However, rounding up to fract_digits could
1429 // make it 10 (including)
1430 //
1431UCS_string mantissa = from_double_to_fixed(v, fract_digits);
1432 if (mantissa.size() >= 2 && mantissa[0] == UNI_1 && mantissa[1] == UNI_0)
1433 {
1434 // mantissa starts with 10
1435 //
1436 if (mantissa.size() == 2) // 10Ee → 1.0yyyEe+1
1437 {
1438 mantissa[0] = UNI_1;
1439 mantissa.resize(1);
1440 if (fract_digits)
1441 {
1442 mantissa += UNI_FULLSTOP;
1443 loop(z, fract_digits) mantissa += UNI_0;
1444 }
1445 ++expo;
1446 }
1447 else if (mantissa[2] == UNI_FULLSTOP) // 10.yyyEe → 1.0yyyEe+1
1448 {
1449 mantissa[1] = UNI_FULLSTOP;
1450 mantissa[2] = UNI_0;
1451 mantissa.pop_back();
1452 ++expo;
1453 }
1454 }
1455
1456 ret.append(mantissa);
1457 ret.append(UNI_E);
1458 ret.append(from_int(expo));
1459
1460 return ret;
1461}
1462//----------------------------------------------------------------------------
1463UCS_string
1464UCS_string::from_double_to_fixed(APL_Float v, int fract_digits)
1465{
1466 // fract_digits shall be the number of digitsd AFTER the decimal point
1467
1468UCS_string ret;
1469 if (v < 0.0) { ret.append(UNI_OVERBAR); v = - v; }
1470
1471 // store the integer part of v in ret, leaving the fract part in v.
1472 //
1473 ret.append(from_big(v)); // from_big() leaves fractional part of v in v
1474 ret.append(UNI_FULLSTOP);
1475
1476 // append one more fractional digit than needed (the last one will be
1477 // rounded
1478 loop(f, fract_digits + 1)
1479 {
1480 v = v * 10.0;
1481 const int vv = v;
1482 ret.append(Unicode(UNI_0 + vv));
1483 v -= vv;
1484 }
1485
1486 // round to last digit may increase the length
1487const int ret_len = ret.size();
1488 ret.round_last_digit();
1489 while (ret.size() > ret_len) ret.pop_back();
1490 return ret;
1491}
1492//----------------------------------------------------------------------------
1493void
1494UCS_string::round_last_digit()
1495{
1496 Assert(size() > 1);
1497
1498 // if the number ends with digits 0..4 then we simply discard
1499 // the trailing digit and return
1500 //
1501const bool round_down = back() <= UNI_4; // remember direction
1502 pop_back(); // discard the digit that is rounded up or down
1503 if (size() && back() == UNI_FULLSTOP) pop_back(); // trailing .
1504
1505 Assert(size() > 0);
1506 if (round_down) // round down (= discard last digit)
1507 {
1508 return;
1509 }
1510
1511bool carry = true; // from the rounded up digit
1512
1513 for (int q = size() - 1; q >= 0; --q)
1514 {
1515 const Unicode cc = at(q);
1516 if (cc < UNI_0) continue; // not a digit, e.g. ¯ or .
1517 if (cc > UNI_9) continue; // not a digit, e.g. ¯ or .
1518 if (cc == UNI_9)
1519 {
1520 at(q) = UNI_0; // 9 → 0 and keep carry
1521 }
1522 else
1523 {
1524 at(q) = Unicode(cc + 1); // round cc up and clear carry
1525 carry = false;
1526 break;
1527 }
1528 }
1529
1530 if (size() && back() == UNI_FULLSTOP) pop_back();
1531 if (carry)
1532 {
1533 insert(at(0) == UNI_OVERBAR || at(0) == UNI_MINUS ? 1 : 0, UNI_1);

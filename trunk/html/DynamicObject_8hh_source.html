32/*
33 The base class for dynamically allocated objects. It remembers where an object
34 was allocated and maintains a doubly linked ring of all allocated objects.
35
36 The doubly linked list has a statically allocated anchor that must not
37 be removed. The existance of the anchor guarantees that the list is never
38 empty (which simplifies programming).
39 */
41class DynamicObject
42{
43 friend class Value; // for rValue()
44
45public:
47 DynamicObject(const char * loc, DynamicObject * anchor)
48 : alloc_loc(loc)
49 {
50 Log(LOG_delete) print_new(CERR, loc);
51
52 next = anchor->next;
53 anchor->next = this;
54
55 prev = anchor;
56 next->prev = this;
57 }
58
60 DynamicObject(const char * loc)
61 : alloc_loc(loc),
62 next(this),
63 prev(this)
64 {}
65
67 DynamicObject * get_next() { return next; }
68
70 const DynamicObject * get_next() const { return next; }
71
73 DynamicObject * get_prev() { return prev; }
74
76 const DynamicObject * get_prev() const { return prev; }
77
79 void unlink()
80 {
81 // print(CERR);
82
83 prev->next = next;
84 next->prev = prev;
85
86 prev = this;
87 next = this;
88 }
89
91 const IndexExpr * pIndexExpr() const;
92
94 IndexExpr * pIndexExpr();
95
98 void print(ostream & out) const;
99
102 void print_new(ostream & out, const char * loc) const;
103
105 void print_chain(ostream & out) const;
106
108 const char * where_allocated() const
109 { return alloc_loc; }
110
112 static const DynamicObject * get_all_values()
113 { return &all_values; }
114
115protected:
117 const Value & rValue() const;
118
120 Value & rValue();
121
123 const char * alloc_loc;
124
126 DynamicObject * next;
127
129 DynamicObject * prev;
130
132 static DynamicObject all_values;
133
135 static DynamicObject all_index_exprs;
136};

28void
29IndexIterator::operator ++()
30{
31 ++pos;
32 if (pos >= get_index_count())
33 {
34 if (upper)
35 {
36 pos = 0;
37 ++*upper;
38 }
39 else
40 {
41 pos = get_index_count(); // so that more() works
42 }
43 }
44
45 // print(CERR);
46}

61TrueIndexIterator::TrueIndexIterator(ShapeItem w, Value_P value,
62 uint32_t qio, ShapeItem max_idx)
63 : IndexIterator(w, value->element_count())
64{
65// CERR << "TrueIndexIterator(w=" << w << ", value=" << *value
66// << ", max=" << max_idx << ")" << endl;
67
68 indices = new ShapeItem[count];
69 loop(v, count)
70 {
71 const ShapeItem idx = value->get_cravel(v).get_near_int() - qio;
72
73 // instead of testing signed < 0 and >= max, we test unsigned >= max.
74 //
75 if (idx >= max_idx) INDEX_ERROR;
76
77 indices[v] = idx * w;
78 }
79}

81MultiIndexIterator::MultiIndexIterator(const Shape & shape,
82 const IndexExpr & IDX)
83 : highest_it(0),
84 lowest_it(0),
85 empty(false)
86{
87 // IDX is parsed from right to left: Value[I2;I1;I0] --> I0 I1 I2
88 // the shapes of Value and IDX are then related as follows:
89 //
90 // Value IDX
91 // ---------------
92 // 0 rank-1 (rank = IDX->value_count())
93 // 1 rank-2
94 // ... ...
95 // rank-2 1
96 // rank-1 0
97 // ---------------
98 //
99
100 if (shape.get_rank() != IDX.get_rank())
101 {
102 Log(LOG_error_throw)
103 {
104 Q1(shape.get_rank())
105 Q1(shape)
106 Q1(IDX)
107 Q1(IDX.get_rank())
108 }
109 INDEX_ERROR;
110 }
111
112ShapeItem weight = 1;
113 loop(idx_r, shape.get_rank())
114 {
115 const sRank val_r = shape.get_rank() - idx_r - 1; // see comment above.
116 const ShapeItem sh_r = shape.get_shape_item(val_r);
117 Value_P I = IDX.values[idx_r];
118 IndexIterator * new_it;
119 if (!I) // elided index
120 {
121 new_it = new ElidedIndexIterator(weight, sh_r);
122 if (sh_r == 0) empty = true;
123 }
124 else
125 {
126 new_it = new TrueIndexIterator(weight, I, IDX.quad_io, sh_r);
127 if (I->element_count() == 0) empty = true;
128 }
129
130 Log(LOG_delete)
131 CERR << "new " << voidP(new_it) << " at " LOC << endl;
132
133 if (highest_it) highest_it->set_upper(new_it);
134 else lowest_it = new_it;
135
136 weight *= sh_r;
137 highest_it = new_it;
138 }
139}
140//----------------------------------------------------------------------------
141MultiIndexIterator::~MultiIndexIterator()
142{
143 for (IndexIterator * it = lowest_it; it;)
144 {
145 IndexIterator * del = it;
146 it = it->get_upper();
147 Log(LOG_delete)
148 CERR << "delete " HEX(uint64_t(del))
149 << " at " LOC << endl;
150 delete del;
151 }
152}
153//----------------------------------------------------------------------------
154ShapeItem
155MultiIndexIterator::operator ++(int)
156{
157 Assert(!lowest_it || more());
158
159ShapeItem ret = 0;
160 for (IndexIterator * it = lowest_it; it; it = it->get_upper())
161 ret += it->get_ivalue();
162
163 if (lowest_it) ++*lowest_it;
164 return ret;
165}

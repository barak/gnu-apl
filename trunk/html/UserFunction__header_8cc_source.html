40UserFunction_header::UserFunction_header(const UCS_string & text,
41 bool macro)
42 : error(E_DEFN_ERROR), // assume bad headr
43 error_info("Bad header"),
44 sym_Z(0),
45 sym_A(0),
46 sym_LO(0),
47 sym_FUN(0),
48 sym_RO(0),
49 sym_X(0),
50 sym_B(0)
51{
52UCS_string signature_text;
53UCS_string lvar_text;
54
55 // split header text into signature and local variables strings
56 {
57 bool in_signature = true;
58 loop(t, text.size())
59 {
60 const Unicode uni = text[t];
61 if (uni == UNI_CR) continue; // ignore CR
62 if (uni == UNI_LF) break; // stop at LF
63 if (uni == UNI_SEMICOLON) in_signature = false;
64 if (in_signature) signature_text.append(uni);
65 else lvar_text.append(uni);
66 }
67 }
68
69 if (signature_text.size() == 0)
70 {
71 error_info = "Empty header line";
72 return;
73 }
74
75 Log(LOG_UserFunction__set_line)
76 {
77 CERR << "[0] " << signature_text << lvar_text << endl;
78 // show_backtrace(__FILE__, __LINE__);
79 }
80
81 if ((error_info = init_signature(signature_text, macro))) return;
82 if ((error_info = init_local_vars(lvar_text, macro))) return;
83
84 error = E_NO_ERROR;
85}
86//-------------------------------------------------------------------------
87UserFunction_header::UserFunction_header(Fun_signature sig, int lambda_num)
88 : error(E_DEFN_ERROR),
89 error_info("Bad header"),
90 sym_Z(0),
91 sym_A(0),
92 sym_LO(0),
93 sym_FUN(0),
94 sym_RO(0),
95 sym_X(0),
96 sym_B(0)
97{
98 function_name.append(UNI_LAMBDA);
99 function_name.append_number(lambda_num);
100
101 if (!signature_is_valid(sig))
102 {
103 error_info = "Invalid signature";
104 return;
105 }
106
107 sym_Z = &Workspace::get_v_LAMBDA();
108 if (sig & SIG_A) sym_A = &Workspace::get_v_ALPHA();
109 if (sig & SIG_LO) sym_LO = &Workspace::get_v_ALPHA_U();
110 if (sig & SIG_RO) sym_RO = &Workspace::get_v_OMEGA_U();
111 if (sig & SIG_B) sym_B = &Workspace::get_v_OMEGA();
112 if (sig & SIG_X) sym_X = &Workspace::get_v_CHI();
113
114 error_info = 0;
115 error = E_NO_ERROR;
116}
117//-------------------------------------------------------------------------
118bool
119UserFunction_header::signature_is_valid(Fun_signature sig)
120{
121 // if sig is valid then (sig | SIG_Z) is also valid. We can therefore
122 // reduce the number of cases.by pretending that SIG_Z is set.
123 //
124 switch(sig | SIG_Z)
125 {
126 // niladic
127 //
128 case SIG_Z_F0:
129
130 // monadic
131 //
132 case SIG_Z_F1_B:
133 case SIG_Z_F1_X_B:
134 case SIG_Z_LO_OP1_B:
135 case SIG_Z_LO_OP1_X_B:
136 case SIG_Z_LO_OP2_RO_B:
137
138 // dyadic
139 //
140 case SIG_Z_A_F2_B:
141 case SIG_Z_A_F2_X_B:
142 case SIG_Z_A_LO_OP1_B:
143 case SIG_Z_A_LO_OP1_X_B:
144 case SIG_Z_A_LO_OP2_RO_B: return true; // valid signature
145 default: return false; // invalid signature
146 }
147}
148//--------------------------------------------------------------------------
149const char *
150UserFunction_header::init_signature(const UCS_string & text, bool macro)
151{
152Token_string tos;
153 {
154 const Tokenizer tokenizer(PM_FUNCTION, LOC, macro);
155 if (const ErrorCode err = tokenizer.tokenize(text, tos))
156 {
157 error = err;
158 return "Tokenize error (function signature)";
159 }
160 }
161
162size_t start = 0;
163size_t len = tos.size();
164
165 if (len >= 2 && tos[1].get_Class() == TC_ASSIGN) // expect Z ← ...
166 {
167 if (tos[0].get_Class() != TC_SYMBOL) return "Bad Z in Z ←";
168
169 sym_Z = tos[0].get_sym_ptr(); // Z ← or λ ←
170 start = 2;
171 len -= 2;
172 }
173
174 if (len <= 3) // F0, F1 B, or A F2 B
175 {
176 if (len == 0) // error: empty signature
177 return sym_Z ? "Empty header (after Z ←)" : "Empty header";
178
179 if (len == 1) // F0
180 {
181 if (tos[start].get_Class() != TC_SYMBOL) return "Bad F0";
182
183 sym_FUN = tos[start].get_sym_ptr();
184 function_name = sym_FUN->get_name();
185 return 0; // OK
186 }
187
188 if (len == 2) // F1 B
189 {
190 if (tos[start].get_Class() != TC_SYMBOL)
191 return "Bad F1 in F1 B";
192
193 if (tos[start + 1].get_Class() != TC_SYMBOL)
194 return "Bad B in F1 B";
195
196 sym_FUN = tos[start].get_sym_ptr();
197 sym_B = tos[start + 1].get_sym_ptr();
198 function_name = sym_FUN->get_name();
199 return 0; // OK
200 }
201
202 // otherwise: A F2 B
203 if (tos[start].get_Class() != TC_SYMBOL)
204 return "Bad A in A F2 B";
205
206 if (tos[start + 1].get_Class() != TC_SYMBOL)
207 return "Bad F2 in A F2 B";
208
209 if (tos[start + 2].get_Class() != TC_SYMBOL)
210 return "Bad B in A F2 B";
211 sym_A = tos[start] .get_sym_ptr();
212 sym_FUN = tos[start + 1].get_sym_ptr();
213 sym_B = tos[start + 2].get_sym_ptr();
214 function_name = sym_FUN->get_name();
215 return 0; // OK
216 }
217
218 // at this point the signature has 4 or more token after the
219 // optional Z← or λ←. Strip of the final B.
220 //
221 if (tos[start + len - 1].get_Class() != TC_SYMBOL)
222 return "Bad B in ... B";
223 sym_B = tos[start + len - 1].get_sym_ptr();
224 len--;
225
226 // maybe strip off the optional axis [X]
227 //
228 if (tos[start + len - 1].get_Class() == TC_R_BRACK)
229 {
230 if (tos[start + len - 2].get_Class() != TC_SYMBOL)
231 return "Bad X in ... [X] B";
232
233 if (tos[start + len - 3].get_tag() != TOK_L_BRACK)
234 return "Bad [ in ... [X] B";
235
236 sym_X = tos[start + len - 2].get_sym_ptr();
237 len -= 3;
238 }
239
240 // at this point we should have one of:
241 //
242 // 1. F1
243 // 2. A F2
244 // 3. A ( LO OP1 )
245 // 4. ( LO OP1 )
246 // 5. A ( LO OP2 RO )
247 // 6. ( LO OP2 RO )
248 //
249 if (len == 1) // case 1.
250 {
251 if (tos[start].get_tag() != TOK_SYMBOL)
252 return "Bad F1 in F1 [X] B";
253
254 sym_FUN = tos[start].get_sym_ptr();
255 function_name = sym_FUN->get_name();
256 return 0; // OK
257 }
258
259 // cases 2-6
260 //
261 if (tos[start].get_Class() == TC_SYMBOL) // case 2, 3, or 5: strip A
262 {
263 sym_A = tos[start].get_sym_ptr();
264 ++start;
265 --len;
266 }
267
268 // cases 2, 4, or 6
269 //
270 if (len == 1) // case 2.
271 {
272 if (tos[start].get_Class() != TC_SYMBOL)
273 return "Bad F2 in A F2 [X] B";
274
275 sym_FUN = tos[start].get_sym_ptr();
276 function_name = sym_FUN->get_name();
277 return 0; // OK
278 }
279
280 // cases 3-6
281 //
282 if (tos[start].get_tag() != TOK_L_PARENT)
283 return "Bad ( in (F2 OP ... )";
284
285 if (tos[start + len - 1].get_tag() != TOK_R_PARENT)
286 return "Bad ) in (F2 OPn ... )";
287
288 if (tos[start + 1].get_Class() != TC_SYMBOL)
289 return "Bad F2 in (F2 OPn ... )";
290
291 sym_LO = tos[start + 1].get_sym_ptr();
292
293 if (tos[start + 2].get_Class() != TC_SYMBOL) // LO
294 return "Bad OPn in (F2 OPn ... )";
295
296 sym_FUN = tos[start + 2].get_sym_ptr();
297 function_name = sym_FUN->get_name();
298
299 if (len == 4) return 0; // OK: ( LO OP1 )
300
301 if (len != 5) // ( LO OP2 RO )
302 return "Bad length in (F2 OPn ... )";
303
304 if (tos[start + 3].get_Class() != TC_SYMBOL)
305 return "Bad G2 in (F2 OP2 G2)";
306
307 sym_RO = tos[start + 3].get_sym_ptr();
308 return 0;
309}
310//--------------------------------------------------------------------------
311const char *
312UserFunction_header::init_local_vars(const UCS_string & text, bool macro)
313{
314Token_string tos;
315 {
316 const Tokenizer tokenizer(PM_FUNCTION, LOC, macro);
317 if (const ErrorCode err = tokenizer.tokenize(text, tos))
318 {
319 error = err;
320 return "Tokenize error (local variables)";
321 }
322 }
323
324 if (tos.size() & 1)
325 {
326 // since each local variable is preceeded by a semicolon, the
327 // number of remaining tokens must be even.
328 //
329 return "local variable list (odd length)";
330 }
331
332 loop(tos_idx, tos.size())
333 {
334 if (tos_idx & 1) // expect variable
335 {
336 const TokenTag tag = tos[tos_idx].get_tag();
337 if (tag != TOK_SYMBOL && tag != TOK_Quad_CT
338 && tag != TOK_Quad_FC
339 && tag != TOK_Quad_IO
340 && tag != TOK_Quad_PP
341 && tag != TOK_Quad_PR
342 && tag != TOK_Quad_PW
343 && tag != TOK_Quad_RL)
344 {
345 CERR << "Offending token at " LOC " is: "
346 << tos[tos_idx] << endl;
347 return "Bad token";
348 }
349
350 local_vars.push_back(tos[tos_idx].get_sym_ptr());
351 }
352 else if (tos[tos_idx].get_tag() != TOK_SEMICOL)
353 {
354 return "Semicolon expected";
355 }
356 }
357
358 // NOTE: duplicated variables are not an APL error,
359 // but we want to localize them only once.
360 //
361 remove_duplicate_local_variables();
362
363 return 0;
364}
365//----------------------------------------------------------------------------
366void
367UserFunction_header::pop_local_vars() const
368{
369 loop(l, label_values.size()) label_values[l].sym->pop();
370
371 loop(l, local_vars.size()) local_vars[l]->pop();
372
373 if (sym_B) sym_B ->pop();
374 if (sym_X) sym_X ->pop();
375 if (sym_RO) sym_RO->pop();
376 if (sym_LO) sym_LO->pop();
377 if (sym_A) sym_A ->pop();
378 if (sym_Z) sym_Z ->pop();
379}

392//----------------------------------------------------------------------------
393void
394UserFunction_header::reverse_local_vars()
395{
396const ShapeItem half = local_vars.size() / 2; // = rounded down!
397 loop(v, half)
398 {
399 Symbol * tmp = local_vars[v];
400 local_vars[v] = local_vars[local_vars.size() - v - 1];
401 local_vars[local_vars.size() - v - 1] = tmp;
402 }
403}
404//----------------------------------------------------------------------------
405void
406UserFunction_header::remove_duplicate_local_variables()
407{
408 // remove local vars that are also labels, arguments or return values.
409 // This is to avoid pushing them twice
410 //
411 remove_duplicate_local_var(sym_Z, 0);
412 remove_duplicate_local_var(sym_A, 0);
413
414 // sym_LO, sym_FUN, and sym_RO are not pushed, so they are never duplicates
415
416 remove_duplicate_local_var(sym_X, 0);
417 remove_duplicate_local_var(sym_B, 0);
418
419 loop(l, label_values.size())
420 remove_duplicate_local_var(label_values[l].sym, 0);
421
422 loop(l, local_vars.size())
423 remove_duplicate_local_var(local_vars[l], l + 1);
424}
425//----------------------------------------------------------------------------
426void
427UserFunction_header::remove_duplicate_local_var(const Symbol * sym, size_t pos)
428{
429 // remove sym from the vector of local variables. Only the local vars
430 // at pos or higher are being removed
431 //
432 if (sym == 0) return; // unused symbol
433
434 while (pos < local_vars.size())
435 {
436 if (sym == local_vars[pos])
437 {
438 local_vars[pos] = local_vars.back();
439 local_vars.pop_back();
440 continue;
441 }
442 ++pos;
443 }
444}
445//----------------------------------------------------------------------------
446UCS_string
447UserFunction_header::lambda_header(Fun_signature sig, int lambda_num)
448{
449UCS_string u;
450
451 if (sig & SIG_Z) u.append_UTF8("λ←");
452 if (sig & SIG_A) u.append_UTF8("⍺ ");
453 if (sig & SIG_LORO) u.append_UTF8("(");
454 if (sig & SIG_LO) u.append_UTF8("⍶ ");
455 u.append_UTF8("λ");
456 u.append_number(lambda_num);
457 if (sig & SIG_RO) u.append_UTF8(" ⍹ ");
458 if (sig & SIG_LORO) u.append_UTF8(")");
459 if (sig & SIG_X) u.append_UTF8("[χ]");
460 if (sig & SIG_B) u.append_UTF8(" ⍵");
461
462 return u;
463}
464//----------------------------------------------------------------------------
465bool
466UserFunction_header::localizes(const Symbol * sym) const
467{
468 if (sym == sym_Z) return true;
469 if (sym == sym_A) return true;
470 if (sym == sym_LO) return true;
471 if (sym == sym_RO) return true;
472 if (sym == sym_X) return true;
473 if (sym == sym_B) return true;
474
475 loop(l, local_vars.size()) if (sym == local_vars[l]) return true;
476 loop(l, label_values.size()) if (sym == label_values[l].sym) return true;
477
478 return false;
479}
480//----------------------------------------------------------------------------
481void
482UserFunction_header::print_properties(ostream & out, int indent) const
483{
484UCS_string ind(indent, UNI_SPACE);
485 if (is_operator()) out << "Operator " << function_name << endl;
486 else out << "Function " << function_name << endl;
487
488 if (sym_Z) out << ind << "Result: " << *sym_Z << endl;
489 if (sym_A) out << ind << "Left Argument: " << *sym_A << endl;
490 if (sym_LO) out << ind << "Left Op Arg: " << *sym_LO << endl;
491 if (sym_RO) out << ind << "Right Op Arg: " << *sym_RO << endl;
492 if (sym_B) out << ind << "Right Argument: " << *sym_B << endl;
493
494 if (local_vars.size())
495 {
496 out << ind << "Local Variables:";
497 loop(l, local_vars.size()) out << " " << *local_vars[l];
498 out << endl;
499 }
500
501 if (label_values.size())
502 {
503 out << ind << "Labels: ";
504 loop(l, label_values.size())
505 {
506 if (l) out << ",";
507 out << " " << *label_values[l].sym
508 << "=" << label_values[l].line;
509 }
510 out << endl;
511 }

516{
517 Log(LOG_UserFunction__enter_leave) CERR << "eval_common()" << endl;
518
519 // push local variables...
520 //
521 loop(l, local_vars.size()) local_vars[l]->push();
522

524 //
525 loop(l, label_values.size())
526 label_values[l].sym->push_label(label_values[l].line);

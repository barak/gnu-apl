44//----------------------------------------------------------------------------
45StateIndicator::~StateIndicator()
46{
47 // flush the FIFO. Do that before delete executable so that values
48 // copied directly from the body of the executable are not killed.
49 //
50 current_stack.clean_up();
51 fun_oper_cache.reset();
52
53 // if executable is a user defined function then pop its local vars.
54 // otherwise delete the body token
55 //
56 if (get_parse_mode() == PM_FUNCTION)
57 {
58 const UserFunction * ufun = get_executable()->get_exec_ufun();
59 if (ufun) ufun->pop_local_vars();
60 }
61 else
62 {
63 Assert1(executable);
64 delete executable;
65 executable = 0;
66 }
67}
68//----------------------------------------------------------------------------
69void
70StateIndicator::goon(Function_Line new_line, const char * loc)
71{
72const Function_PC pc = get_executable()->get_exec_ufun()->pc_for_line(new_line);
73
74 Log(LOG_StateIndicator__push_pop)
75 CERR << "Continue SI[" << level << "] at line " << new_line
76 << " pc=" << pc << " at " << loc << endl;
77
78 if (get_executable()->get_body()[pc].get_tag() == TOK_STOP_LINE) // S∆
79 {
80 // pc points to a S∆ token. We are jumping back from immediate
81 // execution, so we don't want to stop again.
82 //
83 set_PC(pc + 2);
84 }
85 else
86 {
87 set_PC(pc);
88 }
89
90 Log(LOG_prefix_parser) CERR << "GOTO [" << get_line() << "]" << endl;
91
92 current_stack.reset(LOC);
93}

101//----------------------------------------------------------------------------
102bool
103StateIndicator::uses_function(const UserFunction * ufun) const
104{
105const Executable * uexec = ufun;
106
107 for (const StateIndicator * si = this; si; si = si->get_parent())
108 {
109 // case 1: ufun is the currently executing function
110 //
111 if (uexec == si->get_executable()) return true;
112
113 // case 2: ufun is used on the prefix parser stack
114 //
115 if (si->current_stack.uses_function(ufun)) return true;
116
117 // case 3: ufun is in the body of si->get_executable(). This crashes
118 // )SAVE since the symbol is already resolved and points to the old
119 // function.
120 //
121 uexec = si->get_executable();
122 Assert(uexec);
123 const Token_string & body = uexec->get_body();
124 loop(b, body.size())
125 {
126 const Token & tok = body[b];
127 if (tok.get_tag() != TOK_SYMBOL) continue;
128
129 if (ufun->get_name().compare(tok.get_sym_ptr()->get_name())
130 == COMP_EQ) return true;
131 }
132 }
133 return false;
134}

210void
211StateIndicator::list(ostream & out, SI_mode mode) const
212{
213 if (mode & SIM_debug) // command ]SI or ]SIS
214 {
215 print(out);
216 return;
217 }
218
219 // pmode column
220 //
221 switch(get_parse_mode())
222 {
223 case PM_FUNCTION:
224 Assert(executable);
225 if (mode == SIM_SI) // )SI
226 {
227 out << executable->get_exec_ufun()->get_name_and_line(get_PC());
228 break;
229 }
230
231 if (mode & SIM_statements) // )SIS
232 {
233 if (error.get_error_code())
234 {
235 out << error.get_error_line_2() << endl
236 << error.get_error_line_3();
237 }
238 else
239 {
240 const UCS_string name_and_line =
241 executable->get_exec_ufun()->get_name_and_line(get_PC());
242 out << name_and_line
243 << " " << executable->statement_text(get_PC())
244 << endl
245 << UCS_string(name_and_line.size(), UNI_SPACE)
246 << " ^"; // ^^^
247 }
248 }
249
250 if (mode & SIM_name_list) // )SINL
251 {
252 const UCS_string name_and_line =
253 executable->get_exec_ufun()->get_name_and_line(get_PC());
254 out << name_and_line << " ";
255 executable->get_exec_ufun()->print_local_vars(out);
256 }
257 break;
258
259 case PM_STATEMENT_LIST:
260 out << "⋆";
261 if (mode & SIM_statements) // )SIS
262 {
263 if (!executable) break;
264
265 // )SIS and we have a statement
266 //
267 out << " "
268 << executable->statement_text(get_PC())
269 << endl << " ^"; // ^^^
270 }
271 break;
272
273 case PM_EXECUTE:
274 out << "⋆⋆ ";
275 if (mode & SIM_statements) // )SIS
276 {
277 if (!executable) break;
278
279 // )SIS and we have a statement
280 //
281 if (error.get_error_code())
282 out << error.get_error_line_2() << endl
283 << error.get_error_line_3();
284 else
285 out << " "
286 << executable->statement_text(get_PC());
287 }
288 break;
289 }
290
291 out << endl;
292}

312//----------------------------------------------------------------------------
313Token
314StateIndicator::jump(Value_P value)
315{
316 // perform a jump. We either remain in the current function (and then
317 // return TOK_VOID), or we (want to) jump into back into the calling
318 // function (and then return TOK_BRANCH.). The jump itself (if any)
319 // is executed in Prefix.cc.
320 //
321 if (value->get_rank() > 1) RANK_ERROR;
322
323 if (value->element_count() == 0) // →''
324 {
325 // →⍬ in immediate execution means resume (retry) suspended function
326 // →⍬ on ⍎ or defined functions means do nothing
327 //
328 if (get_parse_mode() == PM_STATEMENT_LIST)
329 return Token(TOK_BRANCH, int64_t(Function_Retry));
330
331 return Token(TOK_NOBRANCH); // stay in context
332 }
333
334const Function_Line line = value->get_line_number();
335
336const UserFunction * ufun = get_executable()->get_exec_ufun();
337
338 if (ufun) // →N in user defined function
339 {
340 set_PC(ufun->pc_for_line(line)); // →N to valid line in user function
341 return Token(TOK_VOID); // stay in context
342 }
343
344 // →N in ⍎ or ◊
345 //
346 return Token(TOK_BRANCH, int64_t(line < 0 ? Function_Line_0 : line));
347}

454int
455StateIndicator::nth_push(const Symbol * sym, int from_tos) const
456{
457 if (from_tos == 0) return 0;
458
459 // collect SI entries in reverse order...
460 //
461std::vector<const StateIndicator *> stack;
462
463 for (const StateIndicator * si = Workspace::SI_top();
464 si; si = si->get_parent())
465 {
466 stack.push_back(si);
467 }
468
469 loop(d, stack.size())
470 {
471 const StateIndicator * si = stack[stack.size() - d - 1];
472 const Executable * exec = si->get_executable();
473 Assert(exec);
474 if (!exec->pushes_sym(sym)) continue;
475 if (0 == --from_tos) return si->get_level();
476 }
477
478 FIXME; // mot reached

499void
500StateIndicator::statement_result(Token & result, bool trace)
501{
502 Log(LOG_StateIndicator__enter_leave)
503 CERR << "StateIndicator::statement_result(pmode="
504 << get_parse_mode_name() << ", result=" << result << endl;
505
506 if (trace)
507 {
508 const UserFunction * ufun = executable->get_exec_ufun();
509 if (ufun && (ufun->get_exec_properties()[0] == 0))
510 {
511 const Function_Line line =
512 executable->get_line(Function_PC(get_PC() - 1));
513 result.show_trace(COUT, ufun->get_name(), line);
514 }
515 }
516
517 fun_oper_cache.reset();
518
519// if (get_parse_mode() == PM_EXECUTE) return;
520
521 // if result is a value then print it, unless it is a committed value
522 // (i.e. TOK_APL_VALUE2)
523 //
524 if (result.get_ValueType() != TV_VAL)
525 {
526#ifdef WANT_PYTHON
527 python_result_callback(result);
528#endif
529
530 return;
531 }
532
533const TokenTag tag = result.get_tag();
534Value_P B(result.get_apl_val());
535 Assert(+B);
536
537 // print TOK_APL_VALUE and TOK_APL_VALUE1, but not TOK_APL_VALUE2
538 //
539bool print_value = tag == TOK_APL_VALUE1 || tag == TOK_APL_VALUE3;
540
541#ifdef WANT_LIBAPL
542 if (res_callback) // callback installed
543 {
544 // the callback decides whether the value shall be printed (even
545 // if it was committed)
546 //
547 print_value = res_callback(B.get(), !print_value);
548 }
549#endif
550
551#ifdef WANT_PYTHON
552 print_value = python_result_callback(result);
553#endif
554
555 if (!print_value) return;
556
557 Quad_QUOTE::done(false, LOC);
558
559const int boxing_format = Command::get_boxing_format();
560 if (boxing_format == 0) // no boxing
561 {
562 if (Quad_SYL::print_length_limit &&
563 B->element_count() >= Quad_SYL::print_length_limit)
564 {
565 // the value exceeds print_length_limit.
566 // We cut the longest dimension in half until we are below the
567 // limit
568 //
569 Shape sh(B->get_shape());
570 while (sh.get_volume() >= Quad_SYL::print_length_limit)
571 {
572 sRank longest = 0;
573 loop(r, sh.get_rank())
574 {
575 if (sh.get_shape_item(r) > sh.get_shape_item(longest))
576 longest = r;
577 }
578
579 sh.set_shape_item(longest, sh.get_shape_item(longest) / 2);
580 }
581
582 Value_P B1 = Bif_F12_TAKE::do_take(sh, *B, false);
583 B1->print(COUT);
584
585 CERR << " *** display of value was truncated (limit "
586 "⎕SYL[⎕IO + " << Quad_SYL::SYL_PRINT_LIMIT
587 << "] reached) ***" << endl;
588 }
589 else // no print length limit or small B
590 {
591 B->print(COUT);
592 }
593 }
594 else if (boxing_format < 0)
595 {
596 const PrintContext pctx = Workspace::get_PrintContext(PST_NONE);
597 Value_P B1 = Quad_CR::do_CR(-boxing_format, B.get(), pctx);
598 if (B1->get_cols() >= Workspace::get_PW()) // too large
599 B->print(COUT); // don't box
600 else
601 B1->print(COUT); // do box
602 }
603 else
604 {
605 const PrintContext pctx = Workspace::get_PrintContext(PST_NONE);
606 Value_P B1 = Quad_CR::do_CR(boxing_format, B.get(), pctx);
607 B1->print(COUT);
608 }
609}

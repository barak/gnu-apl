35Token
36Bif_REDUCE::replicate(Value_P A, Value_P B, uAxis axis)
37{
38 // turn scalar B into ,B
39 //
40Shape shape_B = B->get_shape();
41 if (shape_B.get_rank() == 0)
42 {
43 shape_B.add_shape_item(1);
44 axis = 0;
45 }
46
47 if (A->get_rank() > 1) RANK_ERROR;
48 if (axis >= shape_B.get_rank()) AXIS_ERROR;
49
50const ShapeItem len_B = shape_B.get_shape_item(axis);
51ShapeItem len_A = A->element_count();
52ShapeItem len_Z = 0;
53std::vector<ShapeItem> rep_counts;
54 rep_counts.reserve(len_B);
55 if (len_A == 1) // single a -> a a ... a (len_B times)
56 {
57 len_A = len_B;
58 APL_Integer rep_A = A->get_cfirst().get_near_int();
59 loop(a, len_A) rep_counts.push_back(rep_A);
60 if (rep_A > 0) len_Z = rep_A*len_B;
61 else if (rep_A < 0) len_Z = -rep_A*len_B;
62 }
63 else
64 {
65 ShapeItem geq_A = 0; // number of items >= 0 in A
66 loop(a, len_A)
67 {
68 APL_Integer rep_A = A->get_cravel(a).get_near_int();
69 rep_counts.push_back(rep_A);
70 if (rep_A > 0) { len_Z += rep_A; ++geq_A; }
71 else if (rep_A < 0) len_Z -= rep_A;
72 else ++geq_A;
73 }
74
75 if (len_B != 1 && geq_A != len_B) LENGTH_ERROR;
76 }
77
78Shape shape_Z(shape_B);
79 shape_Z.set_shape_item(axis, len_Z);
80
81Value_P Z(shape_Z, LOC);
82
83const Shape3 shape_B3(shape_B, axis);
84
85 loop(h, shape_B3.h())
86 {
87 ShapeItem bm = 0;
88 loop(m, rep_counts.size())
89 {
90 const ShapeItem rep = rep_counts[m];
91
92 if (rep >= 0) // copy l*rep items
93 {
94 loop(r, rep)
95 loop(l, shape_B3.l())
96 {
97 const ShapeItem src = shape_B3.hml(h, bm, l);
98 Z->next_ravel_Cell(B->get_cravel(src));
99 }
100 if (shape_B3.m() > 1) ++bm;
101 }
102 else // init l*-rep items with the fill item
103 {
104 loop(r, -rep)
105 loop(l, shape_B3.l())
106 {
107 const ShapeItem src = shape_B3.hml(h, 0, l);
108 Z->next_ravel_Proto(B->get_cravel(src));
109 }
110
111 // cB is not incremented when fill item is used.
112 }
113 }
114 }
115
116 Z->set_default(*B.get(), LOC);
117 Z->check_value(LOC);
118 return Token(TOK_APL_VALUE1, Z);
119}

121Token
122Bif_REDUCE::reduce(Token & tok_LO, Value_P B, uAxis axis)
123{
124 // if B is a scalar, then Z is B.
125 //
126 if (B->get_rank() == 0) return Token(TOK_APL_VALUE1, CLONE_P(B, LOC));
127
128 if (!tok_LO.is_function())
129 {
130 MORE_ERROR()
131 << "The left argument of operator / resp. ⌿ is not a function";
132 DOMAIN_ERROR;
133 }
134
135Function_P LO = tok_LO.get_function();
136 Assert1(LO);
137 if (!LO->has_result())
138 {
139 MORE_ERROR() << "The left argument of operator / resp. ⌿"
140 " is a function that returns no result";
141 DOMAIN_ERROR;
142 }
143
144 if (LO->get_fun_valence() != 2)
145 {
146 MORE_ERROR() << "The left argument of operator / resp. ⌿"
147 << " is a function that is not dyadic";
148 SYNTAX_ERROR;
149 }
150
151 if (axis >= B->get_rank()) AXIS_ERROR;
152
153const ShapeItem m_len = B->get_shape_item(axis);
154 if (m_len == 0) // apply the identity function
155 {
156 /*
157 Theory: A) f/B₁ B₂ ... Bₙ ←→ (f/B₁, B₂, ... Bₙ₋₁) f Bₙ
158 ------- B) f/B₁ ←→ B₁
159
160 A) B)
161 therefore: B₁ ←→ f/B₁ ←→ B₀ f B₁
162 if there is such a B₀ (i.e. with B₀ f B₁ ←→ B₁).
163 */
164 return LO->eval_identity_fun(B, axis);
165 }
166
167const Shape shape_Z = B->get_shape().without_axis(axis);
168 if (m_len == 1) return Bif_F12_RHO::do_reshape(shape_Z, *B);
169
170 // non-trivial reduce (len > 1)
171 //
172const Shape3 B3(B->get_shape(), axis);
173 if (LO->may_push_SI()) // user defined LO
174 {
175 Value_P X4(4, LOC);
176 X4->next_ravel_Int(axis + Workspace::get_IO());
177 X4->next_ravel_Int(B3.h());
178 X4->next_ravel_Int(B3.m());
179 X4->next_ravel_Int(B3.l());
180 X4->check_value(LOC);
181 return Macro::get_macro(Macro::MAC_Z__LO_REDUCE_X4_B)
182 ->eval_LXB(tok_LO, X4, B);
183 }
184
185const Shape3 Z3(B3.h(), 1, B3.l());
186 return do_reduce(shape_Z, Z3, B3.m(), LO, B, B->get_shape_item(axis));
187}

189Token
190Bif_REDUCE::reduce_n_wise(Value_P A, Token & tok_LO,
191 Value_P B, uAxis axis) const
192{
193 if (!tok_LO.is_function())
194 {
195 MORE_ERROR() << "Function argument f of monadic operator A f/ B"
196 " resp. A f⌿ B is not a function";
197 DOMAIN_ERROR;
198 }
199
200Function_P LO = tok_LO.get_function();
201 if (!LO->has_result())
202 {
203 MORE_ERROR() << "In A " << LO->get_name() << get_name() << " B: "
204 << LO->get_name() << " returns no result";
205 DOMAIN_ERROR;
206 }
207
208 if (LO->get_fun_valence() != 2)
209 {
210 MORE_ERROR() << "In A " << LO->get_name() << get_name() << " B: "
211 << LO->get_name() << " is not a dyadic function";
212 VALENCE_ERROR;
213 }
214
215 if (A->element_count() != 1) LENGTH_ERROR;
216const APL_Integer A0 = A->get_cfirst().get_int_value();
217const int n_wise = A0 < 0 ? -A0 : A0; // the number of items (= M1 in ISO)
218
219 if (B->is_scalar())
220 {
221 if (n_wise > 2) DOMAIN_ERROR;
222 if (n_wise == 0)
223 {
224 Token ident = LO->eval_identity_fun(B, axis);
225 Value_P Z(2, LOC);
226 Z->next_ravel_Cell(ident.get_apl_val()->get_cfirst());
227 Z->next_ravel_Cell(ident.get_apl_val()->get_cfirst());
228 return Token(TOK_APL_VALUE1, Z);
229 }
230
231 // n_wise is 1 or 2: return (2 - n_wise) ⍴ B
232 //
233 Shape sh;
234 if (n_wise == 2) sh.add_shape_item(0);
235 return Bif_F12_RHO::do_reshape(sh, *B);
236 }
237 else
238 {
239 if (n_wise > (1 + B->get_shape_item(axis))) DOMAIN_ERROR;
240 }
241
242 Assert1(LO);
243
244 if (B->get_rank() == 0) return Token(TOK_APL_VALUE1, CLONE_P(B, LOC));
245
246 if (axis >= B->get_rank()) AXIS_ERROR;
247
248 if (n_wise == 0) // apply the identity function
249 {
250 Shape shape_B1 = B->get_shape().insert_axis(axis, 0);
251 shape_B1.increment_shape_item(axis + 1);
252 Value_P val(shape_B1, LOC);
253 val->set_ravel_Cell(0, B->get_cproto()); // prototype
254
255 Token result = LO->eval_identity_fun(val, axis);
256 return result;
257 }
258
259 if (n_wise == (1 + B->get_shape_item(axis))) // empty result
260 {
261 // the examples in ISO/IEC 13751 (2000) p. 116, as well as lrm,
262 // suggest an empty result, while the algorithm on pp. 115/116 seems
263 // not to work for n_wise == (1 + B->get_shape_item(axis)).
264 // We follow the examples and lrm.
265 //
266 Shape shape_Z = B->get_shape();
267 shape_Z.set_shape_item(axis, 0);
268 Value_P Z(shape_Z, LOC);
269 Z->set_ravel_Cell(0, B->get_cproto()); // prototype
270 Z->check_value(LOC);
271 return Token(TOK_APL_VALUE1, Z);
272 }
273
274Shape shape_Z(B->get_shape());
275 shape_Z.set_shape_item(axis, shape_Z.get_shape_item(axis) - n_wise + 1);
276 if (shape_Z.is_empty()) return LO->eval_identity_fun(B, axis);
277
278 if (n_wise == 1) return Bif_F12_RHO::do_reshape(shape_Z, *B);
279
280const Shape3 Z3(shape_Z, axis);
281const Shape3 B3(B->get_shape(), axis);
282 if (LO->may_push_SI()) // user defined LO
283 {
284 Value_P A1 = IntScalar(A0 < 0 ? A0 + 1 : 1 - A0, LOC);
285 Value_P vsh_Z(LOC, &shape_Z);
286 Value_P vsh_Z3(LOC, &Z3);
287 Value_P vsh_B3(LOC, &B3);
288 Value_P X4(4, LOC);
289 X4->next_ravel_Int(axis + Workspace::get_IO()); // X
290 X4->next_ravel_Pointer(vsh_Z.get()); // ⍴Z
291 X4->next_ravel_Pointer(vsh_Z3.get()); // ⍴Z3
292 X4->next_ravel_Pointer(vsh_B3.get()); // ⍴B3
293 X4->check_value(LOC);
294 if (A0 < 0) return Macro::get_macro(Macro::MAC_Z__nA_LO_REDUCE_X4_B)
295 ->eval_ALXB(A1, tok_LO, X4, B);
296 else return Macro::get_macro(Macro::MAC_Z__pA_LO_REDUCE_X4_B)
297 ->eval_ALXB(A1, tok_LO, X4, B);
298 }
299
300 return do_reduce(shape_Z, Z3, A0, LO, B, B->get_shape_item(axis));
301}
302//----------------------------------------------------------------------------
303Token
304Bif_REDUCE::do_reduce(const Shape & shape_Z, const Shape3 & Z3, ShapeItem nwise,
305 Function_P LO, Value_P B, ShapeItem bm)
306{
307Value_P Z(shape_Z, LOC);
308
309 // constants that are valid for the entire reduction
310 //
311const ShapeItem len_Z = Z->element_count();
312const ShapeItem len_L = Z3.get_last_shape_item();
313const ShapeItem len_BML = bm * len_L;
314const ShapeItem len_ZM = Z3.m();
315const ShapeItem len_ZML = len_L * len_ZM;
316prim_f2 scalar_LO = LO->get_scalar_f2();
317
318 loop(z, len_Z)
319 {
320 // break down z into H, M, and L components
321 //
322 const ShapeItem z_L = z % len_L;
323 const ShapeItem z_M = (z / len_L) % len_ZM;
324 const ShapeItem z_H = z / len_ZML;
325
326 // compute LO_count (= the number of LO-reductions), and b (= the
327 // starting point of the LO-reductions). They depend on z and nwise
328 //
329 const ShapeItem LO_count = (nwise == -1) ? z_M
330 : (nwise < -1) ? - nwise - 1
331 : nwise - 1;
332
333 ShapeItem b = z_L + z_H * len_BML // start of row in B
334 + LO_count * len_L; // end of beam in B
335
336 // for reverse (i,e. nwise < -1) reduction direction go to the
337 // start of the beam instead of the end of the beam.
338 //
339 if (nwise < -1) b += (nwise + 1) * len_L;
340 if (nwise != -1) b += z_M * len_L; // start of beam
341
342 // Z[z] = B[b]. We use Z[z] as accumulator for the reduction
343 //
344 Cell & accu = Z->get_wravel(z);
345 accu.init(B->get_cravel(b), *Z, LOC);
346
347 loop(lo, LO_count)
348 {
349 if (nwise < -1) b += len_L; // move forward in B
350 else b -= len_L; // move backward in B
351
352 // one reduction step (one call of LO)
353 //
354 const Cell & cB = B->get_cravel(b);
355 if (scalar_LO && accu.is_simple_cell() && cB.is_simple_cell())
356 {
357 const ErrorCode ec = (accu.*scalar_LO)(&accu, &cB);
358 if (ec) throw_apl_error(ec, LOC);
359 }
360 else
361 {
362 Value_P LO_A = cB.to_value(LOC);
363 Value_P LO_B = accu.to_value(LOC);
364 Token result = LO->eval_AB(LO_A, LO_B);
365 accu.release(LOC);
366
367 if (result.get_tag() == TOK_ERROR)
368 throw_apl_error(result.get_ErrorCode(), LOC);
369
370 Assert(result.get_Class() == TC_VALUE);
371 Value_P ZZ = result.get_apl_val();
372 accu.init_from_value(ZZ.get(), *Z, LOC);
373 }
374 }
375 }
376
377 Z->set_default(*B.get(), LOC);
378 Z->check_value(LOC);
379 return Token(TOK_APL_VALUE1, Z);
380}

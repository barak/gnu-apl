42//----------------------------------------------------------------------------
43Token
44Bif_F12_DOMINO::eval_B(Value_P B) const
45{
46 if (B->is_scalar())
47 {
48 Value_P Z(LOC);
49
50 B->get_cscalar().bif_reciprocal(&Z->get_wscalar());
51 Z->check_value(LOC);
52 return Token(TOK_APL_VALUE1, Z);
53 }
54
55 if (B->get_rank() == 1) // inversion at the unit sphere
56 {
57 const double qct = Workspace::get_CT();
58 const ShapeItem len = B->get_shape_item(0);
59 APL_Complex r2(0.0);
60 loop(l, len)
61 {
62 const APL_Complex b = B->get_cravel(l).get_complex_value();
63 r2 += b*b;
64 }
65
66 if (r2.real() < qct && r2.real() > -qct &&
67 r2.imag() < qct && r2.imag() > -qct)
68 DOMAIN_ERROR;
69
70 Value_P Z(len, LOC);
71
72 if (r2.imag() < qct && r2.imag() > -qct) // real result
73 {
74 loop(l, len)
75 {
76 const APL_Float b = B->get_cravel(l).get_real_value();
77 Z->next_ravel_Float(b / r2.real());
78 }
79 }
80 else // complex result
81 {
82 loop(l, len)
83 {
84 const APL_Complex b = B->get_cravel(l).get_complex_value();
85 Z->next_ravel_Complex(b / r2);
86 }
87 }
88
89 Z->set_default(*B.get(), LOC);
90 Z->check_value(LOC);
91 return Token(TOK_APL_VALUE1, Z);
92 }
93
94 if (B->get_rank() > 2) RANK_ERROR;
95
96const ShapeItem rows = B->get_shape_item(0);
97const ShapeItem cols = B->get_shape_item(1);
98 if (cols > rows) LENGTH_ERROR;
99
100 // create an identity matrix I and call eval_AB(I, B).
101 //
102const Shape shape_I(rows, rows);
103Value_P I(shape_I, LOC);
104
105 loop(y, rows)
106 loop(x, rows) I->next_ravel_Float(y == x ? 1.0 : 0.0);
107
108Token result = eval_AB(I, B);
109 return result;
110}
111//----------------------------------------------------------------------------
112Token
113Bif_F12_DOMINO::eval_XB(Value_P X, Value_P B) const
114{
115 if (!X->is_scalar()) RANK_ERROR;
116const double EPS = X->get_cfirst().get_real_value();
117
118 if (B->get_rank() != 2) RANK_ERROR;
119
120 // if rank of A or B is < 2 then treat it as a
121 // 1 by n (or 1 by 1) matrix..
122 //
123const ShapeItem rows_B = B->get_rows();
124const ShapeItem cols_B = B->get_cols();
125 if (rows_B < cols_B) LENGTH_ERROR;
126 if (rows_B*cols_B == 0) LENGTH_ERROR;
127
128
129const bool need_complex = B->is_complex(true);
130Value_P Z(3, LOC);
131 QR_factorization(Z, need_complex, rows_B, cols_B, &B->get_cfirst(), EPS);
132
133 Z->set_default(*B.get(), LOC); // not needed
134 Z->check_value(LOC);
135 return Token(TOK_APL_VALUE1, Z);
136}
137//----------------------------------------------------------------------------
138Token
139Bif_F12_DOMINO::eval_AB(Value_P A, Value_P B) const
140{
141ShapeItem rows_A = 1;
142ShapeItem cols_A = 1;
143
144Shape shape_Z;
145
146 // if rank of A or B is < 2 then treat it as a
147 // 1 by n (or 1 by 1) matrix..
148 //
149ShapeItem rows_B = 1;
150ShapeItem cols_B = 1;
151 switch(B->get_rank())
152 {
153 case 0: break;
154
155 case 1: rows_B = B->get_shape_item(0);
156 break;
157
158 case 2: cols_B = B->get_shape_item(1);
159 rows_B = B->get_shape_item(0);
160 shape_Z.add_shape_item(cols_B);
161 break;
162
163 default: RANK_ERROR;
164 }
165
166 switch(A->get_rank())
167 {
168 case 0: break;
169
170 case 1: rows_A = A->get_shape_item(0);
171 break;
172
173 case 2: cols_A = A->get_shape_item(1);
174 rows_A = A->get_shape_item(0);
175 shape_Z.add_shape_item(cols_A);
176 break;
177
178 default: RANK_ERROR;
179 }
180
181 if (rows_B < cols_B) LENGTH_ERROR;
182 if (rows_A != rows_B) LENGTH_ERROR;
183
184const bool need_complex = A->is_complex(true) || B->is_complex(true);
185Value_P Z(shape_Z, LOC);
186 divide_matrix(*Z, need_complex, rows_A, cols_A, &A->get_cfirst(),
187 cols_B, &B->get_cfirst());
188
189 Z->set_default(*B.get(), LOC);
190
191 Z->check_value(LOC);
192 return Token(TOK_APL_VALUE1, Z);
193}

203{
204Shape shape_Z;
205 loop(r, A->get_rank() - 1) shape_Z.add_shape_item(A->get_shape_item(r + 1));
206 loop(r, B->get_rank() - 1) shape_Z.add_shape_item(B->get_shape_item(r + 1));

267template<>
268Value_P Bif_F12_DOMINO::invert_upper_triangle_matrix<false>(const ShapeItem M,
269 const ShapeItem N,
270 double * utm,
271 double * tmp)
272{
273 // ignore rows ≥ N; they are 0 and then cols ≥ N of the result as well.
274 // the 0s will be filled in at the end.
275
276matrix_assert(M >= N);
277Matrix<false> UTM(utm, N, N); // the matrix to be inverted
278Matrix<false> AUG(tmp, N, N); // the augmented matrix for the result
279
280 AUG.init_identity(N); // start with the identity matrix
281
282UTM.debug("UTM orig");
283AUG.debug("AUG orig");
284
285 // divide every row by its diagonal element, so that the diagonal
286 // elements of UTM become 1.0.
287 //
288 loop(y, N) // for every row y
289 {
290 const double diag_y = UTM.real(y, y);
291 if (diag_y == 0.0) DOMAIN_ERROR;
292
293 // divide off-diagonal UTM | AUG[y;] by diag_y
294 for (ShapeItem x = y + 1; x < N; ++x) UTM.real(y, x) /= diag_y;
295
296 // divide diagonal of UTM | AUG by diag_y
297 UTM.real(y, y) = 1.0;
298 AUG.real(y, y) /= diag_y;
299 }
300
301UTM.debug("UTM unity " LOC);
302AUG.debug("AUG unity " LOC);
303
304 // subtract a multiple of the diagonal element from the items above it
305 // so that the item becomes 0.0
306 //
307 loop(y, N) // for every row y
308 loop (y1, y) // for every row y1 above y
309 {
310Q1(y) Q1(y1)
311 const double factor = UTM.real(y1, y);
312 if (factor == 0) continue; // already 0.0
313Q1(factor)
314 for (ShapeItem x1 = y1; x1 < N; ++x1)
315 {
316 // make UTM[y1;x1] zero by subtracting UTM[y;x] × UTM[y;]
317 UTM.real(y1, x1) -= UTM.real(y, x1) * factor;
318 AUG.real(y1, x1) -= AUG.real(y, x1) * factor;
319 }
320UTM.debug("UTM at " LOC);
321AUG.debug("AUG at " LOC);
322 }
323
324AUG.debug("AUG at " LOC);
325
326 // create the result value
327const Shape shape_INV(N, M);
328Value_P INV(shape_INV, LOC);
329 loop(y, N) // for every row y
330 {
331 loop(x, M) // for every column y
332 {
333 const double flt = (x < N) ? AUG.real(y, x) : 0.0;
334 if (!isfinite(flt)) DOMAIN_ERROR;
335 INV->next_ravel_Float(flt);
336 }
337 }
338
339 INV->check_value(LOC);
340 return INV;
341}
342//----------------------------------------------------------------------------
344template<>
345Value_P Bif_F12_DOMINO::invert_upper_triangle_matrix<true>(ShapeItem M,
346 ShapeItem N,
347 double * utm,
348 double * tmp)
349{
350 // ignore rows ≥ N; they are 0 and then cols ≥ N of the result as well.
351 // the 0s will be filled in at the end.
352
353matrix_assert(M >= N);
354Matrix<true> UTM(utm, N, N); // the matrix to be inverted
355Matrix<true> AUG(tmp, N, N); // the augmented matrix for the result
356
357 AUG.init_identity(N); // start with the identity matrix
358
359UTM.debug("UTM orig");
360AUG.debug("AUG orig");
361
362 // divide every row by its diagonal element, so that the diagonal
363 // elements of UTM become 1.0.
364 //
365 loop(y, N) // for every row y
366 {
367 const complex<double> diag_y = UTM.get_Z(y, y);
368 if (diag_y.real() == 0.0 && diag_y.imag()) DOMAIN_ERROR;
369
370 // divide off-diagonal UTM | AUG[y;] by diag_y
371 for (ShapeItem x = y + 1; x < N; ++x)
372 UTM.div_Z(y, x, diag_y);
373
374 // divide diagonal of UTM | AUG by diag_y
375 UTM.set_Z(y, y, complex<double>(1.0, 0.0));
376 AUG.div_Z(y, y, diag_y);
377 }
378
379UTM.debug("UTM unity " LOC);
380AUG.debug("AUG unity " LOC);
381
382 // subtract a multiple of the diagonal element from the items above it
383 // so that the item becomes 0.0
384 //
385 loop(y, N) // for every row y
386 loop (y1, y) // for every row y1 above y
387 {
388Q1(y) Q1(y1)
389 const complex<double> factor = UTM.get_Z(y1, y);
390 if (factor.real() == 0 && factor.imag() == 0) continue;
391Q1(factor)
392 for (ShapeItem x1 = y1; x1 < N; ++x1)
393 {
394 // make UTM[y1;x1] zero by subtracting UTM[y;x] × UTM[y;]
395 UTM.sub_Z(y1, x1, UTM.get_Z(y, x1) * factor);
396 AUG.sub_Z(y1, x1, AUG.get_Z(y, x1) * factor);
397 }
398UTM.debug("UTM at " LOC);
399AUG.debug("AUG at " LOC);
400 }
401
402AUG.debug("AUG at " LOC);
403
404 // create the result value
405const Shape shape_INV(N, M);
406Value_P INV(shape_INV, LOC);
407 loop(y, N) // for every row y
408 {
409 loop(x, M) // for every column y
410 {
411 if (x < N) // diagonal or above
412 {
413 const double re = AUG.real(y, x);
414 const double im = AUG.imag(y, x);
415 if (!(isfinite(re) && isfinite(im))) DOMAIN_ERROR;
416 INV->next_ravel_Complex(re, im);
417 }
418 else // below diagonal
419 {
420 INV->next_ravel_Complex(0.0, 0.0);
421 }
422 }
423 }
424
425 INV->check_value(LOC);
426 return INV;
427}
428//----------------------------------------------------------------------------
429void
430Bif_F12_DOMINO::QR_factorization(Value_P Z, bool need_complex, ShapeItem rows,
431 ShapeItem cols, const Cell * cB, double EPS)
432{
433 /* We want to store all floating point variables (including complex ones)
434 in a single double[]. Before and after each variable we leave one double
435 for storing numbers 42.0, 43.0, ...51.0. These numbers are used to check
436 for overrides of the allocated space.
437
438 Complex numbers are stored as real followed by imag part.
439
440 The variables are B, Q, and R with B = Q +.× R, with Q real orthogonal
441 and R real or complex upper triangular. Since being R is computed after
442 Q, it can be used as a temporary variable in the computation of Q (i,e,
443 in function householder()).
444 */
445
446 // start with the base addresses of the variables. All variables are
447 // allocated as rows * rows so that we can freely rorate them...
448 //
449const int CPLX = need_complex ? 2 : 1; // number of doubles per variable item
450const ShapeItem len_B = rows * cols;
451const ShapeItem len = rows * rows;
452const ShapeItem base_B = 1;
453const ShapeItem base_Q = base_B + CPLX*len + 2;
454const ShapeItem base_Qi = base_Q + CPLX*len + 2;
455const ShapeItem base_R = base_Qi + CPLX*len + 2;
456const ShapeItem base_S = base_R + CPLX*len + 2;
457const ShapeItem end = base_S + CPLX*len + 2;
458
459double * data = new double[end*CPLX]; if (data == 0) WS_FULL;
460 memset(data, 0, end*sizeof(double));
461 data[base_B - 1] = 42.0; data[base_B + CPLX*len_B] = 43.0;
462 data[base_Q - 1] = 44.0; data[base_Q + CPLX*len] = 45.0;
463 data[base_Qi - 1] = 46.0; data[base_Qi + CPLX*len] = 47.0;
464 data[base_R - 1] = 48.0; data[base_R + CPLX*len] = 49.0;
465 data[base_S - 1] = 50.0; data[base_S + CPLX*len] = 51.0;
466
467 if (need_complex) // complex B
468 {
469 setup_complex_B(cB, data + base_B, len_B);
470 double * Q = householder<true>(data + base_B, rows, cols, data + base_Q,
471 data + base_Qi, data + base_R, data + base_S, EPS);
472
473 setup_complex_B(cB, data + base_B, len_B); // restore B
474 const Matrix<true> Bm(data + base_B, rows, cols);
475 Matrix<true> Qm(Q, rows, rows);
476 Qm.transpose(rows);
477 Matrix<true> Rm(data + base_R, rows, cols);
478 Rm.init_inner_product(Qm, Bm);
479 Qm.debug("final Q");
480 Rm.debug("final R");
481 }
482 else // real B
483 {
484 Assert(data[base_B + CPLX*len_B] == 43.0);
485 setup_real_B(cB, data + base_B, len_B);
486 Assert(data[base_B + CPLX*len_B] == 43.0);
487 double * Q = householder<false>(data + base_B, rows,cols, data + base_Q,
488 data + base_Qi, data + base_R, data + base_S, EPS);
489
490 setup_real_B(cB, data + base_B, len_B); // restore B
491 const Matrix<false> Bm(data + base_B, rows, cols);
492 Matrix<false> Qm(Q, rows, rows);
493 Qm.transpose(rows);
494 Matrix<false> Rm(data + base_R, rows, cols);
495 Rm.init_inner_product(Qm, Bm);
496 Assert(data[base_B + CPLX*len_B] == 43.0);
497 }
498
499 // check that the memory areas were not overridden
500 Assert(data[base_B - 1] == 42.0);
501 Assert(data[base_B + CPLX*len_B] == 43.0);
502 Assert(data[base_Q - 1] == 44.0);
503 Assert(data[base_Q + CPLX*len] == 45.0);
504 Assert(data[base_Qi - 1] == 46.0);
505 Assert(data[base_Qi + CPLX*len] == 47.0);
506 Assert(data[base_R - 1] == 48.0);
507 Assert(data[base_R + CPLX*len] == 49.0);
508 Assert(data[base_S - 1] == 50.0);
509 Assert(data[base_S + CPLX*len] == 51.0);
510
511 {
512 const Shape Q_shape(rows, rows);
513 Value_P Qv(Q_shape, LOC);
514 if (need_complex)
515 {
516 loop(q, len)
517 {
518 const double re = data[base_Q + 2*q];
519 const double im = data[base_Q + 2*q + 1];
520 if (!(isfinite(re) && isfinite(im))) DOMAIN_ERROR;
521 Qv->next_ravel_Complex(re, im);
522 }
523 }
524 else
525 {
526 loop(q, len)
527 {
528 const double re = data[base_Q + q];
529 if (!isfinite(re)) DOMAIN_ERROR;
530 Qv->next_ravel_Float(re);
531 }
532 }
533 Qv->check_value(LOC);
534 Z->next_ravel_Pointer(Qv.get());
535 }
536 {
537 const Shape shape_R(rows, cols);
538 Value_P vR(shape_R, LOC);
539 if (need_complex)
540 {
541 loop(r, rows*cols)
542 {
543 const double re = data[base_R + 2*r];
544 const double im = data[base_R + 2*r + 1];
545 if (!(isfinite(re) && isfinite(im))) DOMAIN_ERROR;
546 vR->next_ravel_Complex(re, im);
547 }
548 }
549 else
550 {
551 loop(r, rows*cols)
552 {
553 const double re = data[base_R + r];
554 if (!isfinite(re)) DOMAIN_ERROR;
555 vR->next_ravel_Float(re);
556 }
557 }
558 vR->check_value(LOC);
559 Z->next_ravel_Pointer(vR.get());
560
561 if (need_complex)
562 {
563 Value_P INV = invert_upper_triangle_matrix<true>(rows, cols,
564 data + base_R,
565 data + base_Q);
566 Z->next_ravel_Pointer(INV.get());
567 }
568 else
569 {
570 Value_P INV = invert_upper_triangle_matrix<false>(rows, cols,
571 data + base_R,
572 data + base_Q);
573 Z->next_ravel_Pointer(INV.get());
574 }
575 }
576
577 delete[] data;
578}

607//----------------------------------------------------------------------------
608template<bool cplx>
609double *
610Bif_F12_DOMINO::householder(double * pB, ShapeItem rows, ShapeItem cols,
611 double * pQ, double * pQi, double * pT, double * pS,
612 double EPS)
613{
614 // pB is the matrix to be factorized, pQ, pQi, and pT were initialized to 0
615 //
616 // the algorithm is essentially the one described in Garry Helzer's paper
617 // "THE HOUSEHOLDER ALGORITHM AND APPLICATIONS" but using complex numbers
618 // when needed.
619
620const double qct = Workspace::get_CT();
621const double qct2 = qct*qct;
622double BMAX = 0.0;
623
624Matrix<cplx> mT (pT, rows, rows); // temporary storage
625Matrix<cplx> mQ (pQ, rows, rows); // keeps size
626Matrix<cplx> mB (pB, rows, cols); // shrinks
627Matrix<cplx> mQi(pQi, rows, rows); // keeps size
628
629 // [0] Q←HSHLDR2 B;N;BMAX;S;L2;QI;COL1
630 // [1] Q←ID N←↑⍴B
631
632 // mQ was cleared, so setting the diagonal suffices
633 loop(x, rows) mQ.real(x, x) = 1.0;
634
635
636 // [2] →(0=(1↓⍴B),BMAX←⌈/∣,B)/0 ⍝ done if no or only near-0 columns
637
638 Q1(cols)
639 if (cols == 0) return pQ;
640 loop(y, rows)
641 loop(x, cols)
642 {
643 const double abs2 = mB.abs2(y, x);
644 if (BMAX < abs2) BMAX = abs2;
645 }
646 BMAX = sqrt(BMAX);
647 if (BMAX < qct)
648 {
649 Q1("B is 0")
650 return pQ; // all B[x;y] = 0
651 }
652
653 for (;;)
654 {
655
656 // [3] SPRFLCTR: S←ID ↑⍴B ◊ Debug 'B'
657
658 Matrix<cplx> mS (pS, rows, rows);
659mB.debug("[3] B");
660
661 // [4] L2←NORM2 COL1←B[;1] ◊ Debug 'COL1' ◊ Debug 'L2'
662
663 Matrix<cplx> mCOL1 (pB, rows, 1, mB.dY); // COL1←B[;1]
664mCOL1.debug("[4] COL1");
665 norm_result L; mB.col1_norm(L);
666Q1(L.norm2_real)
667Q1(L.norm2_imag)
668 const bool significant = mCOL1.significant(BMAX, EPS);
669
670 // [5] IMBED → L2=0
671 // [6] IMBED → ∼0ϵ0=(1↓COL1) CMP_TOL EPS BMAX
672
673 if (significant || (L.norm2_real + L.norm2_imag) > qct2)
674 {
675Q1("SIGNIFICANT")
676 // [7] B[1;1]←(↑B) + (L2⋆÷2)×(0≤↑B)-0>↑B
677
678#if 0
679 Matrix<cplx>::add_sub(&mB.real(0, 0), &L.norm2_real);
680#else
681 double * B11_real = &mB.real(0, 0);
682 double * B11_imag = B11_real + 1;
683 if (*B11_real < 0)
684 {
685Q1("SIGN ¯1")
686Q1(*B11_real)
687Q1(*B11_imag)
688 *B11_real -= L.norm_real;
689 *B11_imag -= L.norm_imag;
690 }
691 else
692 {
693Q1("SIGN 1")
694Q1(*B11_real)
695Q1(*B11_imag)
696 *B11_real += L.norm_real;
697 *B11_imag += L.norm_imag;
698 }
699#endif
700
701 // [8] COL1←B[;1] ◊ Debug 'COL1'
702 // [9] SCALE←2÷NORM2 COL1 ◊ Debug 'NORM2 COL1' ◊ Debug 'SCALE'·
703 // [10] S←COL1∘.×COL1×SCALE ◊ (1 1⍉S)←1 1⍉S - 1.0
704
705
706 // COL1←B[;1] changes nothing, so only S←... remains.
707 // We have moved the initialization of S to the else
708 // clause below. Therefore -S on the right does not work
709 // here, but we can simply subtract 1.0 from the diagonal.
710 //
711mCOL1.debug("[8] COL1");
712 norm_result scale; mCOL1.col1_norm(scale);
713 mS.init_outer_product(scale, mCOL1);
714Q1(scale.norm__2_real)
715Q1(scale.norm__2_imag)
716 loop(y, rows) mS.real(y, y) -= 1.0; // subtract 1.0
717 }
718 else // →IMBED but do [3] here
719 {
720 mS.init_identity(rows);
721 }
722
723 // [9] QI←ID N ◊ ((-2/↑⍴S)↑QI)←S ◊ Debug 'QI'
724
725 mQi.imbed(mS);
726mQi.debug("[11] QI");
727mS .debug("[11] S");
728
729mQ.debug("[12] Q before Q←Q+.×QI");
730 // [12] Q←Q+.×QI ◊ Debug 'Q'
731
732 // use mT as temporary buffer for mB, so that init_inner_product() works
733 mT.resize(mQ.M, mQ.N); mT = mQ;
734
735mT.debug("[12] T←Q before Q←Q+.×QI");
736 mQ.init_inner_product(mT, mQi);
737mQ.debug("[12] Q after Q←Q+.×QI");
738
739 // since we are only interested in Q we can skip the final B←1 1↓S+.×B
740 //
741 if (0 == --cols)
742 {
743 mQ.debug("[end] Q");
744 return pQ;
745 }
746
747 // [13] Debug 'B' ◊ Debug 'S' ◊ B←1 1↓S+.×B ◊ Debug 'B'
748
749 // use mT as temporary buffer for mB, so that init_inner_product() works
750 mT.resize(mB.M, mB.N); mT = mB;
751mT.debug("[13] B");
752mS.debug("[13] S");
753 mB.resize(mS.M, mT.N);
754 mB.init_inner_product(mS, mT);
755
756 pB = mB.drop_1_1(); // 1 1↓B
757mB.debug("[13] B");
758
759 // [14] →(0≠1↓⍴B)/SPRFLCTR
760
761 --rows;
762 }

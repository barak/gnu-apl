28/*
29 8 ⎕CR A←5 2⍴'eEwWaAzZ92' ◊ 8 ⎕CR B←'Halloween' ◊ A ⎕MAP B
30 */
31Token
32Quad_MAP::eval_AB(Value_P A, Value_P B) const
33{
34bool recursive = false;
35
36 /* a nested scalar A indicates recursive ⎕MAP. For example
37
38 Map←6 2⍴'aAbBcCdDeEfF' ⍝ lowercase to uppercase for a-f
39 Map ⎕MAP B ⍝ flat (top-level only) mapping of B
40 (⊂Map) ⎕Map B ⍝ recursive mapping of B
41
42 if A is a nested scalar, then setrecursive and proceed with ⊃A.
43 */
44 if (A->get_rank() == 0 && A->get_cfirst().is_pointer_cell())
45 {
46 recursive = true;
47 A = A->get_cfirst().get_pointer_value();
48 }
49
50ShapeItem map_len = A->get_rows(); // the number of mappings
51 if (A->get_rank() == 0) RANK_ERROR;
52 else if (A->get_rank() == 1)
53 {
54 // A is a vector, accepted to avoid the need for (N 2⍴A)
55 // A[1 3 5...] are the keys while A[2 4 6...] are the mapped values
56 //
57 map_len = A->element_count();
58 if (map_len & 1)
59 {
60 MORE_ERROR() << "Odd length of A in A ⎕MAP B";
61 LENGTH_ERROR;
62 }
63 map_len = map_len >> 1;
64 }
65 else if (A->get_rank() > 2) RANK_ERROR;
66
67 if (map_len == 0)
68 {
69 MORE_ERROR() << "A with length 0 in A ⎕MAP B";
70 LENGTH_ERROR;
71 }
72
73ShapeItem * indices = new ShapeItem[map_len];
74 if (indices == 0) WS_FULL;
75
76 loop(m, map_len) indices[m] = m;
77
78const ravel_comp_len ctx = { &A->get_cfirst(), 1};
79 Heapsort<ShapeItem>::sort(indices, map_len, &ctx, &Quad_MAP::greater_map);
80
81 // complain about duplicated keys
82 //
83const double qct = Workspace::get_CT();
84 for (ShapeItem m = 1; m < map_len; ++m)
85 {
86 const Cell & cm1 = A->get_cravel(2*indices[m - 1]);
87 const Cell & cm = A->get_cravel(2*indices[m ]);
88 if (cm1.equal(cm, qct))
89 {
90 const int qio = Workspace::get_IO();
91 MORE_ERROR() << "Duplicate keys (e.g. A["
92 << (qio + indices[m - 1]) << "] and A["
93 << (qio + indices[m]) << "]) in 'A ⎕MAP B'";
94 delete[] indices;
95 DOMAIN_ERROR;
96 }
97 }
98
99Value_P Z = do_map(*A, indices, B.get(), recursive);
100 delete[] indices;
101 return Token(TOK_APL_VALUE1, Z);
102}
103//----------------------------------------------------------------------------
104bool
105Quad_MAP::greater_map(const ShapeItem & a, const ShapeItem & b,
106 const void * ctx)
107{
108const ravel_comp_len * rcl = reinterpret_cast<const ravel_comp_len *>(ctx);
109const Cell * cells = rcl->ravel;
110
111 if (const Comp_result cr = cells[2*a].compare(cells[2*b])) // A[1] ≠ B[1]
112 return cr == COMP_GT;
113
114 return a > b;
115}
116//----------------------------------------------------------------------------
117int
118compare_MAP(const Cell & key, const ShapeItem & item, const void * ctx)
119{
120const ravel_comp_len * rcl = reinterpret_cast<const ravel_comp_len *>(ctx);
121const Cell * cells_A = rcl->ravel;
122
123 return key.compare(cells_A[2*item]);
124}

127Value_P
128Quad_MAP::do_map(const Value & A, const ShapeItem * sorted_indices_A,
129 const Value * B, bool recursive)
130{
131Value_P Z(B->get_shape(), LOC); // the result, ⍴Z ←→ ⍴B
132
133const ravel_comp_len ctx = { &A.get_cfirst(), // start of the ravel
134 1 // number of chars to compare
135 };
136
137const ShapeItem len_B = B->element_count();
138const ShapeItem map_len = (A.get_rank() == 1) ? A.element_count() >> 1
139 : A.get_rows();
140 if (len_B == 0) // empty value
141 {
142 const Cell & cell_B = B->get_cfirst();
143 if (const ShapeItem * map =
144 Heapsort<ShapeItem>::search<const Cell &>
145 (cell_B,
146 sorted_indices_A,
147 map_len,
148 compare_MAP,
149 &ctx))
150 {
151 const Cell & cell_A = A.get_cravel(*map*2 + 1);
152 if (cell_A.is_pointer_cell())
153 {
154 Cell & cell_Z0 = Z->get_wproto();
155 cell_Z0.init(cell_A, *Z, LOC);
156 cell_Z0.get_pointer_value()->to_type(false);
157 }
158 else if (cell_A.is_character_cell())
159 Z->set_proto_Spc();
160 else if (cell_A.is_numeric())
161 Z->set_proto_Int();
162 else
163 DOMAIN_ERROR;
164 }
165 else // not mapped, simple
166 {
167 Z->set_default(cell_B, LOC);
168 }
169 }
170
171 loop(b, len_B)
172 {
173 const Cell & cell_B = B->get_cravel(b);
174 if (const ShapeItem * map = Heapsort<ShapeItem>::search<const Cell &>
175 (cell_B, sorted_indices_A, map_len, compare_MAP, &ctx))
176 {
177 Z->next_ravel_Cell(A.get_cravel(*map*2 + 1));
178 }
179 else // cell_B shall not be mapped
180 {
181 if (recursive && cell_B.is_pointer_cell()) // nested: recursive
182 {
183 Value_P sub_B = cell_B.get_pointer_value();
184 Value_P sub_Z = do_map(A, sorted_indices_A,
185 sub_B.get(), true);
186 Z->next_ravel_Pointer(sub_Z.get());
187 }
188 else // not mapped, simple
189 {
190 Z->next_ravel_Cell(cell_B);
191 }
192 }
193 }
194
195 Z->check_value(LOC);
196 return Z;
197}

31class Unicode_source
32{
33public:
35 Unicode_source(const UCS_string & s)
36 : str(s),
37 idx(0),
38 end(s.size())
39 {}
40
42 Unicode_source(const Unicode_source & src, int32_t from, int32_t to)
43 : str(src.str),
44 idx(src.idx + from),
45 end(src.idx + from + to)
46 {
47 if (end > src.str.size()) end = src.str.size();
48 if (idx > end) idx = end;
49 }
50
52 int32_t rest() const
53 { return end - idx; }
54
56 const Unicode & operator[](int32_t i) const
57 { i += idx; Assert(uint32_t(i) < uint32_t(end)); return str[i]; }
58
60 const Unicode & get()
61 { Assert(idx < end); return str[idx++]; }
62
64 const Unicode & operator *() const
65 { Assert(idx < end); return str[idx]; }
66
68 void operator ++()
69 { Assert(idx < end); ++idx; }
70
72 void operator --()
73 { Assert(idx > 0); --idx; }
74
76 void set_rest(int32_t new_rest)
77 { Assert(new_rest <= rest()); end = idx + new_rest; }
78
80 void skip(int32_t count)
81 { idx += count; if (idx > end) idx = end; }
82
83protected:
85 const UCS_string & str;
86
88 int32_t idx;
89
91 int32_t end;
92};

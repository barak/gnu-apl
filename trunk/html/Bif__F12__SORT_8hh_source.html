44struct CollatingCacheEntry
45{
48 CollatingCacheEntry()
49 : ce_char(Invalid_Unicode),
50 ce_shape()
51 {}
52
54 CollatingCacheEntry(Unicode uni, const Shape & shape_A)
55 : ce_char(uni),
56 ce_shape(shape_A) // shape_A means not found
57 {}
58
60 const Unicode ce_char;
61
63 Shape ce_shape;
64
66 void operator =(const CollatingCacheEntry & other)
67 { new (this) CollatingCacheEntry(other); }
68
70 int compare_axis(const CollatingCacheEntry & other, sAxis axis) const
71 {
72 return ce_shape.get_shape_item(axis)
73 - other.ce_shape.get_shape_item(axis);
74 }
75
77 static int compare_chars(const Unicode & key,
78 const CollatingCacheEntry & entry,
79 const void * unused_ctx)
80 { return key - entry.ce_char; }
81};
82//----------------------------------------------------------------------------
83inline ostream &
84operator << (ostream & out, const CollatingCacheEntry & entry)
85{
86 return out << "CC-entry(" << entry.ce_char << ")";
87}

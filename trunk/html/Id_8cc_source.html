104const UTF8 *
105ID::get_name(Id id)
106{
107void * result =
108 bsearch(&id, id2ucs, sizeof(id2ucs) / sizeof(Id_name),
109 sizeof(Id_name), Id_name::compare);
110
111 Assert(result);
112Id_name * idn = static_cast<Id_name *>(result);
113 if (const UTF8 * utf = idn->utf_name) return utf;
114
115 // the name was not yet constructed. Do it now
116 //
117const char * name = "unknown ID";
118 switch(id)
119 {
120#define pp(i, _u, _v) case ID_ ## i: name = #i; break;
121#define qf(i, u, _v) case ID_Quad_ ## i: name = u; break;
122#define qv(i, u, _v) case ID_Quad_ ## i: name = u; break;
123#define sf(i, u, _v) case ID_ ## i: name = u; break;
124#define st(i, u, _v) case ID_ ## i: name = u; break;
125
126#include "Id.def"
127 }
128
129 return reinterpret_cast<const UTF8 *>(name);
130}

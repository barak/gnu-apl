34//----------------------------------------------------------------------------
35NativeFunction::NativeFunction(const UCS_string & so_name,
36 const UCS_string & apl_name)
37 : Function(ID_USER_SYMBOL, TOK_FUN2),
38 handle(0),
39 name(apl_name),
40 original_so_path(so_name),
41 so_path(so_name),
42 valid(false),
43 close_fun(0)
44{
45UCS_string t4;
46 handle = open_so_file(t4, so_path);
47 if (handle == 0)
48 {
49 MORE_ERROR() << t4;
50 return;
51 }
52
53 t4 = UCS_string("shared library ");
54 t4 << so_name << " ";
55
56 // get the function multiplexer
57 //
58void * fmux = dlsym(handle, "get_function_mux");
59 if (!fmux)
60 {
61 CERR << "shared library " << so_name << " is lacking the mandatory "
62 "function get_function_mux() !" << endl;
63 t4 << "is invalid (no get_function_mux())";
64 MORE_ERROR() = t4;
65 return;
66 }
67
68void * (*get_function_mux)(const char *) =
69 reinterpret_cast<void * (*)(const char *)>(fmux);
70
71 // get the mandatory function get_signature() which returns
72 // the function signature
73 //
74 {
75 void * get_sig = get_function_mux("get_signature");
76 if (!get_sig)
77 {
78 CERR << "shared library is lacking the mandatory "
79 "function signature() !" << endl;
80 t4 << "is invalid (no get_signature())";
81 MORE_ERROR() << t4;
82 return;
83 }
84
85
86 signature = reinterpret_cast<Fun_signature (*)()>(get_sig)();
87 }
88
89 // get the optional function close_fun(), which is called before
90 // this function disappears
91 {
92 void * cfun = get_function_mux("close_fun");
93 if (cfun) close_fun = reinterpret_cast
94 <bool (*)(Cause, const NativeFunction *)>(cfun);
95 else close_fun = 0;
96 }
97
98 // create an entry in the symbol table
99 //
100Symbol * sym = Workspace::lookup_symbol(apl_name);
101 Assert(sym);
102
103const char * why = sym->cant_be_defined();
104 if (why)
105 {
106 MORE_ERROR() << why;
107 return;
108 }
109
111 //
112#define Th const NativeFunction * th
113
114#define ev(fun, args) \
115 f_ ## fun = reinterpret_cast<Token (*) args>(get_function_mux(#fun))
116
117 ev(eval_ , ( Th));
118
119 ev(eval_B , ( Vr B, Th));
120 ev(eval_AB , (Vr A, Vr B, Th));
121 ev(eval_XB , ( Vr X, Vr B, Th));
122 ev(eval_AXB , (Vr A, Vr X, Vr B, Th));
123
124 ev(eval_LB , ( Fr LO, Vr B, Th));
125 ev(eval_ALB , (Vr A, Fr LO, Vr B, Th));
126 ev(eval_LXB , ( Fr LO, Vr X, Vr B, Th));
127 ev(eval_ALXB , (Vr A, Fr LO, Vr X, Vr B, Th));
128
129 ev(eval_LRB , ( Fr LO, Fr RO, Vr B, Th));
130 ev(eval_LRXB , ( Fr LO, Fr RO, Vr X, Vr B, Th));
131 ev(eval_ALRB , (Vr A, Fr LO, Fr RO, Vr B, Th));
132 ev(eval_ALRXB , (Vr A, Fr LO, Fr RO, Vr X, Vr B, Th));
133
134 ev(eval_fill_B , ( Vr B, Th));
135 ev(eval_fill_AB , (Vr A, Vr B, Th));
136 ev(eval_ident_Bx, (Vr B, sAxis x, Th));
137#undef ev
138
139 // compute function tag based on the signature
140 //
141 if (signature & SIG_RO) tag = TOK_OPER2;
142 else if (signature & SIG_LO) tag = TOK_OPER1;
143 else if (signature & SIG_B ) tag = TOK_FUN2;
144 else tag = TOK_FUN0;
145
146 if (is_operator()) sym->set_NC(NC_OPERATOR, this);
147 else sym->set_NC(NC_FUNCTION, this);
148
149 Workspace::more_error().clear();
150 valid = true;
151 valid_functions.push_back(this);
152}
153//----------------------------------------------------------------------------
154NativeFunction::~NativeFunction()
155{
156 Log(LOG_UserFunction__enter_leave)
157 get_CERR() << "Native function " << get_name() << " deleted." << endl;
158
159 loop(v, valid_functions.size())
160 {
161 if (valid_functions[v] == this)
162 {
163 valid_functions.erase(valid_functions.begin() + v);
164 }
165 }

167//----------------------------------------------------------------------------
168void *
169NativeFunction::try_one_file(const char * filename, UCS_string & t4)
170{
171const int t4_len = t4.size();
172 t4.append_UTF8(" file ");
173 t4.append_UTF8(filename);
174
175 if (access(filename, R_OK) != 0)
176 {
177 while (t4.size() < t4_len + 44) t4.append_UTF8(" ");
178 t4.append_UTF8(" (");
179 t4.append_UTF8(strerror(errno));
180 t4.append_UTF8(")\n");
181 return 0;
182 }
183
184void * handle = dlopen(filename, RTLD_NOW);
185 if (handle == 0)
186 {
187 const char * err = dlerror();
188 if (strrchr(err, ':')) err = 1 + strrchr(err, ':');
189
190 while (t4.size() < t4_len + 44) t4.append_UTF8(" ");
191 t4.append_UTF8(" (");
192 t4.append_UTF8(err);
193 t4.append_UTF8(" )\n");
194 }
195
196 return handle;
197}
198//----------------------------------------------------------------------------
199void *
200NativeFunction::open_so_file(UCS_string & t4, UCS_string & so_path)
201{
202 // prepare a )MORE error message containing the file names tried.
203 //
204 t4.clear();
205 t4.append_UTF8("Could not find shared library '");
206 t4.append(so_path);
207 t4.append_UTF8("'\n"
208 "The following directories and file names were tried:\n");
209
210 // if the name starts with / (or \ on Windows) or .
211 // then take it as is without changes.
212 //
213 if (so_path[0] == UNI_SLASH ||
214 so_path[0] == UNI_BACKSLASH ||
215 so_path[0] == UNI_FULLSTOP)
216 {
217 UTF8_string filename(so_path);
218 void * handle = try_one_file(filename.c_str(), t4);
219
220 if (handle == 0)
221 {
222 t4.append_UTF8(
223 "NOTE: Filename extensions are NOT automatically added "
224 "when a full path\n"
225 " (i.e. a path starting with / or .) is used.");
226 }
227 return handle;
228 }
229
230 // otherwise try Makefile__pkglibdir . /usr/lib/apl and /usr/local/lib/apl,
231 // avoiding duplicates
232 //
233UTF8_string utf_so_path(so_path);
234const char * dirs[] =
235{
236 Makefile__pkglibdir, // the normal case
237 "/usr/lib/apl",
238 "/usr/local/lib/apl",
239 ".",
240 "./native", // if make install was not performed
241 "./emacs_mode", // if make install was not performed
242};
243
244 // most likely Makefile__pkglibdir is /usr/lib/apl or /usr/local/lib/apl.
245 // don't try them twice.
246 //
247 if (!strcmp(Makefile__pkglibdir, dirs[1])) dirs[1] = 0;
248 if (!strcmp(Makefile__pkglibdir, dirs[2])) dirs[2] = 0;
249
250 loop(d, sizeof(dirs) / sizeof(*dirs))
251 {
252 if (dirs[d] == 0) continue;
253
254 UTF8_string dir_so_path(dirs[d]);
255 dir_so_path += '/';
256 dir_so_path.append_UTF8(utf_so_path);
257
258 UTF8_string dir_only(dir_so_path);
259 dir_only[strrchr(dir_only.c_str(), '/') - dir_only.c_str()] = 0;
260 if (access(dir_only.c_str(), R_OK | X_OK))
261 {
262 t4.append_UTF8(" directory ");
263 t4.append_UTF8(dir_only.c_str());
264 t4.append_UTF8("\n");
265 continue; // new directory
266 }
267
268 // try filename, then filename.so, and then filename.dylib
269 // unless the filename has an extension already.
270 //
271 const char * exts[] = { ".so", ".dylib", "" };
272
273 bool has_extension = false;
274 loop(e, sizeof(exts)/sizeof(*exts))
275 {
276 if (*exts[e]) // not "no extension"
277 {
278 const char * ext = exts[e];
279 const char * end = dir_so_path.c_str();
280 end += strlen(end) - strlen(ext);
281 if (!strcmp(exts[e], end))
282 {
283 has_extension = true;
284 break;
285 }
286 }
287 }
288
289 loop(e, sizeof(exts)/sizeof(*exts))
290 {
291 if (has_extension && *exts[e]) continue;
292
293 UTF8_string filename(dir_so_path);
294 if (exts[e]) filename.append_UTF8(UTF8_string(exts[e]));
295
296 void * handle = try_one_file(filename.c_str(), t4);
297 if (handle) // found library
298 {
299 so_path = UCS_string(filename); // update so_path
300 return handle;
301 }
302 }
303 }
304
305 return 0;
306}
307//----------------------------------------------------------------------------
308void
309NativeFunction::cleanup()
310{
311 // delete in reverse construction order
312 //
313 while(valid_functions.size())
314 {
315 NativeFunction * fun = valid_functions.back();
316 valid_functions.pop_back();
317
318 if (fun->close_fun && fun->handle)
319 {
320 const bool do_dlclose = (*fun->close_fun)(CAUSE_SHUTDOWN, fun);
321 if (do_dlclose)
322 {
323 dlclose(fun->handle);
324 fun->handle = 0;
325 }
326 }
327
328 // don't delete fun since this will be done when the symbol for the
329 // fun is being deleted
330 }
331}
332//----------------------------------------------------------------------------
333void
334NativeFunction::destroy()
335{
336 if (close_fun && handle)
337 {
338 const bool do_dlclose = (*close_fun)(CAUSE_ERASED, this);
339 if (do_dlclose)
340 {
341 dlclose(handle);
342 handle = 0;
343 close_fun = 0;
344 }
345 }
346 else if (handle)
347 {
348 dlclose(handle);
349 handle = 0;
350 }
351
352 delete this;
353}
354//----------------------------------------------------------------------------
355NativeFunction *
356NativeFunction::fix(const UCS_string & so_name,
357 const UCS_string & function_name)
358{
359 // if the function already exists then return it.
360 //
361 loop(v, valid_functions.size())
362 {
363 NativeFunction * fun = valid_functions[v];
364 if (so_name == fun->so_path)
365 {
366 // the NativeFunction object exists, but may have been
367 // )ERASEd at APL level. If so, then re-install it.
368 //
369 Symbol * sym = Workspace::lookup_symbol(fun->get_name());
370 Assert(sym);
371
372 const char * why = sym->cant_be_defined();
373 if (why)
374 {
375 MORE_ERROR() << why;
376 return 0;
377 }
378
379 if (fun->is_operator()) sym->set_NC(NC_OPERATOR, fun);
380 else sym->set_NC(NC_FUNCTION, fun);
381 return fun;
382 }
383 }
384
385NativeFunction * new_function = new NativeFunction(so_name, function_name);
386 Log(LOG_delete)
387 CERR << "new " << voidP(new_function) << " at " LOC << endl;
388
389
390 if (!new_function->valid) // something went wrong
391 {
392 Log(LOG_delete)
393 CERR << "delete " << voidP(new_function) << " at " LOC << endl;
394 delete new_function;
395 return 0;
396 }
397
398 return new_function;
399}

444//----------------------------------------------------------------------------
445UCS_string
446NativeFunction::load_emacs_library(const char * emacs_arg)
447{
448UCS_string so_path("libemacs");
449UCS_string t4;
450
451void * handle = open_so_file(t4, so_path);
452 if (handle == 0) return t4;
453
454 t4 = UCS_string("found emacs library ");
455 t4.append(so_path);
456
457void * emacs_start = dlsym(handle, "emacs_start");
458 if (emacs_start == 0)
459 {
460 t4.append_UTF8(", but it\n it is lacking the mandatory "
461 "function emacs_start()\n");
462 return t4;
463 }
464
465UTF8_string so_path_utf(so_path);
466const int error =
467 reinterpret_cast<int (*)(const char *, const char *)>(emacs_start)
468 (emacs_arg, charP(so_path_utf.c_str()));
469 if (error)
470 {
471 t4.append_UTF8(", but emacs_start() returned error ");
472 t4.append_number(error);
473 t4.append_UTF8("\n");
474 return t4;
475 }
476
477 t4.clear(); // success
478 return t4;
479}

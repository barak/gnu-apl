35/*
36 general comment: for operations between an IntCell and a "higher" numeric
37 Cell type (i.e. FloatCell or ComplexCell), we try to call the corresponding
38 function (or its opposite) of the higher type.
39
40 For example:
41 INT ≠ FLOAT --> FLOAT ≠ INT
42 INT ≤ FLOAT --> FLOAT ≥ INT (≤ is the opposite of ≥)
43
44 */
45//----------------------------------------------------------------------------
46CellType
47IntCell::get_cell_subtype() const
48{
49 if (value.ival < 0) // negative integer (only fits in signed containers)
50 {
51 if (-value.ival <= 0x80)
52 return CellType(CT_INT | CTS_S8 | CTS_S16 | CTS_S32 | CTS_S64);
53
54 if (-value.ival <= 0x8000)
55 return CellType(CT_INT | CTS_S16 | CTS_S32 | CTS_S64);
56
57 if (-value.ival <= 0x80000000)
58 return CellType(CT_INT | CTS_S32 | CTS_S64);
59
60 return CellType(CT_INT | CTS_S64);
61 }
62
63 // positive integer
64 //
65 if (value.ival == 0) // 0: bit (fits in all containers)
66 return CellType(CT_INT | CTS_BIT | CTS_X8 | CTS_S8 | CTS_U8 |
67 CTS_X16 | CTS_S16 | CTS_U16 |
68 CTS_X32 | CTS_S32 | CTS_U32 |
69 CTS_X64 | CTS_S64 | CTS_U64);
70
71 if (value.ival == 1) // 1: bit (fits in all containers)
72 return CellType(CT_INT | CTS_BIT | CTS_X8 | CTS_S8 | CTS_U8 |
73 CTS_X16 | CTS_S16 | CTS_U16 |
74 CTS_X32 | CTS_S32 | CTS_U32 |
75 CTS_X64 | CTS_S64 | CTS_U64);
76
77 if (value.ival <= 0x7F)
78 return CellType(CT_INT | CTS_X8 | CTS_S8 | CTS_U8 |
79 CTS_X16 | CTS_S16 | CTS_U16 |
80 CTS_X32 | CTS_S32 | CTS_U32 |
81 CTS_X64 | CTS_S64 | CTS_U64);
82
83 if (value.ival <= 0xFF)
84 return CellType(CT_INT | CTS_U8 |
85 CTS_X16 | CTS_S16 | CTS_U16 |
86 CTS_X32 | CTS_S32 | CTS_U32 |
87 CTS_X64 | CTS_S64 | CTS_U64);
88
89 if (value.ival <= 0x7FFF)
90 return CellType(CT_INT | CTS_X16 | CTS_S16 | CTS_U16 |
91 CTS_X32 | CTS_S32 | CTS_U32 |
92 CTS_X64 | CTS_S64 | CTS_U64);
93
94 if (value.ival <= 0xFFFF)
95 return CellType(CT_INT | CTS_U16 |
96 CTS_X32 | CTS_S32 | CTS_U32 |
97 CTS_X64 | CTS_S64 | CTS_U64);
98
99 if (value.ival <= 0x7FFFFFFF)
100 return CellType(CT_INT | CTS_X32 | CTS_S32 | CTS_U32 |
101 CTS_X64 | CTS_S64 | CTS_U64);
102
103 if (value.ival <= 0xFFFFFFFF)
104 return CellType(CT_INT | CTS_U32 |
105 CTS_X64 | CTS_S64 | CTS_U64);
106
107 if (value.ival <= 0x7FFFFFFFFFFFFFFFLL) // note: this is always the case
108 return CellType(CT_INT | CTS_X64 | CTS_S64 | CTS_U64);
109
110 return CellType(CT_INT | CTS_U64);
111}
112//----------------------------------------------------------------------------
113bool
114IntCell::equal(const Cell & other, double qct) const
115{
116 if (other.is_integer_cell()) return value.ival == other.get_int_value();
117 if (other.is_numeric()) return other.equal(*this, qct);
118 return false;
119}

329ErrorCode
330IntCell::bif_factorial(Cell * Z) const
331{
332const APL_Integer iv = value.ival;
333
334 // N! fits into:
335 // 32-bit signed integer for N <= 12
336 // 64-bit signed integer for N <= 20
337 // 64-bit float for N <= 170
338 //
339 if (iv < 0) return E_DOMAIN_ERROR;
340
341 if (iv < int_fact_max) // integer result that fits into a 64-bit int
342 {
343 return IntCell::zI(Z, int_factorials[iv]);
344 }
345
346 if (iv < float_fact_max) // float result that fits into a 64 bit double
347 {
348 return FloatCell::zF(Z, float_factorials[iv]);
349 }
350
351 return E_DOMAIN_ERROR;
352}

360ErrorCode
361IntCell::bif_negative(Cell * Z) const
362{
363 if (uint64_t(value.ival) == 0x8000000000000000LL) // integer overflow
364 return IntCell::zI(Z, - value.ival);
365 return IntCell::zI(Z, - value.ival);
366}
367//----------------------------------------------------------------------------
368ErrorCode
369IntCell::bif_direction(Cell * Z) const
370{
371APL_Integer result = 0;
372 if (value.ival > 0) result = 1;
373 else if (value.ival < 0) result = -1;
374 return IntCell::zI(Z, result);
375}
376//----------------------------------------------------------------------------
377ErrorCode
378IntCell::bif_magnitude(Cell * Z) const
379{
380 if (value.ival >= 0) IntCell::zI(Z, value.ival);
381 else bif_negative(Z);
382 return E_NO_ERROR;
383}
384//----------------------------------------------------------------------------
385ErrorCode
386IntCell::bif_reciprocal(Cell * Z) const
387{
388 switch(value.ival)
389 {
390 case 0: return E_DOMAIN_ERROR;
391 case 1: return IntCell::z1(Z);
392 case -1: return IntCell::z_1(Z);
393 }
394
395#ifdef RATIONAL_NUMBERS_WANTED
396 if (value.ival < 0) // make denominator positive
397 return FloatCell::zR(Z, -1, -value.ival); // 1 ÷ X = ¯1 ÷ -X
398 else
399 return FloatCell::zR(Z, 1, value.ival); // 1 ÷ X
400#endif
401
402 return FloatCell::zF(Z, 1.0/value.ival);
403}
404//----------------------------------------------------------------------------
405ErrorCode
406IntCell::bif_roll(Cell * Z) const
407{
408const APL_Integer qio = Workspace::get_IO();
409const APL_Integer set_size = value.ival;
410 if (set_size <= 0) return E_DOMAIN_ERROR;
411
412const uint64_t rnd = Workspace::get_RL(set_size);
413 return IntCell::zI(Z, qio + (rnd % set_size));
414}

461//----------------------------------------------------------------------------
462// dyadic built-in functions...
463//
464// where possible a function with non-int A is delegated to the corresponding
465// member function of A. For numeric cells that is the FloatCell or ComplexCell
466// function and otherwise the default function (that returns E_DOMAIN_ERROR.
467//
468//----------------------------------------------------------------------------
469ErrorCode
470IntCell::bif_add(Cell * Z, const Cell * A) const
471{
472 if (A->is_integer_cell()) // integer A + integer B
473 {
474 // both cells are integers.
475 //
476 const APL_Integer a = A->get_int_value();
477 const APL_Integer b = get_int_value();
478 const APL_Integer z = a + b;
479 if (sum_overflow(z, a, b))
480 return FloatCell::zF(Z, APL_Float(a) + APL_Float(b));
481 else
482 return IntCell::zI(Z,z);
483 }
484
485 // delegate to A
486 //
487 return A->bif_add(Z, this);
488}
489//----------------------------------------------------------------------------
490ErrorCode
491IntCell::bif_add_inverse(Cell * Z, const Cell * A) const
492{
493 return A->bif_subtract(Z, this);
494}
495//----------------------------------------------------------------------------
496ErrorCode
497IntCell::bif_subtract(Cell * Z, const Cell * A) const
498{
499 if (A->is_integer_cell()) // integer A - integer B
500 {
501 // both cells are integers.
502 //
503 const APL_Integer a = A->get_int_value();
504 const APL_Integer b = get_int_value();
505 const APL_Integer z = a - b;
506
507 if (diff_overflow(z, a, b))
508 return FloatCell::zF(Z, APL_Float(a) + APL_Float(b));
509 else
510 return IntCell::zI(Z,z);
511 }
512
513 // delegate to A
514 //
515 this->bif_negative(Z); // Z = -B
516 return A->bif_add(Z, Z); // Z = A + Z = A + -B
517}
518//----------------------------------------------------------------------------
519ErrorCode
520IntCell::bif_divide(Cell * Z, const Cell * A) const
521{
522 if (!A->is_numeric()) return E_DOMAIN_ERROR;
523
524APL_Integer b = get_int_value();
525const bool A_zero = A->is_near_zero();
526 if (b == 0) // division by 0
527 {
528 if (A_zero) return IntCell::z1(Z); // 0÷0 is 1 in APL
529 return E_DOMAIN_ERROR;
530 }
531
532 // at this point B is non-zero
533 //
534 if (A->is_integer_cell())
535 {
536 // both cells are integers.
537 //
538 APL_Integer a = A->get_int_value();
539
540#ifdef RATIONAL_NUMBERS_WANTED
541 if (b < 0) // make denominator positive
542 {
543 a = -a;
544 b = -b;
545 }
546 const APL_Integer gcd = FloatCell::gcd(b, a);
547 if (b == gcd) return IntCell::zI(Z, a/gcd);
548 return FloatCell::zR(Z, a/gcd, b/gcd);
549#else
550 const APL_Float i_quot = a / b;
551 const APL_Float r_quot = a / APL_Float(b);
552
553 if (a != i_quot * b) return FloatCell::zF(Z, r_quot);
554 else return FloatCell::zF(Z, i_quot);
555#endif
556 }
557
558 // delegate to A
559 //
560const ErrorCode ec = this->bif_reciprocal(Z); // Z = ÷B
561 return ec ? ec : A->bif_multiply(Z, Z); // Z = A × Z = A × ÷B
562}
563//----------------------------------------------------------------------------
564ErrorCode
565IntCell::bif_multiply(Cell * Z, const Cell * A) const
566{
567 if (A->is_integer_cell())
568 {
569 // both cells are integers.
570 //
571 const APL_Integer a = A->get_int_value();
572 const APL_Integer b = get_int_value();
573 const APL_Float prod = APL_Float(a) * APL_Float(b);
574
575 if (prod > LARGE_INT ||
576 prod < SMALL_INT) FloatCell::zF(Z, prod);
577 else IntCell::zI(Z, a * b);
578 return E_NO_ERROR;
579 }
580
581 // delegate to A
582 //
583 return A->bif_multiply(Z, this);
584}
585//----------------------------------------------------------------------------
586ErrorCode
587IntCell::bif_multiply_inverse(Cell * Z, const Cell * A) const
588{
589 return A->bif_divide(Z, this);
590}
591//----------------------------------------------------------------------------
592ErrorCode
593IntCell::bif_power(Cell * Z, const Cell * A) const
594{
595 // some A to the integer B-th power
596 //
597 if (!A->is_numeric()) return E_DOMAIN_ERROR;
598
599APL_Integer b = get_int_value();
600const bool invert_Z = b < 0;
601 if (invert_Z) b = - b;
602
603 // at this point, b ≥ 0
604 //
605 if (b <= 1) // special cases: A⋆1, A⋆0, and A⋆¯1
606 {
607 if (b == 0) return IntCell::z1(Z); // A⋆0 is 1
608
609 if (invert_Z) return A->bif_reciprocal(Z);
610 if (A->is_real_cell()) return A->bif_conjugate(Z);
611 return ComplexCell::zC(Z, A->get_complex_value());
612 }
613
614 if (A->is_integer_cell())
615 {
616 APL_Integer a = A->get_int_value();
617 if (a == 0)
618 {
619 if (invert_Z) return E_DOMAIN_ERROR;
620
621 return IntCell::z0(Z); // 0^b = 0
622 }
623 if (a == 1) return IntCell::z1(Z); // 1^b = 1
624 if (a == -1) // -1^b = 1 or -1
625 {
626 if (b & 1) return IntCell::z_1(Z); // -1^(2N+2) = -1
627 else return IntCell::z1(Z); // -1^(2N+1) = 1
628 }
629
630 // try to compute A⋆B as integers
631 //
632 const bool negate_Z = (a < 0) && (b & 1);
633 if (a < 0) a = -a;
634
635 // z is an int small enough for int_64
636 // a⋆b = a⋆(bn + bn-1 + ... + b0) with bj = 0 or 2⋆j
637 // = a⋆bn × a⋆bn-1 × ... × a⋆b0 with a⋆bj = 1 or a⋆2⋆j
638 //
639 bool overflow = false;
640 APL_Integer a_2_n = a; // a^(2^n) = a^(2^(n-1)) × a^(2^(n-1))
641 APL_Integer zi = 1;
642 for (int b1 = b; b1; b1 >>= 1)
643 {
644 if (b1 & 1)
645 {
646 if (uint64_t(zi) >= 0x7FFFFFFFFFFFFFFFULL/a_2_n)
647 {
648 overflow = true;
649 break;
650 }
651 zi *= a_2_n;
652 if (b1 == 1) break;
653 }
654
655 if (a_2_n >= 100000000LL)
656 {
657 overflow = true;
658 break;
659 }
660
661 a_2_n *= a_2_n;
662 }
663
664 if (!overflow)
665 {
666 if (negate_Z) zi = - zi;
667 if (!invert_Z) return IntCell::zI(Z, zi);
668
669 if (zi != 0) return FloatCell::zF(Z, 1.0 / zi);
670 return E_DOMAIN_ERROR;
671 }
672
673 // A and B are integers, but Z is too big for integers
674 //
675 const APL_Float af(a);
676 const APL_Float bf(b);
677 APL_Float z = pow(af, bf);
678
679 if (negate_Z) z = - z;
680 if (invert_Z) z = 1.0 / z;
681 return FloatCell::zF(Z, z);
682 }
683
684 if (A->is_float_cell())
685 {
686 APL_Float af = A->get_real_value();
687 const APL_Float bf(b);
688 const bool negate_Z = (af < 0.0) && (b & 1);
689 if (af < 0.0) af = -af;
690
691 APL_Float z = pow(af, bf);
692 if (!isfinite(z)) return E_DOMAIN_ERROR;
693
694 if (negate_Z) z = - z;
695 if (invert_Z)
696 {
697 if (z == 0.0) return E_DOMAIN_ERROR;
698 z = 1.0 / z;
699 }
700 return FloatCell::zF(Z, z);
701 }
702
703 // complex A to the B-th power
704 //
705 {
706 APL_Complex ca = A->get_complex_value();
707 APL_Complex cb(APL_Float(b), 0.0);
708
709 APL_Complex z = complex_power(ca, cb);
710 if (!isfinite(z.real())) return E_DOMAIN_ERROR;
711 if (!isfinite(z.imag())) return E_DOMAIN_ERROR;
712
713 const APL_Float denom = ComplexCell::mag2(z);
714 if (!invert_Z) return ComplexCell::zC(Z, z);
715
716 if (denom == 0.0) return E_DOMAIN_ERROR;
717 return ComplexCell::zC(Z, z.real() / denom, - z.imag() / denom);
718 }
719}
720//----------------------------------------------------------------------------
721ErrorCode
722IntCell::bif_residue(Cell * Z, const Cell * A) const
723{
724 if (!A->is_numeric()) return E_DOMAIN_ERROR;
725
726 if (A->get_imag_value() != 0.0) // complex A
727 {
728 ComplexCell B(value.ival, 0);
729 return B.bif_residue(Z, A);
730 }
731
732 // if A is zero , return B
733 //
734const APL_Float a = A->get_real_value();
735 if (a == 0.0) return IntCell::zI(Z, value.ival);
736 if (a > (BIG_INT64_F - 1E10) || a < (1E10 - BIG_INT64_F))
737 {
738 // for large a compute a⍝b as for floats
739 //
740 const FloatCell cB(value.ival);
741 return cB.bif_residue(Z, A);
742 }
743
744
745 // IBM: if B is zero , return B
746 //
747 if (value.ival == 0) return IntCell::z0(Z);
748
749 // if ⎕CT != 0 and B ÷ A is close to an integer within ⎕CT then return 0.
750 //
751const APL_Float quot = value.ival / a;
752const APL_Float qf = floor(quot);
753const double qct = Workspace::get_CT();
754 if (qct != 0)
755 {
756 const APL_Float qc = ceil(quot);
757 if (quot > qc - qct) return IntCell::z0(Z);
758 if (quot < qf + qct) return IntCell::z0(Z);
759 }
760
761 // Otherwise return B mod A
762 //
763 // ISO: R←B-(×B)×|A×⌊|B÷A and return R if (×A)=×B or R+A otherwise
764 // IBM: if A > 0 then 0 ≤ Z < A
765 // else 0 ≥ Z > A
766 //
767 if (A->is_integer_cell())
768 {
769 APL_Integer rest = value.ival % A->get_int_value();
770 if (A->get_int_value() < 0) // A negative -> Z negative
771 {
772 if (rest > 0) rest += A->get_int_value(); // += since A < 0
773 }
774 else // A positive -> Z positive
775 {
776 if (rest < 0) rest += A->get_int_value();
777 }
778 return IntCell::zI(Z, rest);
779 }
780
781APL_Float z = value.ival - a * qf;
782 if (Cell::is_near_zero(z)) return IntCell::z0(Z);
783
784 // make ×Z == ×A
785 //
786 if (a < 0.0 && z > 0.0) z = z + a; // += since a < 0
787 else if (a > 0.0 && z < 0.0) z = z + a;
788
789 return FloatCell::zF(Z, z);
790}

862//----------------------------------------------------------------------------
863bool
864IntCell::greater(const Cell & other) const
865{
866const APL_Integer this_val = get_int_value();
867
868 switch(other.get_cell_type())
869 {
870 case CT_INT:
871 {
872 const APL_Integer other_val = other.get_int_value();
873 if (this_val == other_val) return this > &other;
874 return this_val > other_val;
875 }
876
877 case CT_FLOAT:
878 {
879 const APL_Float other_val = other.get_real_value();
880 if (this_val == other_val) return this > &other;
881 return this_val > other_val;
882 }
883
884 case CT_COMPLEX: break;
885 case CT_CHAR: return true; // int is always greater than char
886 case CT_POINTER: return false; // int is always smaller than nested
887 case CT_CELLREF: DOMAIN_ERROR;
888 default: Assert(0 && "Bad celltype");
889 }
890
891const Comp_result comp = compare(other);
892 if (comp == COMP_EQ) return this > &other;
893 return (comp == COMP_GT);
894}

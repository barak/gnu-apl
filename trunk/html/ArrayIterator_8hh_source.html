30class AxisIterator
31{
32public:
34 AxisIterator()
35 {}
36
38 AxisIterator(ShapeItem len, ShapeItem weight, bool _wrap)
39 : axis_length(len),
40 axis_weight(weight),
41 shape_offset(0),
42 ravel_offset(0),
43 wrap(_wrap)
44 {}
45
51 bool next()
52 {
53 ravel_offset += axis_weight;
54
55 // NOTE: increment of scalar shape_offset needed for more() below.
56 if (++shape_offset < axis_length || is_scalar_iterator()) return false;
57
58 // carry has occurred.
59 if (wrap) shape_offset = ravel_offset = 0;
60 return true; // carry: increment next (if any) iterator
61 }
62
64 bool more() const
65 { return is_scalar_iterator() ? shape_offset == 0 // before next()
66 : shape_offset < axis_length; }
67
69 ShapeItem get_shape_offset() const
70 { return shape_offset; }
71
73 ShapeItem get_ravel_offset() const
74 { return ravel_offset; }
75
77 ShapeItem get_weight() const
78 { return axis_weight; }
79
81 ShapeItem get_total_weight() const
82 { return axis_weight * axis_length; }
83
85 bool is_scalar_iterator() const
86 { return axis_weight == 0; }
87
88protected:
90 ShapeItem axis_length;
91
94 ShapeItem axis_weight;
95
97 ShapeItem shape_offset;
98
100 ShapeItem ravel_offset;
101
106 bool wrap;
107};

110class ArrayIterator
111{
112public:
114 ArrayIterator(const Shape & shape)
115 : rank(shape.get_rank()),
116 total_ravel_offset(0)
117 {
118 if (rank == 0) // scalar
119 {
120 new (&get_iterator(0)) AxisIterator(1, 0, false);
121 return;
122 }
123
124 ShapeItem weight = 1;
125 for (sAxis a = rank - 1; a >= 0; --a)
126 {
127 // axis_iterators[0] is the one with the highest weight, and
128 // axis_iterators[rank - 1 the one with the lowest weight (1).
129 //
130 const ShapeItem len = shape.get_shape_item(a);
131 new (&get_iterator(a)) AxisIterator(len, weight, a > 0);
132 weight *= len;
133 }
134 }
135
137 ArrayIterator(const Shape & shape, const Shape & perm)
138 : rank(shape.get_rank()),
139 total_ravel_offset(0)
140 {
141 if (rank == 0) // scalar
142 {
143 new (&get_iterator(0)) AxisIterator(1, 0, false);
144 return;
145 }
146
147 ShapeItem weight = 1;
148 for (sAxis a = rank - 1; a >= 0; --a)
149 {
150 const sAxis perm_a = perm.get_shape_item(a);
151
152 // axis_iterators[0] is the one with the highest weight, and
153 // axis_iterators[rank - 1 the one with the lowest weight (1).
154 //
155 const ShapeItem len = shape.get_shape_item(perm_a);
156 new (&get_iterator(perm_a)) AxisIterator(len, weight, perm_a);
157 weight *= len;
158 }
159 }
160
161 // Note: The axis_iterators[a] for a > 0 wrap at the end and therefore
162 // axis_iterators[a].more() is always true for a > 0.
163 // As a consequence we only need to check axis_iterators[0].more() to see
164 // if the entire iteration is done.
165 //
167 bool more() const
168 {
169 return get_iterator(0).more();
170 }
171
173 void operator++()
174 {
175 if (rank == 0) // scalar
176 {
177 get_iterator(0).next();
178 return;
179 }
180
181 for (sAxis a = rank - 1; a >= 0; --a)
182 {
183 AxisIterator & iterator = get_iterator(a);
184 total_ravel_offset += iterator.get_weight();
185 if (!iterator.next()) break; // if no carry
186 total_ravel_offset -= iterator.get_total_weight();
187 }
188 }
189
191 ShapeItem get_shape_offset(sAxis a) const
192 { return get_iterator(a).get_shape_offset(); }
193
195 ShapeItem get_ravel_offset() const
196 { return total_ravel_offset; }
197
199 Shape get_shape_offsets()
200 {
201 Shape ret;
202 loop(r, rank)
203 ret.add_shape_item(get_iterator(r).get_shape_offset());
204 return ret;
205 }
206
207protected:
209 const AxisIterator & get_iterator(sAxis a) const
210 { return axis_iterators[a]; }
211
213 AxisIterator & get_iterator(sAxis a)
214 { return axis_iterators[a]; }
215
217 const uRank rank;
218
220 ShapeItem total_ravel_offset;
221
227 AxisIterator axis_iterators[MAX_RANK];
228};

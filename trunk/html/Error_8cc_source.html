31Error::Error(ErrorCode ec, const char * loc)
32 : error_code(ec),
33 throw_loc(loc),
34 parser_loc(0),
35 show_locked(false),
36 left_caret(-1),
37 right_caret(-1),
38 print_loc(0)
39{
40const char more = Workspace::more_error().size() ? '+' : 0;
41 snprintf(error_message_1, sizeof(error_message_1), "%s%c",
42 error_name(error_code), more);
43
44 *symbol_name = 0;
45 *error_message_2 = 0;
46}
47//----------------------------------------------------------------------------
48void
49Error::print(ostream & out, const char * loc) const
50{
51 if (print_loc)
52 {
53 CERR << endl << "*** Error printed twice; first printed at "
54 << print_loc << endl
55 << "now printed at " << loc << endl;
56
57 return;
58 }
59
60 Log(LOG_verbose_error)
61 {
62 out << endl
63 << "--------------------------" << endl;
64
65 if (error_code == E_NO_ERROR)
66 {
67 out << error_name(E_NO_ERROR) << endl
68 << "--------------------------" << endl;
69
70 return;
71 }
72
73 if (*error_message_1)
74 {
75 out << error_message_1 << endl;
76 }
77 else
78 {
79 out << error_name(error_code);
80 if (Workspace::more_error().size()) out << UNI_PLUS;
81 out << endl;
82 }
83
84 if (parser_loc) out << " Parser LOC: " << parser_loc << endl;
85 if (print_loc) out << " Print LOC: " << print_loc << endl;
86 out << " loc: " << loc << endl;
87 loc = print_loc;
88
89 if (*symbol_name)
90 out << " Symbol: " << symbol_name << endl;
91
92 out << " Thrown at: " << throw_loc << endl
93 << "--------------------------" << endl
94 << endl;
95 }
96}

143//----------------------------------------------------------------------------
144UCS_string
145Error::get_error_line_3() const
146{
147 if (error_code == E_NO_ERROR) return UCS_string();
148
149 if (left_caret < 0) return UCS_string();
150
151UCS_string ret;
152 if (left_caret > 0) ret.append(UCS_string(left_caret, UNI_SPACE));
153 ret.append(UNI_CIRCUMFLEX);
154
155const int diff = right_caret - left_caret;
156 if (diff <= 0) return ret;
157 if (diff > 1) ret.append(UCS_string(diff - 1, UNI_SPACE));
158 ret.append(UNI_CIRCUMFLEX);
159
160 return ret;
161}

181//----------------------------------------------------------------------------
182void
183throw_apl_error(ErrorCode code, const char * loc)
184{
185 ADD_EVENT(0, VHE_Error, code, loc);
186
187StateIndicator * si = Workspace::SI_top();
188
189 Log(LOG_error_throw)
190 {
191 CERR << endl
192 << "throwing " << Error::error_name(code)
193 << " at " << loc << endl;
194 }
195
196 Log(LOG_verbose_error)
197 {
198 if (!(si && si->get_safe_execution())) BACKTRACE
199 }
200
201 // maybe map error to DOMAIN ERROR.
202 //
203 if (si)
204 {
205 const UserFunction * ufun = si->get_executable()->get_exec_ufun();
206 if (ufun && ufun->get_exec_properties()[3]) code = E_DOMAIN_ERROR;
207 }
208
209Error error(code, loc);
210 if (si) error.update_error_info(si);
211
212Error & eref = error;
213 throw eref;
214}

237void
238Error::throw_symbol_error(const UCS_string & sym_name, const char * loc)
239{
240 Log(LOG_error_throw)
241 {
242 CERR << "throwing VALUE ERROR at " << loc;
243 if (sym_name.size()) CERR << " (symbol is: '" << sym_name << "')";
244 CERR << endl;
245 }
246
247 Log(LOG_verbose_error) BACKTRACE
248
249Error err(E_VALUE_ERROR, loc);
250UTF8_string sym_name_utf(sym_name);
251 snprintf(err.symbol_name, sizeof(err.symbol_name), "%s",
252 sym_name_utf.c_str());
253 if (StateIndicator * si = Workspace::SI_top()) err.update_error_info(si);
254
255Error & eref = err;
256 throw eref;
257}
258//----------------------------------------------------------------------------
259void
260Error::throw_define_error(const UCS_string & fun_name, const UCS_string & cmd,
261 const char * loc)
262{
263 Log(LOG_error_throw)
264 {
265 CERR << "throwing DEFN ERROR at " << loc
266 << " (function is " << fun_name << ")" << endl;
267 }
268
269 Log(LOG_verbose_error) BACKTRACE
270
271Error err(E_DEFN_ERROR, loc);
272Error & eref = err;
273UTF8_string fun_name_utf(fun_name);
274 snprintf(err.symbol_name, sizeof(err.symbol_name), "%s",
275 fun_name_utf.c_str());
276
277UTF8_string cmd_utf(cmd); // cmd is something like ∇FUN[⎕]∇
278 snprintf(err.error_message_2, sizeof(err.error_message_2),
279 " aaa %s", cmd_utf.c_str());
280 eref.left_caret = 5 + cmd.size();
281 if (Workspace::SI_top()) *Workspace::get_error() = eref;
282 throw eref;
283}
284//----------------------------------------------------------------------------
285void
286Error::update_error_info(StateIndicator * si)
287{
288bool locked = false;
289const UserFunction * ufun = si->get_executable()->get_exec_ufun();
290
291 // prepare second error line (failed statement)
292 //
293 if (ufun)
294 {
295 if (get_show_locked() || ufun->get_exec_properties()[1])
296 {
297 locked = true;
298 set_error_line_2(" ");
299 set_left_caret(6);
300 }
301 else
302 {
303 UCS_string ucs(ufun->get_name_and_line(si->get_PC()));
304 ucs.append(UNI_SPACE);
305 ucs.append(UNI_SPACE);
306 UTF8_string utf(ucs);
307 set_error_line_2(utf.c_str());
308 set_left_caret(ucs.size());
309 }
310 }
311 else
312 {
313 set_error_line_2(" ");
314 set_left_caret(6);
315 }
316
317 // prepare third line (carets)
318 //
319 set_right_caret(-1);
320
321 if (locked)
322 {
323 si->get_executable()->get_exec_ufun()->set_locked_error_info(*this);
324 }
325 else
326 {
327 const Function_PC from = si->get_prefix().get_range_low();
328 Function_PC to = si->get_prefix().get_range_high();
329 const Function_PC2 error_range(from, to);
330 si->get_executable()->set_error_info(*this, error_range);
331 }
332
333 // print error, unless we are in safe execution mode.
334 //
335 {
336 bool print_error = true;
337 for (const StateIndicator * si1 = si; si1; si1 = si1->get_parent())
338 {
339 if (si1->get_safe_execution())
340 {
341 print_error = false;
342 break;
343 }
344 }
345 if (print_error) print_em(UERR, LOC);
346 }
347
348 StateIndicator::get_error(si) = *this;
349}

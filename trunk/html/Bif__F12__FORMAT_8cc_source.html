35//----------------------------------------------------------------------------
36Token
37Bif_F12_FORMAT::eval_B(Value_P B) const
38{
39 // ISO and lrm: If B is a character array, then Z is B
40 //
41 if (!B->NOTCHAR()) return Token(TOK_APL_VALUE1, B);
42
43 if (B->is_empty())
44 {
45 Value_P Z(B->get_shape(), LOC);
46 Z->set_proto_Spc();
47 Z->check_value(LOC);
48 return Token(TOK_APL_VALUE1, Z);
49 }
50
51 if (!B->is_simple())
52 {
53 PrintBuffer pb(*B, Workspace::get_PrintContext(PR_APL), 0);
54 Assert(pb.is_rectangular());
55 const ShapeItem cols = pb.get_column_count();
56 const ShapeItem rows = pb.get_row_count();
57 Shape shape_Z(rows, cols);
58 Value_P Z(shape_Z, LOC);
59 loop(y, rows)
60 {
61 UCS_string row = pb.get_line(y);
62 row.map_pad();
63 loop(x, cols) Z->next_ravel_Char(row[x]);
64 }
65
66 Z->check_value(LOC);
67
68 // turn 1-line matrices into vectors
69 //
70 if (Z->get_rank() == 2 && Z->get_shape().get_shape_item(0) == 1)
71 {
72 Shape sh(Z->get_shape().get_shape_item(1));
73 Z->set_shape(sh);
74 }
75
76 return Token(TOK_APL_VALUE1, Z);
77 }
78
79Value_P Z;
80 if (B->get_rank() > 2)
81 {
82 // temporarily reduce the N > 2 dimensions of B
83 // to N = 2 dimensions of B1 (with the same ravel).
84 //
85 const Shape shape_B = B->get_shape();
86 const Shape shape_B1(B->get_rows(), B->get_cols());
87
88 try
89 {
90 B->set_shape(shape_B1);
91 Z = monadic_format(B);
92 B->set_shape(shape_B);
93 }
94 catch (...)
95 {
96 B->set_shape(shape_B);
97 throw; // rethrow error
98 }
99
100 // ¯1↓⍴B ←→ ¯1↓⍴Z i.e. the leading axes of B have the same lengths as
101 // the leading axes of Z. monadic ⍕ changes (increases) only the length
102 // of the last axis. We reshape Z to ¯1↓⍴B , ¯1⍴⍴Z.
103 //
104 Shape shape_Z = B->get_shape().without_last_axis(); // ¯1↓⍴B
105 shape_Z.add_shape_item(Z->get_last_shape_item()); // ¯1↓⍴B , ¯1↑⍴Z
106 Z->set_shape(shape_Z);
107 }
108 else // B->get_rank() is 0, 1, or 2
109 {
110 Z = monadic_format(B);
111 }
112
113 // adjust rank as needed:
114 // ρρZ ←→ ,1⌈⍴ρB if B is simple
115 // ρρZ ←→ ,1 or ,2 if B is nested
116 //
117const APL_types::Depth depth = B->compute_depth();
118Shape sZ;
119 if (depth > 1) // B is nested, therefore ⍴⍴R is 1 or 2
120 {
121 // the examples in lrm contradict the text in lrm.
122 //
123 // Page 136 shows: R←2 3ρ'ONE' 1 1 'TWO' 2 22 which contains only
124 // scalar and vector items (R itself being a matrix) and therefore
125 // ⍕R should be a vector according to page 137:
126 //
127 // "Nested Arrays: When R is a nested array, Z is a vector if all"
128 // items of R at any depth are scalars or vectors."
129 //
130 // lrm also contradicts the ISO standard regarding the rank of Z.
131 //
132 // We try our best...
133 //
134 if (B->is_one_dimensional())
135 {
136 sZ = Shape(Z->element_count());
137 }
138 else
139 {
140 sZ = Z->get_shape(); // leave Z as is (2-dimensional)
141 }
142 }
143 else // B is simple: result rank: ,1⌈⍴ρB
144 {
145 if (B->get_rank() < 2)
146 {
147 sZ = Shape(Z->get_cols());
148 }
149 else
150 {
151 sZ = B->get_shape();
152 sZ.set_last_shape_item(Z->get_last_shape_item());
153 }
154 }
155
156 if (sZ != Z->get_shape())
157 {
158 Assert(sZ.get_volume() == Z->element_count());
159 Z->set_shape(sZ);
160 }
161
162 Z->check_value(LOC);
163 return Token(TOK_APL_VALUE1, Z);
164}
165//----------------------------------------------------------------------------
166Token
167Bif_F12_FORMAT::eval_AB(Value_P A, Value_P B) const
168{
169Value_P Z;
170
171 // any A should be a scalar or a vcector
172 //
173 if (A->get_rank() > 1) RANK_ERROR;
174
175 if (A->is_char_array()) Z = format_by_example(A, B);
176 else if (A->is_int_array()) Z = format_by_specification(A, B);
177 else
178 {
179 MORE_ERROR() = "Bad left argument of ⍕";
180 DOMAIN_ERROR;
181 }
182
183 Z->set_proto_Spc();
184 Z->check_value(LOC);
185 return Token(TOK_APL_VALUE1, Z);
186}
187//----------------------------------------------------------------------------
188Value_P
189Bif_F12_FORMAT::monadic_format(Value_P B)
190{
191 Assert(B->get_rank() <= 2);
192
193const PrintStyle style(PrintStyle(PR_APL | PST_NO_FRACT_0));
194const PrintContext pctx = Workspace::get_PrintContext(style);
195
196const PrintBuffer pb(*B, pctx, 0);
197
198const ShapeItem width = pb.get_column_count();
199const ShapeItem height = pb.get_row_count();
200
201 // monadic_format() returns the Value with rank 1 or 2 which may be changed
202 // in Bif_F12_FORMAT::eval_B() later on to match the rather arbitrary rules
203 // in lrm.
204 //
205Value_P Z;
206 if (height == 1) Z = Value_P(width, LOC);
207 else Z = Value_P(Shape(height, width), LOC);
208
209 loop(h, height)
210 loop(w, width)
211 {
212 const Unicode uni = pb.get_char(w, h);
213 if (is_iPAD_char(uni)) Z->next_ravel_Char(UNI_SPACE);
214 else Z->next_ravel_Char(uni);
215 }
216
217 Z->check_value(LOC);
218 return Z;
219}
220//----------------------------------------------------------------------------
221Value_P
222Bif_F12_FORMAT::format_by_example(Value_P A, Value_P B)
223{
224 // convert the ravel of char vector A into UCS_string 'format'.
225 //
226UCS_string all_formats = A->get_UCS_ravel();
227 if (all_formats.size() == 0) LENGTH_ERROR;
228
229const ShapeItem cols = B->get_cols();
230const ShapeItem rows = B->get_rows();
231
232 // split string all_formats into individual format fields, one per column.
233 // If there is only one format field, then repeat it cols times.
234 //
235vector<UCS_string> col_formats;
236 split_example_into_columns(all_formats, col_formats);
237 if (col_formats. size() == 1)
238 {
239 const UCS_string f = all_formats;
240 const UCS_string col0 = col_formats[0]; // keep it out of loop!
241 loop(c, cols - 1)
242 {
243 all_formats.append(f);
244 col_formats.push_back(col0);
245 }
246 }
247
248 if (cols != ShapeItem(col_formats.size())) LENGTH_ERROR;
249
250 // convert each column format string into a Format_LIFER
251 //
252vector<Format_LIFER> col_items;
253 loop(c, col_formats.size())
254 col_items.push_back(Format_LIFER(col_formats[c]));
255
256 Log(LOG_Bif_F12_FORMAT)
257 {
258 Q1(all_formats)
259 Q1(all_formats.size())
260 Q1(col_formats.size())
261
262 loop(c, col_items.size())
263 {
264 CERR << "At " LOC " in format_by_example()" << endl
265 << " col_items[c].left_deco: '"
266 << col_items[c].left_deco << "'" << endl
267 << " col_items[c].int_part: '"
268 << col_items[c].int_part << "'" << endl
269 << " col_items[c].fract_part: '"
270 << col_items[c].fract_part << "'" << endl
271 << " col_items[c].expo_deco: '"
272 << col_items[c].expo_deco << "'" << endl
273 << " col_items[c].expo_negative: "
274 << col_items[c].expo_negative << endl
275 << " col_items[c].right_deco: '"
276 << col_items[c].right_deco << "'" << endl;
277 }
278 }
279
280Shape shape_Z(B->get_shape());
281 if (B->is_scalar()) shape_Z.add_shape_item(1);
282 shape_Z.set_last_shape_item(all_formats.size());
283
284Value_P Z(shape_Z, LOC);
285
286 try
287 {
288 loop(r, rows)
289 {
290 UCS_string row;
291 loop(c, cols)
292 {
293 const Cell & cB = B->get_cravel(c + r*cols);
294 if (!cB.is_real_cell()) DOMAIN_ERROR;
295
296 const APL_Float value = cB.get_real_value();
297
298 const UCS_string item = col_items[c].format_example(value);
299 Log(LOG_Bif_F12_FORMAT) Q1(item)
300 row.append(item);
301 }
302
303 Log(LOG_Bif_F12_FORMAT) { Q1(row) Q1(all_formats) }
304
305 Assert(row.size() == all_formats.size());
306 loop(c, row.size())
307 Z->set_ravel_Char(r*all_formats.size() + c, row[c]);
308 }
309 }
310 catch (Error err)
311 {
312 throw err; // rethrow
313 }
314 catch (...)
315 {
316 }
317
318 return Z;
319}
320//----------------------------------------------------------------------------
321void
322Bif_F12_FORMAT::split_example_into_columns(const UCS_string & all_formats,
323 vector<UCS_string> & col_formats)
324{
325 // split string 'all_formats' into fields, where:
326 //
327 // 1. a field contains at least one digit and is delimited by either
328 // 1a. a space (which belongs to the next field), or
329 // 1b. digit '6' (which belongs to the field).
330 //
331bool digit_seen = false;
332UCS_string current_format;
333 loop(f, all_formats.size())
334 {
335 const Unicode cc = all_formats[f];
336 current_format.append(cc);
337
338 if (Avec::is_digit(cc)) digit_seen = true;
339
340 if ((cc == UNI_SPACE) && digit_seen) // end of field, case 1a.
341 {
342 col_formats.push_back(current_format);
343 current_format.clear(); // start a new field;
344 digit_seen = false;
345 }
346 else if (cc == UNI_6) // end of field, case 1b.
347 {
348 ++f; // next char is right decorator (and end of field)
349 if (f < all_formats.size())
350 current_format.append(all_formats[f]);
351
352 col_formats.push_back(current_format);
353 current_format.clear(); // start a new field;
354 digit_seen = false;
355 continue; // next char
356 }
357 }
358
359 if (col_formats.size() && !digit_seen)
360 {
361 col_formats.back().append(current_format);
362 }
363 else
364 {
365 col_formats.push_back(current_format);
366 }
367}
368//----------------------------------------------------------------------------
369UCS_string
370Bif_F12_FORMAT::Format_LIFER::format_example(APL_Float value)
371{
372UCS_string data_int;
373UCS_string data_fract;
374UCS_string data_expo;
375const APL_Float val_1 = (value < 0.0) ? APL_Float(-value) : value;
376
377bool overflow = false;
378 fill_data_fields(val_1, data_int, data_fract, data_expo, overflow);
379 Log(LOG_Bif_F12_FORMAT)
380 {
381 Q1(overflow)
382 Q1(data_int)
383 Q1(data_fract)
384 Q1(data_expo)
385 }
386 if (overflow) return UCS_string(out_size(), Workspace::get_FC(3));
387
388const UCS_string left = format_left_side(data_int, value < 0.0, overflow);
389 if (overflow) return UCS_string(out_size(), Workspace::get_FC(3));
390 Assert(left.size() == (left_deco.out_len + int_part.out_len));
391
392const UCS_string right = format_right_side(data_fract, value < 0.0, data_expo);
393 Assert(right.size() == (fract_part.out_len + expo_deco.out_len +
394 exponent.out_len + right_deco.out_len));
395
396 return left + right;
397}
398//----------------------------------------------------------------------------
399UCS_string
400Bif_F12_FORMAT::Format_LIFER::format_left_side(const UCS_string data_int,
401 bool negative, bool & overflow)
402{
403const UCS_string data = int_part.insert_int_commas(data_int, overflow);
404
405 if (overflow) return UCS_string();
406
407 Assert(int_part.size() >= data.size());
408const Unicode pad_char = int_part.pad_char(Workspace::get_FC(2));
409const UCS_string pad(int_part.size() - data.size(), pad_char);
410
411UCS_string ucs;
412 if (int_part.no_float()) // floating disabled.
413 {
414 ucs.append(left_deco.format);
415 ucs.append(pad);
416 }
417 else if (int_part.do_float(negative)) // floating enabled, deco shown
418 {
419 ucs.append(pad);
420 ucs.append(left_deco.format);
421 }
422 else // floating enabled, deco hidden
423 {
424 ucs.append(pad);
425 ucs.append(UCS_string(left_deco.size(), UNI_SPACE));
426 }
427
428 ucs.append(data);
429 return ucs;
430}
431//----------------------------------------------------------------------------
432UCS_string
433Bif_F12_FORMAT::Format_LIFER::format_right_side(const UCS_string data_fract,
434 bool negative,
435 const UCS_string data_expo)
436{
437int pad_count = 0;
438UCS_string ucs;
439
440 if (fract_part.out_len)
441 {
442 const UCS_string data = fract_part.insert_fract_commas(data_fract);
443 Assert(fract_part.size() >= data.size());
444 pad_count += fract_part.size() - data.size();
445
446 // print nothing instead of .
447 if (data.size() == 0) ++pad_count;
448 else ucs.append(Workspace::get_FC(0));
449 ucs.append(data);
450 }
451
452 if (exponent.size())
453 {
454 UCS_string data(1, exponent_char); // the 'E'
455
456 const Unicode pad_char = exponent.pad_char(Workspace::get_FC(2));
457 if (exponent.no_float()) // floating disabled.
458 {
459 data.append(expo_deco.format);
460 data.append(data_expo);
461 }
462 else if (exponent.do_float(expo_negative)) // floating, deco shown
463 {
464 data.append(expo_deco.format);
465 data.append(data_expo);
466 }
467 else // floating enabled and deco hidden
468 {
469 data.append(UCS_string(expo_deco.format.size(), pad_char));
470 data.append(data_expo);
471 }
472
473 pad_count += expo_deco.out_len + exponent.out_len - data.size();
474 ucs.append(data);
475 }
476
477 // now do the right side padding.
478 {
479 const Unicode pad_char = fract_part.pad_char(Workspace::get_FC(2));
480 const UCS_string pad(pad_count, pad_char);
481
482 if (fract_part.no_float()) // floating disabled.
483 {
484 ucs.append(right_deco.format);
485 ucs.append(pad);
486 }
487 else if (fract_part.do_float(negative)) // floating, deco shown
488 {
489 ucs.append(pad);
490 ucs.append(right_deco.format);
491 }
492 else // floating, deco hidden
493 {
494 ucs.append(pad);
495 ucs.append(UCS_string(right_deco.size(), pad_char));
496 }
497 }
498
499 return ucs;
500}
501//----------------------------------------------------------------------------
502Bif_F12_FORMAT::Format_LIFER::Format_LIFER(const UCS_string format)
503 : exponent_char(UNI_E),
504 expo_negative(false)
505{
506 // split one column format into sub-format chunks...
507 //
508int f = 0;
509bool exponent_pending = false;
510bool have_decimal_point = false;
511
512// left_decorator:
513 while (f < format.size())
514 {
515 if (is_control_char(format[f])) goto integral_part;
516 else left_deco.format.append(format[f++]);
517 }
518 goto fields_done;
519
520integral_part:
521 while (f < format.size())
522 {
523 const Unicode cc = format[f++];
524
525 if (cc == UNI_FULLSTOP)
526 {
527 have_decimal_point = true;
528 goto fractional_part;
529 }
530 if (cc == UNI_E) goto exponent_part;
531 if (is_control_char(cc))
532 {
533 int_part.format.append(cc);
534 if (cc == UNI_6) goto right_decorator;
535 }
536 else
537 {
538 --f;
539 goto right_decorator;
540 }
541 }
542 goto fields_done;
543
544fractional_part:
545 while (f < format.size())
546 {
547 const Unicode cc = format[f++];
548 if (cc == UNI_7) exponent_pending = true;
549
550 if (is_control_char(cc))
551 {
552 fract_part.format.append(cc);
553 if (cc == UNI_6) goto right_decorator;
554 }
555 else
556 {
557 exponent_char = cc;
558 goto exponent_decorator;
559 }
560 }
561 goto right_decorator;
562
563exponent_decorator:
564 if (!exponent_pending) { --f; goto right_decorator; }
565
566 while (f < format.size())
567 {
568 const Unicode cc = format[f++];
569
570 if (!is_control_char(cc))
571 {
572 expo_deco.format.append(cc);
573 }
574 else
575 {
576 --f;
577 goto exponent_part;
578 }
579 }
580
581exponent_part:
582 while (f < format.size())
583 {
584 const Unicode cc = format[f++];
585
586 if (is_control_char(cc))
587 {
588 exponent.format.append(cc);
589 if (cc == UNI_6) goto right_decorator;
590 }
591 else
592 {
593 --f;
594 goto right_decorator;
595 }
596 }
597
598right_decorator:
599
600 if (have_decimal_point &&
601 exponent.format.size() == 0 &&
602 fract_part.format.size() == 0) // quirk!
603 {
604 // this is ambiguous as to whether the decimal point belongs to
605 // the number or to the right decorator. lrm says all non-digits
606 // are decorators but that seems to be wrong anyhow. (e.g. ',').
607 //
608 // we put a trailing decimat point into the right decorator.
609 //
610 right_deco.format.append(UNI_FULLSTOP);
611 }
612
613 while (f < format.size())
614 right_deco.format.append(format[f++]);
615
616fields_done:
617 left_deco.out_len = left_deco .format.size();
618
619 int_part .out_len = int_part .format.size();
620
621 if (fract_part.format.size())
622 fract_part.out_len = fract_part.format.size() + 1;
623
624 if (exponent.size())
625 {
626 expo_deco.out_len = expo_deco.size();
627 exponent.out_len = exponent.format.size() + 1;
628 }
629
630 right_deco.out_len = right_deco.format.size();
631
632 // compute floating flags.
633 //
634 {
635 const int count = int_part .map_field(0)
636 + fract_part.map_field(1);
637
638 if (count == 1) // flags apply to both sides.
639 {
640 int_part. flt_mask |= fract_part.flt_mask & (BIT_1 | BIT_2 | BIT_3);
641 fract_part.flt_mask |= int_part .flt_mask & (BIT_1 | BIT_2 | BIT_3);
642 }
643
644 exponent.map_field(2);
645 }
646}
647//----------------------------------------------------------------------------
648int
649Format_sub::map_field(int type)
650{
651int flt_cnt = 0;
652
653 loop(f, format.size())
654 {
655 switch(format[f])
656 {
657 case UNI_0: flt_mask |= BIT_0; break;
658 case UNI_1: ++flt_cnt; flt_mask |= BIT_1; break;
659 case UNI_2: ++flt_cnt; flt_mask |= BIT_2; break;
660 case UNI_3: ++flt_cnt; flt_mask |= BIT_3; break;
661 case UNI_4: ++flt_cnt; flt_mask |= BIT_4; break;
662 case UNI_5: flt_mask |= BIT_5; break;
663 case UNI_6: flt_mask |= BIT_6; break;
664 case UNI_7: flt_mask |= BIT_7; break;
665 case UNI_8: flt_mask |= BIT_8; break;
666 case UNI_9: flt_mask |= BIT_9; break;
667 default: break;
668 }
669 }
670
671 if (flt_mask & (BIT_0 | BIT_9))
672 {
673 min_len = format.size();
674 loop(f, format.size())
675 {
676 const size_t pos = (type == 1) ? format.size() - f - 1 : f;
677 const Unicode uni = format[pos];
678 if (uni == UNI_0) break;
679 if (uni == UNI_9) break;
680 --min_len;
681 }
682 }
683
684 return flt_cnt;
685}
686//----------------------------------------------------------------------------
687ostream &
688Format_sub::print(ostream & out) const
689{
690 out << "format: '" << format << "',"
691 " min: " << min_len << ", out_len " << out_len << ", flags: ";
692 loop(d, 32) if (flt_mask & (1 << d)) out << char('0' + d);
693 return out;
694}
695//----------------------------------------------------------------------------
696void
697Bif_F12_FORMAT::Format_LIFER::fill_data_fields(APL_Float value,
698 UCS_string & data_int, UCS_string & data_fract,
699 UCS_string & data_expo, bool & overflow)
700{
701char format[40];
702const int data_buf_len = int_part.out_len + 1 // 123.
703 + fract_part.out_len // 456
704 + 1 + exponent.out_len + 1; // E-22
705
706 if (data_buf_len > 100) DOMAIN_ERROR;
707
708char data_buf[101];
709char * fract_end = 0;
710
711 if (exponent.size()) // E in format string
712 {
713 // create a format like %.5E (if fract_part has 5 digits)
714 //
715 const int flen = snprintf(format, sizeof(format), "%%.%luE",
716 unsigned_long(fract_part.size()));
717 Assert(flen < int(sizeof(format))); // format was big enough.
718
719 const int dlen = snprintf(&data_buf[0], data_buf_len, format, value);
720 Assert(dlen < data_buf_len);
721
722 char * ep = strchr(&data_buf[0], 'E');
723 Assert(ep);
724 fract_end = ep++;
725 if (*ep == '+') ++ep; // skip +
726 else if (*ep == '-') { expo_negative = true; ++ep; } // skip -
727 if (*ep == '0' && ep[1]) ++ep; // skip leading 0 in exponent
728
729 int elen = strlen(ep);
730
731 // insert leading zeros until we have at least min_len digits.
732 //
733 for (; elen < exponent.min_len; ++elen) data_expo.append(UNI_0);
734
735 data_expo.append(UCS_string(UTF8_string(ep)));
736 }
737 else // no exponent in format string.
738 {
739 const int flen = snprintf(format, sizeof(format), "%%.%luf",
740 unsigned_long(fract_part.size()));
741 Assert(flen < int(sizeof(format))); // assume no snprintf() overflow
742
743 const int dlen = snprintf(&data_buf[0], data_buf_len, format, value);
744 data_buf[data_buf_len - 1] = 0;
745
746 // the int part could be longer than allowed by the example string.
747 //
748 const char * dot = strchr(data_buf, '.');
749 const int ilen = dot ? (dot - data_buf)
750 : strlen(data_buf);
751 if (ilen > int_part.out_len)
752 {
753 if (Workspace::get_FC(3) == UNI_0)
754 {
755 MORE_ERROR() << "Overflow in integer part";
756 DOMAIN_ERROR;
757 }
758 overflow = true;
759 return ;
760 }
761
762 Assert(dlen < data_buf_len);
763 fract_end = &data_buf[dlen];
764 }
765
766char * int_end = strchr(&data_buf[0], '.');
767 if (fract_part.size() == 0)
768 {
769 Assert(int_end == 0);
770 int_end = fract_end;
771 }
772 else
773 {
774 Assert(int_end);
775 char * fract_digits = int_end + 1;
776
777 int flen = fract_end - fract_digits;
778
779 // remove trailing zeros, but leave at least min_len chars.
780 while (fract_digits[flen - 1] == '0' && flen > fract_part.min_len)
781 fract_digits[--flen] = 0;
782
783 loop(f, flen) data_fract.append(Unicode(fract_digits[f]));
784 }
785
786const int ilen = int_end - &data_buf[0];
787
788 // insert leading zeros so that we will have at least min_len digits
789 // after appending the integer data.
790 //
791 loop(d, int_part.min_len - ilen) data_int.append(UNI_0);
792
793 loop(i, ilen) data_int.append(Unicode(data_buf[i]));
794
795 // convert 0.xxx to .xxx
796 //
797 if ( data_int.size() == 1
798 && data_int[0] == UNI_0
799 && int_part.min_len == 0
800 && fract_part.size()) data_int.clear();
801
802 Log(LOG_Bif_F12_FORMAT)
803 {
804 CERR << "At " LOC " in Format_LIFER::fill_data_fields()" << endl
805 << " format: '" << format << "'" << endl
806 << " value: " << value << endl
807 << " data_int: '" << data_int << "'" << endl
808 << " data_fract: '" << data_fract << "'" << endl
809 << " data_expo: '" << data_expo << "'" << endl;
810 }
811}
812//----------------------------------------------------------------------------
813UCS_string
814Format_sub::insert_int_commas(const UCS_string & data, bool & overflow) const
815{
816size_t fill_pos = -1;
817Unicode fill_char = UNI_SPACE;
818
819 // BIT_0: pad with 0
820 // BIT_8: fille with ⎕FC[3] (* by default)
821 // BIT_9: pad with 0
822 if (flt_mask & (BIT_0 | BIT_8 | BIT_9)) // format has a '0', '8', or '9'
823 {
824 loop(f, format.size())
825 {
826 const Unicode format_char = format[f];
827 if (format_char == UNI_0 || format_char == UNI_9)
828 {
829 fill_pos = f;
830 fill_char = UNI_0;
831 break;
832 }
833
834 if (format_char == UNI_8)
835 {
836 fill_pos = f;
837 fill_char = Workspace::get_FC(2);
838 break;
839 }
840 }
841 }
842
843 // we move backwards from the end of the format field, filling it
844 // with chars from data of from format.
845 //
846UCS_string ucs;
847size_t d = data.size();
848
849 // d runs downwards from data.size() to 0
850 // f1 runs upwards from 0 to format.size()
851 // f runs downwards from format.size() to 0
852 //
853 loop(f1, format.size())
854 {
855 const size_t f = format.size() - f1 - 1;
856 const Unicode format_char = format[f];
857 if (format_char == UNI_COMMA)
858 {
859 // Workspace::get_FC(1) is ⎕FC[2] when ⎕IO is 1
860 //
861 if (d) ucs.append(Workspace::get_FC(1));
862 else if (f >= fill_pos) ucs.append(fill_char);
863 else break;
864 }
865 else if (Avec::is_digit(format_char))
866 {
867 if (d) ucs.append(data[--d]);
868 else if (f >= fill_pos) ucs.append(fill_char);
869 else break;
870 }
871 else
872 {
873 CERR << "Offending format char [" << f << "] : '"
874 << format_char << "' at " << LOC << endl;
875 break;
876 }
877 }
878
879 if (d) // format too short
880 {
881 if (Workspace::get_FC(3) == UNI_0) DOMAIN_ERROR;
882 else overflow = true;
883 }
884
885 if ((flt_mask & BIT_9) && ucs.all_zeroes()) ucs.clear();
886
887 return ucs.reverse();
888}
889//----------------------------------------------------------------------------
890UCS_string
891Format_sub::insert_fract_commas(const UCS_string & data) const
892{
893UCS_string ucs;
894int d = 0;
895
896 loop(f, format.size())
897 {
898 const Unicode format_char = format[f];
899 if (format_char == UNI_COMMA)
900 {
901 ucs.append(Workspace::get_FC(1));
902 }
903 else if (Avec::is_digit(format_char))
904 {
905 if (d < data.size()) ucs.append(data[d++]);
906 else break;
907 }
908 else
909 {
910 CERR << "Offending format char [" << f << "] : '"
911 << format_char << "' at " << LOC << endl;
912 }
913 }
914
915 if ((flt_mask & BIT_9) && ucs.all_zeroes()) ucs.clear();
916
917 return ucs;
918}

927//----------------------------------------------------------------------------
928Value_P
929Bif_F12_FORMAT::format_by_specification(Value_P A, Value_P B)
930{
931 // A is a near-int scalar or vector.
932
933const Shape shape_1(1);
934const Shape & shape_B = B->get_rank() ? B->get_shape() : shape_1;
935
936const ShapeItem rows_B = shape_B.get_rows();
937const ShapeItem cols_B = shape_B.get_cols();
938
939const ShapeItem len_A = A->element_count();
940
941 if (len_A != 1 && len_A != 2 && len_A != 2*cols_B) LENGTH_ERROR;
942
943 if (shape_B.get_volume() == 0) // empty B
944 {
945 ShapeItem W = 0;
946 loop(c, cols_B)
947 {
948 if (len_A <= 2) W += A->get_cfirst().get_near_int();
949 else W += A->get_cravel(2*c).get_near_int();
950 }
951
952 Shape shape_Z = shape_B.without_last_axis();
953 shape_Z.add_shape_item(W);
954 const ShapeItem ec_Z = shape_Z.get_volume();
955
956 Value_P Z(shape_Z, LOC);
957 loop(z, ec_Z) Z->next_ravel_Char(UNI_SPACE);
958
959 Z->set_proto_Spc();
960 return Z;
961 }
962
963PrintBuffer pb;
964 loop(col, cols_B)
965 {
966 APL_Integer col_width;
967 APL_Integer precision;
968
969 if (len_A == 1)
970 {
971 col_width = 0;
972 precision = A->get_cfirst().get_near_int();
973 }
974 else if (len_A == 2)
975 {
976 col_width = A->get_cfirst().get_near_int();
977 precision = A->get_cravel(1).get_near_int();
978 }
979 else
980 {
981 col_width = A->get_cravel(2*col) .get_near_int();
982 precision = A->get_cravel(2*col + 1).get_near_int();
983 }
984
985 PrintBuffer pb_col(format_one_col_by_spec(col_width, precision,
986 &B->get_cravel(col), cols_B, rows_B));
987
988 if (col_width == 0) pb_col.pad_l(UNI_SPACE, 1);
989
990 if (col) pb.append_col(pb_col);
991 else pb = pb_col;
992 }
993
994const ShapeItem pb_w = pb.get_column_count();
995const ShapeItem pb_h = pb.get_row_count();
996Shape shape_Z(shape_B);
997 shape_Z.set_last_shape_item(pb_w);
998
999Value_P Z(shape_Z, LOC);
1000
1001 loop(h, pb_h)
1002 loop(w, pb_w) Z->next_ravel_Char(pb.get_char(w, h));
1003
1004 return Z;
1005}
1006//----------------------------------------------------------------------------
1007PrintBuffer
1008Bif_F12_FORMAT::format_one_col_by_spec(int width, int precision,
1009 const Cell * cB, ShapeItem cols,
1010 ShapeItem rows)
1011{
1012PrintBuffer ret;
1013
1014bool has_char = false;
1015bool has_num = false;
1016bool has_complex = false;
1017
1018 // determine the data types (character/numbers/complex) in B
1019 loop(r, rows)
1020 {
1021 const Cell & cell = cB[r*cols];
1022 if (cell.is_numeric())
1023 {
1024 has_num = true;
1025 if (cell.is_complex_cell()) has_complex = true;
1026 }
1027 else
1028 {
1029 has_char = true;
1030 }
1031 }
1032
1033 if (has_complex)
1034 {
1035 // split cB into vectors real and imag...
1036 Value_P real(rows, LOC);
1037 Value_P imag(rows, LOC);
1038 loop(r, rows)
1039 {
1040 const Cell & cell = cB[r*cols];
1041 if (cell.is_complex_cell())
1042 {
1043 real->next_ravel_Float(cell.get_real_value());
1044 imag->next_ravel_Float(cell.get_imag_value());
1045 }
1046 else
1047 {
1048 real->next_ravel_Cell(cell);
1049 imag->next_ravel_Char(UNI_SPACE);
1050 }
1051 }
1052
1053 PrintBuffer pb_real = format_one_col_by_spec(width, precision,
1054 &real->get_cfirst(), 1,
1055 rows);
1056 PrintBuffer pb_imag = format_one_col_by_spec(width, precision,
1057 &imag->get_cfirst(), 1,
1058 rows);
1059
1060 PrintBuffer pb_real_imag;
1061 loop(r,rows)
1062 {
1063 UCS_string real_imag = pb_real.get_line(r);
1064 UCS_string imag = pb_imag.get_line(r);
1065 imag.remove_leading_whitespaces();
1066 if (imag.size())
1067 {
1068 real_imag += UNI_J;
1069 real_imag.append(imag);
1070 }
1071 pb_real_imag.append_ucs(real_imag);
1072 }
1073 return pb_real_imag;
1074 }
1075
1076 // real B. (if B was initially complex then we arrive here twice: once
1077 // for the real parts and once for the imag parts).
1078 //
1079 loop(r, rows)
1080 {
1081 const Cell & cell = cB[r*cols];
1082 if (cell.is_character_cell())
1083 {
1084 UCS_string data = UCS_string(cell.get_char_value());
1085
1086 add_row(ret, r, has_char, has_num, UNI_E, data);
1087 continue;
1088 }
1089
1090 if (cell.is_pointer_cell())
1091 {
1092 Value_P value = cell.get_pointer_value();
1093 UCS_string data = value->get_UCS_ravel();
1094
1095 if (width && data.size() > width) // overflow
1096 {
1097 if (Workspace::get_FC(3) == UNI_0) DOMAIN_ERROR;
1098
1099 data = UCS_string(width, Workspace::get_FC(3));
1100 }
1101
1102 add_row(ret, r, has_char, has_num, UNI_E, data);
1103 continue;
1104 }
1105
1106 if (!cell.is_real_cell()) DOMAIN_ERROR;
1107
1108 APL_Float value = cB[r*cols].get_real_value();
1109
1110 if (precision >= 0) // integer or floating format
1111 {
1112 // precision == 0 means integer;
1113 // precision > 0 is the number of digits AFTER the decimal point
1114
1115 // fix values close to 0 as 0 so that, for example, we never
1116 // see ¯.0000 in the formatted output.
1117 //
1118 double minval = 0.01;
1119 loop(p, precision) minval /= 10;
1120 if (value < minval && value > -minval) value = 0.0;
1121
1122 UCS_string data = format_float_by_spec(value, precision);
1123 if (width && data.size() > width) // overflow
1124 {
1125 if (Workspace::get_FC(3) == UNI_0) DOMAIN_ERROR;
1126 data = UCS_string(width, Workspace::get_FC(3));
1127 }
1128
1129 add_row(ret, r, has_char, has_num, UNI_E, data);
1130 }
1131 else // exponential format
1132 {
1133 // precision < 0 is the TOTAL (!) number of digits, which is
1134 // one more than the digits AFTER the decimal point.
1135 //
1136 const int fract_digits = (-precision) - 1;
1137 UCS_string data = UCS_string::from_double_to_expo(value,
1138 fract_digits);
1139 add_row(ret, r, has_char, has_num, UNI_E, data);
1140 }
1141 }
1142
1143 if (width && ret.get_column_count() < width)
1144 ret.pad_l(UNI_SPACE, width - ret.get_column_count());
1145
1146 return ret;
1147}
1148//----------------------------------------------------------------------------
1149void
1150Bif_F12_FORMAT::add_row(PrintBuffer & ret, int row, bool has_char,
1151 bool has_num, Unicode align_char, UCS_string & data)
1152{
1153 if (row == 0) // first row: don't align
1154 {
1155 ret.append_ucs(data);
1156 }
1157 else if (!has_char) // only numbers: align at '.' or 'E'
1158 {
1159 ret.append_aligned(data, align_char);
1160 }
1161 else if (!has_num) // only chars: align left
1162 {
1163 const int d = ret.get_column_count() - data.size();
1164 if (d < 0) ret.pad_r(UNI_SPACE, -d);
1165 else if (d > 0) data.append(UCS_string(d, UNI_SPACE));
1166 ret.append_ucs(data);
1167 }
1168 else // chars and numbers: align right
1169 {
1170 const int d = ret.get_column_count() - data.size();
1171 if (d < 0) ret.pad_l(UNI_SPACE, -d);
1172 else if (d > 0) data = UCS_string(d, UNI_SPACE) + data;
1173 ret.append_ucs(data);
1174 }
1175}
1176//----------------------------------------------------------------------------
1177UCS_string
1178Bif_F12_FORMAT::format_float_by_spec(APL_Float value, int precision)
1179{
1180UCS_string ret = UCS_string::from_double_to_fixed(value, precision);
1181
1182 // Note: the examples in the apl standard use a leading 0 (like 0.00)
1183 // while lrm shows .00 instead. We follow lrm and remove a leading 0.
1184 // Interestingly, the exponential format leaves a leading 0 in both
1185 // the standard and lrm
1186 //
1187 if (precision) // we have a '.'
1188 {
1189 if (ret[0] == UNI_0 &&
1190 ret[1] == UNI_FULLSTOP) // 0.xxx → .xxx
1191 {
1192 ret.erase(0);
1193 }
1194 else if (ret[0] == UNI_OVERBAR &&
1195 ret[1] == UNI_0 &&
1196 ret[2] == UNI_FULLSTOP) // ¯0.x → ¯.x
1197 {
1198 ret[1] = UNI_OVERBAR;
1199 ret.erase(0);
1200 }
1201 }
1202 else if (ret.size() == 2 &&
1203 ret[0] == UNI_OVERBAR &&
1204 ret[1] == UNI_0) // ¯0 → 0
1205 {
1206 ret.erase(0);
1207 }
1208
1209 return ret;
1210}

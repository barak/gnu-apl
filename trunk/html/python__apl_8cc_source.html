72//----------------------------------------------------------------------------
73static PyObject *
74apl_to_python(const Value * value)
75{
76PyObject * shape = PyList_New(value->get_rank());
77 loop(r, value->get_rank())
78 PyList_SetItem(shape, r, PyLong_FromLong(value->get_shape_item(r)));
79
80const ShapeItem ravel_len = value->nz_element_count();
81PyObject * ravel = PyList_New(ravel_len);
82 loop(i, ravel_len)
83 {
84 const Cell & cell = value->get_cravel(i);
85 if (cell.is_integer_cell())
86 PyList_SetItem(ravel, i, PyLong_FromLong(cell.get_int_value()));
87 else if (cell.is_float_cell())
88 PyList_SetItem(ravel, i, PyFloat_FromDouble(cell.get_real_value()));
89 else if (cell.is_complex_cell())
90 PyList_SetItem(ravel, i,
91 PyComplex_FromDoubles(cell.get_real_value(),
92 cell.get_imag_value()));
93 else if (cell.is_character_cell())
94 {
95 const UCS_string ucs(cell.get_char_value()); // 1-element string
96 UTF8_string utf(ucs);
97 PyList_SetItem(ravel, i, PyUnicode_FromStringAndSize(utf.c_str(),
98 utf.size()));
99 }
100 else if (cell.is_pointer_cell())
101 {
102 Value_P sub_val = cell.get_pointer_value();
103 PyList_SetItem(ravel, i, apl_to_python(sub_val.get()));
104 }
105 else
106 {
107 CERR << "*** Bad cell type at " LOC << endl;
108 return 0;
109 }
110 }
111
112 return PyTuple_Pack(2, shape, ravel);
113}
114//----------------------------------------------------------------------------
115static PyObject * exec_result = 0;
116
117bool
118python_result_callback(Token & result)
119{
120const TokenTag tag = result.get_tag();
121bool do_display = false;
122
123 if (tag == TOK_APL_VALUE1 || tag == TOK_APL_VALUE3) // non-committed value
124 {
125 do_display = display_mode > 0;
126 exec_result = PyTuple_Pack(2, PyLong_FromLong(1),
127 apl_to_python(result.get_apl_val().get()));
128
129 }
130 else if (tag == TOK_APL_VALUE2) // committed value
131 {
132 do_display = display_mode > 1;
133 exec_result = PyTuple_Pack(2, PyLong_FromLong(2),
134 apl_to_python(result.get_apl_val().get()));
135 }
136 else if (tag == TOK_VOID) // void
137 {
138 exec_result = PyTuple_Pack(2, PyLong_FromLong(3), Py_None);
139 }
140 else if (tag == TOK_BRANCH) // →N
141 {
142 exec_result = PyTuple_Pack(2, PyLong_FromLong(4),
143 PyLong_FromLong(result.get_int_val()));
144 }
145 else if (tag == TOK_NOBRANCH) // →''
146 {
147 // never called because TOK_NOBRANCH is a NOOP
148 exec_result = PyTuple_Pack(2, PyLong_FromLong(4), Py_None);
149 }
150 else if (tag == TOK_ESCAPE) // →
151 {
152 exec_result = PyTuple_Pack(2, PyLong_FromLong(5), Py_None);
153 }
154 else
155 {
156 CERR << "Unexpected Tag << HEX(tag) at " LOC << endl;
157 FIXME;
158 }
159
160 return do_display;
161}
162//----------------------------------------------------------------------------
163static PyObject *
164apl_exec(PyObject * self, PyObject * args)
165{
166const char * line = 0;
167 if (!PyArg_ParseTuple(args, "s", &line))
168 {
169 CERR << "*** argument of exec() is not a string." << endl;
170 return 0;
171 }
172
173 init_if_necessary();
174
175UTF8_string line_utf8(line);
176UCS_string line_ucs(line_utf8);
177
178 exec_result = 0;
179 try { Command::process_line(line_ucs); }
180 catch (const Error & error)
181 {
182 CERR << "cought Error" << endl;
183 return PyLong_FromLong(error.get_error_code());
184 }
185 catch (ErrorCode error_code)
186 {
187 CERR << "cought ErrorCode" << endl;
188 return PyLong_FromLong(error_code);
189 }
190 catch (...)
191 {
192 CERR << "cought something else" << endl;
193 return PyLong_FromLong(-1);
194 }
195
196 if (exec_result == 0) // error or defined function without return value
197 {
198 if (const StateIndicator * si = Workspace::SI_top_error())
199 {
200 const Error & err = StateIndicator::get_error(si);
201 const ErrorCode ec = err.get_error_code();
202 return PyTuple_Pack(2, PyLong_FromLong(0), PyLong_FromLong(ec));
203 }
204
205 // defined function without result
206 //
207 return PyTuple_Pack(2, PyLong_FromLong(3), Py_None);
208 }
209
210PyObject * ret = exec_result;
211 exec_result = 0;
212 return ret;
213}

215static const Value *
216apl_get_var_value(PyObject * args)
217{
218const char * varname = 0;
219 if (!PyArg_ParseTuple(args, "s", &varname))
220 {
221 CERR << "*** argument of get_shape(varname) is not a string." << endl;
222 return 0;
223 }
224
225 init_if_necessary();
226
227UTF8_string varname_utf8(varname);
228UCS_string varname_ucs(varname_utf8);
229Symbol * sym = Workspace::lookup_existing_symbol(varname_ucs);
230 if (sym == 0)
231 {
232 CERR << "*** " << varname_ucs << "is not an APL symbol." << endl;
233 return 0;
234 }
235
236const ValueStackItem * top = sym->top_of_stack();
237 if (top == 0 || top->get_NC() != NC_VARIABLE)
238 {
239 CERR << "*** " << varname_ucs << "is not an APL variable." << endl;
240 return 0;
241 }
242
243 return top->get_val_cptr();
244}

264//----------------------------------------------------------------------------
265static PyObject *
266make_ravel(const Value * value)
267{
268const ShapeItem len = value->nz_element_count();
269PyObject * result = PyList_New(len);
270
271 loop(l, len)
272 {
273 const Cell & cell = value->get_cravel(l);
274 PyObject * item = Py_None;
275 if (cell.is_integer_cell())
276 {
277 item = PyLong_FromLong(cell.get_int_value());
278 }
279 else if (cell.is_float_cell())
280 {
281 item = PyFloat_FromDouble(cell.get_real_value());
282 }
283 else
284 {
285 CERR << "*** Warning: unsupported Cell type" << endl;
286 }
287 PyList_SetItem(result, l, item);
288 }
289
290 return result;
291}

311//----------------------------------------------------------------------------
312static Shape
313list_to_shape(PyObject * shape)
314{
315Shape ret;
316 if (!PyList_Check(shape))
317 {
318 CERR << "*** tuple[1] is not a list at " LOC << endl;
319 return ret;
320 }
321
322const uRank rank = PyList_Size(shape);
323 loop(r, rank)
324 {
325 PyObject * axis = PyList_GetItem(shape, r);
326 ret.add_shape_item(PyLong_AsLong(axis));
327 }
328 return ret;
329}
330//----------------------------------------------------------------------------
331Shape
332shape_for_item(PyObject * item)
333{
334 // compute the shape of a given item. The shape is:
335 //
336 // [ len(list) ] for python lists, and
337 // shape for tuples (ravel, shape)
338 // [] otherwise (primitive python types)
339 //
340 if (PyList_Check(item)) return Shape(PyList_Size(item));
341 if (!PyTuple_Check(item)) return Shape();
342 if (PyTuple_Size(item) != 2)
343 {
344 CERR << "*** tuple len ≠ 2 at " LOC << endl;
345 return Shape();
346 }
347
348PyObject * shape = PyTuple_GetItem(item, 1);
349 return list_to_shape(shape);
350}
351//----------------------------------------------------------------------------
352static Value_P python_to_apl(PyObject * ravel, PyObject * shape);
353
354static Value_P
355python_to_apl(PyObject * ravel, const Shape & shape)
356{
357Value_P Z(shape, LOC);
358const ShapeItem len_Z = Z->nz_element_count();
359
360 if (PyComplex_Check(ravel)) // ravel is a complex scalar
361 {
362 const double real = PyComplex_RealAsDouble(ravel);
363 const double imag = PyComplex_ImagAsDouble(ravel);
364 loop(z, len_Z) Z->next_ravel_Complex(real, imag);
365 }
366 else if (PyFloat_Check(ravel)) // ravel is a floating point scalar
367 {
368 const double real = PyFloat_AsDouble(ravel);
369 loop(z, len_Z) Z->next_ravel_Float(real);
370 }
371 else if (PyLong_Check(ravel)) // ravel is an integer scalar
372 {
373 const APL_Integer ival = PyLong_AsLong(ravel);
374 loop(z, len_Z) Z->next_ravel_Int(ival);
375 }
376 else if (PyUnicode_Check(ravel)) // ravel is a char scalar or a string
377 {
378 Py_ssize_t len = 0;
379 const UTF8 * data = reinterpret_cast<const UTF8 * >
380 (PyUnicode_AsUTF8AndSize(ravel, &len));
381 UTF8_string utf(data, len);
382 UCS_string ucs(utf);
383 loop(z, len_Z) Z->next_ravel_Char(ucs[z % ucs.size()]);
384 }
385 else if (PyTuple_Check(ravel)) // item is a tuple (val, shape)
386 {
387 PyObject * sub_shape = PyTuple_GetItem(ravel, 1);
388 PyObject * sub_ravel = PyTuple_GetItem(ravel, 0);
389 Value_P sub = python_to_apl(sub_ravel, sub_shape);
390 loop(z, len_Z) Z->next_ravel_Pointer(sub.get());
391 }
392 else if (PyList_Check(ravel)) // ravel is a list
393 {
394 const ShapeItem src_len = PyList_Size(ravel);
395 loop(z, len_Z)
396 {
397 PyObject * src = PyList_GetItem(ravel, z % src_len);
398 Value_P sub = python_to_apl(src, 0);
399 Z->next_ravel_Value(sub.get());
400 }
401 }
402 else
403 {
404 CERR << "*** " << "ravel is something else" << endl;
405 FIXME;
406 }
407
408 Z->check_value(LOC);
409 return Z;
410}
411//----------------------------------------------------------------------------
412static Value_P
413python_to_apl(PyObject * ravel, PyObject * shape)
414{
415 if (shape) return python_to_apl(ravel, list_to_shape(shape));
416 else return python_to_apl(ravel, shape_for_item(ravel));
417}
418//----------------------------------------------------------------------------
419static PyObject *
420apl_set_value(PyObject * self, PyObject * args)
421{
422const char * varname = 0;
423PyObject * ravel = 0;
424PyObject * shape = 0; // optional
425
426 if (!PyTuple_Check(args)) return 0; // never
427
428const int arg_count = PyTuple_Size(args);
429 if (arg_count == 2) // set_value(varname, ravel)
430 {
431 if (!PyArg_ParseTuple(args, "sO", &varname, &ravel))
432 {
433 CERR << "*** Bad arguments in set_value(varname, ravel) ."
434 << endl;
435 return 0;
436 }
437 }
438 else if (arg_count == 3) // set_value(varname, ravel, shape)
439 {
440 if (!PyArg_ParseTuple(args, "sOO", &varname, &ravel, shape))
441 {
442 CERR << "*** Bad arguments in set_value(varname, ravel, shape) ."
443 << endl;
444 return 0;
445 }
446 }
447 else if (arg_count > 3)
448 {
449 CERR << "*** Too many (" << arg_count << ") arguments in "
450 "set_value(varname, ravel, [shape]) ." << endl;
451 return 0;
452 }
453 else
454 {
455 CERR << "*** Too few (" << arg_count << ") arguments in "
456 "set_value(varname, ravel, [shape]) ." << endl;
457 return 0;
458 }
459
460 init_if_necessary();
461
462UTF8_string varname_utf8(varname);
463UCS_string varname_ucs(varname_utf8);
464Symbol * sym = Workspace::lookup_symbol(varname_ucs);
465 if (sym == 0)
466 {
467 CERR << "*** " << varname_ucs << "cannot be assigned." << endl;
468 return 0;
469 }
470
471Value_P value = python_to_apl(ravel, shape);
472 if (!value)
473 {
474 CERR << "*** " << ravel << "could not be assigned." << endl;
475 return 0;
476 }
477
478 sym->assign(value, true, LOC);
479 return Py_None;
480}

482static PyObject *
483apl_fix_function(PyObject * self, PyObject * args)
484{
485const char * text = 0;
486 if (!PyArg_ParseTuple(args, "s", &text))
487 {
488 CERR << "*** argument of fix_function() is not a string." << endl;
489 return 0;
490 }
491
492 init_if_necessary();
493
494const UTF8_string text_utf8(text);
495const UCS_string text_ucs(text_utf8);
496int error_line = 0;
497
498const UTF8_string creator("python");
499UserFunction * fun = UserFunction::fix(text_ucs, error_line, false, LOC,
500 creator, true);
501
502 if (fun) return Py_None;
503
504 CERR << "*** invalid string argument of fix_function()" << endl
505 << "*** offending function line: " << error_line << endl;
506
507 return PyLong_FromLong(error_line);
508}

511apl_set_display(PyObject * self, PyObject * args)

524 return 0;

526
527const int oldmode = display_mode;
528 display_mode = mode;
529 return PyLong_FromLong(oldmode);
530}

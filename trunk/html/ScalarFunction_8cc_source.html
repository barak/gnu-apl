126//----------------------------------------------------------------------------
127Token
128ScalarFunction::eval_scalar_B(Value_P B, prim_f1 fun) const
129{
130const ShapeItem len_Z = B->element_count();
131 if (len_Z == 0) return do_eval_fill_B(B);
132
133PERFORMANCE_START(start)
134
135ErrorCode ec = E_NO_ERROR;
136Value_P Z = do_scalar_B(ec, B, fun); // sets ec
137 if (ec != E_NO_ERROR)
138 {
139 Thread_context::cancel_all_monadic_jobs();
140 throw_apl_error(ec, LOC);
141 }
142
143PERFORMANCE_END(fs_SCALAR_B, start, Z->nz_element_count());
144
145 loop(a, Thread_context::get_active_core_count())
146 Assert(Thread_context::get_context(CoreNumber(a))
147 ->joblist_B.get_size() == 0);
148
149 return Token(TOK_APL_VALUE1, Z);
150}
151//----------------------------------------------------------------------------
152Value_P
153ScalarFunction::do_scalar_B(ErrorCode & ec, Value_P B, prim_f1 fun) const
154{
155Value_P Z(B->get_shape(), LOC);
156
157 // create a worklist with one top-level item job_B that computes Z.
158 // If nested values are detected while computing Z, then new jobs for
159 // them are added to the worklist.
160 //
161 {
162 const PJob_scalar_B job_B(Z, B);
163 Thread_context::get_master().joblist_B.start(job_B, LOC);
164 }
165
166#if PARALLEL_ENABLED
167const bool maybe_parallel = Parallel::run_parallel &&
168 may_parallel() &&
169 Thread_context::get_active_core_count() > 1;
170#endif // PARALLEL_ENABLED
171
172 for (;;)
173 {
174 loop(a, Thread_context::get_active_core_count())
175 {
176 job_B = Thread_context::get_context(CoreNumber(a))
177 ->joblist_B.next_job();
178 if (job_B) break;
179 }
180 if (job_B == 0) break; // all jobs done
181
182#if PARALLEL_ENABLED
183 if (maybe_parallel && job_B->len_Z > get_monadic_threshold())
184 {
185 // parallel execution...
186 //
187 job_B->fun = this;
188 job_B->fun1 = fun;
189 Thread_context::do_work = PF_scalar_B;
190 Thread_context::M_fork("eval_scalar_B"); // start pool
191 PF_scalar_B(Thread_context::get_master());
192PERFORMANCE_START(start_M_join)
193 Thread_context::M_join();
194 if (job_B->error != E_NO_ERROR)
195 {
196 ec = job_B->error;
197 return Value_P();
198 }
199PERFORMANCE_END(fs_M_join_B, start_M_join, 1);
200 }
201 else
202#endif // PARALLEL_ENABLED
203 {
204 // sequential execution...
205 //
206 loop(z, job_B->len_Z)
207 {
208 const Cell & cell_B = job_B->B_at(z);
209 Cell & cell_Z = job_B->Z_at(z);
210
211 if (cell_B.is_pointer_cell()) // nested B-item
212 {
213 Value_P B1 = cell_B.get_pointer_value();
214 Value_P Z1(B1->get_shape(), LOC);
215 new (&cell_Z) PointerCell(Z1.get(), *job_B->value_Z);
216
217 const PJob_scalar_B j1(Z1, B1);
218 Thread_context::get_master().joblist_B.add_job(j1);
219 }
220 else // simple B-item
221 {
222PERFORMANCE_START(start_2)
223 ec = (cell_B.*fun)(&cell_Z);
224 if (ec != E_NO_ERROR)
225 {
226 job_B->~PJob_scalar_B(); // ownership of B, and Z
227 job_B = 0;
228 return Value_P();
229 }
230CELL_PERFORMANCE_END(get_statistics_B(), start_2, z)
231 }
232 }
233 }
234 job_B->value_Z->check_value(LOC);
235 job_B->~PJob_scalar_B(); // give up ownership of B, and Z.
236 job_B = 0;
237 }
238
239 Z->check_value(LOC);
240
241 return Z;
242}
243//----------------------------------------------------------------------------
244void
245ScalarFunction::PF_scalar_B(Thread_context & tctx)
246{
247#if PARALLEL_ENABLED
248
249 /*
250 the following functions are not thread-safe and must therefore be locked.
251 We use different locks so that independent items do not lock each other.
252
253 * DynamicObject (and therefore Value() and Value_P() constructors
254 * PJob_scalar_AB::add_job(), PJob_scalar_B::add_job()
255 * PointerCell constructors (change their parents)
256 */
257
258const int cores = Thread_context::get_active_core_count();
259
260const ShapeItem slice_len = (job_B->len_Z + cores - 1) / cores;
261ShapeItem z = tctx.get_N() * slice_len;
262ShapeItem end_z = z + slice_len;
263 if (end_z > job_B->len_Z) end_z = job_B->len_Z;
264
265 for (; z < end_z; ++z)
266 {
267 const Cell & cell_B = job_B->B_at(z);
268 Cell & cell_Z = job_B->Z_at(z);
269
270 const bool scalar_B = ! cell_B.is_pointer_cell();
271
272 if (scalar_B)
273 {
274PERFORMANCE_START(start_2)
275
276 job_B->error = (cell_B.*job_B->fun1)(&cell_Z);
277 if (job_B->error != E_NO_ERROR) return;
278
279CELL_PERFORMANCE_END(job_B->fun->get_statistics_B(), start_2, z)
280 }
281 else // B is nested
282 {
283 Parallel::acquire_lock(jobs_lock);
284
285 // B is nested
286 //
287 Value_P B1 = cell_B.get_pointer_value();
288
289 if (const ShapeItem len_Z1 = B1->element_count())
290 {
291 Value_P Z1(B1->get_shape(), LOC);
292 new (&cell_Z) PointerCell(Z1.get(), *job_B->value_Z);
293
294 const PJob_scalar_B j1(Z1, B1);
295 tctx.joblist_B.add_job(j1);
296 }
297 else
298 {
299 Value_P Z1 = CLONE_P(B1, LOC);
300 Z1->to_type(/* numeric */ true);
301 new (&cell_Z) PointerCell(Z1.get(), *job_B->value_Z);
302 }
303 Parallel::release_lock(jobs_lock);
304 }
305 }
306#endif // PARALLEL_ENABLED
307}
308//----------------------------------------------------------------------------
309void
310ScalarFunction::expand_nested(Value * Z, const Cell * cell_A,
311 const Cell * cell_B, prim_f2 fun) const
312{
313 if (cell_A->is_pointer_cell())
314 {
315 if (cell_B->is_pointer_cell()) // nested A and nested B
316 {
317 Value_P value_A = cell_A->get_pointer_value();
318 Value_P value_B = cell_B->get_pointer_value();
319 Token token = eval_scalar_AB(value_A, value_B, fun);
320 Z->next_ravel_Pointer(token.get_apl_val().get());
321 }
322 else // nested A and simple B
323 {
324 Value_P value_A = cell_A->get_pointer_value();
325 Value_P scalar_B(*cell_B, LOC);
326 Token token = eval_scalar_AB(value_A, scalar_B, fun);
327 Z->next_ravel_Pointer(token.get_apl_val().get());
328 }
329 }
330 else // A is simple
331 {
332 if (cell_B->is_pointer_cell()) // simple A and nested B
333 {
334 Value_P scalar_A(*cell_A, LOC);
335 Value_P value_B = cell_B->get_pointer_value();
336 Token token = eval_scalar_AB(scalar_A, value_B, fun);
337 Z->next_ravel_Pointer(token.get_apl_val().get());
338 }
339 else // simple A and simple B
340 {
341 const ShapeItem pos = Z->get_valid_item_count();
342 Z->next_ravel_0(); // pre-init with 0
343 (cell_B->*fun)(&Z->get_wravel(pos), cell_A);
344 }
345 }

369//----------------------------------------------------------------------------
370Value_P
371ScalarFunction::do_scalar_AB(ErrorCode & ec, Value_P A, Value_P B,
372 prim_f2 fun) const
373{
374const Shape * shape_Z = conforming_shape(ec, A->get_shape(), B->get_shape());
375 if (ec) return Value_P();
376
377const ShapeItem len_Z = shape_Z->get_volume();
378 if (len_Z == 0) return do_eval_fill_AB(A, B).get_apl_val();
379
380Value_P Z(*shape_Z, LOC);
381
382 // create a worklist with one top-level item that computes Z. If nested
383 // values are detected while computing Z, then jobs for them are added
384 // to the worklist.
385 //
386 {
387 const PJob_scalar_AB job(Z, A, B);
388 Thread_context::get_master().joblist_AB.start(job, LOC);
389 }
390
391#if PARALLEL_ENABLED
392const bool maybe_parallel = Parallel::run_parallel &&
393 Thread_context::get_active_core_count() > 1;
394#endif
395
396 for (;;)
397 {
398 loop(a, Thread_context::get_active_core_count())
399 {
400 job_AB = Thread_context::get_context(CoreNumber(a))
401 ->joblist_AB.next_job();
402 if (job_AB) break;
403 }
404 if (job_AB == 0) break; // all jobs done
405
406#if PARALLEL_ENABLED
407 if (maybe_parallel && job_AB->len_Z > get_dyadic_threshold())
408 {
409 // parallel execution...
410 //
411 job_AB->fun = this;
412 job_AB->fun2 = fun;
413 Thread_context::do_work = PF_scalar_AB;
414 Thread_context::M_fork("eval_scalar_AB"); // start pool
415 PF_scalar_AB(Thread_context::get_master());
416PERFORMANCE_START(start_M_join)
417 Thread_context::M_join();
418 ec = job_AB->error;
419 if (ec != E_NO_ERROR) return Value_P();
420PERFORMANCE_END(fs_M_join_AB, start_M_join, 1);
421 }
422 else
423#endif // PARALLEL_ENABLED
424 {
425 // sequential execution...
426 //
427 loop(z, job_AB->len_Z)
428 {
429 const Cell & cell_A = job_AB->A_at(z);
430 const Cell & cell_B = job_AB->B_at(z);
431 Cell & cell_Z = job_AB->Z_at(z);
432
433 if (cell_A.is_pointer_cell() && cell_B.is_pointer_cell())
434 {
435 /* both cell_A and cell_B are nested, pointing to
436 Values A1 and B1 respectively.
437 cell_Z is nested, pointing to Z1←A1 fun B1.
438 */
439 Value_P A1 = cell_A.get_pointer_value();
440 Value_P B1 = cell_B.get_pointer_value();
441 const Shape * sh_Z1 = conforming_shape(ec,
442 B1->get_shape(),
443 A1->get_shape());
444 if (ec) return Value_P();
445
446 const ShapeItem len_Z1 = sh_Z1->get_volume();
447 if (len_Z1 == 0)
448 {
449 Value_P Z1 = do_eval_fill_AB(A1, B1).get_apl_val();
450 job_AB->value_Z->next_ravel_Pointer(Z1.get());
451 continue;
452 }
453
454 Value_P Z1(*sh_Z1, LOC);
455 new (&cell_Z)
456 PointerCell(Z1.get(), *job_AB->value_Z, 0x6B616769);
457
458 const PJob_scalar_AB j1(Z1, A1, B1);
459 Thread_context::get_master().joblist_AB.add_job(j1);
460 }
461 else if (cell_A.is_pointer_cell())
462 {
463 /* cell_A is nested, pointing to Values A1.
464 cell_Z is nested, pointing to Z1←A1 fun B where B
465 is a scalar according to cell_B.
466 */
467 Value_P A1 = cell_A.get_pointer_value();
468 Value_P B1(cell_B, LOC);
469
470 const ShapeItem len_Z1 = A1->element_count();
471 if (len_Z1 == 0)
472 {
473 Value_P Z1 = do_eval_fill_B(A1).get_apl_val();
474 new (&cell_Z) PointerCell(Z1.get(),
475 *job_AB->value_Z);
476 }
477 else
478 {
479 Value_P Z1(A1->get_shape(), LOC);
480 if (!Z1) WS_FULL;
481 new (&cell_Z)
482 PointerCell(Z1.get(),*job_AB->value_Z,
483 0x6B616769);
484
485 const PJob_scalar_AB j1(Z1, A1, B1);
486 Thread_context::get_master().joblist_AB.add_job(j1);
487 }
488 }
489 else if (cell_B.is_pointer_cell())
490 {
491 /* cell_B is nested, pointing to Values B1.
492 cell_Z is nested, pointing to Z1←A fun B1 where A
493 is a scalar according to cell_A.
494 */
495 // B is nested, A is simple
496 //
497 Value_P B1 = cell_B.get_pointer_value();
498
499 const ShapeItem len_Z1 = B1->element_count();
500 if (len_Z1 == 0)
501 {
502 Value_P Z1 =
503 do_eval_fill_B(B1).get_apl_val();
504 new (&cell_Z) PointerCell(Z1.get(),
505 *job_AB->value_Z);
506 }
507 else
508 {
509 Value_P A1(cell_A, LOC);
510 Value_P Z1(B1->get_shape(), LOC);
511
512 new (&cell_Z)
513 PointerCell(Z1.get(), *job_AB->value_Z,
514 0x6B616769);
515
516 const PJob_scalar_AB j1(Z1, A1, B1);
517 Thread_context::get_master().joblist_AB.add_job(j1);
518 }
519 }
520 else
521 {
522 /* cell_A and cell_B are both simple.
523 Compute cell_Z = cell_A fun cell_B
524 */
525PERFORMANCE_START(start_2)
526
527 ec = (cell_B.*fun)(&cell_Z, &cell_A);
528 if (ec != E_NO_ERROR)
529 {
530 job_AB->~PJob_scalar_AB(); // ownership of A, B, Z.
531 job_AB = 0;
532 return Value_P();
533 }
534CELL_PERFORMANCE_END(get_statistics_AB(), start_2, z)
535 }
536 }
537 }
538 job_AB->value_Z->check_value(LOC);
539 job_AB->~PJob_scalar_AB(); // give up ownership of A, B, and Z.
540 job_AB = 0;
541 }
542
543 Z->set_default(*B, LOC);
544 Z->check_value(LOC);
545
546 return Z;
547}
548//----------------------------------------------------------------------------
549void
550ScalarFunction::PF_scalar_AB(Thread_context & tctx)
551{
552#if PARALLEL_ENABLED
553
554 /*
555 the following functions are not thread-safe and must therefore be locked.
556 We use different locks so that independent items do not lock each other.
557
558 * DynamicObject (and therefore Value() and Value_P() constructors
559 * PJob_scalar_AB::add_job(), PJob_scalar_B::add_job()
560 * PointerCell constructors (change their parents)
561 */
562
563const CoreCount cores = Thread_context::get_active_core_count();
564
565const ShapeItem slice_len = (job_AB->len_Z + cores - 1) / cores;
566ShapeItem z = tctx.get_N() * slice_len;
567
568ShapeItem end_z = z + slice_len;
569 if (end_z > job_AB->len_Z) end_z = job_AB->len_Z;
570
571 for (; z < end_z; ++z)
572 {
573 const Cell & cell_A = job_AB->A_at(z);
574 const Cell & cell_B = job_AB->B_at(z);
575 Cell & cell_Z = job_AB->Z_at(z);
576
577 const bool scalar_A = ! cell_A.is_pointer_cell();
578 const bool scalar_B = ! cell_B.is_pointer_cell();
579
580 if (scalar_A && scalar_B) // simple A and simple B
581 {
582PERFORMANCE_START(start_2)
583
584 job_AB->error = (cell_B.*job_AB->fun2)(&cell_Z, &cell_A);
585 if (job_AB->error != E_NO_ERROR) return;
586
587CELL_PERFORMANCE_END(job_AB->fun->get_statistics_AB(), start_2, z)
588 }
589 else // nested A and/or nested B
590 {
591 Parallel::acquire_lock(jobs_lock);
592 if (scalar_B) // nested A and simple B
593 {
594 Value_P A1 = cell_A.get_pointer_value();

597 {
598 Value_P Z1(A1->get_shape(), LOC);

600
601 const PJob_scalar_AB j1(Z1, A1, B1);
602 tctx.joblist_AB.add_job(j1);
603 }
604 else // empty A1 and simple B
605 {
606 Value_P Z1 = job_AB->fun->eval_fill_AB(A1, B1)
607 .get_apl_val();
608 new (&cell_Z) PointerCell(Z1.get(), *job_AB->value_Z);
609 }
610 }
611 else if (scalar_A) // simple A and nested B
612 {
613 Value_P A1(cell_A, LOC);
614 Value_P B1 = cell_B.get_pointer_value();
615 if (const ShapeItem len_Z1 = B1->element_count())
616 {
617 Value_P Z1(B1->get_shape(), LOC);
618 new (&cell_Z) PointerCell(Z1.get(), *job_AB->value_Z);
619
620 const PJob_scalar_AB j1(Z1, A1, B1);
621 tctx.joblist_AB.add_job(j1);
622 }
623 else // simple A and empty B1
624 {
625 Value_P Z1 = job_AB->fun->eval_fill_AB(A1, B1)
626 .get_apl_val();
627 new (&cell_Z) PointerCell(Z1.get(), *job_AB->value_Z);
628 }
629 }
630 else // nested A and nested B
631 {
632 Value_P A1(cell_A, LOC);
633 Value_P B1 = cell_B.get_pointer_value();
634 const Shape * sh_Z1 = conforming_shape(job_AB->error,
635 B1->get_shape(),
636 A1->get_shape());
637 if (job_AB->error)
638 {
639 Parallel::release_lock(jobs_lock);
640 return;
641 }
642
643 if (const ShapeItem len_Z1 = sh_Z1->get_volume())
644 {
645 Value_P Z1(*sh_Z1, LOC);
646 new (&cell_Z) PointerCell(Z1.get(), *job_AB->value_Z);
647
648 const PJob_scalar_AB j1(Z1, A1, B1);
649 tctx.joblist_AB.add_job(j1);
650 }
651 else // empty B1/Z1
652 {
653 Token result = job_AB->fun->eval_fill_AB(A1, B1);

655 {
656 job_AB->error = ErrorCode(result.get_int_val());
657 Parallel::release_lock(jobs_lock);
658 return;
659 }
660 }
661 }
662
663 Parallel::release_lock(jobs_lock);
664 }
665 }

667#endif // PARALLEL_ENABLED

669//----------------------------------------------------------------------------
670Token
671ScalarFunction::do_eval_fill_AB(Value_P A, Value_P B) const
672{
673 /* eval_fill_AB() is called when A and/or B is empty and the non-empty
674 argument (if any) is non-scalar. The scalar case for the non-empty
675 argument of a dyadic scalar function is handled by do_eval_fill_B().
676
677 NOTE however, that even though non-empty arguments of scalar functions
678 must be scalars (and are therefore are not handled here but instead in
679 do_eval_fill_B() below) A or B may still be non-scalar and non-empty
680 when called from other places (such as Bif_OPER2_INNER::fill()).
681
682 NOTE also that there seems to be some confusion in lrm regarding
683 fill functions. On p. 56 a description and examples for the fill
684 function is given and IBM APL2 seem to follow that description.
685
686 On p. 110/Figure 20 of lrm a different definition for the fill
687 function for scalar functions is given, i.e.:
688
689 Z←(R) ≠ (L) with R←↑A and L←↑B (*)
690
691 The example on page 56 of lrm, i.e.:
692
693 W←(ι0) ⌈ ι0
694 DISPALY W
695
696 gives:
697
698 ┌⊖┐
699 │0│
700 └─┘
701
702 in both APL2 (PC version) and in GNU APL while
703
704 (↑⍳0) ≠ (↑⍳0) (according to the definition on lrm p. 110/Figure 20)
705
706 gives the simple numeric scalar
707
708 0
709
710 in both APL2 (PC version) and in GNU APL. Therefore the definition
711 given on p. 110/Figure 20 looks wrong.
712 */
713
714 if (B->element_count() == 0) return do_eval_fill_B(B);
715 if (A->element_count() == 0) return do_eval_fill_B(A);
716 return do_eval_fill_B(A);
717}
718//----------------------------------------------------------------------------
719Token
720ScalarFunction::do_eval_fill_B(Value_P B) const
721{
722 /* eval_fill_B() is called for:
723
724 1. a monadic scalar function with empty B, or
725 2a. a dyadic (!) scalar function with empty A and scalar B, or
726 2b. a dyadic (!) scalar function with scalar A and empty B.
727 */
728
729 // lrm p. 56: When the prototypes of the empty arguments are simple
730 // scalars, return a zero prototype
731 //
732 if (B->get_cfirst().is_numeric() || B->get_cfirst().is_character_cell())
733 {
734 Value_P Z(B->get_shape(), LOC);
735 Z->check_value(LOC);
736 return Token(TOK_APL_VALUE1, Z);
737 }
738
739 /* Apply do_eval_fill_B() recursively. Value::prototype() does not work
740 here, so we clone() and to_type(true) where true forces numeric 0
741 for character Cells.
742 */
743Value_P Z = B->clone(LOC);
744 Z->to_type(/* numeric */ true);
745 Z->check_value(LOC);
746
747 return Token(TOK_APL_VALUE1, Z);
748}

780//----------------------------------------------------------------------------
781Token
782ScalarFunction::eval_scalar_AXB(Value_P A, Value_P X, Value_P B,
783 prim_f2 fun) const
784{
785PERFORMANCE_START(start_1)
786
787 {
788 int sec = 0;
789 if (A->is_scalar_extensible()) ++sec;
790 if (B->is_scalar_extensible()) ++sec;
791
792 /* avoid a conflict between scalar extension and axis of 1-element
793 values with different ranks, e.g.
794
795 (1 1⍴'A') = [1] (1⍴'B')
796
797 which could lead to the wrong (smaller rank) shape of the result
798 */
799 if (sec == 1 ||
800 (sec == 2 && A->same_rank(*B)) // ← conflict if ranks differ
801 ) return eval_scalar_AB(A, B, fun);
802 }
803
804 if (X->get_rank() > 1) AXIS_ERROR;
805
806const APL_Integer qio = Workspace::get_IO();
807const sRank rank_A = A->get_rank();
808const sRank rank_B = B->get_rank();
809AxesBitmap axes_X = 0; // bitmap of axes in X
810const ShapeItem len_X = X->element_count();
811
812 loop(iX, len_X)
813 {
814 APL_Integer i = X->get_cravel(iX).get_near_int() - qio;
815 if (i < 0) AXIS_ERROR; // axis i too small
816 if (i >= rank_A && i >= rank_B) AXIS_ERROR; // axis i too large
817 if (axes_X & 1 << i) AXIS_ERROR; // axis i used twice
818 axes_X |= 1 << i;
819 }
820
821 // if A and B have the same rank, then all axes must be in and A f[X} B
822 // is the same as A f B.
823 //
824 if (rank_A == rank_B)
825 {
826 if (rank_A != len_X) AXIS_ERROR;
827 return eval_scalar_AB(A, B, fun);
828 }
829
830 if (rank_A < rank_B)
831 {
832 if (rank_A != len_X) AXIS_ERROR;
833
834 Value_P Z = eval_scalar_AXB(A, axes_X, B, fun, false);
835PERFORMANCE_END(fs_SCALAR_AB, start_1, Z->nz_element_count())
836 return Token(TOK_APL_VALUE1, Z);
837 }
838 else
839 {
840 if (rank_B != len_X) AXIS_ERROR;
841
842 Value_P Z = eval_scalar_AXB(B, axes_X, A, fun, true);
843PERFORMANCE_END(fs_SCALAR_AB, start_1, Z->nz_element_count())
844 return Token(TOK_APL_VALUE1, Z);
845 }
846}
847//----------------------------------------------------------------------------
848Value_P
849ScalarFunction::eval_scalar_AXB(Value_P A, AxesBitmap axes_X, Value_P B,
850 prim_f2 fun, bool reversed) const
851{
852 // A is the value with the smaller rank (possibly a scalar).
853 // B the value with the larger rank (neve a scalar).
854 //
855 // If (reversed) then A and B have changed roles (in order to
856 // make A the value with the smaller rank).
857
858 // 1. check that A and B agree on the axes in X (common axes of A and B
859 // nust have the same length).
860 //
861 {
862 sRank rA = 0;
863 loop(rB, B->get_rank())
864 {
865 if (axes_X & 1 << rB)
866 {
867 // if the axis is in X then the corresponding shape items in
868 // A and B must agree.
869 //
870 if (B->get_shape_item(rB) != A->get_shape_item(rA++))
871 LENGTH_ERROR;
872 }
873 }
874 }
875
876Shape weights_A = A->get_shape().get_weights();
877
878Value_P Z(B->get_shape(), LOC);
879
880const Cell * cB = &B->get_cfirst();
881
882 for (ArrayIterator it_B(B->get_shape()); it_B.more(); ++it_B)
883 {
884 ShapeItem wA = 0; // weigth of the A axes in X
885 sRank rA = 0;
886 loop(rB, B->get_rank())
887 {
888 if (axes_X & 1 << rB)
889 {
890 wA += weights_A.get_shape_item(rA++)
891 * it_B.get_shape_offset(rB);
892 }
893 }
894
895 const Cell * cA = &A->get_cravel(wA);
896
897 // restore the original order of A and B
898 //
899 if (reversed) expand_nested(Z.get(), cB++, cA, fun);
900 else expand_nested(Z.get(), cA, cB++, fun);
901 }
902
903 Z->set_default(*B, LOC);
904 Z->check_value(LOC);
905 return Z;
906}
907//---------------------------------------------------------------------------
908const Shape *
909ScalarFunction::conforming_shape(ErrorCode & ec, const Shape & shape_A,
910 const Shape & shape_B)
911{
912 // scalar shape_A or shape_B. If one is scalar and the other is 1-element
913 // then 1-element wins.
914 if (shape_A.get_rank() == 0) return &shape_B;
915 if (shape_B.get_rank() == 0) return &shape_A;
916
917 // 1-element shape_A or shape_B
918 if (shape_A.get_volume() == 1) return &shape_B;
919 if (shape_B.get_volume() == 1) return &shape_A;
920
921 if (shape_A == shape_B) return &shape_A;
922
923 if (shape_A.get_rank() != shape_B.get_rank()) ec = E_RANK_ERROR;
924 else ec = E_LENGTH_ERROR;
925 return 0;

928Token
929Bif_F2_FIND::eval_AB(Value_P A, Value_P B) const
930{
931PERFORMANCE_START(start_1)
932
933const double qct = Workspace::get_CT();
934Value_P Z(B->get_shape(), LOC);
935Shape shape_A;
936
937const ShapeItem len_Z = Z->element_count();
938
939 if (A->get_rank() > B->get_rank()) // then Z is all zeros.
940 {
941 loop(z, len_Z) Z->next_ravel_0();
942 goto done;
943 }
944
945 // Reshape A to match rank B if necessary...
946 //
947 {
948 const sRank rank_diff = B->get_rank() - A->get_rank();
949 loop(d, rank_diff) shape_A.add_shape_item(1);
950 loop(r, A->get_rank()) shape_A.add_shape_item(A->get_shape_item(r));
951 }
952
953 // if any dimension of A is longer than that of B, then A cannot be found.
954 //
955 loop(r, B->get_rank())
956 {
957 if (shape_A.get_shape_item(r) > B->get_shape_item(r))
958 {
959 loop(z, len_Z) Z->next_ravel_0();
960 goto done;
961 }
962 }
963
964 for (ArrayIterator zi(B->get_shape()); zi.more(); ++zi)
965 {
966PERFORMANCE_START(start_2)
967 if (contained(shape_A, &A->get_cfirst(), B, zi.get_shape_offsets(), qct))
968 Z->next_ravel_1();
969 else
970 Z->next_ravel_0();
971
972CELL_PERFORMANCE_END(get_statistics_AB(), start_2, B->get_shape().get_volume())
973 }
974
975done:
976 Z->check_value(LOC);
977
978PERFORMANCE_END(fs_SCALAR_AB, start_1, len_Z);
979 return Token(TOK_APL_VALUE1, Z);
980}
981//============================================================================
982bool
983Bif_F2_FIND::contained(const Shape & shape_A, const Cell * cA,
984 Value_P B, const Shape & idx_B, double qct)
985{
986 /* quick check (along each axis): before comparing any ravel elements,
987 we check that A, when offset by idx_B, fits into B:
988
989 ├─── idx B ───┤├────── A ──────┤
990 ├───────────────── B ─────────────────┤
991
992
993 */
994 loop(r, B->get_rank())
995 {
996 if ((idx_B.get_shape_item(r) + shape_A.get_shape_item(r))
997 > B->get_shape_item(r)) return false;
998 }
999
1000const Shape weights_B = B->get_shape().get_weights();
1001
1002 for (ArrayIterator ai(shape_A); ai.more(); ++ai)
1003 {
1004 const Shape & pos_A = ai.get_shape_offsets();
1005 ShapeItem pos_B = 0;
1006 loop(r, B->get_rank()) pos_B += weights_B.get_shape_item(r)
1007 * (idx_B.get_shape_item(r)
1008 + pos_A.get_shape_item(r));
1009
1010 if (!cA[ai.get_ravel_offset()].equal(B->get_cravel(pos_B), qct))
1011 return false;
1012 }
1013
1014 return true;
1015}

1017Token
1018Bif_F12_ROLL::eval_AB(Value_P A, Value_P B) const
1019{
1020 // draw A items from the set { ⎕IO ... ⎕IO+ B } (i.e. without repetitions)
1021 //
1022 if (!A->is_scalar_extensible()) RANK_ERROR;
1023 if (!B->is_scalar_extensible()) RANK_ERROR;
1024
1025const ShapeItem zlen = A->get_cfirst().get_near_int();
1026APL_Integer set_size = B->get_cfirst().get_near_int();
1027 if (zlen > set_size) DOMAIN_ERROR;
1028 if (zlen < 0) DOMAIN_ERROR;
1029 if (set_size < 0) DOMAIN_ERROR;
1030 if (set_size > 0x7FFFFFFF) DOMAIN_ERROR;
1031
1032 // at this point, A and B are OK. We use a bitmap 'used' to keep track
1033 // of items alrady drawn.
1034 //
1035Value_P Z(zlen, LOC);
1036
1037 // set_size can be rather big, so we new/delete it
1038 //
1039uint8_t * used = new uint8_t[(set_size + 7)/8];
1040 if (used == 0) throw_apl_error(E_WS_FULL, LOC);
1041 memset(used, 0, (set_size + 7)/8);
1042
1043 loop(z, zlen)
1044 {
1045 const uint64_t rnd = Workspace::get_RL(set_size);
1046
1047 if (used[rnd >> 3] & 1 << (rnd & 7)) // already drawn: try again
1048 {
1049 --z;
1050 continue;
1051 }
1052 used[rnd >> 3] |= 1 << (rnd & 7); // remember rnd
1053 Z->next_ravel_Int(rnd + Workspace::get_IO());
1054 }
1055
1056 delete [] used;
1057
1058 Z->check_value(LOC);
1059 return Token(TOK_APL_VALUE1, Z);
1060}

1172//----------------------------------------------------------------------------
1173Value_P
1174Bif_F12_WITHOUT::large_eval_AB(const Value & A, const Value & B)
1175{
1176const ShapeItem len_A = A.element_count();
1177const ShapeItem len_B = B.element_count();
1178
1179 /* pack pointers to the cells of the arguments A and B and of the
1180 result Z into one big array:
1181
1182 len_A len_Z len_B
1183 ┌─────────┬─────────┬─────────┐
1184 │ cells_A │ cells_Z │ cells_B │
1185 └─────────┴─────────┴─────────┘
1186 */
1187const Cell ** cells_A = new const Cell *[2*len_A + len_B];
1188const Cell ** cells_Z = cells_A + len_A;
1189const Cell ** cells_B = cells_A + 2*len_A;
1190
1191 loop(a, len_A) cells_A[a] = &A.get_cravel(a);
1192 loop(b, len_B) cells_B[b] = &B.get_cravel(b);
1193
1194 // sort the A-cells and the B-cells ascendingly
1195 //
1196 Heapsort<const Cell *>::sort(cells_A, len_A, 0, Cell::compare_stable);
1197 Heapsort<const Cell *>::sort(cells_B, len_B, 0, Cell::compare_stable);
1198
1199 // store those cells_A pointers that are not in cells_B into cells_Z. Use
1200 // the fact that cells_A and cells_B are sorted.
1201ShapeItem len_Z = 0;
1202 {
1203 ShapeItem idx_A = 0;
1204 ShapeItem idx_B = 0;
1205 const double qct = Workspace::get_CT();
1206
1207 while (idx_A < len_A && idx_B < len_B)
1208 {
1209 const Cell & ref_A = *cells_A[idx_A];
1210 const Cell & ref_B = *cells_B[idx_B];
1211 if (ref_A.equal(ref_B, qct)) ++idx_A; // A is in B → not in Z
1212 else if (ref_A.greater(ref_B)) ++idx_B;
1213 else cells_Z[len_Z++] = cells_A[idx_A++]; // A is in Z
1214 }
1215
1216 // the rest of A is in Z
1217 //
1218 while (idx_A < len_A) cells_Z[len_Z++] = cells_A[idx_A++];
1219 }
1220
1221 // sort cells_Z by position so that the original order in A is reconstructed
1222 //
1223 Heapsort<const Cell *>::sort(cells_Z, len_Z, 0, Cell::compare_ptr);
1224
1225Value_P Z(len_Z, LOC);
1226
1227 loop(z, len_Z) Z->next_ravel_Cell(*cells_Z[z]);
1228
1229 delete[] cells_A; // incl. cells_Z and cells_B
1230
1231 Z->check_value(LOC);
1232 return Z;
1233}

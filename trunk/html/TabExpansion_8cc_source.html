32//----------------------------------------------------------------------------
33ExpandResult
34TabExpansion::expand_tab(UCS_string & user)
35{
36 // skip leading and trailing blanks
37 //
38 user.remove_leading_and_trailing_whitespaces();
39
40 if (user.size() == 0) // nothing entered yet
41 return expand_user_name(user);
42
43 if (Avec::is_first_symbol_char(user[0])) // start of user defined name
44 return expand_user_name(user);
45
46 if (user[0] == ')' || user[0] == ']') // APL command
47 return expand_APL_command(user);
48
49 return expand_distinguished_name(user);
50}
51//----------------------------------------------------------------------------
52ExpandResult
53TabExpansion::expand_user_name(UCS_string & user)
54{
55std::vector<const Symbol *> symbols = Workspace::get_all_symbols();
56
57UCS_string_vector matches;
58 loop(s, symbols.size())
59 {
60 const Symbol * sym = symbols[s];
61 if (sym->is_erased()) continue;
62
63 const UCS_string & sym_name = sym->get_name();
64 if (!sym_name.starts_with(user)) continue;
65 matches.push_back(sym_name);
66 }
67
68 if (matches.size() == 0) return ER_IGNORE; // no match
69
70 if (matches.size() > 1) // multiple names match user input
71 {
72 const int user_len = user.size();
73 user.clear();
74 return show_alternatives(user, user_len, matches);
75 }
76
77 // unique match
78 //
79 if (user.size() < matches[0].size())
80 {
81 // the name is longer than user, so we expand it.
82 //
83 user = matches[0];
84 return ER_REPLACE;
85 }
86
87 return ER_IGNORE;
88}
89//----------------------------------------------------------------------------
90ExpandResult
91TabExpansion::expand_APL_command(UCS_string & user)
92{
93ExpandHint ehint = EH_NO_PARAM;
94const char * shint = 0;
95UCS_string_vector matches;
96UCS_string cmd = user;
97UCS_string arg;
98 cmd.split_ws(arg);
99
100#define cmd_def(cmd_str, code, arg, hint) \
101 { UCS_string ustr(cmd_str); \
102 if (ustr.starts_iwith(cmd)) \
103 { matches.push_back(ustr); ehint = hint; shint = arg; } }
104#include "Command.def"
105
106 // no match was found: ignore the TAB
107 //
108 if (matches.size() == 0) return ER_IGNORE;
109
110 // if we have multiple matches but the user has provided a command
111 // argument then some matches were wrong. For example:
112 //
113 // LIB 3 matches LIB and LIBS
114 //
115 // remove wrong matches
116 //
117 if (matches.size() > 1 && arg.size() > 0)
118 {
119 again:
120 if (matches.size() > 1)
121 {
122 loop(m, matches.size())
123 {
124 if (matches[m].size() != cmd.size()) // wrong match
125 {
126 matches[m].swap(matches.back());
127 matches.pop_back();
128 goto again;
129 }
130 }
131 }
132 }
133
134 // if multiple matches were found then either expand the common part
135 // or list all matches
136 //
137 if (matches.size() > 1) // multiple commands match cmd
138 {
139 user.clear();
140 return show_alternatives(user, cmd.size(), matches);
141 }
142
143 // unique match
144 //
145 if (cmd.size() < matches[0].size())
146 {
147 // the command is longer than user, so we expand it.
148 //
149 user = matches[0];
150 if (ehint != EH_NO_PARAM) user.append(UNI_SPACE);
151 return ER_REPLACE;
152 }
153
154 if (cmd.size() == matches[0].size() &&
155 ehint != EH_NO_PARAM &&
156 arg.size() == 0 &&
157 !have_trailing_blank) // no blank entered
158 {
159 // the entire command was entered but without a blank. If the
160 // command has arguments then append a space to indicate that.
161 // Otherwiese fall throught to expand_command_arg();
162 //
163 user = matches[0];
164 user.append(UNI_SPACE);
165 return ER_REPLACE;
166 }
167
168 return expand_command_arg(user, ehint, shint, matches[0], arg);
169}

171ExpandResult
172TabExpansion::expand_command_arg(UCS_string & user,
173 ExpandHint ehint, const char * shint,
174 const UCS_string cmd, const UCS_string arg)
175{
176 switch(ehint)
177 {
178 case EH_NO_PARAM:
179 return ER_IGNORE;
180
181 case EH_oWSNAME:
182 case EH_oLIB_WSNAME:
183 case EH_oLIB_oPATH:
184 case EH_oPATH:
185 case EH_FILENAME:
186 case EH_DIR_OR_LIB:
187 case EH_WSNAME:
188 return expand_filename(user, ehint, shint, cmd, arg);
189
190 case EH_PRIMITIVE:
191 return expand_help_topics(user);
192
193 default:
194 CIN << endl;
195 CERR << cmd << " " << shint << endl;
196 return ER_AGAIN;
197 }

200ExpandResult
201TabExpansion::expand_help_topics(UCS_string & user)
202{
203 if (user.size() <= 5) return expand_help_topics(); // initial display
204
205const UCS_string help(user, 0, 6);
206UCS_string prefix(user, 5, user.size() - 5); // the name prefix
207 prefix.remove_leading_whitespaces();
208
209std::vector<const Symbol *> symbols = Workspace::get_all_symbols();
210
211UCS_string_vector matches;
212 loop(s, symbols.size())
213 {
214 const Symbol * sym = symbols[s];
215 if (sym->is_erased()) continue;
216
217 const UCS_string & sym_name = sym->get_name();
218 if (!sym_name.starts_with(prefix)) continue;
219 matches.push_back(sym_name);
220 }
221
222 if (matches.size() == 0) return ER_IGNORE; // no match
223 if (matches.size() > 1) // multiple matches
224 {
225 matches.sort();
226
227 const int common_len = compute_common_length(prefix.size(), matches);
228 if (common_len > prefix.size())
229 {
230 // all matches can be extended in a unique way
231 //
232 matches[0].resize(common_len);
233 user = help + matches[0];
234 return ER_REPLACE;
235 }
236
237 // all possible extensions are different
238 //
239 return show_alternatives(user, prefix.size(), matches);
240 }
241
242 // unique match
243 //
244 user = help + matches[0];
245 return ER_REPLACE;
246}

263ExpandResult
264TabExpansion::expand_help_topics()
265{
266 CIN << "\n";
267 CERR << "Help topics (APL primitives and user-defined names) are:" << endl;
268
269 // show APL primitives (but only once). For that Help.def must be sorted.
270 // Many primitives occur twice (once for their monadic and once for their
271 // dyadic form. We filter those duplicates out with strcmp(prim, last)
272 //
273 // We also remove some duplicates caused by different Unicodes (like ∣ and |)
274 //
275const char * last = "";
276int col = 0;
277const int max_col = Workspace::get_PW() - 4;
278
279 loop(h, HELP_count)
280 {
281 const char * prim = help_texts[h].prim;
282 if (!strcmp(prim, last)) continue;
283 last = prim; // remember it.
284 if (strchr("|~", *prim)) continue;
285 CERR << " " << prim;
286 col += 2;
287 if (col > max_col) { CERR << endl; col = 0; } \
288 }
289
290std::vector<const Symbol *> symbols = Workspace::get_all_symbols();
291
292UCS_string_vector names;
293 loop(s, symbols.size())
294 {
295 const Symbol * sym = symbols[s];
296 if (sym->is_erased()) continue;
297
298 const UCS_string & sym_name = sym->get_name();
299 names.push_back(sym_name);
300 }
301
302 names.sort();
303
304 // see if printing full names takes more than 5 lines. If so then
305 // only print first characters
306 //
307int rows = 1; // the current row
308int c1 = col;
309 loop(n, names.size())
310 {
311 const int len = 1 + names[n].size();
312 c1 += len;
313 if (c1 > max_col) { ++rows; c1 = len; }
314 }
315
316 if (names.size() == 0) /* nothing to do */;
317 else if (rows < 6)
318 {
319 loop(n, names.size())
320 {
321 const int len = 1 + names[n].size();
322 col += len;
323 if (col > max_col) { CERR << endl; col = len; }
324 CERR << " " << names[n];
325 }
326 }
327 else
328 {
329 Unicode uni = names[0][0];
330 CERR << " " << uni;
331 for (ShapeItem n = 1; n < ShapeItem(names.size()); ++n)
332 {
333 if (names[n][0] == uni) continue; // same first character
334 uni = names[n][0];
335 CERR << " " << uni; col += 2;
336 if (col > max_col) { CERR << endl; col = 0; }
337 }
338 }
339
340 CERR << endl;
341 return ER_AGAIN;
342}

345ExpandResult
346TabExpansion::expand_distinguished_name(UCS_string & user)
347{
348 // figure the length of longest ⎕xxx name (probably ⎕TRACE == 5)
349 //
350unsigned int max_e = 2;
351#define ro_sv_def(_q, str, _txt) if (max_e < strlen(str)) max_e = strlen(str);
352#define rw_sv_def(_q, str, _txt) if (max_e < strlen(str)) max_e = strlen(str);
353#define sf_def(_q, str, _txt) if (max_e < strlen(str)) max_e = strlen(str);
354#include "SystemVariable.def"
355
356 // Search for ⎕ backwards from the end
357 //
358int qpos = -1;
359
360 loop(e, max_e)
361 {
362 if (e >= user.size()) break;
363 if (user[user.size() - e - 1] == UNI_Quad_Quad)
364 {
365 qpos = user.size() - e;
366 break;
367 }
368 }
369
370 if (qpos != -1) // ⎕xxx at end
371 {
372 UCS_string qxx(user, qpos, user.size() - qpos);
373 UCS_string_vector matches;
374
375#define ro_sv_def(_q, str, _txt) { UCS_string ustr(str); \
376 if (ustr.size() && ustr.starts_iwith(qxx)) matches.push_back(ustr); }
377
378#define rw_sv_def(_q, str, _txt) { UCS_string ustr(str); \
379 if (ustr.size() && ustr.starts_iwith(qxx)) matches.push_back(ustr); }
380
381#define sf_def(_q, str, _txt) { UCS_string ustr(str); \
382 if (ustr.size() && ustr.starts_iwith(qxx)) matches.push_back(ustr); }
383
384#include "SystemVariable.def"
385
386 if (matches.size() == 0) return ER_IGNORE;
387 if (matches.size() > 1)
388 {
389 matches.sort();
390
391 const int common_len = compute_common_length(qxx.size(), matches);
392 if (common_len == qxx.size())
393 {
394 // qxx is already the common part of all matching ⎕xx
395 // display matching ⎕xx
396 //
397 CIN << endl;
398 loop(m, matches.size())
399 {
400 CERR << "⎕" << matches[m] << " ";
401 }
402 CERR << endl;
403 return ER_AGAIN;
404 }
405 else
406 {
407 // qxx is a prefix of the common part of all matching ⎕xx.
408 // expand to common part.
409 //
410 user = matches[0];
411 user.resize(common_len);
412 return ER_REPLACE;
413 }
414 }
415
416 // unique match
417 //
418 user = UCS_string(UNI_Quad_Quad);
419 user.append(matches[0]);
420 return ER_REPLACE;
421 }
422
423 return ER_IGNORE;
424}
425//----------------------------------------------------------------------------
426ExpandResult
427TabExpansion::expand_filename(UCS_string & user,
428 ExpandHint ehint, const char * shint,
429 const UCS_string cmd, UCS_string arg)
430{
431 if (arg.size() == 0)
432 {
433 // the user has entered a command but nothing else.
434 // If the command accepts a library number then we propose one of
435 // the existing libraries.
436 //
437 if (ehint == EH_oLIB_WSNAME || ehint == EH_DIR_OR_LIB)
438 {
439 // the command accepts a library reference number
440 //
441 UCS_string libs_present;
442 loop(lib, 10)
443 {
444 if (!LibPaths::is_present(LibRef(lib)))
445 continue;
446
447 libs_present.append(UNI_SPACE);
448 libs_present.append(Unicode(UNI_0 + lib));
449 }
450 if (libs_present.size() == 0) goto nothing;
451
452 CIN << endl;
453 CERR << cmd << libs_present << " <workspace-name>" << endl;
454 return ER_AGAIN;
455 }
456
457 goto nothing;
458 }
459
460 if (arg[0] >= '0' && arg[0] <= '9')
461 {
462 // library number 0-9. EH_DIR_OR_LIB is complete already so
463 // EH_oLIB_WSNAME is the only expansion case left
464 //
465 if (ehint != EH_oLIB_WSNAME) goto nothing;
466
467 const LibRef lib = LibRef(arg[0] - '0');
468
469 if (arg.size() == 1 && !have_trailing_blank) // no space yet
470 {
471 user = cmd;
472 user.append(UNI_SPACE);
473 user.append(Unicode(arg[0]));
474 user.append(UNI_SPACE);
475 return ER_REPLACE;
476 }
477
478 // discard library reference number
479 //
480 if (arg.size() == 1) arg.erase(0);
481 if (arg.size()) arg.erase(0);
482 return expand_wsname(user, cmd, lib, arg);
483 }
484
485 // otherwise: real file name
486 {
487 UCS_string dir_ucs;
488 const bool slash_at_1 = arg.size() > 1 && arg[1] == UNI_SLASH;
489 const bool tilde_at_0 = arg[0] == UNI_TILDE ||
490 arg[0] == UNI_TILDE_OPERATOR;
491
492 if (arg[0] == UNI_SLASH) // absolute path /xxx
493 {
494 dir_ucs = arg;
495 }
496 else if (arg[0] == UNI_FULLSTOP && slash_at_1) // relative path ./xxx
497 {
498 const char * pwd = getenv("PWD");
499 if (pwd == 0) goto nothing;
500 dir_ucs = UCS_string(pwd);
501 dir_ucs.append(arg.drop(1));
502 }
503 else if (tilde_at_0 && slash_at_1) // user's home ~/
504 {
505 const char * home = getenv("HOME");
506 if (home == 0) goto nothing;
507 dir_ucs = UCS_string(home);
508 dir_ucs.append(arg.drop(1));
509 }
510 else if (tilde_at_0) // somebody's home
511 {
512 dir_ucs = UCS_string("/home/");
513 dir_ucs.append(arg.drop(1));
514 }
515 else goto nothing;
516
517 UTF8_string dir_utf = UTF8_string(dir_ucs);
518
519 const char * dir_dirname = dir_utf.c_str();
520 const char * dir_basename = strrchr(dir_dirname, '/');
521 if (dir_basename == 0) goto nothing;
522
523 UTF8_string base_utf = UTF8_string(++dir_basename);
524 dir_utf.resize(dir_basename - dir_dirname);
525 dir_ucs = UCS_string(dir_utf);
526
527 DIR * dir = opendir(dir_utf.c_str());
528 if (dir == 0) goto nothing;
529
530 UCS_string_vector matches;
531 read_matching_filenames(dir, dir_utf, base_utf, ehint, matches);
532 closedir(dir);
533
534 if (matches.size() == 0)
535 {
536 CIN << endl;
537 CERR << " no matching filesnames" << endl;
538 return ER_AGAIN;
539 }
540
541 if (matches.size() > 1)
542 {
543 UCS_string prefix(base_utf);
544 user = cmd; // e.g. )LOAD
545 user.append(UNI_SPACE);
546 user.append(dir_ucs); // e.g. )LOAD /usr/apl/
547 return show_alternatives(user, prefix.size(), matches);
548 }
549
550 // unique match
551 //
552 user = cmd;
553 user.append(UNI_SPACE);
554 user.append(dir_ucs);
555 user.append(matches[0]);
556 return ER_REPLACE;
557 }
558
559nothing:
560 CIN << endl;
561 CERR << cmd << " " << shint << endl;
562 return ER_AGAIN;
563}

565ExpandResult
566TabExpansion::expand_wsname(UCS_string & user, const UCS_string cmd,
567 LibRef lib, const UCS_string filename)
568{
569UTF8_string path = LibPaths::get_lib_dir(lib);
570 if (path.size() == 0)
571 {
572 CIN << endl;
573 CERR << "Invalib library reference " << lib << endl;
574 return ER_AGAIN;
575 }
576
577DIR * dir = opendir(path.c_str());
578 if (dir == 0)
579 {
580 CIN << endl;
581 CERR
582<< " library reference " << lib
583<< " is a valid number, but the corresponding directory " << endl
584<< " " << path << " does not exist" << endl
585<< " or is not readable. " << endl
586<< " The relation between library reference numbers and filenames" << endl
587<< " (aka. paths) can be configured file 'preferences'." << endl << endl
588<< " At this point, you can use a path instead of the optional" << endl
589<< " library reference number and the workspace name." << endl;
590
591 user = cmd;
592 user.append(UNI_SPACE);
593 return ER_REPLACE;
594 }
595
596UCS_string_vector matches;
597UTF8_string arg_utf(filename);
598 read_matching_filenames(dir, path, arg_utf, EH_oLIB_WSNAME, matches);
599 closedir(dir);
600
601 if (matches.size() == 0) goto nothing;
602 if (matches.size() > 1)
603 {
604 user = cmd;
605 user.append(UNI_SPACE);
606 user.append_number(lib);
607 user.append(UNI_SPACE);
608 return show_alternatives(user, filename.size(), matches);
609 }
610
611 // unique match
612 //
613 user = cmd;
614 user.append(UNI_SPACE);
615 user.append_UTF8(path.c_str());
616 user.append(UNI_SLASH);
617 user.append(matches[0]);
618 return ER_REPLACE;
619
620nothing:
621 CIN << endl;
622 CERR << cmd << " " << lib << " '" << filename << "'" << endl;
623 return ER_AGAIN;
624}
625//----------------------------------------------------------------------------
626int
627TabExpansion::compute_common_length(int len, const UCS_string_vector & matches)
628{
629 // we assume that all matches have the same case
630
631 for (;; ++len)
632 {
633 loop(m, matches.size())
634 {
635 if (len >= matches[m].size()) return matches[m].size();
636 if (matches[0][len] != matches[m][len]) return len;
637 }
638 }
639}
640//----------------------------------------------------------------------------
641void
642TabExpansion::read_matching_filenames(DIR * dir, UTF8_string dirname,
643 UTF8_string prefix, ExpandHint ehint,
644 UCS_string_vector & matches)
645{
646const bool only_workspaces = (ehint == EH_oLIB_WSNAME) ||
647 (ehint == EH_WSNAME ) ||
648 (ehint == EH_oWSNAME );
649
650 for (;;)
651 {
652 struct dirent * dent = readdir(dir);
653 if (dent == 0) break;
654
655 const size_t dlen = strlen(dent->d_name);
656 if (dlen == 1 && dent->d_name[0] == '.') continue;
657 if (dlen == 2 && dent->d_name[0] == '.'
658 && dent->d_name[1] == '.') continue;
659
660 if (strncmp(dent->d_name, prefix.c_str(), prefix.size())) continue;
661
662 UCS_string name(dent->d_name);
663
664 const bool is_dir = Command::is_directory(dent, dirname);
665 if (is_dir) name.append(UNI_SLASH);
666 else if (only_workspaces)
667 {
668 const UTF8_string filename(dent->d_name);
669 bool is_wsname = false;
670 if (filename.ends_with(".apl")) is_wsname = true;
671 if (filename.ends_with(".xml")) is_wsname = true;
672 if (!is_wsname) continue;
673 }
674
675 matches.push_back(name);
676 }
677}
678//----------------------------------------------------------------------------
679ExpandResult
680TabExpansion::show_alternatives(UCS_string & user, int prefix_len,
681 UCS_string_vector & matches)
682{
683const int common_len = compute_common_length(prefix_len, matches);
684
685 if (common_len == prefix_len)
686 {
687 // prefix is already the common part of all matching files
688 // display matching items
689 //
690 CIN << endl;
691 loop(m, matches.size())
692 {
693 CERR << matches[m] << " ";
694 }
695 CERR << endl;
696 return ER_AGAIN;
697 }
698 else
699 {
700 // prefix is a prefix of the common part of all matching files.
701 // expand to common part.
702 //
703 const int usize = user.size();
704 user.append(matches[0]);
705 user.resize(usize + common_len);
706 return ER_REPLACE;
707 }
708}

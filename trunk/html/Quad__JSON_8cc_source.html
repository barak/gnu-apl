39Token
40Quad_JSON::eval_AB(Value_P A, Value_P B) const
41{
42 if (A->get_rank() > 0) RANK_ERROR;
43
44const int function_number = A->get_cfirst().get_int_value();
45 switch(function_number)
46 {
47 case 0: // same as monadic ⎕JSON
48 {
49 return eval_B(B);
50 }
51
52 case 1: // read and convert a JSOM file
53 {
54 return convert_file(*B);
55 }
56
57 case 2: // read and convert a JSOM file (unsorted)
58 {
59 return Token(TOK_APL_VALUE1, APL_to_JSON(*B, false));
60 }
61
62 case 3: // read and convert a JSOM file (sorted)
63 {
64 return Token(TOK_APL_VALUE1, APL_to_JSON(*B, true));
65 }
66
67 }
68
69 MORE_ERROR() << "A ⎕JSON B: Bad function number A=" << function_number;
70 DOMAIN_ERROR;
71}
72//----------------------------------------------------------------------------
73Token
74Quad_JSON::convert_file(const Value & B) const
75{
76const UCS_string filename_ucs(B);
77const UTF8_string filename_utf(filename_ucs);
78
79 errno = 0;
80const int fd = open(filename_utf.c_str(), O_RDONLY);
81 if (fd == -1)
82 {
83 MORE_ERROR() << "1 ⎕JSON B: error reading " << B
84 << ": " << strerror(errno);
85 DOMAIN_ERROR;
86 }
87
88struct stat st;
89 if (fstat(fd, &st))
90 {
91 MORE_ERROR() << "1 ⎕JSON B: error in fstat(" << B
92 << "): " << strerror(errno);
93 ::close(fd);
94 DOMAIN_ERROR;
95 }
96
97UTF8 * buffer = new UTF8[st.st_size];
98 if (buffer == 0)
99 {
100 ::close(fd);
101 WS_FULL;
102 }
103
104const ssize_t bytes_read = read(fd, buffer, st.st_size);
105 if (bytes_read != st.st_size)
106 {
107 MORE_ERROR() << "1 ⎕JSON B: error in reading " << B
108 << "): " << strerror(errno);
109 ::close(fd);
110 DOMAIN_ERROR;
111 ::close(fd);
112 delete [] buffer;
113 }
114
115const UTF8_string json_string_utf8(buffer, bytes_read);
116 delete[] buffer;
117 ::close(fd);
118
119UCS_string json_string_ucs(json_string_utf8);
120Value_P json_string_value(json_string_ucs, LOC);
121
122 return eval_B(json_string_value);
123}

146//----------------------------------------------------------------------------
147void
148Quad_JSON::APL_to_JSON_string(UCS_string & result, const Cell & cell,
149 bool level, bool sorted)
150{
151 if (cell.is_integer_cell())
152 {
153 char cc[40];
154 snprintf(cc, sizeof(cc), "%lld", long_long(cell.get_int_value()));
155 result.append_UTF8(cc);
156 return;
157 }
158
159 if (cell.is_float_cell())
160 {
161 char cc[40];
162 snprintf(cc, sizeof(cc), "%lg", cell.get_real_value());
163 result.append_UTF8(cc);
164 return;
165 }
166
167 if (cell.is_complex_cell())
168 {
169 char cc[50];
170 snprintf(cc, sizeof(cc), "%lgJ%lg", cell.get_real_value(),
171 cell.get_imag_value());
172 result.append_UTF8(cc);
173 return;
174 }
175
176 if (!cell.is_pointer_cell())
177 {
178 MORE_ERROR() << "2 ⎕JSON B: Unexpected/unsupported Celltype "
179 << Cell::get_cell_type_name(cell.get_cell_type());
180 DOMAIN_ERROR;
181 }
182
183 // at this point the cell should end up as char vector. Determint its depth.
184 //
185const Value * Z = cell.get_pointer_value().get();
186 if (!(Z->get_cfirst().is_pointer_cell() && Z->is_scalar()))
187 {
188 APL_to_JSON_string(result, *Z, level, sorted);
189 return;
190 }
191
192 Z = Z->get_cfirst().get_pointer_value().get();
193 if (!Z->is_char_vector())
194 {
195FIXME;
196 }
197
198const UCS_string lit_ucs(*Z);
199 if (lit_ucs.compare(UCS_string("null")) == COMP_EQ ||
200 lit_ucs.compare(UCS_string("true")) == COMP_EQ ||
201 lit_ucs.compare(UCS_string("false")) == COMP_EQ)
202 {
203 result.append(lit_ucs);
204 return;
205 }
206
207 MORE_ERROR() << "2 ⎕JSON B: bad JSON literal ⊂'" << lit_ucs
208 << "'. Expecting ⊂'true', ⊂'false', or ⊂'null'";
209 result.clear(); // indicate error
210}
211//----------------------------------------------------------------------------
212void
213Quad_JSON::APL_to_JSON_string(UCS_string & result, const Value & B,
214 bool level, bool sorted)
215{
216 if (B.is_scalar()) // number or literal
217 {
218 APL_to_JSON_string(result, B.get_cfirst(), level, sorted);
219 return;
220 }
221
222 if (B.is_char_vector())
223 {
224 const UCS_string ucs_string(B);
225 result += UNI_DOUBLE_QUOTE;
226 loop(u, ucs_string.size())
227 {
228 switch(const Unicode uni = ucs_string[u])
229 {
230 case UNI_BS: result.append_UTF8("\\b"); break;
231 case UNI_HT: result.append_UTF8("\\t"); break;
232 case UNI_LF: result.append_UTF8("\\n"); break;
233 case UNI_FF: result.append_UTF8("\\f"); break;
234 case UNI_CR: result.append_UTF8("\\r"); break;
235 case UNI_DOUBLE_QUOTE: result.append_UTF8("\\\""); break;
236
237 default: if (uni < UNI_SPACE)
238 {
239 char cc[10];
240 snprintf(cc, sizeof(cc), "\\u%4.4X",
241 int(uni));
242 result.append_UTF8(cc);
243 }
244 else
245 {
246 result += uni;
247 }
248 }
249 }
250 result += UNI_DOUBLE_QUOTE;
251 return;
252 }
253
254 if (B.get_rank() == 1) // JSON array
255 {
256 UCS_string array("[ ");
257 const ShapeItem ec = B.element_count();
258 loop(e, ec)
259 {
260 if (e) array.append(", ");
261 if (array.size() > 60)
262 {
263 array.back() = UNI_LF;
264 result.append(array);
265 array = UCS_string(2*level + 2, UNI_SPACE);
266 }
267
268 APL_to_JSON_string(array, B.get_cravel(e), level + 1, sorted);
269 if (array.size() == 0) // error in APL_to_JSON_string()
270 {
271 result.clear(); // indicate error
272 return;
273 }
274 }
275 array.append_UTF8(" ]");
276 result.append(array);
277 return;
278 }
279
280 if (B.is_structured()) // JSON object
281 {
282 std::vector<ShapeItem> member_indices;
283 B.used_members(member_indices, sorted);
284
285 loop(m, member_indices.size())
286 {
287 const Cell & member_name = B.get_cravel(2*member_indices[m]);
288 const Cell & member_data = B.get_cravel(2*member_indices[m] + 1);
289
290 result.append(UCS_string(2*level, UNI_SPACE)); // level indent
291 if (m) result.append_UTF8(" \"");
292 else result.append_UTF8("{ \"");
293 UCS_string member(*member_name.get_pointer_value());
294 result.append(member_name);
295 result.append_UTF8("\": ");
296 APL_to_JSON_string(result, member_data, level + 1, sorted);
297 if (result.size() == 0) return; // error converting member_data
298 if (size_t(m) < (member_indices.size() - 1))
299 {
300 result += UNI_COMMA;
301 result += UNI_LF;
302 }
303 else
304 {
305 result += UNI_SPACE;
306 }
307 }
308
309 result.append_UTF8("}");
310 return;
311 }
312
313 // unexpected rank for a JSON value
314 //
315 MORE_ERROR() << "⎕JSON B: bad rank " << B.get_rank();
316 result.clear(); // indicate error
317}
318//----------------------------------------------------------------------------
319size_t
320Quad_JSON::skip_string(const UCS_string & ucs_B, ShapeItem & b)
321{
322const ShapeItem B0 = b++; // the leading "
323 Assert(ucs_B[B0] == UNI_DOUBLE_QUOTE);
324
325ShapeItem content_len = 0;
326 for (; b < ucs_B.size(); ++b)
327 {
328 const Unicode uni = ucs_B[b];
329 if (uni == UNI_DOUBLE_QUOTE) return content_len;
330
331 ++content_len;
332 if (uni == UNI_BACKSLASH) // skip the escaped part...
333 {
334 if (ucs_B[b + 1] == UNI_u) // \uUUUU
335 {
336 const Unicode u1 = decode_UUUU(ucs_B, b);
337 if (u1 == Unicode_0) // decode_UUUU() failed
338 {
339 MORE_ERROR() << "⎕JSON B: bad escape sequence "
340 << ucs_B[b ] << ucs_B[b + 1]
341 << ucs_B[b + 2] << ucs_B[b + 3]
342 << ucs_B[b + 4] << ucs_B[b + 5]
343 << " at " << b << "↓B";
344 DOMAIN_ERROR;
345 }
346
347 if (is_high_surrogate(u1))
348 {
349 const Unicode u2 = decode_UUUU(ucs_B, b + 6);
350 if (u2 == Unicode_0) // decode_UUUU() failed
351 {
352 MORE_ERROR() << "⎕JSON B: bad escape sequence "
353 << ucs_B[b + 6] << ucs_B[b + 7]
354 << ucs_B[b + 8] << ucs_B[b + 9]
355 << ucs_B[b + 10] << ucs_B[b + 11]
356 << " at " << (b + 6) << "↓B";
357 DOMAIN_ERROR;
358 }
359 b += 11;
360 }
361 else // u1 is not a high surrogate (normal \uUUUU)
362 {
363 b += 5;
364 }
365 }
366 else // 1-character escape
367 {
368 ++b; // the escaped character
369 }
370 }
371 }
372
373 // end of string reached without seeing "
374 //
375 MORE_ERROR() <<
376 "⎕JSON B: No string end for string starting at " << B0 << "↓B";
377 DOMAIN_ERROR;
378}
379//----------------------------------------------------------------------------
380size_t
381Quad_JSON::number_len(const UCS_string & ucs_B, ShapeItem b)
382{
383const ShapeItem B0 = b;
384
385 // leading sign
386 if (ucs_B[b] == UNI_MINUS) ++b;
387 else if (ucs_B[b] == UNI_OVERBAR) ++b;
388
389 // mandatory integral part
390 while (uint32_t(ucs_B[b] - UNI_0) < 10) ++b;
391
392 if (ucs_B[b] == UNI_FULLSTOP) // optional fractional part
393 {
394 ++b; // skip UNI_FULLSTOP
395 while (uint32_t(ucs_B[b] - UNI_0) < 10) ++b;
396 }
397
398 if (ucs_B[b] == UNI_E || ucs_B[b] == UNI_e) // optional exponent
399 {
400 ++b; // skip UNI_E / UNI_e
401 if (ucs_B[b] == UNI_MINUS) ++b;
402 else if (ucs_B[b] == UNI_OVERBAR) ++b;
403 while (uint32_t(ucs_B[b] - UNI_0) < 10) ++b;
404 }
405
406 return b - B0;
407}
408//----------------------------------------------------------------------------
409Value_P
410Quad_JSON::JSON_to_APL(const Value & B)
411{
412const ShapeItem len_B = B.element_count();
413
414UCS_string ucs_B;
415 ucs_B.reserve(len_B + 1);
416 loop(b, len_B) ucs_B += B.get_cravel(b).get_char_value();
417 ucs_B += Unicode_0; // 0-terminate ucs_B to avoid too many length checks
418
419 // tokenize ucs_B
420 //
421std::vector<ShapeItem> tokens_B;
422 tokens_B.reserve(10 + len_B/2);
423
424 loop(b, ucs_B.size())
425 {
426 const Unicode uni = ucs_B[b];
427 if (uni <= ' ') continue; // skip leading whitespace
428
429 tokens_B.push_back(b);
430 switch(uni)
431 {
432 case UNI_DOUBLE_QUOTE:
433 skip_string(ucs_B, b);
434 continue;
435
436 case UNI_COMMA:
437 case UNI_COLON:
438 case UNI_L_BRACK:
439 case UNI_R_BRACK:
440 case UNI_L_CURLY:
441 case UNI_R_CURLY: continue; // the six structural chars
442
443 case UNI_f: b += 4; continue; // literal false
444 case UNI_n: // literal true
445 case UNI_t: b += 3; continue; // literal null
446
447 case UNI_MINUS:
448 case UNI_OVERBAR:
449 case UNI_FULLSTOP:
450 case UNI_0 ... UNI_9:
451 b += number_len(ucs_B, b) - 1;
452 continue;
453
454 default:
455 MORE_ERROR() << "⎕JSON B: Got '" << uni <<
456 "' when expecting a JSON token at " << b << "↓B";
457 DOMAIN_ERROR;
458 MORE_ERROR() << "⎕JSON B: Got '" << uni <<
459 "' when expecting a JSON token at " << b << "↓B";
460 DOMAIN_ERROR;
461 }
462 }
463
464Value_P Z(LOC);
465size_t token0 = 0;
466 parse_value(*Z, ucs_B, tokens_B, token0);
467 Z->check_value(LOC);
468
469 if (token0 != tokens_B.size())
470 {
471 MORE_ERROR() <<
472 "⎕JSON B: there were extra token in B (tokenized: " << tokens_B.size()
473 << ", but processed: " << token0 <<
474 ").\n The JSON string must be one serialized value.";
475 LENGTH_ERROR;
476 }
477
478 if (Z->is_simple_scalar())
479 {
480 Assert(Z->get_cfirst().is_numeric());
481 return Z; // number
482 }
483
484 Assert(Z->is_scalar());
485 Assert(Z->get_cfirst().is_pointer_cell());
486 return Z->get_cfirst().get_pointer_value();
487}
488//----------------------------------------------------------------------------
489void
490Quad_JSON::parse_value(Value & Z, const UCS_string & ucs_B,
491 const std::vector<ShapeItem> & tokens_B, size_t & token0)
492{
493const ShapeItem b = tokens_B.at(token0);
494 switch(ucs_B[b])
495 {
496 // unexpected token...
497 //
498 case UNI_COMMA:
499 MORE_ERROR() << "⎕JSON: Got ',' when expecting a value";
500 DOMAIN_ERROR;
501
502 case UNI_COLON:
503 MORE_ERROR() << "⎕JSON: Got ':' when expecting a value";
504 DOMAIN_ERROR;
505
506 case UNI_R_BRACK:
507 MORE_ERROR() << "⎕JSON: Got ']' when expecting a value";
508 DOMAIN_ERROR;
509
510 case UNI_R_CURLY:
511 MORE_ERROR() << "⎕JSON: Got '}' when expecting a value";
512 DOMAIN_ERROR;
513
514 // single token (we increment token0)...
515 //
516 case '-':
517 case UNI_OVERBAR:
518 case UNI_FULLSTOP:
519 case '0' ... '9': parse_number(Z, ucs_B, b); ++token0; break;
520 case UNI_DOUBLE_QUOTE: parse_string(Z, ucs_B, b); ++token0; break;
521 case UNI_f: parse_literal(Z, ucs_B, b, "false"); ++token0; break;
522 case UNI_n: parse_literal(Z, ucs_B, b, "null"); ++token0; break;
523 case UNI_t: parse_literal(Z, ucs_B, b, "true"); ++token0; break;
524
525 // multi token (they increment token0)...
526 //
527 case UNI_L_BRACK: parse_array (Z, ucs_B, tokens_B, token0); break;
528 case UNI_L_CURLY: parse_object(Z, ucs_B, tokens_B, token0); break;
529
530 default: FIXME;
531 }
532}
533//----------------------------------------------------------------------------
534void
535Quad_JSON::parse_array(Value & Z, const UCS_string & ucs_B,
536 const std::vector<ShapeItem> & tokens_B,
537 size_t & token0)
538{
539size_t token_from = token0;
540 Assert(ucs_B[tokens_B[token_from]] == UNI_L_BRACK); // [
541
542const size_t commas = comma_count(ucs_B, tokens_B, token0);
543 Assert(ucs_B[tokens_B[token0]] == UNI_R_BRACK); // always ]
544
545 ++token_from; // skip [
546 if (commas == 0) // [ ] or [ item ]
547 {
548 if ((token0 - token_from) == 0) // [ ]
549 {
550 // CERR << "empty ARRAY" << std::endl;
551 Value_P Zsub = Idx0(LOC);
552 Z.next_ravel_Pointer(Zsub.get());
553 }
554 else
555 {
556 // CERR << "One element ARRAY" << std::endl;
557 Value_P Zsub(1, LOC);
558 parse_value(*Zsub, ucs_B, tokens_B, token_from);
559 Zsub->check_value(LOC);
560 Z.next_ravel_Value(Zsub.get());
561 }
562 ++token_from; // skip ]
563 }
564 else // [ item , item... ]
565 {
566 const size_t len = commas + 1;
567 // CERR << "ARRAY with " << len << " elements" << endl;
568
569 Value_P Zsub(len, LOC);
570 loop(l, len)
571 {
572 parse_value(*Zsub, ucs_B, tokens_B, token_from);
573 const Unicode uni = ucs_B[tokens_B[token_from]];
574 if (uni == UNI_COMMA) ++token_from;
575 else if (uni == UNI_R_BRACK) ++token_from;
576 else
577 {
578 MORE_ERROR() << "⎕JSON B: Got '" << uni
579 << "' when expecting ',' or ']' at "
580 << token_from << "↓B";
581 DOMAIN_ERROR;
582 }
583 }
584 Zsub->check_value(LOC);
585 Z.next_ravel_Value(Zsub.get());
586 }
587
588 ++token0; // skip final ]
589 Assert(token0 == token_from);
590}
591//----------------------------------------------------------------------------
592void
593Quad_JSON::parse_object(Value & Z, const UCS_string & ucs_B,
594 const std::vector<ShapeItem> & tokens_B,
595 size_t & token0)
596{
597size_t token_from = token0;
598 Assert(ucs_B[tokens_B[token_from]] == UNI_L_CURLY); // {
599
600const size_t commas = comma_count(ucs_B, tokens_B, token0);
601 Assert(ucs_B[tokens_B[token0]] == UNI_R_CURLY); // always }
602
603Value_P assoc_array = EmptyStruct(LOC);
604 Z.next_ravel_Pointer(assoc_array.get());
605
606 ++token_from; // skip {
607 if (commas == 0) // { } or { 'name' : value }
608 {
609 if ((token0 - token_from) == 0) // { }
610 {
611 // CERR << "empty OBJECT" << std::endl;
612 ++token_from; // skip }
613 }
614 else
615 {
616 // CERR << "One element OBJECT" << std::endl;
617 parse_object_member(*assoc_array, ucs_B, tokens_B, token_from);
618 }
619 }
620 else // { 'name' : value , 'name' : value... }
621 {
622 const size_t items = commas + 1;
623 // CERR << "OBJECT with " << items << " elements" << endl;
624
625 loop(it, items)
626 {
627 parse_object_member(*assoc_array, ucs_B, tokens_B, token_from);
628 }
629 }
630
631 ++token0; // skip final }
632 Assert(token0 == token_from);
633}
634//----------------------------------------------------------------------------
635void
636Quad_JSON::parse_object_member(Value & Z, const UCS_string & ucs_B,
637 const std::vector<ShapeItem> & tokens_B,
638 size_t & token_from)
639{
640const size_t B_start = tokens_B[token_from];
641 if (ucs_B[B_start] != UNI_DOUBLE_QUOTE)
642 {
643 MORE_ERROR() << "⎕JSON B: Got '" << ucs_B[B_start] <<
644 "' when expecting (double-quoted) object member name at " <<
645 B_start << "↓B";
646 DOMAIN_ERROR;
647 }
648 ++token_from; // skip "
649
650 // parse "member-name" :
651 //
652UCS_string member_name;
653 {
654 const ShapeItem ucs_B_name = B_start + 1;
655
656 const ShapeItem ucs_B_colon = tokens_B[token_from];
657 if (ucs_B[ucs_B_colon] != UNI_COLON)
658 {
659 MORE_ERROR() <<
660 "⎕JSON B: Got '" << ucs_B[ucs_B_colon] <<
661 "' when expecting ':' after object member name at " <<
662 B_start << "↓B";
663 DOMAIN_ERROR;
664 }
665 ++token_from; // skip :
666
667 // find trailing " of "member-name" : by searching backwards from ':'
668 //
669 ShapeItem name_len = ucs_B_colon - ucs_B_name;
670 while (name_len && ucs_B[ucs_B_name + name_len - 1] != UNI_DOUBLE_QUOTE)
671 --name_len;
672 --name_len; // skip the trailing "
673 if (name_len < 1)
674 {
675 FIXME; // since skip_string() should have failed
676 }
677
678 member_name = UCS_string(ucs_B, ucs_B_name, name_len);
679 }
680
681 // check that member_name does not yet exist in Z.
682 //
683 if (Z.get_member_data(member_name))
684 {
685 // member name exists already
686 //
687 MORE_ERROR() << "⎕JSON: duplicate member name '" << member_name
688 << "' at " << B_start << "↓B";
689 DOMAIN_ERROR;
690 }
691
692 // parse the member value
693 {
694 Value_P Zsub(LOC);
695 parse_value(*Zsub, ucs_B, tokens_B, token_from);
696 Zsub->check_value(LOC);
697
698 Cell * member_data = Z.get_new_member(member_name);
699 member_data->init(Zsub->get_cfirst(), Z, LOC);
700 }
701
702 // check member-seperator (or end of object).
703 {
704 const ShapeItem ucs_B_sepa = tokens_B[token_from++]; // skip , or }
705 const Unicode sepa = ucs_B[ucs_B_sepa];
706 if (sepa != UNI_COMMA && sepa != UNI_R_CURLY)
707 {
708 MORE_ERROR() << "⎕JSON B: Got '" << sepa <<
709 "' when expecting ',' or '}' after \"" << member_name
710 << "\" : at " << B_start << "↓B";
711 DOMAIN_ERROR;
712 }
713 }
714}
715//----------------------------------------------------------------------------
716void
717Quad_JSON::parse_number(Value & Z, const UCS_string & ucs_B, ShapeItem b)
718{
719 enum { MAX_NUMLEN = 90 };
720char cc[MAX_NUMLEN + 10];
721size_t cc_len = 0;
722bool need_fract = false;
723bool have_expo = false;
724double dval = 0;
725
726 if (ucs_B[b] == UNI_MINUS || ucs_B[b] == UNI_OVERBAR)
727 { ++b; cc[cc_len++] = '-'; }
728
729 // copy integer part to cc...
730 //
731 while (ucs_B[b] >= UNI_0 && ucs_B[b] <= UNI_9)
732 {
733 if (cc_len >= MAX_NUMLEN) goto number_too_long;
734 cc[cc_len++] = ucs_B[b++];
735 }
736
737 // maybe copy fractional part to cc...
738 //
739 if (ucs_B[b] == UNI_FULLSTOP) // fractional part
740 {
741 need_fract = true; // force FloatCell
742 if (cc_len >= MAX_NUMLEN) goto number_too_long;
743 cc[cc_len++] = '.';
744 ++b;
745 while (ucs_B[b] >= UNI_0 && ucs_B[b] <= UNI_9)
746 {
747 if (cc_len >= MAX_NUMLEN) goto number_too_long;
748 cc[cc_len++] = ucs_B[b++];
749 }
750 }
751
752 // maybe copy exponent part to cc...
753 //
754 if (ucs_B[b] == UNI_E || ucs_B[b] == UNI_e) // exponent part
755 {
756 have_expo = true;
757 if (cc_len >= MAX_NUMLEN) goto number_too_long;
758 cc[cc_len++] = 'e';
759 ++b;
760 if (ucs_B[b] == UNI_MINUS || ucs_B[b] == UNI_OVERBAR) // negative expo
761 {
762 need_fract = true;
763 ++b; cc[cc_len++] = '-'; }
764 while (ucs_B[b] >= UNI_0 && ucs_B[b] <= UNI_9)
765 {
766 if (cc_len >= MAX_NUMLEN) goto number_too_long;
767 cc[cc_len++] = ucs_B[b++];
768 }
769 }
770 cc[cc_len++] = 0;
771
772 // always scan the double value
773 //
774 dval = strtod(cc, 0);
775
776 if (dval > LARGE_INT || dval < SMALL_INT) need_fract = true;
777
778 if (need_fract)
779 {
780 Z.next_ravel_Float(dval);
781 return;
782 }
783 else if (have_expo) // so strtoll wont work
784 {
785 if (dval < 0) Z.next_ravel_Int(dval - 0.5); // round negative → 0
786 else Z.next_ravel_Int(dval + 0.5); // round positive → 0
787 return;
788 }
789 else
790 {
791 const long long int ival = strtoll(cc, 0, 10);
792 Z.next_ravel_Int(ival);
793 return;
794 }
795
796number_too_long:
797 MORE_ERROR() << "⎕JSON B: number too long at " << b
798 << "↓B (max. length is " << MAX_NUMLEN << ")";
799 DOMAIN_ERROR;
800}
801//----------------------------------------------------------------------------
802void
803Quad_JSON::parse_string(Value & Z, const UCS_string & ucs_B, ShapeItem b)
804{
805ShapeItem bb = b + 1; // NOTE: skip_string() will increment b
806const ShapeItem content_len = skip_string(ucs_B, b);
807
808Value_P Zsub(content_len, LOC);
809 loop(l, content_len)
810 {
811 Unicode uni = ucs_B[bb++];
812 if (uni == UNI_BACKSLASH)
813 {
814 Unicode surr = Unicode_0;
815
816 uni = ucs_B[bb++]; // default: escaped = character itself
817 switch(uni)
818 {
819 case UNI_b: uni = UNI_BS; break; // \b
820 case UNI_f: uni = UNI_FF; break; // \f
821 case UNI_n: uni = UNI_LF; break; // \n
822 case UNI_r: uni = UNI_CR; break; // \r
823 case UNI_t: uni = UNI_HT; break; // \t
824 case UNI_u: // \uUUUU
825 uni = decode_UUUU(ucs_B, bb - 2);
826 surr = decode_UUUU(ucs_B, bb + 4);
827 if (is_high_surrogate(uni) && is_low_surrogate(surr))
828 {
829 uni = Unicode(0x10000 + (surr & 0x03FF)
830 + ((uni & 0x03FF) << 10));
831 bb += 10;
832 }
833 else if (is_high_surrogate(uni))
834 {
835 MORE_ERROR() <<
836 "⎕JSON B: No low surrogate at " << bb << "↓B";
837 DOMAIN_ERROR;
838 }
839 else if (uni == Unicode_0) // decode_UUUU() failed
840 {
841 FIXME; // since skip_string() should have failed
842 }
843 else // normal (non-surrogate) \uUUUU
844 {
845 bb += 4;
846 }
847 break; // case UNI_u
848
849 default: break;
850 }
851 }
852
853 Zsub->next_ravel_Char(uni);
854 }
855
856 Zsub->check_value(LOC);
857 Z.next_ravel_Pointer(Zsub.get());
858}
859//----------------------------------------------------------------------------
860Unicode
861Quad_JSON::decode_UUUU(const UCS_string & ucs_B, ShapeItem b)
862{
863 if (ucs_B[b++] != UNI_BACKSLASH) return Unicode_0;
864 if (ucs_B[b++] != UNI_u) return Unicode_0;
865
866char cc[5];
867 cc[0] = ucs_B[b++];
868 cc[1] = ucs_B[b++];
869 cc[2] = ucs_B[b++];
870 cc[3] = ucs_B[b++];
871 cc[4] = 0;
872
873 return Unicode(strtol(cc, 0, 16));
874}
875//----------------------------------------------------------------------------
876void
877Quad_JSON::parse_literal(Value & Z, const UCS_string & ucs_B,
878 ShapeItem b, const char * expected_literal)
879{
880const size_t len = strlen(expected_literal);
881 loop(l, len)
882 {
883 if (ucs_B[b + l] != expected_literal[l])
884 {
885 MORE_ERROR() << "⎕JSON B: misspelled JSON literal "
886 << UCS_string(ucs_B, b, len) << " at " << b
887 << "↓B (expecting " << expected_literal << ")";
888 DOMAIN_ERROR;
889 }
890 }
891
892Value_P Zsubsub(len, LOC);
893 loop(l, len)
894 Zsubsub->next_ravel_Char(Unicode(expected_literal[l]));
895 Zsubsub->check_value(LOC);
896
897Value_P Zsub(LOC);
898 Zsub->next_ravel_Pointer(Zsubsub.get());
899 Zsub->check_value(LOC);
900
901 Z.next_ravel_Pointer(Zsub.get());
902}
903//----------------------------------------------------------------------------
904size_t
905Quad_JSON::comma_count(const UCS_string & ucs_B,
906 const std::vector<ShapeItem> & tokens_B,
907 size_t & token0)
908{
909 // ucs_B is the JSON text string being parsed,
910 // tokens_B are the start positions (in ucs_B) of the tokenized ucs_B,
911 // token0 is the current token in tokens_B which is either [ (start of a
912 // JSON array) or { (start of a JSON object).
913 //
914const Unicode start = ucs_B[tokens_B[token0]]; // either [ or {/
915Unicode end;
916 if (start == UNI_L_BRACK) end = UNI_R_BRACK;
917 else if (start == UNI_L_CURLY) end = UNI_R_CURLY;
918 else FIXME;
919
920UCS_string stack(end); // a stack of ] amd } to track nested [...] and {...}
921size_t commas = 0; // the numebr of (top-level-) commas
922bool expect_comma = false;
923bool expect_colon = true;
924
925 for (++token0; token0 < tokens_B.size(); ++token0)
926 {
927 const Unicode uni = ucs_B[tokens_B[token0]];
928 switch(uni)
929 {
930 case UNI_L_BRACK: // [
931 stack += UNI_R_BRACK; // push ]
932 continue;
933
934 case UNI_L_CURLY: // {
935 stack += UNI_R_CURLY; // push }
936 continue;
937
938 case UNI_COMMA:
939 if (stack.size() > 1) continue;
940 if (!expect_comma)
941 {
942 MORE_ERROR() << "⎕JSON B: Got unexpected ',' at "
943 << tokens_B[token0] << "↓B";
944 DOMAIN_ERROR;
945 }
946
947 ++commas;
948 expect_comma = false;
949 expect_colon = true;
950 continue;
951
952 case UNI_COLON:
953 if (stack.size() > 1) continue;
954 if (!expect_comma)
955 {
956 MORE_ERROR() << "⎕JSON B: Got unexpected ':' at "
957 << tokens_B[token0] << "↓B";
958 DOMAIN_ERROR;
959 }
960
961 expect_comma = false;
962 if (!expect_colon)
963 {
964 MORE_ERROR() << "⎕JSON B: Got ':' "
965 "instead of ',' at "
966 << tokens_B[token0] << "↓B";
967 DOMAIN_ERROR;
968 }
969 expect_colon = false;
970 continue;
971
972 case UNI_R_BRACK:
973 //
974 // UNI_R_BRACK must match stack.back()
975 //
976 if (stack.back() != UNI_R_BRACK)
977 {
978 MORE_ERROR() << "Mismatch ]";
979 DOMAIN_ERROR;
980 }
981 stack.pop_back();
982 if (stack.size() == 0) return commas;
983 expect_comma = true;
984 continue;
985
986 case UNI_R_CURLY:
987 //
988 // UNI_R_CURLY must match stack.back()
989 //
990 if (stack.back() != UNI_R_CURLY)
991 {
992 MORE_ERROR() << "Mismatch ]";
993 DOMAIN_ERROR;
994 }
995 stack.pop_back();
996 if (stack.size() == 0) return commas;
997 expect_comma = true;
998 continue;
999
1000 default:
1001 if (stack.size() > 1) continue; // not of interest
1002 if (expect_comma)
1003 {
1004 MORE_ERROR() << "⎕JSON B: Got '" << uni
1005 << "' when expecting ',' at "
1006 << tokens_B[token0] << "↓B";
1007 DOMAIN_ERROR;
1008 }
1009 expect_comma = true;
1010 }
1011
1012 }
1013
1014 MORE_ERROR() << "⎕JSON B: No matching " << end << " for " << start
1015 << " at " << tokens_B[token0] << "↓B ";
1016 DOMAIN_ERROR;
1017}

83bool
84ESCmap::need_more(const char * seq, int len)
85{
86 loop(e, ESCmap_entry_count)
87 {
88 if (ESCmap::the_ESCmap[e].has_prefix(seq, len)) return true;
89 }
90
91 return false;
92}
93//----------------------------------------------------------------------------
94bool
95ESCmap::has_prefix(const char * seq, int seq_len) const
96{
97 if (seq_len >= len) return false;
98 loop(s, seq_len)
99 {
100 if ((seq[s] != seqence[s]) && seqence[s]) return false;
101 }
102
103 return true;
104}
105//----------------------------------------------------------------------------
106bool
107ESCmap::is_equal(const char * seq, int seq_len) const
108{
109 if (len != seq_len) return false;
110 loop(s, seq_len)
111 {
112 if ((seq[s] != seqence[s]) && seqence[s]) return false;
113 }
114
115 return true;
116}

157//----------------------------------------------------------------------------
158void
159LineHistory::read_history(const char * filename)
160{
161FILE * hist = fopen(filename, "r");
162 if (hist == 0)
163 {
164 Log(LOG_get_line)
165 CERR << "Cannot open history file " << filename
166 << ": " << strerror(errno) << endl;
167 return;
168 }
169
170int count = 0;
171 for (;;)
172 {
173 char buffer[4000];
174 const char * s = fgets(buffer, sizeof(buffer) - 1, hist);
175 if (s == 0) break; // end of file
176 buffer[sizeof(buffer) - 1] = 0;
177
178 int slen = strlen(buffer);
179 if (slen && (buffer[slen - 1] == '\n')) buffer[--slen] = 0;
180 if (slen && (buffer[slen - 1] == '\r')) buffer[--slen] = 0;
181 ++count;
182 UTF8_string utf(buffer);
183 UCS_string ucs(utf);
184 add_line(ucs);
185 }
186
187 next();
188}
189//----------------------------------------------------------------------------
190void
191LineHistory::save_history(const char * filename)
192{
193 if (hist_lines.size() == 0) return;
194
195ofstream outf(filename);
196 if (!outf.is_open())
197 {
198 CERR << "Cannot write history file " << filename
199 << ": " << strerror(errno) << endl;
200 return;
201 }
202
203int count = 0;
204 for (size_t p = put + 1; p < hist_lines.size(); ++p)
205 {
206 outf << hist_lines[p] << endl;
207 ++count;
208 }
209 for (int p = 0; p < put; ++p)
210 {
211 outf << hist_lines[p] << endl;
212 ++count;
213 }
214
215 Log(LOG_get_line)
216 cerr << count << " history lines written to " << filename << endl;
217}

241//----------------------------------------------------------------------------
242void
243LineHistory::add_line(const UCS_string & line)
244{
245 if (max_lines == 0) return; // no history
246 if (!line.has_black()) return; // almost empty
247
248 // remove leading blanks
249 //
250int blanks = 0;
251 while (blanks < line.size() && line[blanks] <= UNI_SPACE) ++blanks;
252
253 // a repeated cut-and-paste of entire lines increases the indentation every
254 // time due to the APL input prompt). We therefore limit this effect
255 // to 6 blanks.
256 //
257UCS_string line1;
258 if (blanks)
259 {
260 line1 = UCS_string(line, blanks, line.size() - blanks);
261 }
262 else
263 {
264 line1 = line;
265 }
266 while (line1.back() <= UNI_SPACE) line1.pop_back();
267
268 if (int(hist_lines.size()) < max_lines) // append
269 {
270 hist_lines.push_back(line1);
271 put = 0;
272 }
273 else // override
274 {
275 if (put >= int(hist_lines.size())) put = 0; // wrap
276 hist_lines[put++] = line1;
277 }
278
279 next(); // update current_line
280}
281//----------------------------------------------------------------------------
282void
283LineHistory::replace_line(const UCS_string & line)
284{
285 if (put > 0) hist_lines[put - 1] = line;
286 else hist_lines.back() = line;
287}
288//----------------------------------------------------------------------------
289const UCS_string *
290LineHistory::up()
291{
292 if (hist_lines.size() == 0) return 0; // no history
293
294int new_current_line = current_line - 1;
295 if (new_current_line < 0) new_current_line += hist_lines.size(); // wrap
296 if (new_current_line == put) return 0;
297
298 return &hist_lines[current_line = new_current_line];
299}
300//----------------------------------------------------------------------------
301const UCS_string *
302LineHistory::down()
303{
304 if (hist_lines.size() == 0) return 0; // no history
305 if (current_line == put) return 0;
306
307int new_current_line = current_line + 1;
308 if (new_current_line >= int(hist_lines.size()))
309 new_current_line = 0; // wrap
310 current_line = new_current_line;
311 if (current_line == put) return 0;
312
313 return &hist_lines[current_line];
314}

327//----------------------------------------------------------------------------
328const UCS_string *
329LineHistory::search(UCS_string &cur_line)
330{
331 if( hist_lines.size() == 0 ) return 0; // no history
332
333 // For now, a simple substring search of hist_lines[]
334 int search_start_line = last_search_line - 1;
335 if( search_start_line < 0) {
336 search_start_line = hist_lines.size()-1;
337 }
338 int idx = search_start_line;
339 bool found = false;
340 do {
341 if( hist_lines[idx].substr_pos(cur_search_substr) >= 0 ) {
342 current_line = idx;
343 found = true;
344 continue;
345 }
346
347 idx--;
348 if( idx < 0 ) {
349 idx = hist_lines.size()-1;
350 }
351 if( idx == search_start_line ) {
352 break;
353 }
354 } while(!found);
355
356 if( !found ) {
357 idx = hist_lines.size()-1;
358 }
359
360 last_search_line = idx;
361 if( idx == 0 ) return 0;
362
363 return &hist_lines[current_line];
364}

366LineEditContext::LineEditContext(LineInputMode mode, int rows, int cols,
367 LineHistory & hist, const UCS_string & prmt)
368 : screen_rows(rows),
369 screen_cols(cols),
370 allocated_height(1),
371 uidx(0),
372 ins_mode(true),
373 history(hist),
374 history_entered(false)
375{
376 if (mode == LIM_Quote_Quad)
377 {
378 // the prompt was printed by ⍞ already. Make it the beginning of
379 // user_line so that it can be edited.
380 //
381 user_line = prmt.no_pad();
382 uidx = user_line.size();
383 }
384 else
385 {
386 prompt = prmt.no_pad();
387 }
388
389 refresh_all();
390}
391//----------------------------------------------------------------------------
392LineEditContext::~LineEditContext()
393{
394 // restore block cursor
395 //
396 if (!ins_mode) CIN << "\x1B[1 q" << flush;
397}
398//----------------------------------------------------------------------------
399void
400LineEditContext::adjust_allocated_height()
401{
402const int rows = 1 + get_total_length() / screen_cols;
403
404 if (allocated_height >= rows) return;
405
406 // scroll some lines so that prior text is not overridden.
407 //
408 CIN.set_cursor(-1, 0);
409 loop(a, rows - allocated_height) CIN << endl;
410
411 allocated_height = rows;
412
413 // redraw screen from -allocated_height:0 onwards
414 //
415 if (CIN.can_clear_EOS())
416 {
417 CIN.set_cursor(-allocated_height, 0);
418 CIN.clear_EOS();
419 }
420 else
421 {
422 loop(a, allocated_height)
423 {
424 CIN.set_cursor(a - allocated_height, 0);
425 CIN.clear_EOL();
426 }
427 }
428
429 refresh_all();
430}
431//----------------------------------------------------------------------------
432void
433LineEditContext::refresh_from_cursor()
434{
435const int saved_uidx = uidx;
436
437 adjust_allocated_height();
438 set_cursor();
439 for (; uidx < user_line.size(); ++uidx)
440 {
441 refresh_wrapped_cursor();
442 CIN << user_line[uidx];
443 }
444
445 // clear from end of user_line
446 //
447 move_idx(user_line.size());
448 if (CIN.can_clear_EOS())
449 {
450 CIN.clear_EOS();
451 }
452 else
453 {
454 CIN.clear_EOL();
455
456 // clear subsequent lines
457 //
458 for (int a = 1; a < allocated_height; ++a)
459 {
460 CIN.set_cursor(a - allocated_height, 0);
461 CIN.clear_EOL();
462 }
463 }
464
465 move_idx(saved_uidx);
466}
467//----------------------------------------------------------------------------
468void
469LineEditContext::refresh_all()
470{
471const int saved_uidx = uidx;
472 uidx = 0;
473
474 CIN.set_cursor(-allocated_height, 0);
475 CIN << prompt << user_line;
476 refresh_from_cursor();
477 move_idx(saved_uidx);
478}
479//----------------------------------------------------------------------------
480void
481LineEditContext::delete_char()
482{
483 if (uidx == (user_line.size() - 1)) // cursor on last char
484 {
485 CIN << ' ' << UNI_BS;
486 user_line.pop_back();
487 }
488 else
489 {
490 user_line.erase(uidx);
491 refresh_from_cursor();
492 }
493
494 if (get_total_length() >= screen_cols) set_cursor();
495}
496//----------------------------------------------------------------------------
497void
498LineEditContext::insert_char(Unicode uni)
499{
500 if (uidx >= user_line.size()) // append char
501 {
502 user_line.append(uni);
503 adjust_allocated_height();
504 refresh_wrapped_cursor();
505 CIN << uni;
506 }
507 else if (ins_mode) // insert char
508 {
509 user_line.insert(uidx, uni);
510 adjust_allocated_height();
511 refresh_wrapped_cursor();
512 refresh_from_cursor();
513 }
514 else // replace char
515 {
516 user_line[uidx] = uni;
517 adjust_allocated_height();
518 refresh_wrapped_cursor();
519 refresh_from_cursor();
520 }
521
522 move_idx(uidx + 1);
523}
524//----------------------------------------------------------------------------
525void
526LineEditContext::cut_to_EOL()
527{
528 if (uidx >= user_line.size()) return; // nothing to cut
529
530 cut_buffer = UCS_string(user_line, uidx, user_line.size() - uidx);
531 user_line.resize(uidx);
532 refresh_from_cursor();
533}
534//----------------------------------------------------------------------------
535void
536LineEditContext::paste()
537{
538 if (cut_buffer.size() == 0) return;
539
540 if (uidx >= user_line.size()) // append cut buffer
541 {
542 user_line.append(cut_buffer);
543 }
544 else // insert cut buffer
545 {
546 const UCS_string rest(user_line, uidx, user_line.size() - uidx);
547 user_line.resize(uidx);
548 user_line.append(cut_buffer);
549 user_line.append(rest);
550 }
551
552 refresh_from_cursor();
553 move_idx(uidx + cut_buffer.size());
554}
555//----------------------------------------------------------------------------
556void
557LineEditContext::toggle_ins_mode()
558{
559 ins_mode = ! ins_mode;
560
561 // CSI [0 q : blinking block
562 // CSI [1 q : blinking block
563 // CSI [2 q : steady block
564 // CSI [3 q : blinking underline
565 // CSI [4 q : steady underline
566 // CSI [5 q : blinking bar (doesn't work)
567 // CSI [6 q : steady bar (doesn't work)
568
569 if (ins_mode) CIN << "\x1B[0 q" << flush;
570 else CIN << "\x1B[3 q" << flush;
571}
572//----------------------------------------------------------------------------
573void
574LineEditContext::tab_expansion(LineInputMode mode)
575{
576 if (mode != LIM_ImmediateExecution) return;
577
578UCS_string line = user_line;
579TabExpansion tab_exp(line);
580const ExpandResult expand_result = tab_exp.expand_tab(line);
581
582 switch(expand_result)
583 {
584 case ER_IGNORE: return;
585
586 case ER_AGAIN:
587 // expand_tab has shown a list of options.
588 // Reset the input window and redisplay.
589 allocated_height = 1;
590 adjust_allocated_height();
591 refresh_all();
592 return;
593
594 case ER_REPLACE:
595 user_line.clear();
596 user_line.append(line);
597 uidx = 0;
598 refresh_from_cursor();
599 move_idx(user_line.size());
600 return;
601
602 default: FIXME;
603 }
604}

612//----------------------------------------------------------------------------
613void
614LineEditContext::cursor_UP()
615{
616 Log(LOG_get_line) history.info(CERR << "cursor_UP()") << endl;
617
618const UCS_string * ucs = history.up();
619 if (ucs == 0) // no line above
620 {
621 Log(LOG_get_line) CERR << "hit top of history()" << endl;
622 Log(LOG_get_line) history.info(CERR << "cursor_UP() done" << endl);
623 return;
624 }
625
626 if (!history_entered) // not yet in history: remember user_line
627 {
628 user_line_before_history = user_line;
629 history_entered = true;
630 }
631
632 user_line = *ucs;
633 adjust_allocated_height();
634
635 uidx = 0;
636 refresh_from_cursor();
637 move_idx(user_line.size());
638 Log(LOG_get_line) history.info(CERR << "cursor_UP() done" << endl);
639}
640//----------------------------------------------------------------------------
641void
642LineEditContext::cursor_DOWN()
643{
644 Log(LOG_get_line) history.info(CERR << "cursor_DOWN()" << endl);
645
646const UCS_string * ucs = history.down();
647 if (ucs == 0) // no line below
648 {
649 Log(LOG_get_line) CERR << "hit bottom of history()" << endl;
650 // if inside history: restore user_line
651 //
652 if (history_entered) user_line = user_line_before_history;
653 history_entered = false;
654 goto refresh;
655 }
656
657 if (!history_entered) // not yet in history: remember user_line
658 {
659 user_line_before_history = user_line;
660 history_entered = true;
661 }
662
663 user_line = *ucs;
664 adjust_allocated_height();
665
666refresh:
667 uidx = 0;
668 refresh_from_cursor();
669 move_idx(user_line.size());
670 Log(LOG_get_line) history.info(CERR << "cursor_DOWN() done" << endl);
671}

678//----------------------------------------------------------------------------
679void
680LineEditContext::cursor_SEARCH()
681{
682 user_line_before_history = user_line;
683 history_entered = true;
684

691 }
692
693 adjust_allocated_height();
694
695 uidx = 0;
696 user_line = *ucs;
697 refresh_from_cursor();
698 move_idx(user_line.size());
699 Log(LOG_get_line) history.info(CERR << "cursor_SEARCH() done" << endl);
700}

702LineInput::LineInput(bool do_read_history)
703 : history(uprefs.line_history_len),
704 write_history(false)
705{
706 initial_termios_errno = 0;
707
708 if (tcgetattr(STDIN_FILENO, &initial_termios))
709 initial_termios_errno = errno;
710
711 if (do_read_history)
712 {
713 history.read_history(uprefs.line_history_path.c_str());
714 write_history = true;
715 }
716
717 current_termios = initial_termios;
718
719 // set current_termios to raw mode
720 //
721 current_termios.c_iflag &= ~( ISTRIP | // don't strip off bit 8
722 INLCR | // don't NL → CR
723 IGNCR); // don't ignore CR
724 current_termios.c_iflag |= IGNBRK | // ignore break
725 IGNPAR |
726 ICRNL ; // CR → NL
727
728 current_termios.c_lflag &= ~(ECHO | ECHONL | ICANON | IEXTEN);
729 current_termios.c_lflag |= ISIG;
730
731#ifndef WANT_LIBAPL
732# ifndef WANT_PYTHON
733 tcsetattr(STDIN_FILENO, TCSANOW, &current_termios);
734# endif // WANT_PYTHON
735#endif // WANT_LIBAPL
736}

756//============================================================================
757void
758InputMux::get_line(LineInputMode mode, const UCS_string & prompt,
759 UCS_string & line, bool & eof, LineHistory & hist)
760{
761 if (get_line_callback)
762 {
763 get_line_callback(mode, prompt, line, eof, hist);
764 return;
765 }
766
767 if (InputFile::is_validating()) Quad_QUOTE::done(true, LOC);
768
769 InputFile::increment_current_line_no();
770
771 // check if we have input from a file. We do NOT use the file if the input
772 // is for ⍞ unless we are in a .tc testcase file.
773 //
774bool interactive = (mode == LIM_Quote_Quad) || (mode == LIM_Quad_Quad);
775 if (InputFile::is_validating()) interactive = false;
776 if (InputFile::pushed_file()) interactive = true;
777
778 if (!interactive)
779 {
780 UTF8_string file_line;
781 bool file_eof = false;
782 IO_Files::get_file_line(file_line, file_eof);
783
784 if (!file_eof)
785 {
786 line = UCS_string(file_line);
787
788 switch(mode)
789 {
790 case LIM_ImmediateExecution:
791 case LIM_Quad_Quad:
792 case LIM_Quad_INP:
793 CIN << prompt << line << endl;
794 break;
795
796 case LIM_Quote_Quad:
797 line = prompt;
798
799 // for each leading backspace in line: discard last
800 // prompt character. This is for testing the user
801 // backspacing over the ⍞ prompt
802 //
803 while (line.size() &&
804 file_line.size() &&
805 file_line[0] == UNI_BS)
806 {
807 file_line.erase(0);
808 line.pop_back();
809 }
810 line.append_UTF8(file_line.c_str());
811 break;
812
813 case LIM_Nabla:
814 break;
815
816 default: FIXME;
817 }
818
819 return;
820 }
821 }
822
823 // no (more) input from files: get line from terminal
824 //
825 if (uprefs.raw_cin)
826 {
827 Quad_QUOTE::done(mode != LIM_Quote_Quad, LOC);
828 CIN << '\r' << prompt;
829 char buffer[4000];
830 const APL_time_us from = now();
831 const char * s = fgets(buffer, sizeof(buffer) - 1, stdin);
832 Workspace::add_wait(now() - from);
833
834 if (s == 0)
835 {
836 eof = true;
837 return;
838 }
839 buffer[sizeof(buffer) - 1] = 0;
840
841 int slen = strlen(buffer);
842 if (slen && buffer[slen - 1] == '\n') buffer[--slen] = 0;
843 if (slen && buffer[slen - 1] == '\r') buffer[--slen] = 0;
844
845 UTF8_string line_utf(buffer);
846 line = UCS_string(line_utf);
847 return;
848 }
849
850 Quad_QUOTE::done(mode != LIM_Quote_Quad, LOC);
851
852const APL_time_us from = now();
853 if (start_input) (*start_input)();
854
855#if PARALLEL_ENABLED
856 CPU_pool::lock_pool(false);
857#endif
858
859 for (int control_D_count = 0;;)
860 {
861 bool _eof = false;
862 LineInput::get_terminal_line(mode, prompt, line, _eof, hist);
863 if (!_eof) break;
864
865 ++control_D_count;
866
867 // ^D or end of file
868 if (uprefs.control_Ds_to_exit) // there is a ^D limit
869 {
870 if (control_D_count >= uprefs.control_Ds_to_exit)
871 {
872 CIN << endl;
873#if PARALLEL_ENABLED
874 Thread_context::kill_pool();
875#endif // PARALLEL_ENABLED
876 uprefs.silent = true; // exit silently
877 Command::cmd_OFF(4); // exit()s
878 return; // not reached
879 }
880 }
881 else if (control_D_count < 5)
882 {
883 CIN << endl;
884 COUT << " ^D or end-of-input detected ("
885 << control_D_count << "). Use )OFF to leave APL!"
886 << endl;
887 }
888
889 eof = true;
890
891 if (control_D_count > 10 && (now() - from)/control_D_count < 10000)
892 {
893 // we got 10 or more times EOF (or possibly ^D) at a rate
894 // of 10 ms or faster. That looks like end-of-input rather
895 // than ^D typed by the user. Abort the interpreter.
896 //
897 CIN << endl;
898 COUT << " *** end of input" << endl;
899#if PARALLEL_ENABLED
900 Thread_context::kill_pool();
901#endif
902 Command::cmd_OFF(2); // exit()s
903 return; // not reached
904 }
905 }
906#if PARALLEL_ENABLED
907 CPU_pool::unlock_pool(false);
908#endif
909
910 Log(LOG_get_line) CERR << " '" << line << "'" << endl;
911
912 Workspace::add_wait(now() - from);
913 if (end_input) (*end_input)();
914
915 if (uprefs.echo_CIN) COUT << prompt << line << endl;
916}

1101 CIN << endl;

1123Unicode
1124LineInput::get_uni()
1125{
1126again:
1127
1128const int b0 = safe_fgetc();
1129 if (b0 == EOF) return UNI_EOF;
1130
1131 if (b0 & 0x80) // non-ASCII unicode
1132 {
1133 int len;
1134 uint32_t bx = b0; // the "significant" bits in b0
1135 if ((b0 & 0xE0) == 0xC0) { len = 2; bx &= 0x1F; }
1136 else if ((b0 & 0xF0) == 0xE0) { len = 3; bx &= 0x0F; }
1137 else if ((b0 & 0xF8) == 0xF0) { len = 4; bx &= 0x0E; }
1138 else if ((b0 & 0xFC) == 0xF8) { len = 5; bx &= 0x0E; }
1139 else if ((b0 & 0xFE) == 0xFC) { len = 6; bx &= 0x0E; }
1140 else
1141 {
1142 CERR << "Bad UTF8 sequence start at " << LOC << endl;
1143 return Invalid_Unicode;
1144 }
1145
1146 uint32_t uni = 0;
1147 loop(l, len - 1)
1148 {
1149 const UTF8 subc = safe_fgetc();
1150 if ((subc & 0xC0) != 0x80)
1151 {
1152 CERR << "Bad UTF8 sequence: " << HEX(b0)
1153 << "... at " LOC << endl;
1154 return Invalid_Unicode;
1155 }
1156
1157 bx <<= 6;
1158 uni <<= 6;
1159 uni |= subc & 0x3F;
1160 }
1161
1162 return Unicode(bx | uni);
1163 }
1164
1165 if (b0 == UNI_ESC)
1166 {
1167 char seq[Output::MAX_ESC_LEN]; seq[0] = UNI_ESC;
1168 for (int s = 1; s < Output::MAX_ESC_LEN; ++s)
1169 {
1170 const int bs = safe_fgetc();
1171 if (bs == EOF) return UNI_EOF;
1172 seq[s] = bs;
1173
1174 // check for exact match
1175 //
1176 loop(e, ESCmap_entry_count)
1177 {
1178 if (ESCmap::the_ESCmap[e].is_equal(seq, s + 1))
1179 return ESCmap::the_ESCmap[e].uni;
1180 }
1181
1182 // check for prefix match
1183 //
1184 if (ESCmap::need_more(seq, s)) continue;
1185
1186// CERR << endl << "Unknown ESC sequence: ESC";
1187// loop(ss, s) CERR << " " << HEX2(seq[ss + 1]);
1188// CERR << endl;
1189
1190 return Invalid_Unicode;
1191 }
1192 }
1193 else if (b0 < UNI_SPACE) // ^something (except ESC)
1194 {
1195 switch(b0)
1196 {
1197 case UNI_SOH: return UNI_CursorHome; // ^A
1198 case UNI_STX: return UNI_CursorLeft; // ^B
1199 case UNI_ETX: return UNI_ETX; // ^C
1200 case UNI_EOT: return UNI_EOT; // ^D
1201 case UNI_ENQ: return UNI_CursorEnd; // ^E
1202 case UNI_ACK: return UNI_CursorRight; // ^F
1203 case UNI_BS: return UNI_BS; // ^H
1204 case UNI_HT: return UNI_HT; // ^I

1206 case UNI_VT: return UNI_VT; // ^K
1207 case UNI_SO: return UNI_CursorDown; // ^N
1208 case UNI_DLE: return UNI_CursorUp; // ^P
1209 case UNI_DC2: return UNI_DC2; // ^R
1210 case UNI_EM: return UNI_EM; // ^Y
1211#ifdef WANT_CTRLD_DEL
1212 case UNI_SUB: return UNI_SUB; // ^Z (as alt EOT, allowing ^D as delete-char)
1213#endif
1214
1215 default: goto again;
1216 }
1217 }
1218 else if (b0 == UNI_DELETE) return UNI_BS;

1220 return Unicode(b0);

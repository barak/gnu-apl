93Token
94Quad_AF::eval_B(Value_P B) const
95{
96const ShapeItem ec = B->element_count();
97Value_P Z(B->get_shape(), LOC);
98
99 loop(v, ec)
100 {
101 const Cell & cell_B = B->get_cravel(v);
102
103 if (cell_B.is_character_cell()) // Unicode to AV index
104 {
105 const Unicode uni = cell_B.get_char_value();
106 int32_t pos = Avec::find_av_pos(uni);
107 if (pos < 0) Z->next_ravel_Int(Avec::MAX_AV);
108 else Z->next_ravel_Int(pos);
109 continue;
110 }
111
112 if (cell_B.is_integer_cell())
113 {
114 const APL_Integer idx = cell_B.get_near_int();
115 Z->next_ravel_Char(Quad_AV::indexed_at(idx));
116 continue;
117 }
118
119 DOMAIN_ERROR;
120 }
121
122 Z->check_value(LOC);
123 return Token(TOK_APL_VALUE1, Z);
124}

126Token
127Quad_AT::eval_AB(Value_P A, Value_P B) const
128{
129 // A is an integer scalar 1, 2, 3, or 4 (the mode)
130 // B is a matrix of symbol names
131
132 if (A->get_rank() > 0) RANK_ERROR;
133
134const APL_Integer mode = A->get_cfirst().get_near_int();
135 if (mode < 1) DOMAIN_ERROR;
136 if (mode > 4) DOMAIN_ERROR;
137
138const ShapeItem cols = B->get_cols();
139const ShapeItem rows = B->get_rows();
140 if (rows == 0) LENGTH_ERROR;
141
142 // mode: 1 2 3 4
143const int mode_vec[] = { 3, 7, 4, 2 };
144const int mode_len = mode_vec[mode - 1];
145Shape shape_Z(rows);
146 shape_Z.add_shape_item(mode_len);
147
148Value_P Z(shape_Z, LOC);
149
150 loop(r, rows)
151 {
152 // get the symbol name by stripping trailing spaces
153 //
154 const ShapeItem b = r*cols; // start of the symbol name
155 UCS_string symbol_name;
156 loop(c, cols)
157 {
158 const Unicode uni = B->get_cravel(b + c).get_char_value();
159 if (uni == UNI_SPACE) break;
160 symbol_name.append(uni);
161 }
162
163 const NamedObject * obj = Workspace::lookup_existing_name(symbol_name);
164 if (obj == 0) Error::throw_symbol_error(symbol_name, LOC);
165
166 if (const Function * function = obj->get_function())
167 {
168 // defined or system function.
169 function->get_attributes(mode, *Z);
170 continue;
171 }
172
173 if (const Symbol * symbol = obj->get_symbol())
174 {
175 // defined or sys var
176 symbol->get_attributes(mode, *Z);
177 continue;
178 }
179
180 // neither function nor variable (e.g. unused name)
181 VALUE_ERROR;
182
183#if 0 // not reached ??
184 if (Avec::is_quad(symbol_name[0])) // system function or variable
185 {
186 int l;
187 const Token t(Workspace::get_quad(symbol_name, l));
188 if (t.get_Class() == TC_SYMBOL) // system variable
189 {
190 // Assert() if symbol_name is not a function
191 t.get_sym_ptr();
192 }
193 else
194 {
195 // throws SYNTAX_ERROR if t is not a function
196 t.get_function();
197 }
198 }
199 else // user defined
200 {
201 Symbol * symbol = Workspace::lookup_existing_symbol(symbol_name);
202 if (symbol == 0) VALUE_ERROR;
203
204 symbol->get_attributes(mode, &Z->get_cravel(r*mode_len));
205 }
206#endif
207 }
208
209 Z->check_value(LOC);
210 return Token(TOK_APL_VALUE1, Z);
211}

213Token
214Quad_DL::eval_B(Value_P B) const
215{
216const APL_time_us start = now();
217
218 // B should be an integer or real scalar
219 //
220 if (B->get_rank() > 0) RANK_ERROR;
221 if (!B->get_cfirst().is_real_cell()) DOMAIN_ERROR;
222
223const APL_time_us end = start + 1000000 * B->get_cfirst().get_real_value();
224 if (end < start) DOMAIN_ERROR;
225 if (end > start + 31*24*60*60*1000000LL) DOMAIN_ERROR; // > 1 month
226
227bool need_CR = false;
228 while (now() < end)
229 {
230 usleep(20000);
231 if (attention_is_raised()) need_CR = true;
232 if (interrupt_is_raised())
233 {
234 need_CR = true;
235 break;
236 }
237 }
238
239 // interrupt or attention may have displayed ^C, start a new line if so.
240 if (need_CR) CERR << endl;
241
242 // we do not clear_attention_raised(LOC) or clear_interrupt_raised(LOC);
243 // so that the user can continue with →''
244
245 // return the time elapsed.
246 //
247 return Token(TOK_APL_VALUE1, FloatScalar(0.000001*(now() - start), LOC));
248}

287Quad_EC::eval_B(Value_P B) const
288{
289const UCS_string statement_B(*B.get());
290
291ExecuteList * fun = 0;
292 try {
293 fun = ExecuteList::fix(statement_B, LOC);
294 }
295 catch (...) {}
296
297 if (fun == 0)
298 {
299 // syntax error in B
300 //
301 Value_P Z2(2, LOC);
302 Z2->next_ravel_Int(Error::error_major(E_SYNTAX_ERROR));
303 Z2->next_ravel_Int(Error::error_minor(E_SYNTAX_ERROR));
304 Z2->check_value(LOC);
305
306 UTF8_string Z3_utf(Error::error_name(E_SYNTAX_ERROR));
307 UCS_string Z3_ucs(Z3_utf);
308 Value_P Z3(Z3_ucs, LOC);
309 Value_P Z(3, LOC);
310 Z->next_ravel_0(); // return code = error
311 Z->next_ravel_Pointer(Z2.get()); // ⎕ET value
312 Z->next_ravel_Pointer(Z3.get()); // ⎕EM
313
314 Z->check_value(LOC);
315 return Token(TOK_APL_VALUE1, Z);
316 }
317
318 Assert(fun);
319
320 Log(LOG_UserFunction__execute) fun->print(CERR);
321
322 Workspace::push_SI(fun, LOC);
323 Workspace::SI_top()->set_safe_execution();
324
325 return Token(TOK_SI_PUSHED);
326}

347//----------------------------------------------------------------------------
348void
349Quad_EC::eoc(Token & result)
350{
351 // set Token result to an APL value Z = (Z1 Z2 Z3) where:
352 //
353 // Z1 is an integer scalar 0-5 (return code):
354 //
355 // 0: Error
356 // 1: Value
357 // 2: Committed Value
358 // 3: Void
359 // 4: Branch
360 // 5: Escape
361 //
362 // Z2 is a two-element integer vector with the value that ⎕ET would have,
363 // ⎕ET is not set while in ⎕EC.
364 //
365 // Z3 is a value that depends on Z1:
366 //
367 // 0: ⎕EM
368 // 1 or 2: the value
369 // 3 or 5: 0⍴0
370 // 4: the branch line
371
372Value_P Z(3, LOC);
373 if (result.get_tag() == TOK_ERROR)
374 {
375 StateIndicator * si = Workspace::SI_top();
376 si->clear_safe_execution();
377
378 const Error & err = StateIndicator::get_error(si);
379 const ErrorCode ec = ErrorCode(result.get_int_val());
380
381 PrintBuffer pb;
382 pb.append_ucs(Error::error_name(ec));
383 pb.append_ucs(UCS_string(UTF8_string(err.get_error_line_2())));
384 pb.append_ucs(err.get_error_line_3());
385
386 Value_P Z2(2, LOC);
387 Z2->next_ravel_Int(Error::error_major(ec));
388 Z2->next_ravel_Int(Error::error_minor(ec));
389 Z2->check_value(LOC);
390
391 Value_P Z3(pb, LOC); // 3 line message like ⎕EM
392
393 Z->next_ravel_0();
394 Z->next_ravel_Pointer(Z2.get());
395 Z->next_ravel_Pointer(Z3.get());
396
397 Z->check_value(LOC);
398 result.move_2(Token(TOK_APL_VALUE1, Z), LOC);
399 return;
400 }
401
402 // all other cases have Z2 = 0 0
403 //
404Value_P Z2(2, LOC);
405 Z2->next_ravel_0();
406 Z2->next_ravel_0();
407 Z2->check_value(LOC);
408
409 switch(result.get_tag())
410 {
411 case TOK_APL_VALUE1:
412 case TOK_APL_VALUE3:
413 Z->next_ravel_1();
414 Z->next_ravel_Pointer(Z2.get());
415 Z->next_ravel_Value(result.get_apl_val().get());
416 break;
417
418 case TOK_APL_VALUE2:
419 Z->next_ravel_Int(2);
420 Z->next_ravel_Pointer(Z2.get());
421 Z->next_ravel_Value(result.get_apl_val().get());
422 break;
423
424 case TOK_NO_VALUE:
425 case TOK_VOID:
426 Z->next_ravel_Int(3);
427 Z->next_ravel_Pointer(Z2.get());
428 Z->next_ravel_Pointer(Idx0_0(LOC).get());
429 break;
430
431 case TOK_BRANCH:
432 Z->next_ravel_Int(4);
433 Z->next_ravel_Pointer(Z2.get());
434 Z->next_ravel_Int(result.get_int_val());
435 break;
436
437 case TOK_ESCAPE:
438 Z->next_ravel_Int(5);
439 Z->next_ravel_Pointer(Z2.get());
440 Z->next_ravel_Pointer(Idx0_0(LOC).get());
441 break;
442
443 default: CERR << "unexpected result tag " << result.get_tag()
444 << " in Quad_EC::eoc()" << endl;
445 Assert(0);
446 }
447
448 Z->check_value(LOC);
449 result.move_2(Token(TOK_APL_VALUE1, Z), LOC);
450}

452Token
453Quad_ENV::eval_B(Value_P B) const
454{
455 if (!B->is_char_string()) DOMAIN_ERROR;
456
457const ShapeItem ec_B = B->element_count();
458
459std::vector<const char *> evars;
460
461 for (char **e = environ; *e; ++e)
462 {
463 const char * env = *e;
464
465 // check if env starts with B.
466 //
467 bool match = true;
468 loop(b, ec_B)
469 {
470 if (B->get_cravel(b).get_char_value() != Unicode(env[b]))
471 {
472 match = false;
473 break;
474 }
475 }
476
477 if (match) evars.push_back(env);
478 }
479
480const Shape sh_Z(evars.size(), 2);
481Value_P Z(sh_Z, LOC);
482
483 loop(e, evars.size())
484 {
485 const char * env = evars[e];
486 UCS_string ucs;
487 while (*env)
488 {
489 if (*env != '=') ucs.append(Unicode(*env++));
490 else break;
491 }
492 ++env; // skip '='
493
494 Value_P varname(ucs, LOC);
495
496 ucs.clear();
497 while (*env) ucs.append(Unicode(*env++));
498
499 Value_P varval(ucs, LOC);
500
501 Z->next_ravel_Pointer(varname.get());
502 Z->next_ravel_Pointer(varval.get());
503 }
504
505 Z->set_proto_Spc();
506 Z->check_value(LOC);
507 return Token(TOK_APL_VALUE1, Z);
508}

531//----------------------------------------------------------------------------
532Token
533Quad_ES::event_simulate(const UCS_string * A, Value_P B, Error & error)
534{
535 // B is empty: no action
536 //
537 if (B->element_count() == 0) return Token();
538
539const ErrorCode ec = get_error_code(B);
540 if (ec == E_QUAD_ES_BRA) return Token(TOK_QUAD_ES_BRA, CLONE_P(B, LOC));
541 if (ec == E_QUAD_ES_COM) return Token(TOK_QUAD_ES_COM, CLONE_P(B, LOC));
542 if (ec == E_QUAD_ES_ERR) return Token(TOK_QUAD_ES_ERR, CLONE_P(B, LOC));
543 if (ec == E_QUAD_ES_ESC) return Token(TOK_QUAD_ES_ESC, CLONE_P(B, LOC));
544
545 new (&error) Error(ec, error.get_throw_loc());
546
547 if (error.get_error_code() == E_NO_ERROR) // B = 0 0: reset ⎕ET and ⎕EM.
548 {
549 Workspace::clear_error(LOC);
550 return Token();
551 }
552
553 if (error.get_error_code() == E_ASSERTION_FAILED) // B = 0 ASSERTION_FAILED
554 {
555 Assert(0 && "simulated ASSERTION_FAILED in ⎕ES");
556 }
557
558 // at this point we shall throw the error. Add some error details.
559 //
560 // set up error message 1
561 //
562 if (A) // A ⎕ES B
563 {
564 UCS_string msg1_ucs(*A);
565 UTF8_string msg1_utf(msg1_ucs);
566 error.set_error_line_1(msg1_utf.c_str());
567 }
568 else if (error.get_error_code() == E_USER_DEFINED_ERROR) // ⎕ES with character B
569 {
570 UCS_string msg1_ucs(*B.get());
571 UTF8_string msg1_utf(msg1_ucs);
572 error.set_error_line_1(msg1_utf.c_str());
573 }
574 else if (error.is_known()) // ⎕ES B with known major/minor B
575 {
576 /* error_message_1 already OK */ ;
577 }
578 else // ⎕ES B with unknown major/minor B
579 {
580 char cc[58];
581 snprintf(cc, sizeof(cc), "Unkown error (major %d, minor %d) in ⎕ES B",
582 error.get_error_code() >> 16,
583 error.get_error_code() & 0xFFFF);
584 error.set_error_line_1(cc);
585 }
586
587 error.set_show_locked(true);
588
589 Assert(Workspace::SI_top());
590 if (StateIndicator * si = Workspace::SI_top()->get_parent())
591 {
592 const UserFunction * ufun = si->get_executable()->get_exec_ufun();
593 if (ufun)
594 {
595 // lrm p 282: When ⎕ES is executed from within a defined function
596 // and B is not empty, the event action is generated as though
597 // the function were primitive.
598 //
599 UCS_string ufun_name(" ");
600 ufun_name.append(ufun->get_name());
601 error.set_error_line_2(ufun_name, 6, -1);
602 Workspace::pop_SI(LOC);
603 StateIndicator::get_error(Workspace::SI_top()) = error;
604 error.print_em(UERR, LOC);
605 return Token();
606 }
607 }
608
609 error.update_error_info(Workspace::SI_top());
610 return Token();
611}
612//----------------------------------------------------------------------------
613ErrorCode
614Quad_ES::get_error_code(Value_P B)
615{
616 // B shall be one of: → Error code
617 //
618 // 1. empty, or → No error
619 // 2. a 2 element integer vector, or → B[0]/N[1]
620 // 3. a simple character vector. → user defined
621 //
622 // Otherwise, throw an error
623
624 // We extend the standard error codes by a markers E_QUAD_ES_* (10-13) which
625 // are only used by the Macros Z__A_Quad_EA_B and Z__A_Quad_EB_B in order to
626 // return branches and the like as a result.
627
628 if (B->get_rank() > 1) RANK_ERROR;
629
630 if (B->element_count() == 0) return E_NO_ERROR;
631 if (B->is_char_string()) return E_USER_DEFINED_ERROR;
632
633const APL_Integer err = (B->get_cfirst().get_near_int() << 16)
634 | (B->get_cravel(1).get_near_int());
635
636 if ((err >> 16) == (E_QUAD_ES_BRA >> 16)) // one of the ⎕EA or ⎕EB events
637 {
638 const ShapeItem len_B = B->element_count();
639 if (err == E_QUAD_ES_COM && len_B == 3) return E_QUAD_ES_COM;
640 if (err == E_QUAD_ES_ERR && len_B == 5) return E_QUAD_ES_ERR;
641 if (err == E_QUAD_ES_BRA && len_B == 3) return E_QUAD_ES_BRA;
642 if (err == E_QUAD_ES_ESC && len_B == 2) return E_QUAD_ES_ESC;
643 DOMAIN_ERROR;
644 }
645
646 if (B->element_count() != 2) LENGTH_ERROR;
647
648 return ErrorCode(err);
649}

651Token
652Quad_EX::eval_B(Value_P B) const
653{
654 if (B->get_rank() > 2) RANK_ERROR;
655
656 // we don't throw a DOMAIN ERROR if B is bad, but provide info for
657 // the user if she asks for it with )MORE.
658 loop(b, B->element_count())
659 {
660 const Cell & cell = B->get_cravel(b);
661 if (!cell.is_character_cell())
662 {
663 MORE_ERROR() << "⎕EX B: non-character in list B (of symbol names)";
664 break;
665 }
666
667 const Unicode uni = cell.get_char_value();
668 if (!( Avec::is_symbol_char(uni) ||
669 Avec::is_white(uni) ||
670 (uni == UNI_FULLSTOP)))
671 {
672 MORE_ERROR() << "⎕EX B: invalid character "
673 << uni << " in list B (of symbol names)";
674 break;
675 }
676 }
677
678const ShapeItem var_count = B->get_rows();
679const UCS_string_vector vars(*B, false);
680
681Shape sh_Z;
682 if (var_count > 1) sh_Z.add_shape_item(var_count);
683Value_P Z(sh_Z, LOC);
684
685 loop(z, var_count)
686 {
687 const UCS_string & var = vars[z];
688 const int erased = expunge(var);
689
690 Z->next_ravel_Int(erased);
691 }
692
693 Z->check_value(LOC);
694 return Token(TOK_APL_VALUE1, Z);
695}
696//----------------------------------------------------------------------------
697int
698Quad_EX::expunge(const UCS_string & name)
699{
700 if (name.size() == 0) return 0;
701
702 if (!name.contains(UNI_FULLSTOP)) // unless member access
703 {
704 Symbol * symbol = Workspace::lookup_existing_symbol(name);
705 if (symbol == 0) return 0;
706 return symbol->expunge();
707 }
708
709 // expunge a member...
710 //
711int ret = 0; // assume ⎕EX failure
712
713 // build vector of member names in reverse order
714 //
715vector<const UCS_string *>members;
716 {
717 int dot = name.size();
718 for (int from = dot - 1; from >= 0; --from)

1026//----------------------------------------------------------------------------
1027void
1028Quad_INP::split_strings()
1029{
1030UCS_string empty;
1031 loop(r, raw_lines.size())
1032 {
1033 UCS_string & line = raw_lines[r];
1034 const ShapeItem epos = line.substr_pos(esc1);
1035 if (esc1.size() == 0 || epos == -1) // no escape in this line
1036 {
1037 prefixes.push_back(line);
1038 escapes.push_back(empty);
1039 suffixes.push_back(empty);
1040 continue;
1041 }
1042
1043 // at this point line did contain an exec string
1044 //
1045 UCS_string pref = line;
1046 pref.resize(epos);
1047 prefixes.push_back(pref);
1048
1049 line = line.drop(epos + esc1.size()); // skip prefix and esc1
1050 if (esc2.size()) // end defined
1051 {
1052 const ShapeItem eend = line.substr_pos(esc2);
1053 if (eend == -1) // no exec end in this line
1054 {
1055 escapes.push_back(line);
1056 suffixes.push_back(empty);
1057 continue;
1058 }
1059 else // found an exec end in this line
1060 {
1061 UCS_string exec = line;
1062 exec.resize(eend);
1063 escapes.push_back(exec);
1064 line = line.drop(eend + esc2.size()); // skip exec and esc2
1065 suffixes.push_back(line);
1066 continue;
1067 }
1068 }
1069 else // no end defined
1070 {
1071 escapes.push_back(line);
1072 suffixes.push_back(empty);
1073 }
1074 }
1075
1076 Assert(prefixes.size() >= raw_lines.size());
1077 Assert(prefixes.size() == escapes.size());
1078 Assert(prefixes.size() == suffixes.size());
1079}

1106{

1108
1109const Unicode uni = ucs[0];
1110
1111 // system name ?
1112 {
1113 const Symbol * sys = 0;
1114
1115 // ⍺, ⍶, ⍵, and ⍹ are local variables of lambdas and may change their
1116 // NC at runtime. We will therefore consult their symbol later on.
1117 //
1118 if (uni == UNI_ALPHA) sys = &Workspace::get_v_ALPHA();
1119 else if (uni == UNI_LAMBDA) sys = &Workspace::get_v_LAMBDA();

1122 else if (uni == UNI_ALPHA_UNDERBAR) sys = &Workspace::get_v_OMEGA_U();
1123 else if (uni == UNI_OMEGA_UNDERBAR) sys = &Workspace::get_v_OMEGA_U();

1126 // their lengths is 1.
1127 //
1128 if (ucs.size() != 1) sys = 0; // more than one char
1129
1130 if (Avec::is_quad(uni)) // distinguished name
1131 {

1136 if (t.get_Class() == TC_SYMBOL) sys = t.get_sym_ptr();

1138 }

1141 {
1142 const NameClass nc = sys->get_NC();
1143 if (nc == NC_UNUSED_USER_NAME) return NC_SYSTEM_FUN;
1144 return nc + 3;
1145 }
1146 }
1147
1148 // user-defined name
1149 //
1150const Symbol * user_sym = Workspace::lookup_existing_symbol(ucs);
1151
1152 if (!user_sym)
1153 {
1154 // user_sym not found. Distinguish between invalid and unused names
1155 //
1156 if (!Avec::is_first_symbol_char(ucs[0])) return NC_INVALID;
1157 loop (u, ucs.size())
1158 {
1159 if (!Avec::is_symbol_char(ucs[u])) return NC_INVALID;
1160 }
1161 return 0; // unused
1162 }
1163
1164 switch(const NameClass nc = user_sym->get_NC())
1165 {
1166 case NC_INVALID: return NC_INVALID;
1167 case NC_SYSTEM_VAR: return NC_VARIABLE;
1168 default: return nc & NC_case_mask;
1169 }

1171//============================================================================
1172Token
1173Quad_NL::do_quad_NL(Value_P A, Value_P B)
1174{
1175 if (+A && !A->is_char_string()) DOMAIN_ERROR;
1176 if (B->get_rank() > 1) RANK_ERROR;
1177 if (B->element_count() == 0) // nothing requested
1178 {
1179 return Token(TOK_APL_VALUE1, Str0_0(LOC));
1180 }
1181
1182UCS_string first_chars;
1183 if (+A) first_chars = UCS_string(*A);
1184
1185 // 1. create a bitmap of ⎕NC values requested in B
1186 //
1187int requested_NCs = 0;
1188 {
1189 loop(b, B->element_count())
1190 {
1191 const APL_Integer bb = B->get_cravel(b).get_near_int();
1192 if (bb < 1) DOMAIN_ERROR;
1193 if (bb > 6) DOMAIN_ERROR;
1194 requested_NCs |= 1 << bb;
1195 }
1196 }
1197
1198 // 2, build a name table, starting with user defined names
1199 //
1200UCS_string_vector names;
1201 {
1202 std::vector<const Symbol *> symbols = Workspace::get_all_symbols();
1203
1204 loop(s, symbols.size())
1205 {
1206 const Symbol * symbol = symbols[s];
1207 if (symbol->is_erased()) continue;
1208
1209 NameClass nc = symbol->get_NC();
1210 if (nc == NC_SYSTEM_VAR) nc = NC_VARIABLE;
1211
1212 if (!(requested_NCs & 1 << nc)) continue; // name class not in B
1213
1214 if (first_chars.size())
1215 {
1216 const Unicode first_char = symbol->get_name()[0];
1217 if (!first_chars.contains(first_char)) continue;
1218 }
1219
1220 names.push_back(symbol->get_name());
1221 }
1222 }
1223
1224 // 3, append ⎕-vars and ⎕-functions to name table (unless prevented by A)
1225 //
1226const bool vars = requested_NCs & 1 << 5;
1227const bool funs = requested_NCs & 1 << 6;
1228
1229 if (first_chars.size() == 0 || // all
1230 first_chars.contains(UNI_Quad_Quad)) // ⎕-variables and -functions
1231 {
1232#define ro_sv_def(x, _str, _txt) \
1233 { const Symbol * symbol = &Workspace::get_v_ ## x(); \
1234 if (vars && symbol->get_NC() != 0) \
1235 names.push_back(symbol->get_name()); }
1236
1237#define rw_sv_def(x, _str, _txt) \
1238 { const Symbol * symbol = &Workspace::get_v_ ## x(); \
1239 if (vars && symbol->get_NC() != 0) \
1240 names.push_back(symbol->get_name()); }
1241
1242#define sf_def(_x, str, _txt) \
1243 { if (funs) names.push_back(UCS_string(UTF8_string("⎕" str))); }
1244#include "SystemVariable.def"
1245 }
1246
1247
1248 // 4. compute length of longest name
1249 //
1250ShapeItem longest = 0;
1251 loop(n, names.size())
1252 {
1253 if (longest < names[n].size())
1254 longest = names[n].size();
1255 }
1256
1257const Shape shZ(names.size(), longest);
1258Value_P Z(shZ, LOC);
1259
1260 // 5. construct result. The number of symbols is small and ⎕NL is
1261 // (or should) not be a perfomance // critical function, so we can
1262 // use a simple (find smallest and print) // approach.
1263 //
1264 for (int count = names.size(); count; --count)
1265 {
1266 // find smallest.
1267 //
1268 uint32_t smallest = count - 1;
1269 loop(t, count - 1)
1270 {
1271 if (names[smallest].compare(names[t]) > 0) smallest = t;
1272 }
1273 // copy name to result, padded with spaces.
1274 //
1275 loop(l, longest)
1276 {
1277 const UCS_string & ucs = names[smallest];
1278 Z->next_ravel_Char(l < ucs.size() ? ucs[l] : UNI_SPACE);
1279 }
1280
1281 // remove smalles from table
1282 //
1283 names[smallest] = names[count - 1];
1284 }
1285
1286 Z->set_proto_Spc();
1287 Z->check_value(LOC);
1288 return Token(TOK_APL_VALUE1, Z);
1289}

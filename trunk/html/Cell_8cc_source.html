72//----------------------------------------------------------------------------
73void
74Cell::init_type(const Cell & other, Value & cell_owner, const char * loc)
75{
76 if (other.is_pointer_cell())
77 {
78 // Somewhat tricky! We must 'Value_P proto' so that proto owns
79 // the cloned value until PointerCell() takes over ownership of it.
80 //
81 Value_P proto = other.get_pointer_value()->clone(loc);
82 Assert(!proto->is_simple_scalar());
83 proto->to_type(false);
84 new (this) PointerCell(proto.get(), cell_owner);
85 }
86 else if (other.is_lval_cell()) new (this) LvalCell(0, 0);
87 else if (other.is_character_cell()) new (this) CharCell(UNI_SPACE);
88 else new (this) IntCell(0);
89}

158//----------------------------------------------------------------------------
159bool
160Cell::is_near_int(APL_Float value)
161{
162 // all large values are considered int because their fractional part
163 // has been rounded off. However, they may not fit into an int64_t,
164 // and if that is the concern then use is_near_int64_t() below.
165
166 if (value > LARGE_INT) return true;
167 if (value < SMALL_INT) return true;
168
169const APL_Float result = nearbyint(value);
170const APL_Float diff = value - result;
171 if (diff >= INTEGER_TOLERANCE) return false;
172 if (diff <= -INTEGER_TOLERANCE) return false;
173
174 return true;
175}
176//----------------------------------------------------------------------------
177bool
178Cell::is_near_int64_t(APL_Float value)
179{
180 if (value > LARGE_INT) return false;
181 if (value < SMALL_INT) return false;
182
183const APL_Float result = nearbyint(value);
184const APL_Float diff = value - result;
185 if (diff >= INTEGER_TOLERANCE) return false;
186 if (diff <= -INTEGER_TOLERANCE) return false;
187
188 return true;
189}
190//----------------------------------------------------------------------------
191APL_Integer
192Cell::near_int(APL_Float value)
193{
194 if (value >= LARGE_INT) DOMAIN_ERROR;
195 if (value <= SMALL_INT) DOMAIN_ERROR;
196
197const APL_Float result = nearbyint(value);
198const APL_Float diff = value - result;
199 if (diff > INTEGER_TOLERANCE) DOMAIN_ERROR;
200 if (diff < -INTEGER_TOLERANCE) DOMAIN_ERROR;
201
202 if (result > 0.0) return APL_Integer(0.3 + result);
203 else return - APL_Integer(0.3 - result);
204}
205//----------------------------------------------------------------------------
206bool
207Cell::greater_cp(const ShapeItem & A, const ShapeItem & B, const void * ctx)
208{
209const ravel_comp_len * rcl = reinterpret_cast<const ravel_comp_len *>(ctx);
210const Cell * cells = rcl->ravel;
211const ShapeItem comp_len = rcl->comp_len;
212const Cell * cell_A = cells + A * comp_len;
213const Cell * cell_B = cells + B * comp_len;
214
215 loop(l, comp_len)
216 {
217 if (const Comp_result cr = cell_A++->compare(*cell_B++))
218 return cr == COMP_GT;
219 }
220
221 // at this point all cells were equal
222 //
223 return A > B;
224}
225//----------------------------------------------------------------------------
226bool
227Cell::smaller_cp(const ShapeItem & A, const ShapeItem & B, const void * ctx)
228{
229const ravel_comp_len * rcl = reinterpret_cast<const ravel_comp_len *>(ctx);
230const Cell * cells = rcl->ravel;
231const ShapeItem comp_len = rcl->comp_len;
232const Cell * cell_A = cells + A * comp_len;
233const Cell * cell_B = cells + B * comp_len;
234
235 loop(l, comp_len)
236 {
237 if (const Comp_result cr = cell_A++->compare(*cell_B++))
238 return cr == COMP_LT;
239 }
240
241 // at this point all cells were equal
242 //
243 return A > B;
244}

297//----------------------------------------------------------------------------
298ShapeItem *
299Cell::sorted_indices(const Cell * ravel, ShapeItem length, Sort_order order,
300 ShapeItem comp_len)
301{
302ShapeItem * indices = new ShapeItem[length];
303 if (indices == 0) return indices;
304
305 // initialize indices with 0, 1, ... length-1
306 loop(l, length) indices[l] = l;
307
308const ravel_comp_len ctx = { ravel, comp_len};
309 if (order == SORT_ASCENDING)
310 Heapsort<ShapeItem>::sort(indices, length, &ctx, &Cell::greater_cp);
311 else
312 Heapsort<ShapeItem>::sort(indices, length, &ctx, &Cell::smaller_cp);
313 return indices;
314}

43class ValueStackItem
44{
45 friend class Symbol;
46
47public:
49 NameClass get_NC() const
50 { return name_class; }
51
53 void set_NC(NameClass nc)
54 { name_class = nc; }
55
57 void set_apl_value(Value_P new_value)
58 {
59 name_class = NC_VARIABLE;
60 apl_val = new_value;
61 }
62
64 void reset_apl_value()
65 { apl_val.reset(); }
66
68 const Function_Line get_label() const
69 { return sym_val.label; }
70
72 Function_P get_function() const
73 { return sym_val.function; }
74
76 void set_function(Function_P fun)
77 {
78 name_class = fun->is_operator() ? NC_OPERATOR : NC_FUNCTION;
79 sym_val.function = fun;
80 }
81
83 void clear_function()
84 {
85 sym_val.function = 0;
86 name_class = NC_UNUSED_USER_NAME;
87 }
88
90 Function_P * get_function_P()
91 { return & sym_val.function; }
92
94 SV_key get_key() const
95 { return sym_val.sv_key; }
96
98 void set_key(SV_key key)
99 { name_class = NC_SYSTEM_VAR; sym_val.sv_key = key; }
100
104 const Value * get_val_cptr() const
105 { return apl_val.get(); }
106
109 Value * get_val_wptr()
110 { return apl_val.get(); }
111
113 enum VS_flags
114 {
115 VSF_NONE = 0,
116 VSF_COW = 0x01,
117 };
118
120 VS_flags get_vs_flags() const
121 { return flags; }
122
124 void set_vs_flags(VS_flags flg)
125 { flags = flg; }
126
128 void isolate(const char * loc)
129 { if (+apl_val) apl_val.isolate(loc); }
130
131protected:
133 ValueStackItem()
134 : name_class(NC_UNUSED_USER_NAME),
135 flags(VSF_NONE)
136 { memset(&sym_val, 0, sizeof(sym_val)); }
137
139 ValueStackItem(Function_Line lab)
140 : name_class(NC_LABEL),
141 flags(VSF_NONE)
142 { sym_val.label = lab; }
143
145 ValueStackItem(Value_P val)
146 : apl_val(val),
147 name_class(NC_VARIABLE),
148 flags(VSF_NONE)
149 {}
150
152 ValueStackItem(SV_key key)
153 : name_class(NC_SYSTEM_VAR),
154 flags(VSF_NONE)
155 { sym_val.sv_key = key; }
156
158 void clear()
159 {
160 // sym_val contains only POD variables, so it can be memset()
161 memset(&sym_val, 0, sizeof(sym_val));
162 if (!!apl_val) apl_val.reset();
163 name_class = NC_UNUSED_USER_NAME;
164 flags = VSF_NONE;
165 }
166
168 union _sym_val
169 {
170 Function_P function;
171 Function_Line label;
172 SV_key sv_key;
173 };
174
176 _sym_val sym_val;
177
179 Value_P apl_val;
180
182 NameClass name_class;
183
185 VS_flags flags;
186};

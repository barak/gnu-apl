47const char *
48event_name(Svar_event ev)
49{
50 switch(ev)
51 {
52 case SVE_NO_EVENTS: return "(no event)";
53 case SVE_OFFER_MISMATCH: return "offer mismatch";
54 case SVE_OFFER_MATCHED: return "offer matched";
55 case SVE_OFFER_RETRACT: return "offer retracted";
56 case SVE_ACCESS_CONTROL: return "access control changed";
57 case SVE_USE_BY_OFF_SUCCESS: return "use by offering successful";
58 case SVE_USE_BY_OFF_FAILED: return "use by offering failed";
59 case SVE_SET_BY_OFF_SUCCESS: return "set by offering successful";
60 case SVE_SET_BY_OFF_FAILED: return "set by offering failed";
61 case SVE_USE_BY_ACC_SUCCESS: return "use by accepting successful";
62 case SVE_USE_BY_ACC_FAILED: return "use by accepting failed";
63 case SVE_SET_BY_ACC_SUCCESS: return "set by accepting successful";
64 case SVE_SET_BY_ACC_FAILED: return "set by accepting failed";
65 }
66
67 return "(unknown event)";
68}

99//----------------------------------------------------------------------------
100SV_Coupling
101Svar_record::retract()
102{
103const SV_Coupling old_coupling = get_coupling();
104
105 if (ProcessorID::get_id() == offering.id) remove_offering();
106 else if (ProcessorID::get_id() == accepting.id) remove_accepting();
107 else
108 {
109 bad_proc("Svar_record::retract", ProcessorID::get_id());
110 return NO_COUPLING;
111 }
112
113 // clear variable if last partner has retracted, or else send signal
114 // to the remaining partner.
115 //
116 if (get_coupling() == NO_COUPLING) // retract of a non-coupled variable
117 {
118 clear();
119 }
120
121 return old_coupling;
122}

132//----------------------------------------------------------------------------
133bool
134Svar_record::match_name(const uint32_t * UCS_other) const
135{
136 // compare INCLUDING terminating 0;
137 for (int v = 0; v < (MAX_SVAR_NAMELEN); ++v)
138 {
139 if (varname[v] != UCS_other[v])
140 {
141 if (varname[v] == 0x22C6) return true; // ⋆
142 if (varname[v] == '*') return true;
143 if (UCS_other[v] == 0x22C6) return true; // ⋆
144 if (UCS_other[v] == '*') return true;
145 return false;
146 }
147 if (varname[v] == 0) return true; // short name
148 }
149
150 return true; // long name
151}
152//----------------------------------------------------------------------------
153Svar_Control
154mirror(int flags)
155{
156int ret = 0;
157 if (flags & SET_BY_OFF) ret |= SET_BY_ACC;
158 if (flags & SET_BY_ACC) ret |= SET_BY_OFF;
159 if (flags & USE_BY_OFF) ret |= USE_BY_ACC;
160 if (flags & USE_BY_ACC) ret |= USE_BY_OFF;
161 return Svar_Control(ret);
162}
163//----------------------------------------------------------------------------
164Svar_Control
165Svar_record::get_control() const
166{
167int ctl = offering.get_control() | accepting.get_control();
168
169 if (ProcessorID::get_id() == accepting.id) ctl = mirror(ctl);
170 return Svar_Control(ctl);
171}
172//----------------------------------------------------------------------------
173void
174Svar_record::set_control(Svar_Control ctl)
175{
176 Log(LOG_shared_variables)
177 {
178 get_CERR() << "set_control(" << ctl << ") on ";
179 print_name(get_CERR());
180 get_CERR() << " by " << ProcessorID::get_id().proc << endl;
181 }
182
183 if (ProcessorID::get_id() == offering.id)
184 {
185 offering.set_control(ctl);
186 if (get_coupling() == SV_COUPLED) // fully coupled: inform peer
187 {
188 accepting.events = Svar_event
189 (accepting.events | SVE_ACCESS_CONTROL);
190 }
191 }
192 else if (ProcessorID::get_id() == accepting.id) // hence fully coupled
193 {
194 accepting.set_control(mirror(ctl));
195 offering.events = Svar_event(offering.events | SVE_ACCESS_CONTROL);
196 }
197 else
198 {
199 bad_proc(__FUNCTION__, ProcessorID::get_id());
200 }
201}

208//----------------------------------------------------------------------------
209void
210Svar_record::set_state(bool used, const char * loc)
211{
212usleep(50000);
213
214 Log(LOG_shared_variables)
215 {
216 const char * op = used ? "used" : "set";
217 get_CERR() << "set_state(" << op << ") on ";
218 print_name(get_CERR());
219 get_CERR() << " by " << ProcessorID::get_id().proc
220 << " at " << loc << endl;
221 }
222
223 // the control vector as seen by the offering side
224 //
225const int control = offering.get_control() | accepting.get_control();
226Svar_event event = SVE_NO_EVENTS;
227Svar_partner * peer = 0;
228
229 if (ProcessorID::get_id() == offering.id)
230 {
231 peer = &offering;
232 offering.events = SVE_NO_EVENTS; // clear events
233
234 if (used) // offering has used the variable (unless read-back)
235 {
236 if (state == SVS_OFF_HAS_SET) ; // read-back
237 else
238 {
239 if (control & USE_BY_OFF) event = SVE_USE_BY_OFF_SUCCESS;
240 state = SVS_IDLE;
241 }
242 }
243 else // offering has set the variable
244 {
245 if (control & SET_BY_OFF) event = SVE_SET_BY_OFF_SUCCESS;
246 state = SVS_OFF_HAS_SET;
247 }
248 }
249 else if (ProcessorID::get_id() == accepting.id)
250 {
251 peer = &offering;
252 accepting.events = SVE_NO_EVENTS; // clear events
253
254 if (used) // accepting has used the variable (unless read-back)
255 {
256 if (state == SVS_ACC_HAS_SET) ; // read-back
257 else
258 {
259 if (control & USE_BY_ACC) event = SVE_USE_BY_ACC_SUCCESS;
260 state = SVS_IDLE;
261 }
262 }
263 else // accepting has set the variable
264 {
265 if (control & SET_BY_ACC) event = SVE_SET_BY_ACC_SUCCESS;
266 state = SVS_ACC_HAS_SET;
267 }
268 }
269 else // only the partners should call set_state();
270 {
271 bad_proc(__FUNCTION__, ProcessorID::get_id());
272 return;
273 }
274
275 // if access was restricted then inform peer
276 //
277 if (peer && event != SVE_NO_EVENTS)
278 {
279 peer->events = Svar_event(peer->events | event);
280 }
281}
282//----------------------------------------------------------------------------
283bool
284Svar_record::may_use(int attempt)
285{
286 // control restriction as seen by the offering partner
287 //
288const int control = offering.get_control() | accepting.get_control();
289const int restriction = control & state;
290
291 if (ProcessorID::get_id() == offering.id)
292 {
293 if ((restriction & USE_BY_OFF) == 0) return true; // no restriction
294
295 if (accepting.is_active() && (attempt == 0))
296 {
297 accepting.events = Svar_event(accepting.events |
298 SVE_USE_BY_OFF_FAILED);
299 }
300 return false;
301 }
302
303 if (ProcessorID::get_id() == accepting.id)
304 {
305 if ((restriction & USE_BY_ACC) == 0) return true; // no restriction
306
307 if (offering.is_active() && (attempt == 0)) // maybe send event to peer
308 {
309 offering.events = Svar_event(offering.events |
310 SVE_USE_BY_ACC_FAILED);
311 }
312 return false;
313 }
314
315 bad_proc(__FUNCTION__, ProcessorID::get_id());
316 return false;
317}

362//----------------------------------------------------------------------------
363void
364Svar_record::print(ostream & out) const
365{
366const Svar_state st = get_state();
367 out << "║" << setw(5) << (key & 0xFFFF) << "│" << get_coupling() << "║";
368 offering.print(out) << "║";
369 accepting.print(out) << "║";
370 if (st & SET_BY_OFF) out << "1"; else out << "0";
371 if (st & SET_BY_ACC) out << "1"; else out << "0";
372 if (st & USE_BY_OFF) out << "1"; else out << "0";
373 if (st & USE_BY_ACC) out << "1│"; else out << "0│";
374 print_name(out, varname, 10) << "║" << endl;
375}
376//----------------------------------------------------------------------------
377ostream &
378Svar_record::print_name(ostream & out, const uint32_t * name, int len)
379{
380 while (*name)
381 {
382 uint32_t uni = *name++;
383 if (uni < 0x80)
384 {
385 out << char(uni);
386 }
387 else if (uni < 0x800)
388 {
389 const uint8_t b1 = uni & 0x3F; uni >>= 6;
390 out << char(uni | 0xC0)
391 << char(b1 | 0x80);
392 }
393 else if (uni < 0x10000)
394 {
395 const uint8_t b2 = uni & 0x3F; uni >>= 6;
396 const uint8_t b1 = uni & 0x3F; uni >>= 6;
397 out << char(uni | 0xE0)
398 << char(b1 | 0x80)
399 << char(b2 | 0x80);
400 }
401 else if (uni < 0x110000)
402 {
403 const uint8_t b3 = uni & 0x3F; uni >>= 6;
404 const uint8_t b2 = uni & 0x3F; uni >>= 6;
405 const uint8_t b1 = uni & 0x3F; uni >>= 6;
406 out << char(uni | 0xE0)
407 << char(b1 | 0x80)
408 << char(b2 | 0x80)
409 << char(b3 | 0x80);
410 }
411
412 --len;
413 }
414
415 while (len-- > 0) out << " ";
416
417 return out;
418}
